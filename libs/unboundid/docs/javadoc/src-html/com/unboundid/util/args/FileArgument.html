042    /**
043     * This class defines an argument that is intended to hold values which refer to
044     * files on the local filesystem.  File arguments must take values, and it is
045     * possible to restrict the values to files that exist, or whose parent exists.
046     */
047    @Mutable()
048    @ThreadSafety(level=ThreadSafetyLevel.NOT_THREADSAFE)
049    public final class FileArgument
050           extends Argument
051    {
052      /**
053       * The serial version UID for this serializable class.
054       */
055      private static final long serialVersionUID = -8478637530068695898L;

059      // Indicates whether values must represent files that exist.
060      private final boolean fileMustExist;
061    
062      // Indicates whether the provided value must be a directory if it exists.
063      private final boolean mustBeDirectory;
064    
065      // Indicates whether the provided value must be a regular file if it exists.
066      private final boolean mustBeFile;
067    
068      // Indicates whether values must represent files with parent directories that
069      // exist.
070      private final boolean parentMustExist;
071    
072      // The set of values assigned to this argument.
073      private final ArrayList<File> values;
074    
075      // The list of default values for this argument.
076      private final List<File> defaultValues;

165      /**
166       * Creates a new file argument with the provided information.
167       *
168       * @param  shortIdentifier   The short identifier for this argument.  It may
169       *                           not be {@code null} if the long identifier is
170       *                           {@code null}.
171       * @param  longIdentifier    The long identifier for this argument.  It may
172       *                           not be {@code null} if the short identifier is
173       *                           {@code null}.
174       * @param  isRequired        Indicates whether this argument is required to
175       *                           be provided.
176       * @param  maxOccurrences    The maximum number of times this argument may be
177       *                           provided on the command line.  A value less than
178       *                           or equal to zero indicates that it may be present
179       *                           any number of times.
180       * @param  valuePlaceholder  A placeholder to display in usage information to
181       *                           indicate that a value must be provided.  It must
182       *                           not be {@code null}.
183       * @param  description       A human-readable description for this argument.
184       *                           It must not be {@code null}.
185       * @param  fileMustExist     Indicates whether each value must refer to a file
186       *                           that exists.
187       * @param  parentMustExist   Indicates whether each value must refer to a file
188       *                           whose parent directory exists.
189       * @param  mustBeFile        Indicates whether each value must refer to a
190       *                           regular file, if it exists.
191       * @param  mustBeDirectory   Indicates whether each value must refer to a
192       *                           directory, if it exists.
193       * @param  defaultValues     The set of default values to use for this
194       *                           argument if no values were provided.
195       *
196       * @throws  ArgumentException  If there is a problem with the definition of
197       *                             this argument.
198       */
199      public FileArgument(final Character shortIdentifier,
200                          final String longIdentifier, final boolean isRequired,
201                          final int maxOccurrences, final String valuePlaceholder,
202                          final String description, final boolean fileMustExist,
203                          final boolean parentMustExist, final boolean mustBeFile,
204                          final boolean mustBeDirectory,
205                          final List<File> defaultValues)
206             throws ArgumentException
207      {
208        super(shortIdentifier, longIdentifier, isRequired,  maxOccurrences,
209              valuePlaceholder, description);
210    
211        if (valuePlaceholder == null)
212        {
213          throw new ArgumentException(ERR_ARG_MUST_TAKE_VALUE.get(
214                                           getIdentifierString()));
215        }
216    
217        if (mustBeFile && mustBeDirectory)
218        {
219          throw new ArgumentException(ERR_FILE_CANNOT_BE_FILE_AND_DIRECTORY.get(
220                                           getIdentifierString()));
221        }
222    
223        this.fileMustExist   = fileMustExist;
224        this.parentMustExist = parentMustExist;
225        this.mustBeFile      = mustBeFile;
226        this.mustBeDirectory = mustBeDirectory;
227    
228        if ((defaultValues == null) || defaultValues.isEmpty())
229        {
230          this.defaultValues = null;
231        }
232        else
233        {
234          this.defaultValues = Collections.unmodifiableList(defaultValues);
235        }
236    
237        values = new ArrayList<File>();
238      }

310      /**
311       * {@inheritDoc}
312       */
313      @Override()
314      protected void addValue(final String valueString)
315                throws ArgumentException
316      {
317        // NOTE:  java.io.File has an extremely weird behavior.  When a File object
318        // is created from a relative path and that path contains only the filename,
319        // then calling getParent or getParentFile will return null even though it
320        // obviously has a parent.  Therefore, you must always create a File using
321        // the absolute path if you might want to get the parent.
322        final File f = new File(new File(valueString).getAbsolutePath());
323    
324        if (f.exists())
325        {
326          if (mustBeFile && (! f.isFile()))
327          {
328            throw new ArgumentException(ERR_FILE_VALUE_NOT_FILE.get(
329                                             getIdentifierString(),
330                                             f.getAbsolutePath()));
331          }
332          else if (mustBeDirectory && (! f.isDirectory()))
333          {
334            throw new ArgumentException(ERR_FILE_VALUE_NOT_DIRECTORY.get(
335                                             getIdentifierString(),
336                                             f.getAbsolutePath()));
337          }
338        }
339        else
340        {
341          if (fileMustExist)
342          {
343            throw new ArgumentException(ERR_FILE_DOESNT_EXIST.get(
344                                             f.getAbsolutePath(),
345                                             getIdentifierString()));
346          }
347          else if (parentMustExist)
348          {
349            final File parentFile = f.getParentFile();
350            if ((parentFile == null) ||
351                (! parentFile.exists()) ||
352                (! parentFile.isDirectory()))
353            {
354              throw new ArgumentException(ERR_FILE_PARENT_DOESNT_EXIST.get(
355                                               f.getAbsolutePath(),
356                                               getIdentifierString()));
357            }
358          }
359        }
360    
361        if (values.size() >= getMaxOccurrences())
362        {
363          throw new ArgumentException(ERR_ARG_MAX_OCCURRENCES_EXCEEDED.get(
364                                           getIdentifierString()));
365        }
366    
367        values.add(f);
368      }

372      /**
373       * Retrieves the value for this argument, or the default value if none was
374       * provided.  If there are multiple values, then the first will be returned.
375       *
376       * @return  The value for this argument, or the default value if none was
377       *          provided, or {@code null} if there is no value and no default
378       *          value.
379       */
380      public File getValue()
381      {
382        if (values.isEmpty())
383        {
384          if ((defaultValues == null) || defaultValues.isEmpty())
385          {
386            return null;
387          }
388          else
389          {
390            return defaultValues.get(0);
391          }
392        }
393        else
394        {
395          return values.get(0);
396        }
397      }

418      /**
419       * Reads the contents of the file specified as the value to this argument and
420       * retrieves a list of the lines contained in it.  If there are multiple
421       * values for this argument, then the file specified as the first value will
422       * be used.
423       *
424       * @return  A list containing the lines of the target file, or {@code null} if
425       *          no values were provided.
426       *
427       * @throws  IOException  If the specified file does not exist or a problem
428       *                       occurs while reading the contents of the file.
429       */
430      public List<String> getFileLines()
431             throws IOException
432      {
433        final File f = getValue();
434        if (f == null)
435        {
436          return null;
437        }
438    
439        final ArrayList<String> lines  = new ArrayList<String>();
440        final BufferedReader    reader = new BufferedReader(new FileReader(f));
441        try
442        {
443          String line = reader.readLine();
444          while (line != null)
445          {
446            lines.add(line);
447            line = reader.readLine();
448          }
449        }
450        finally
451        {
452          reader.close();
453        }
454    
455        return lines;
456      }

460      /**
461       * Reads the contents of the file specified as the value to this argument and
462       * retrieves a list of the non-blank lines contained in it.  If there are
463       * multiple values for this argument, then the file specified as the first
464       * value will be used.
465       *
466       * @return  A list containing the non-blank lines of the target file, or
467       *          {@code null} if no values were provided.
468       *
469       * @throws  IOException  If the specified file does not exist or a problem
470       *                       occurs while reading the contents of the file.
471       */
472      public List<String> getNonBlankFileLines()
473             throws IOException
474      {
475        final File f = getValue();
476        if (f == null)
477        {
478          return null;
479        }
480    
481        final ArrayList<String> lines = new ArrayList<String>();
482        final BufferedReader reader = new BufferedReader(new FileReader(f));
483        try
484        {
485          String line = reader.readLine();
486          while (line != null)
487          {
488            if (line.length() > 0)
489            {
490              lines.add(line);
491            }
492            line = reader.readLine();
493          }
494        }
495        finally
496        {
497          reader.close();
498        }
499    
500        return lines;
501      }

505      /**
506       * Reads the contents of the file specified as the value to this argument.  If
507       * there are multiple values for this argument, then the file specified as the
508       * first value will be used.
509       *
510       * @return  A byte array containing the contents of the target file, or
511       *          {@code null} if no values were provided.
512       *
513       * @throws  IOException  If the specified file does not exist or a problem
514       *                       occurs while reading the contents of the file.
515       */
516      public byte[] getFileBytes()
517             throws IOException
518      {
519        final File f = getValue();
520        if (f == null)
521        {
522          return null;
523        }
524    
525        final byte[] fileData = new byte[(int) f.length()];
526        final FileInputStream inputStream = new FileInputStream(f);
527        try
528        {
529          int startPos  = 0;
530          int length    = fileData.length;
531          int bytesRead = inputStream.read(fileData, startPos, length);
532          while ((bytesRead > 0) && (startPos < fileData.length))
533          {
534            startPos += bytesRead;
535            length   -= bytesRead;
536            bytesRead = inputStream.read(fileData, startPos, length);
537          }
538    
539          if (startPos < fileData.length)
540          {
541            throw new IOException(ERR_FILE_CANNOT_READ_FULLY.get(
542                                       f.getAbsolutePath(), getIdentifierString()));
543          }
544    
545          return fileData;
546        }
547        finally
548        {
549          inputStream.close();
550        }
551      }
