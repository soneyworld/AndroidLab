034    /**
035     * This class provides a utility for formatting output in multiple columns.
036     * Each column will have a defined width and alignment.  It can alternately
037     * generate output as tab-delimited text or comma-separated values (CSV).
038     */
039    @NotMutable()
040    @ThreadSafety(level=ThreadSafetyLevel.NOT_THREADSAFE)
041    public final class ColumnFormatter
042           implements Serializable
043    {
044      /**
045       * The default output format to use.
046       */
047      private static final OutputFormat DEFAULT_OUTPUT_FORMAT =
048           OutputFormat.COLUMNS;
049    
050    
051    
052      /**
053       * The default spacer to use between columns.
054       */
055      private static final String DEFAULT_SPACER = " ";
056    
057    
058    
059      /**
060       * The default date format string that will be used for timestamps.
061       */
062      private static final String DEFAULT_TIMESTAMP_FORMAT = "HH:mm:ss";
063    
064

066      /**
067       * The serial version UID for this serializable class.
068       */
069      private static final long serialVersionUID = -2524398424293401200L;

074      private final boolean includeTimestamp;
075    
076      // The column to use for the timestamp.
077      private final FormattableColumn timestampColumn;
078    
079      // The columns to be formatted.
080      private final FormattableColumn[] columns;
081    
082      // The output format to use.
083      private final OutputFormat outputFormat;
084    
085      // The string to insert between columns.
086      private final String spacer;
087    
088      // The format string to use for the timestamp.
089      private final String timestampFormat;
090    
091      // The thread-local formatter to use for floating-point values.
092      private final transient ThreadLocal<DecimalFormat> decimalFormatter;
093    
094      // The thread-local formatter to use when formatting timestamps.
095      private final transient ThreadLocal<SimpleDateFormat> timestampFormatter;

103       * @param  columns  The columns to be formatted.  At least one column must be
104       *                  provided.

106      public ColumnFormatter(final FormattableColumn... columns)
107      {
108        this(false, null, null, null, columns);
109      }

113      /**
114       * Creates a column formatter that will format the provided columns.
115       *
116       * @param  includeTimestamp  Indicates whether to insert a timestamp before
117       *                           the first column when generating data lines
118       * @param  timestampFormat   The format string to use for the timestamp.  It
119       *                           may be {@code null} if no timestamp should be
120       *                           included or the default format should be used.
121       *                           If a format is provided, then it should be one
122       *                           that will always generate timestamps with a
123       *                           constant width.
124       * @param  outputFormat      The output format to use.
125       * @param  spacer            The spacer to use between columns.  It may be
126       *                           {@code null} if the default spacer should be
127       *                           used.  This will only apply for an output format
128       *                           of {@code COLUMNS}.
129       * @param  columns           The columns to be formatted.  At least one
130       *                           column must be provided.
131       */
132      public ColumnFormatter(final boolean includeTimestamp,
133                             final String timestampFormat,
134                             final OutputFormat outputFormat, final String spacer,
135                             final FormattableColumn... columns)
136      {
137        Validator.ensureNotNull(columns);
138        Validator.ensureTrue(columns.length > 0);
139    
140        this.includeTimestamp = includeTimestamp;
141        this.columns          = columns;
142    
143        decimalFormatter   = new ThreadLocal<DecimalFormat>();
144        timestampFormatter = new ThreadLocal<SimpleDateFormat>();
145    
146        if (timestampFormat == null)
147        {
148          this.timestampFormat = DEFAULT_TIMESTAMP_FORMAT;
149        }
150        else
151        {
152          this.timestampFormat = timestampFormat;
153        }
154    
155        if (outputFormat == null)
156        {
157          this.outputFormat = DEFAULT_OUTPUT_FORMAT;
158        }
159        else
160        {
161          this.outputFormat = outputFormat;
162        }
163    
164        if (spacer == null)
165        {
166          this.spacer = DEFAULT_SPACER;
167        }
168        else
169        {
170          this.spacer = spacer;
171        }
172    
173        if (includeTimestamp)
174        {
175          final SimpleDateFormat dateFormat =
176               new SimpleDateFormat(this.timestampFormat);
177          final String timestamp = dateFormat.format(new Date());
178          final String label = INFO_COLUMN_LABEL_TIMESTAMP.get();
179          final int width = Math.max(label.length(), timestamp.length());
180    
181          timestampFormatter.set(dateFormat);
182          timestampColumn =
183               new FormattableColumn(width, HorizontalAlignment.LEFT, label);
184        }
185        else
186        {
187          timestampColumn = null;
188        }
189      }

256      /**
257       * Obtains the lines that should comprise the column headers.
258       *
259       * @param  includeDashes  Indicates whether to include a row of dashes below
260       *                        the headers if appropriate for the output format.
261       *
262       * @return  The lines that should comprise the column headers.
263       */
264      public String[] getHeaderLines(final boolean includeDashes)
265      {
266        if (outputFormat == OutputFormat.COLUMNS)
267        {
268          int maxColumns = 1;
269          final String[][] headerLines = new String[columns.length][];
270          for (int i=0; i < columns.length; i++)
271          {
272            headerLines[i] = columns[i].getLabelLines();
273            maxColumns = Math.max(maxColumns, headerLines[i].length);
274          }
275    
276          final StringBuilder[] buffers = new StringBuilder[maxColumns];
277          for (int i=0; i < maxColumns; i++)
278          {
279            final StringBuilder buffer = new StringBuilder();
280            buffers[i] = buffer;
281            if (includeTimestamp)
282            {
283              if (i == (maxColumns - 1))
284              {
285                timestampColumn.format(buffer, timestampColumn.getSingleLabelLine(),
286                     outputFormat);
287              }
288              else
289              {
290                timestampColumn.format(buffer, "", outputFormat);
291              }
292            }
293    
294            for (int j=0; j < columns.length; j++)
295            {
296              if (includeTimestamp || (j > 0))
297              {
298                buffer.append(spacer);
299              }
300    
301              final int rowNumber = i + headerLines[j].length - maxColumns;
302              if (rowNumber < 0)
303              {
304                columns[j].format(buffer, "", outputFormat);
305              }
306              else
307              {
308                columns[j].format(buffer, headerLines[j][rowNumber], outputFormat);
309              }
310            }
311          }
312    
313          final String[] returnArray;
314          if (includeDashes)
315          {
316            returnArray = new String[maxColumns+1];
317          }
318          else
319          {
320            returnArray = new String[maxColumns];
321          }
322    
323          for (int i=0; i < maxColumns; i++)
324          {
325            returnArray[i] = buffers[i].toString();
326          }
327    
328          if (includeDashes)
329          {
330            final StringBuilder buffer = new StringBuilder();
331            if (timestampColumn != null)
332            {
333              for (int i=0; i < timestampColumn.getWidth(); i++)
334              {
335                buffer.append('-');
336              }
337            }
338    
339            for (int i=0; i < columns.length; i++)
340            {
341              if (includeTimestamp || (i > 0))
342              {
343                buffer.append(spacer);
344              }
345    
346              for (int j=0; j < columns[i].getWidth(); j++)
347              {
348                buffer.append('-');
349              }
350            }
351    
352            returnArray[returnArray.length - 1] = buffer.toString();
353          }
354    
355          return returnArray;
356        }
357        else
358        {
359          final StringBuilder buffer = new StringBuilder();
360          if (timestampColumn != null)
361          {
362            timestampColumn.format(buffer, timestampColumn.getSingleLabelLine(),
363                 outputFormat);
364          }
365    
366          for (int i=0; i < columns.length; i++)
367          {
368            if (includeTimestamp || (i > 0))
369            {
370              if (outputFormat == OutputFormat.TAB_DELIMITED_TEXT)
371              {
372                buffer.append('\t');
373              }
374              else if (outputFormat == OutputFormat.CSV)
375              {
376                buffer.append(',');
377              }
378            }
379    
380            final FormattableColumn c = columns[i];
381            c.format(buffer, c.getSingleLabelLine(), outputFormat);
382          }
383    
384          return new String[] { buffer.toString() };
385        }
386      }

390      /**
391       * Formats a row of data.  The provided data must correspond to the columns
392       * used when creating this formatter.
393       *
394       * @param  columnData  The elements to include in each row of the data.
395       *
396       * @return  A string containing the formatted row.
397       */
398      public String formatRow(final Object... columnData)
399      {
400        final StringBuilder buffer = new StringBuilder();
401    
402        if (includeTimestamp)
403        {
404          SimpleDateFormat dateFormat = timestampFormatter.get();
405          if (dateFormat == null)
406          {
407            dateFormat = new SimpleDateFormat(timestampFormat);
408            timestampFormatter.set(dateFormat);
409          }
410    
411          timestampColumn.format(buffer, dateFormat.format(new Date()),
412               outputFormat);
413        }
414    
415        for (int i=0; i < columns.length; i++)
416        {
417          if (includeTimestamp || (i > 0))
418          {
419            switch (outputFormat)
420            {
421              case TAB_DELIMITED_TEXT:
422                buffer.append('\t');
423                break;
424              case CSV:
425                buffer.append(',');
426                break;
427              case COLUMNS:
428                buffer.append(spacer);
429                break;
430            }
431          }
432    
433          if (i >= columnData.length)
434          {
435            columns[i].format(buffer, "", outputFormat);
436          }
437          else
438          {
439            columns[i].format(buffer, toString(columnData[i]), outputFormat);
440          }
441        }
442    
443        return buffer.toString();
444      }

448      /**
449       * Retrieves a string representation of the provided object.  If the object
450       * is {@code null}, then the empty string will be returned.  If the object is
451       * a {@code Float} or {@code Double}, then it will be formatted using a
452       * DecimalFormat with a format string of "0.000".  Otherwise, the
453       * {@code String.valueOf} method will be used to obtain the string
454       * representation.
455       *
456       * @param  o  The object for which to retrieve the string representation.
457       *
458       * @return  A string representation of the provided object.
459       */
460      private String toString(final Object o)
461      {
462        if (o == null)
463        {
464          return "";
465        }
466    
467        if ((o instanceof Float) || (o instanceof Double))
468        {
469          DecimalFormat f = decimalFormatter.get();
470          if (f == null)
471          {
472            f = new DecimalFormat("0.000");
473            decimalFormatter.set(f);
474          }
475    
476          final double d;
477          if (o instanceof Float)
478          {
479            d = ((Float) o).doubleValue();
480          }
481          else
482          {
483            d = ((Double) o);
484          }
485    
486          return f.format(d);
487        }
488    
489        return String.valueOf(o);
490      }
