089      /**
090       * Retrieves a UTF-8 byte representation of the provided string.
091       *
092       * @param  s  The string for which to retrieve the UTF-8 byte representation.
093       *
094       * @return  The UTF-8 byte representation for the provided string.
095       */
096      public static byte[] getBytes(final String s)
097      {
098        final int length;
099        if ((s == null) || ((length = s.length()) == 0))
100        {
101          return NO_BYTES;
102        }
103    
104        final byte[] b = new byte[length];
105        for (int i=0; i < length; i++)
106        {
107          final char c = s.charAt(i);
108          if (c <= 0x7F)
109          {
110            b[i] = (byte) (c & 0x7F);
111          }
112          else
113          {
114            try
115            {
116              return s.getBytes("UTF-8");
117            }
118            catch (Exception e)
119            {
120              // This should never happen.
121              debugException(e);
122              return s.getBytes();
123            }
124          }
125        }
126    
127        return b;
128      }

568       * Appends a single-line string representation of the stack trace for the
569       * provided {@code Throwable} to the given buffer.  It will include the
570       * unqualified name of the {@code Throwable} class, a list of source files and
571       * line numbers (if available) for the stack trace, and will also include the

574       * @param  t       The {@code Throwable} for which to retrieve the stack

576       * @param  buffer  The buffer to which the information should be appended.
577       */
578      public static void getStackTrace(final Throwable t,
579                                       final StringBuilder buffer)
580      {
581        buffer.append(getUnqualifiedClassName(t.getClass()));
582        buffer.append('(');
583    
584        final String message = t.getMessage();
585        if (message != null)
586        {
587          buffer.append("message='");
588          buffer.append(message);
589          buffer.append("', ");
590        }
591    
592        buffer.append("trace='");
593        getStackTrace(t.getStackTrace(), buffer);
594        buffer.append('\'');
595    
596        final Throwable cause = t.getCause();
597        if (cause != null)
598        {
599          buffer.append(", cause=");
600          getStackTrace(cause, buffer);
601        }
602        buffer.append(", revision=");
603        buffer.append(Version.REVISION_NUMBER);
604        buffer.append(')');
605      }

661      /**
662       * Retrieves a string representation of the provided {@code Throwable} object
663       * suitable for use in a message.  For runtime exceptions and errors, then a
664       * full stack trace for the exception will be provided.  For exception types
665       * defined in the LDAP SDK, then its {@code getExceptionMessage} method will
666       * be used to get the string representation.  For all other types of
667       * exceptions, then the standard string representation will be used.
668       * <BR><BR>
669       * For all types of exceptions, the message will also include the cause if one
670       * exists.
671       *
672       * @param  t  The {@code Throwable} for which to generate the exception
673       *            message.
674       *
675       * @return  A string representation of the provided {@code Throwable} object
676       *          suitable for use in a message.
677       */
678      public static String getExceptionMessage(final Throwable t)
679      {
680        if (t == null)
681        {
682          return ERR_NO_EXCEPTION.get();
683        }
684    
685        final StringBuilder buffer = new StringBuilder();
686        if (t instanceof LDAPSDKException)
687        {
688          buffer.append(((LDAPSDKException) t).getExceptionMessage());
689        }
690        else if (t instanceof LDAPSDKRuntimeException)
691        {
692          buffer.append(((LDAPSDKRuntimeException) t).getExceptionMessage());
693        }
694        if ((t instanceof RuntimeException) || (t instanceof Error))
695        {
696          return getStackTrace(t);
697        }
698        else
699        {
700          buffer.append(String.valueOf(t));
701        }
702    
703        final Throwable cause = t.getCause();
704        if (cause != null)
705        {
706          buffer.append(" caused by ");
707          buffer.append(getExceptionMessage(cause));
708        }
709    
710        return buffer.toString();
711      }

762      /**
763       * Decodes the provided string as a timestamp in generalized time format.
764       *
765       * @param  t  The timestamp to be decoded.  It must not be {@code null}.
766       *
767       * @return  The {@code Date} object decoded from the provided timestamp.
768       *
769       * @throws  ParseException  If the provided string could not be decoded as a
770       *                          timestamp in generalized time format.
771       */
772      public static Date decodeGeneralizedTime(final String t)
773             throws ParseException
774      {
775        ensureNotNull(t);
776    
777        // Extract the time zone information from the end of the value.
778        int tzPos;
779        final TimeZone tz;
780        if (t.endsWith("Z"))
781        {
782          tz = TimeZone.getTimeZone("UTC");
783          tzPos = t.length() - 1;
784        }
785        else
786        {
787          tzPos = t.lastIndexOf('-');
788          if (tzPos < 0)
789          {
790            tzPos = t.lastIndexOf('+');
791            if (tzPos < 0)
792            {
793              throw new ParseException(ERR_GENTIME_DECODE_CANNOT_PARSE_TZ.get(t),
794                                       0);
795            }
796          }
797    
798          tz = TimeZone.getTimeZone("GMT" + t.substring(tzPos));
799          if (tz.getRawOffset() == 0)
800          {
801            // This is the default time zone that will be returned if the value
802            // cannot be parsed.  If it's valid, then it will end in "+0000" or
803            // "-0000".  Otherwise, it's invalid and GMT was just a fallback.
804            if (! (t.endsWith("+0000") || t.endsWith("-0000")))
805            {
806              throw new ParseException(ERR_GENTIME_DECODE_CANNOT_PARSE_TZ.get(t),
807                                       tzPos);
808            }
809          }
810        }
811    
812    
813        // See if the timestamp has a sub-second portion.  Note that if there is a
814        // sub-second portion, then we may need to massage the value so that there
815        // are exactly three sub-second characters so that it can be interpreted as
816        // milliseconds.
817        final String subSecFormatStr;
818        final String trimmedTimestamp;
819        int periodPos = t.lastIndexOf('.', tzPos);
820        if (periodPos > 0)
821        {
822          final int subSecondLength = tzPos - periodPos - 1;
823          switch (subSecondLength)
824          {
825            case 0:
826              subSecFormatStr  = "";
827              trimmedTimestamp = t.substring(0, periodPos);
828              break;
829            case 1:
830              subSecFormatStr  = ".SSS";
831              trimmedTimestamp = t.substring(0, (periodPos+2)) + "00";
832              break;
833            case 2:
834              subSecFormatStr  = ".SSS";
835              trimmedTimestamp = t.substring(0, (periodPos+3)) + '0';
836              break;
837            default:
838              subSecFormatStr  = ".SSS";
839              trimmedTimestamp = t.substring(0, periodPos+4);
840              break;
841          }
842        }
843        else
844        {
845          subSecFormatStr  = "";
846          periodPos        = tzPos;
847          trimmedTimestamp = t.substring(0, tzPos);
848        }
849    
850    
851        // Look at where the period is (or would be if it existed) to see how many
852        // characters are in the integer portion.  This will give us what we need
853        // for the rest of the format string.
854        final String formatStr;
855        switch (periodPos)
856        {
857          case 10:
858            formatStr = "yyyyMMddHH" + subSecFormatStr;
859            break;
860          case 12:
861            formatStr = "yyyyMMddHHmm" + subSecFormatStr;
862            break;
863          case 14:
864            formatStr = "yyyyMMddHHmmss" + subSecFormatStr;
865            break;
866          default:
867            throw new ParseException(ERR_GENTIME_CANNOT_PARSE_INVALID_LENGTH.get(t),
868                                     periodPos);
869        }
870    
871    
872        // We should finally be able to create an appropriate date format object
873        // to parse the trimmed version of the timestamp.
874        final SimpleDateFormat dateFormat = new SimpleDateFormat(formatStr);
875        dateFormat.setTimeZone(tz);
876        dateFormat.setLenient(false);
877        return dateFormat.parse(trimmedTimestamp);
878      }

883       * Trims only leading spaces from the provided string, leaving any trailing
884       * spaces intact.
885       *
886       * @param  s  The string to be processed.  It must not be {@code null}.
887       *
888       * @return  The original string if no trimming was required, or a new string
889       *          without leading spaces if the provided string had one or more.  It
890       *          may be an empty string if the provided string was an empty string
891       *          or contained only spaces.
892       */
893      public static String trimLeading(final String s)
894      {
895        ensureNotNull(s);
896    
897        int nonSpacePos = 0;
898        final int length = s.length();
899        while ((nonSpacePos < length) && (s.charAt(nonSpacePos) == ' '))
900        {
901          nonSpacePos++;
902        }
903    
904        if (nonSpacePos == 0)
905        {
906          // There were no leading spaces.
907          return s;
908        }
909        else if (nonSpacePos >= length)
910        {
911          // There were no non-space characters.
912          return "";
913        }
914        else
915        {
916          // There were leading spaces, so return the string without them.
917          return s.substring(nonSpacePos, length);
918        }
919      }

923      /**
924       * Trims only trailing spaces from the provided string, leaving any leading
925       * spaces intact.
926       *
927       * @param  s  The string to be processed.  It must not be {@code null}.
928       *
929       * @return  The original string if no trimming was required, or a new string
930       *          without trailing spaces if the provided string had one or more.
931       *          It may be an empty string if the provided string was an empty
932       *          string or contained only spaces.
933       */
934      public static String trimTrailing(final String s)
935      {
936        ensureNotNull(s);
937    
938        final int lastPos = s.length() - 1;
939        int nonSpacePos = lastPos;
940        while ((nonSpacePos >= 0) && (s.charAt(nonSpacePos) == ' '))
941        {
942          nonSpacePos--;
943        }
944    
945        if (nonSpacePos < 0)
946        {
947          // There were no non-space characters.
948          return "";
949        }
950        else if (nonSpacePos == lastPos)
951        {
952          // There were no trailing spaces.
953          return s;
954        }
955        else
956        {
957          // There were trailing spaces, so return the string without them.
958          return s.substring(0, (nonSpacePos+1));
959        }
960      }

964      /**
965       * Wraps the contents of the specified line using the given width.  It will
966       * attempt to wrap at spaces to preserve words, but if that is not possible
967       * (because a single "word" is longer than the maximum width), then it will
968       * wrap in the middle of the word at the specified maximum width.
969       *
970       * @param  line      The line to be wrapped.  It must not be {@code null}.
971       * @param  maxWidth  The maximum width for lines in the resulting list.  A
972       *                   value less than or equal to zero will cause no wrapping
973       *                   to be performed.
974       *
975       * @return  A list of the wrapped lines.  It may be empty if the provided line
976       *          contained only spaces.
977       */
978      public static List<String> wrapLine(final String line, final int maxWidth)
979      {
980        final int length = line.length();
981        if ((maxWidth <= 0) || (length < maxWidth))
982        {
983          return Arrays.asList(line);
984        }
985    
986    
987        int wrapPos = maxWidth;
988        int lastWrapPos = 0;
989        final ArrayList<String> lineList = new ArrayList<String>(5);
990        while (true)
991        {
992          final int spacePos = line.lastIndexOf(' ', wrapPos);
993          if (spacePos > lastWrapPos)
994          {
995            // We found a space in an acceptable location, so use it after trimming
996            // any trailing spaces.
997            final String s = trimTrailing(line.substring(lastWrapPos, spacePos));
998    
999            // Don't bother adding the line if it contained only spaces.
1000            if (s.length() > 0)
1001            {
1002              lineList.add(s);
1003            }
1004    
1005            wrapPos = spacePos;
1006          }
1007          else
1008          {
1009            // We didn't find any spaces, so we'll have to insert a hard break at
1010            // the specified wrap column.
1011            lineList.add(line.substring(lastWrapPos, wrapPos));
1012          }
1013    
1014          // Skip over any spaces before the next non-space character.
1015          while ((wrapPos < length) && (line.charAt(wrapPos) == ' '))
1016          {
1017            wrapPos++;
1018          }
1019    
1020          lastWrapPos = wrapPos;
1021          wrapPos += maxWidth;
1022          if (wrapPos >= length)
1023          {
1024            // The last fragment can fit on the line, so we can handle that now and
1025            // break.
1026            if (lastWrapPos >= length)
1027            {
1028              break;
1029            }
1030            else
1031            {
1032              final String s = trimTrailing(line.substring(lastWrapPos));
1033              if (s.length() > 0)
1034              {
1035                lineList.add(s);
1036              }
1037              break;
1038            }
1039          }
1040        }
1041    
1042        return lineList;
1043      }

1047      /**
1048       * Formats the provided string in a manner that will allow it to be properly
1049       * displayed as the value of a command-line argument in an example.  The
1050       * processing that will be performed includes:
1051       * <UL>
1052       *   <LI>If the provided string has a length of zero characters, then the
1053       *       value returned will be two consecutive double quotes (i.e., "").</LI>
1054       *   <LI>All instances of the following characters will be escaped by
1055       *       prefixing them with backslash characters:  backslash (\\), double
1056       *       quote ("), percent (%), vertical pipe (|), dollar sign ($),
1057       *       exclamation point (!), and backwards single quote (`).</LI>
1058       *   <LI>The string resulting from the above escaping will be surrounded by
1059       *       double quote (") characters if it contains any characters other than
1060       *       the following characters:  uppercase or lowercase ASCII letters,
1061       *       numeric digits, dash (-), underscore (_), colon (:), period (.),
1062       *       backslash (\\), and forward slash (/).</LI>
1063       * </UL>
1064       *
1065       * @param  s  The string to be processed.  It must not be {@code null}.
1066       *
1067       * @return  A cleaned version of the provided string in a form that will allow
1068       *          it to be displayed as the value of a command-line argument.
1069       */
1070      public static String cleanExampleCommandLineArgument(final String s)
1071      {
1072        ensureNotNull(s);
1073    
1074        if (s.length() == 0)
1075        {
1076          return "\"\"";
1077        }
1078    
1079        boolean needsQuotes = false;
1080        final StringBuilder buffer = new StringBuilder(2 * s.length() + 2);
1081    
1082        for (int i=0; i < s.length(); i++)
1083        {
1084          final char c = s.charAt(i);
1085          switch (c)
1086          {
1087            case '\\':
1088              // This needs to be escaped, but the value does not necessarily need
1089              // to be quoted.
1090              buffer.append("\\\\");
1091              break;
1092    
1093            case '"':
1094            case '%':
1095            case '|':
1096            case '$':
1097            case '!':
1098            case '`':
1099              // This needs to be escaped and the value needs to be quoted.
1100              needsQuotes = true;
1101              buffer.append('\\');
1102              buffer.append(c);
1103              break;
1104    
1105            case '-':
1106            case '_':
1107            case ':':
1108            case '.':
1109            case '/':
1110              // This does not need to be escaped and does not indicate that the
1111              // string needs to be quoted.
1112              buffer.append(c);
1113              break;
1114    
1115            default:
1116              if (((c >= 'a') && (c <= 'z')) ||
1117                  ((c >= 'A') && (c <= 'Z')) ||
1118                  ((c >= '0') && (c <= '9')))
1119              {
1120                buffer.append(c);
1121              }
1122              else
1123              {
1124                needsQuotes = true;
1125                buffer.append(c);
1126              }
1127          }
1128        }
1129    
1130        if (needsQuotes)
1131        {
1132          buffer.insert(0, '"');
1133          buffer.append('"');
1134        }
1135    
1136        return buffer.toString();
1137      }

1189      /**
1190       * Converts a duration in seconds to a string with a human-readable duration
1191       * which may include days, hours, minutes, and seconds, to the extent that
1192       * they are needed.
1193       *
1194       * @param  m  The number of milliseconds to be represented.
1195       *
1196       * @return  A string containing a human-readable representation of the
1197       *          provided time.
1198       */
1199      public static String millisToHumanReadableDuration(final long m)
1200      {
1201        final StringBuilder buffer = new StringBuilder();
1202        long numMillis = m;
1203    
1204        final long numDays = numMillis / 86400000L;
1205        if (numDays > 0)
1206        {
1207          numMillis -= (numDays * 86400000L);
1208          if (numDays == 1)
1209          {
1210            buffer.append(INFO_NUM_DAYS_SINGULAR.get(numDays));
1211          }
1212          else
1213          {
1214            buffer.append(INFO_NUM_DAYS_PLURAL.get(numDays));
1215          }
1216        }
1217    
1218        final long numHours = numMillis / 3600000L;
1219        if (numHours > 0)
1220        {
1221          numMillis -= (numHours * 3600000L);
1222          if (buffer.length() > 0)
1223          {
1224            buffer.append(", ");
1225          }
1226    
1227          if (numHours == 1)
1228          {
1229            buffer.append(INFO_NUM_HOURS_SINGULAR.get(numHours));
1230          }
1231          else
1232          {
1233            buffer.append(INFO_NUM_HOURS_PLURAL.get(numHours));
1234          }
1235        }
1236    
1237        final long numMinutes = numMillis / 60000L;
1238        if (numMinutes > 0)
1239        {
1240          numMillis -= (numMinutes * 60000L);
1241          if (buffer.length() > 0)
1242          {
1243            buffer.append(", ");
1244          }
1245    
1246          if (numMinutes == 1)
1247          {
1248            buffer.append(INFO_NUM_MINUTES_SINGULAR.get(numMinutes));
1249          }
1250          else
1251          {
1252            buffer.append(INFO_NUM_MINUTES_PLURAL.get(numMinutes));
1253          }
1254        }
1255    
1256        if (numMillis == 1000)
1257        {
1258          if (buffer.length() > 0)
1259          {
1260            buffer.append(", ");
1261          }
1262    
1263          buffer.append(INFO_NUM_SECONDS_SINGULAR.get(1));
1264        }
1265        else if ((numMillis > 0) || (buffer.length() == 0))
1266        {
1267          if (buffer.length() > 0)
1268          {
1269            buffer.append(", ");
1270          }
1271    
1272          final long numSeconds = numMillis / 1000L;
1273          numMillis -= (numSeconds * 1000L);
1274          if ((numMillis % 1000L) != 0L)
1275          {
1276            final double numSecondsDouble = numSeconds + (numMillis / 1000.0);
1277            final DecimalFormat decimalFormat = new DecimalFormat("0.000");
1278            buffer.append(INFO_NUM_SECONDS_WITH_DECIMAL.get(
1279                 decimalFormat.format(numSecondsDouble)));
1280          }
1281          else
1282          {
1283            buffer.append(INFO_NUM_SECONDS_PLURAL.get(numSeconds));
1284          }
1285        }
1286    
1287        return buffer.toString();
1288      }

1322      /**
1323       * Encodes the provided UUID to a byte array containing its 128-bit
1324       * representation.
1325       *
1326       * @param  uuid  The UUID to be encoded.  It must not be {@code null}.
1327       *
1328       * @return  The byte array containing the 128-bit encoded UUID.
1329       */
1330      public static byte[] encodeUUID(final UUID uuid)
1331      {
1332        final byte[] b = new byte[16];
1333    
1334        final long mostSignificantBits  = uuid.getMostSignificantBits();
1335        b[0]  = (byte) ((mostSignificantBits >> 56) & 0xFF);
1336        b[1]  = (byte) ((mostSignificantBits >> 48) & 0xFF);
1337        b[2]  = (byte) ((mostSignificantBits >> 40) & 0xFF);
1338        b[3]  = (byte) ((mostSignificantBits >> 32) & 0xFF);
1339        b[4]  = (byte) ((mostSignificantBits >> 24) & 0xFF);
1340        b[5]  = (byte) ((mostSignificantBits >> 16) & 0xFF);
1341        b[6]  = (byte) ((mostSignificantBits >> 8) & 0xFF);
1342        b[7]  = (byte) (mostSignificantBits & 0xFF);
1343    
1344        final long leastSignificantBits = uuid.getLeastSignificantBits();
1345        b[8]  = (byte) ((leastSignificantBits >> 56) & 0xFF);
1346        b[9]  = (byte) ((leastSignificantBits >> 48) & 0xFF);
1347        b[10] = (byte) ((leastSignificantBits >> 40) & 0xFF);
1348        b[11] = (byte) ((leastSignificantBits >> 32) & 0xFF);
1349        b[12] = (byte) ((leastSignificantBits >> 24) & 0xFF);
1350        b[13] = (byte) ((leastSignificantBits >> 16) & 0xFF);
1351        b[14] = (byte) ((leastSignificantBits >> 8) & 0xFF);
1352        b[15] = (byte) (leastSignificantBits & 0xFF);
1353    
1354        return b;
1355      }

1359      /**
1360       * Decodes the value of the provided byte array as a Java UUID.
1361       *
1362       * @param  b  The byte array to be decoded as a UUID.  It must not be
1363       *            {@code null}.
1364       *
1365       * @return  The decoded UUID.
1366       *
1367       * @throws  ParseException  If the provided byte array cannot be parsed as a
1368       *                         UUID.
1369       */
1370      public static UUID decodeUUID(final byte[] b)
1371             throws ParseException
1372      {
1373        if (b.length != 16)
1374        {
1375          throw new ParseException(ERR_DECODE_UUID_INVALID_LENGTH.get(toHex(b)), 0);
1376        }
1377    
1378        long mostSignificantBits = 0L;
1379        for (int i=0; i < 8; i++)
1380        {
1381          mostSignificantBits = (mostSignificantBits << 8) | (b[i] & 0xFF);
1382        }
1383    
1384        long leastSignificantBits = 0L;
1385        for (int i=8; i < 16; i++)
1386        {
1387          leastSignificantBits = (leastSignificantBits << 8) | (b[i] & 0xFF);
1388        }
1389    
1390        return new UUID(mostSignificantBits, leastSignificantBits);
1391      }
