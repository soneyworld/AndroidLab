042    /**
043     * This class provides a framework for developing command-line tools that use
044     * the argument parser provided as part of the UnboundID LDAP SDK for Java.
045     * This tool adds only a "-H" or "--help" option, which can be used to display
046     * usage information for the program.
047     * <BR><BR>
048     * Subclasses should include their own {@code main} method that creates an
049     * instance of a {@code CommandLineTool} and should invoke the
050     * {@link CommandLineTool#runTool} method with the provided arguments.  For
051     * example:
052     * <PRE>
053     *   public class ExampleCommandLineTool
054     *          extends CommandLineTool
055     *   {
056     *     public static void main(String[] args)
057     *     {
058     *       ExampleCommandLineTool tool = new ExampleCommandLineTool();
059     *       ResultCode resultCode = tool.runTool(args);
060     *       if (resultCode != ResultCode.SUCCESS)
061     *       {
062     *         System.exit(resultCode.intValue());
063     *       }
064     *     |
065     *
066     *     public ExampleCommandLineTool()
067     *     {
068     *       super(System.out, System.err);
069     *     }
070     *
071     *     // The rest of the tool implementation goes here.
072     *     ...
073     *   }
074     * </PRE>.
075     * <BR><BR>
076     * Note that in general, methods in this class are not threadsafe.  However, the
077     * {@link #out(Object[])} and {@link #err(Object[])} methods may be invoked
078     * concurrently by any number of threads.
079     */
080    @Extensible()
081    @ThreadSafety(level=ThreadSafetyLevel.INTERFACE_NOT_THREADSAFE)
082    public abstract class CommandLineTool
083    {
084      // The print stream to use for messages written to standard output.
085      private final PrintStream out;
086    
087      // The print stream to use for messages written to standard error.
088      private final PrintStream err;
089    
090      // The argument used to request tool help.
091      private BooleanArgument helpArgument = null;
092    
093    
094    
095      /**
096       * Creates a new instance of this command-line tool with the provided
097       * information.
098       *
099       * @param  outStream  The output stream to use for standard output.  It may be
100       *                    {@code System.out} for the JVM's default standard output
101       *                    stream, {@code null} if no output should be generated,
102       *                    or a custom output stream if the output should be sent
103       *                    to an alternate location.
104       * @param  errStream  The output stream to use for standard error.  It may be
105       *                    {@code System.err} for the JVM's default standard error
106       *                    stream, {@code null} if no output should be generated,
107       *                    or a custom output stream if the output should be sent
108       *                    to an alternate location.
109       */
110      public CommandLineTool(final OutputStream outStream,
111                             final OutputStream errStream)
112      {
113        if (outStream == null)
114        {
115          out = NullOutputStream.getPrintStream();
116        }
117        else
118        {
119          out = new PrintStream(outStream);
120        }
121    
122        if (errStream == null)
123        {
124          err = NullOutputStream.getPrintStream();
125        }
126        else
127        {
128          err = new PrintStream(errStream);
129        }
130      }

134      /**
135       * Performs all processing for this command-line tool.  This includes:
136       * <UL>
137       *   <LI>Creating the argument parser and populating it using the
138       *       {@link #addToolArguments} method.</LI>
139       *   <LI>Parsing the provided set of command line arguments, including any
140       *       additional validation using the {@link #doExtendedArgumentValidation}
141       *       method.</LI>
142       *   <LI>Invoking the {@link #doToolProcessing} method to do the appropriate
143       *       work for this tool.</LI>
144       * </UL>
145       *
146       * @param  args  The command-line arguments provided to this program.
147       *
148       * @return  The result of processing this tool.  It should be
149       *          {@link ResultCode#SUCCESS} if the tool completed its work
150       *          successfully, or some other result if a problem occurred.
151       */
152      public final ResultCode runTool(final String... args)
153      {
154        try
155        {
156          final ArgumentParser parser = createArgumentParser();
157          parser.parse(args);
158    
159          if (helpArgument.isPresent())
160          {
161            out(parser.getUsageString(79));
162            displayExampleUsages();
163            return ResultCode.SUCCESS;
164          }
165    
166          doExtendedArgumentValidation();
167        }
168        catch (ArgumentException ae)
169        {
170          debugException(ae);
171          err(ae.getMessage());
172          return ResultCode.PARAM_ERROR;
173        }
174    
175        try
176        {
177          return doToolProcessing();
178        }
179        catch (Exception e)
180        {
181          debugException(e);
182          err(getExceptionMessage(e));
183          return ResultCode.LOCAL_ERROR;
184        }
185      }

189      /**
190       * Writes example usage information for this tool to the standard output
191       * stream.
192       */
193      private void displayExampleUsages()
194      {
195        final LinkedHashMap<String[],String> examples = getExampleUsages();
196        if ((examples == null) || examples.isEmpty())
197        {
198          return;
199        }
200    
201        out(INFO_CL_TOOL_LABEL_EXAMPLES);
202    
203        for (final Map.Entry<String[],String> e : examples.entrySet())
204        {
205          out();
206          wrapOut(2, 79, e.getValue());
207          out();
208    
209          final StringBuilder buffer = new StringBuilder();
210          buffer.append("    ");
211          buffer.append(getToolName());
212    
213          final String[] args = e.getKey();
214          for (int i=0; i < args.length; i++)
215          {
216            buffer.append(' ');
217    
218            // If the argument has a value, then make sure to keep it on the same
219            // line as the argument name.  This may introduce false positives due to
220            // unnamed trailing arguments, but the worst that will happen that case
221            // is that the output may be wrapped earlier than necessary one time.
222            String arg = args[i];
223            if (arg.startsWith("-"))
224            {
225              if ((i < (args.length - 1)) && (! args[i+1].startsWith("-")))
226              {
227                arg += ' ' + cleanExampleCommandLineArgument(args[i+1]);
228                i++;
229              }
230            }
231            else
232            {
233              arg = cleanExampleCommandLineArgument(arg);
234            }
235    
236            if ((buffer.length() + arg.length() + 2) < 79)
237            {
238              buffer.append(arg);
239            }
240            else
241            {
242              buffer.append('\\');
243              out(buffer.toString());
244              buffer.setLength(0);
245              buffer.append("         ");
246              buffer.append(arg);
247            }
248          }
249    
250          out(buffer.toString());
251        }
252      }

310      /**
311       * Creates a parser that can be used to to parse arguments accepted by
312       * this tool.
313       *
314       * @return ArgumentParser that can be used to parse arguments for this
315       *         tool.
316       *
317       * @throws ArgumentException  If there was a problem initializing the
318       *                            parser for this tool.
319       */
320      public final ArgumentParser createArgumentParser()
321             throws ArgumentException
322      {
323        final ArgumentParser parser = new ArgumentParser(getToolName(),
324             getToolDescription(), getMaxTrailingArguments(),
325             getTrailingArgumentsPlaceholder());
326    
327        addToolArguments(parser);
328    
329        helpArgument = new BooleanArgument('H', "help",
330             INFO_CL_TOOL_DESCRIPTION_HELP.get());
331        helpArgument.addShortIdentifier('?');
332        helpArgument.setUsageArgument(true);
333        parser.addArgument(helpArgument);
334    
335        return parser;
336      }

522      /**
523       * Writes the provided message to the given print stream, optionally wrapping
524       * and/or indenting the text in the process.
525       *
526       * @param  stream      The stream to which the message should be written.
527       * @param  indent      The number of spaces each line should be indented.  A
528       *                     value less than or equal to zero indicates that no
529       *                     indent should be used.
530       * @param  wrapColumn  The column at which to wrap long lines.  A value less
531       *                     than or equal to two indicates that no wrapping should
532       *                     be performed.  If both an indent and a wrap column are
533       *                     to be used, then the wrap column must be greater than
534       *                     the indent.
535       * @param  msg         The message components that will be written to the
536       *                     standard output stream.  They will be concatenated
537       *                     together on the same line, and that line will be
538       *                     followed by an end-of-line sequence.
539       */
540      private static void write(final PrintStream stream, final int indent,
541                                final int wrapColumn, final Object... msg)
542      {
543        final StringBuilder buffer = new StringBuilder();
544        for (final Object o : msg)
545        {
546          buffer.append(o);
547        }
548    
549        if (wrapColumn > 2)
550        {
551          final List<String> lines;
552          if (indent > 0)
553          {
554            for (final String line :
555                 wrapLine(buffer.toString(), (wrapColumn - indent)))
556            {
557              for (int i=0; i < indent; i++)
558              {
559                stream.print(' ');
560              }
561              stream.println(line);
562            }
563          }
564          else
565          {
566            for (final String line : wrapLine(buffer.toString(), wrapColumn))
567            {
568              stream.println(line);
569            }
570          }
571        }
572        else
573        {
574          if (indent > 0)
575          {
576            for (int i=0; i < indent; i++)
577            {
578              stream.print(' ');
579            }
580          }
581          stream.println(buffer.toString());
582        }
583      }
