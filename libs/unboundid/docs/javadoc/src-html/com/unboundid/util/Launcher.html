032    import com.unboundid.ldap.sdk.examples.LDAPModify;
033    import com.unboundid.ldap.sdk.examples.LDAPSearch;
034    import com.unboundid.ldap.sdk.examples.ModRate;

037    import com.unboundid.ldap.sdk.examples.ValidateLDIF;

041    /**
042     * This class provides an entry point that may be used to launch other tools
043     * provided as part of the LDAP SDK.  This is primarily a convenience for
044     * someone who just has the jar file and none of the scripts, since you can run
045     * "<CODE>java -jar unboundid-ldapsdk-se.jar {tool-name} {tool-args}</CODE>"
046     * in order to invoke any of the example tools.  Running just
047     * "<CODE>java -jar unboundid-ldapsdk-se.jar</CODE>" will display version
048     * information about the LDAP SDK.
049     * <BR><BR>
050     * The tool names are case-insensitive.  Supported tool names include:
051     * <UL>
052     *   <LI>authrate -- Launch the {@link AuthRate} tool.</LI>
053     *   <LI>ldapcompare -- Launch the {@link LDAPCompare} tool.</LI>
054     *   <LI>ldapmodify -- Launch the {@link LDAPModify} tool.</LI>
055     *   <LI>ldapsearch -- Launch the {@link LDAPSearch} tool.</LI>
056     *   <LI>modrate -- Launch the {@link ModRate} tool.</LI>
057     *   <LI>searchrate -- Launch the {@link SearchRate} tool.</LI>
058     *   <LI>search-and-mod-rate -- Launch the {@link SearchAndModRate} tool.</LI>
059     *   <LI>validate-ldif -- Launch the {@link ValidateLDIF} tool.</LI>
060     *   <LI>version -- Display version information for the LDAP SDK.</LI>
061     * </UL>
062     */
063    @ThreadSafety(level=ThreadSafetyLevel.COMPLETELY_THREADSAFE)
064    public final class Launcher
065    {
066      /**
067       * Prevent this utility class from being externally instantiated.
068       */
069      Launcher()
070      {
071        // No implementation required.
072      }
073    
074    
075    
076      /**
077       * Parses the command-line arguments and performs any appropriate processing
078       * for this program.
079       *
080       * @param  args  The command-line arguments provided to this program.
081       */
082      public static void main(final String... args)
083      {
084        main(System.out, System.err, args);
085      }
086    
087    
088    
089      /**
090       * Parses the command-line arguments and performs any appropriate processing
091       * for this program.
092       *
093       * @param  outStream  The output stream to which standard out should be
094       *                    written.  It may be {@code null} if output should be
095       *                    suppressed.
096       * @param  errStream  The output stream to which standard error should be
097       *                    written.  It may be {@code null} if error messages
098       *                    should be suppressed.
099       * @param  args       The command-line arguments provided to this program.
100       *
101       * @return  A result code with information about the status of processing.
102       */
103      public static ResultCode main(final OutputStream outStream,
104                                    final OutputStream errStream,
105                                    final String... args)
106      {
107        if ((args == null) || (args.length == 0) ||
108            args[0].equalsIgnoreCase("version"))
109        {
110          if (outStream != null)
111          {
112            final PrintStream out = new PrintStream(outStream);
113            for (final String line : Version.getVersionLines())
114            {
115              out.println(line);
116            }
117          }
118    
119          return ResultCode.SUCCESS;
120        }
121    
122        final String firstArg = StaticUtils.toLowerCase(args[0]);
123        final String[] remainingArgs = new String[args.length - 1];
124        System.arraycopy(args, 1, remainingArgs, 0, remainingArgs.length);
125    
126        if (firstArg.equals("authrate"))
127        {
128          return AuthRate.main(remainingArgs, outStream, errStream);
129        }
130        else if (firstArg.equals("ldapcompare"))
131        {
132          return LDAPCompare.main(remainingArgs, outStream, errStream);
133        }
134        else if (firstArg.equals("ldapmodify"))
135        {
136          return LDAPModify.main(remainingArgs, outStream, errStream);
137        }
138        else if (firstArg.equals("ldapsearch"))
139        {
140          return LDAPSearch.main(remainingArgs, outStream, errStream);
141        }
142        else if (firstArg.equals("modrate"))
143        {
144          return ModRate.main(remainingArgs, outStream, errStream);
145        }
146        else if (firstArg.equals("searchrate"))
147        {
148          return SearchRate.main(remainingArgs, outStream, errStream);
149        }
150        else if (firstArg.equals("search-and-mod-rate"))
151        {
152          return SearchAndModRate.main(remainingArgs, outStream, errStream);
153        }
154        else if (firstArg.equals("validate-ldif"))
155        {
156          return ValidateLDIF.main(remainingArgs, outStream, errStream);
157        }
158        else
159        {
160          if (errStream != null)
161          {
162            final PrintStream err = new PrintStream(errStream);
163            err.println("Unrecognized tool name '" + args[0] + '\'');
164            err.println("Supported tool names include:");
165            err.println("     authrate");
166            err.println("     ldapcompare");
167            err.println("     ldapmodify");
168            err.println("     ldapsearch");
169            err.println("     modrate");
170            err.println("     searchrate");
171            err.println("     search-and-mod-rate");
172            err.println("     validate-ldif");
173            err.println("     version");
174          }
175    
176          return ResultCode.PARAM_ERROR;
177        }
178      }
179    }
