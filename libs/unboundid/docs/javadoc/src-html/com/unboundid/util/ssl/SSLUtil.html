038    /**
039     * This class provides a simple interface for creating {@code SSLContext} and
040     * {@code SSLSocketFactory} instances, which may be used to create SSL-based
041     * connections, or secure existing connections with StartTLS.
042     * <BR><BR>
043     * <H2>Example 1</H2>
044     * The following example demonstrates the use of the SSL helper to create an
045     * SSL-based LDAP connection that will blindly trust any certificate that the
046     * server presents:
047     * <PRE>
048     *   SSLUtil sslUtil = new SSLUtil(new TrustAllTrustManager());
049     *
050     *   LDAPConnection connection =
051     *        new LDAPConnection(sslUtil.createSSLSocketFactory());
052     *   connection.connect("server.example.com", 636);
053     * </PRE>
054     * <BR>
055     * <H2>Example 2</H2>
056     * The following example demonstrates the use of the SSL helper to create a
057     * non-secure LDAP connection and then use the StartTLS extended operation to
058     * secure it.  It will use a trust store to determine whether to trust the
059     * server certificate.
060     * <PRE>
061     *   LDAPConnection connection = new LDAPConnection();
062     *   connection.connect("server.example.com", 389);
063     *
064     *   String trustStoreFile  = "/path/to/trust/store/file";
065     *   SSLUtil sslUtil = new SSLUtil(new TrustStoreTrustManager(trustStoreFile));
066     *
067     *   ExtendedResult extendedResult = conn.processExtendedOperation(
068     *        new StartTLSExtendedRequest(h.createSSLContext()));
069     * </PRE>
070     */
071    @ThreadSafety(level=ThreadSafetyLevel.COMPLETELY_THREADSAFE)
072    public final class SSLUtil
073    {
074      // The set of key managers to be used.
075      private final KeyManager[] keyManagers;
076    
077      // The set of trust managers to be used.
078      private final TrustManager[] trustManagers;
079    
080    
081    
082      /**
083       * Creates a new SSLUtil instance that will not have a custom key manager or
084       * trust manager.  It will not be able to provide a certificate to the server
085       * if one is requested, and it will only trust certificates signed by a
086       * predefined set of authorities.
087       */
088      public SSLUtil()
089      {
090        keyManagers   = null;
091        trustManagers = null;
092      }
093    
094    
095    
096      /**
097       * Creates a new SSLUtil instance that will use the provided trust manager to
098       * determine whether to trust server certificates presented to the client.
099       * It will not be able to provide a certificate to the server if one is
100       * requested.
101       *
102       * @param  trustManager  The trust manager to use to determine whether to
103       *                       trust server certificates presented to the client.
104       *                       It may be {@code null} if the default set of trust
105       *                       managers should be used.
106       */
107      public SSLUtil(final TrustManager trustManager)
108      {
109        keyManagers = null;
110    
111        if (trustManager == null)
112        {
113          trustManagers = null;
114        }
115        else
116        {
117          trustManagers = new TrustManager[] { trustManager };
118        }
119      }
120    
121    
122    
123      /**
124       * Creates a new SSLUtil instance that will use the provided trust managers
125       * to determine whether to trust server certificates presented to the client.
126       * It will not be able to provide a certificate to the server if one is
127       * requested.
128       *
129       * @param  trustManagers  The set of trust managers to use to determine
130       *                        whether to trust server certificates presented to
131       *                        the client.  It may be {@code null} or empty if the
132       *                        default set of trust managers should be used.
133       */
134      public SSLUtil(final TrustManager[] trustManagers)
135      {
136        keyManagers = null;
137    
138        if ((trustManagers == null) || (trustManagers.length == 0))
139        {
140          this.trustManagers = null;
141        }
142        else
143        {
144          this.trustManagers = trustManagers;
145        }
146      }
147    
148    
149    
150      /**
151       * Creates a new SSLUtil instance that will use the provided key manager to
152       * obtain certificates to present to the server, and the provided trust
153       * manager to determine whether to trust server certificates presented to the
154       * client.
155       *
156       * @param  keyManager    The key manager to use to obtain certificates to
157       *                       present to the server if requested.  It may be
158       *                       {@code null} if no client certificates will be
159       *                       required or should be provided.
160       * @param  trustManager  The trust manager to use to determine whether to
161       *                       trust server certificates presented to the client.
162       *                       It may be {@code null} if the default set of trust
163       *                       managers should be used.
164       */
165      public SSLUtil(final KeyManager keyManager, final TrustManager trustManager)
166      {
167        if (keyManager == null)
168        {
169          keyManagers = null;
170        }
171        else
172        {
173          keyManagers = new KeyManager[] { keyManager };
174        }
175    
176        if (trustManager == null)
177        {
178          trustManagers = null;
179        }
180        else
181        {
182          trustManagers = new TrustManager[] { trustManager };
183        }
184      }
185    
186    
187    
188      /**
189       * Creates a new SSLUtil instance that will use the provided key managers to
190       * obtain certificates to present to the server, and the provided trust
191       * managers to determine whether to trust server certificates presented to the
192       * client.
193       *
194       * @param  keyManagers    The set of key managers to use to obtain
195       *                        certificates to present to the server if requested.
196       *                        It may be {@code null} or empty if no client
197       *                        certificates will be required or should be provided.
198       * @param  trustManagers  The set of trust managers to use to determine
199       *                        whether to trust server certificates presented to
200       *                        the client.  It may be {@code null} or empty if the
201       *                        default set of trust managers should be used.
202       */
203      public SSLUtil(final KeyManager[] keyManagers,
204                     final TrustManager[] trustManagers)
205      {
206        if ((keyManagers == null) || (keyManagers.length == 0))
207        {
208          this.keyManagers = null;
209        }
210        else
211        {
212          this.keyManagers = keyManagers;
213        }
214    
215        if ((trustManagers == null) || (trustManagers.length == 0))
216        {
217          this.trustManagers = null;
218        }
219        else
220        {
221          this.trustManagers = trustManagers;
222        }
223      }
224    
225    
226    
227      /**
228       * Creates an initialized SSL context created with the configured key and
229       * trust managers.  It will use the "TLSv1" protocol and the default provider.
230       *
231       * @return  The created SSL context.
232       *
233       * @throws  GeneralSecurityException  If a problem occurs while creating or
234       *                                    initializing the SSL context.
235       */
236      public SSLContext createSSLContext()
237             throws GeneralSecurityException
238      {
239        return createSSLContext("TLSv1");
240      }
241    
242    
243    
244      /**
245       * Creates an initialized SSL context created with the configured key and
246       * trust managers.  It will use the default provider.
247       *
248       * @param  protocol  The protocol to use.  As per the Java SE 6 Cryptography
249       *                   Architecture document, the set of supported protocols
250       *                   should include at least "SSLv3", "TLSv1", "TLSv1.1", and
251       *                   "SSLv2Hello".  It must not be {@code null}.
252       *
253       * @return  The created SSL context.
254       *
255       * @throws  GeneralSecurityException  If a problem occurs while creating or
256       *                                    initializing the SSL context.
257       */
258      public SSLContext createSSLContext(final String protocol)
259             throws GeneralSecurityException
260      {
261        ensureNotNull(protocol);
262    
263        final SSLContext sslContext = SSLContext.getInstance(protocol);
264        sslContext.init(keyManagers, trustManagers, null);
265        return sslContext;
266      }
267    
268    
269    
270      /**
271       * Creates an initialized SSL context created with the configured key and
272       * trust managers.
273       *
274       * @param  protocol  The protocol to use.  As per the Java SE 6 Cryptography
275       *                   Architecture document, the set of supported protocols
276       *                   should include at least "SSLv3", "TLSv1", "TLSv1.1", and
277       *                   "SSLv2Hello".  It must not be {@code null}.
278       * @param  provider  The name of the provider to use for cryptographic
279       *                   operations.  It must not be {@code null}.
280       *
281       * @return  The created SSL context.
282       *
283       * @throws  GeneralSecurityException  If a problem occurs while creating or
284       *                                    initializing the SSL context.
285       */
286      public SSLContext createSSLContext(final String protocol,
287                                         final String provider)
288             throws GeneralSecurityException
289      {
290        ensureNotNull(protocol, provider);
291    
292        final SSLContext sslContext = SSLContext.getInstance(protocol, provider);
293        sslContext.init(keyManagers, trustManagers, null);
294        return sslContext;
295      }
296    
297    
298    
299      /**
300       * Creates an SSL socket factory using the configured key and trust manager
301       * providers.  It will use the "TLSv1" protocol and the default provider.
302       *
303       * @return  The created SSL socket factory.
304       *
305       * @throws  GeneralSecurityException  If a problem occurs while creating or
306       *                                    initializing the SSL socket factory.
307       */
308      public SSLSocketFactory createSSLSocketFactory()
309             throws GeneralSecurityException
310      {
311        return createSSLContext().getSocketFactory();
312      }
313    
314    
315    
316      /**
317       * Creates an SSL socket factory with the configured key and trust managers.
318       * It will use the default provider.
319       *
320       * @param  protocol  The protocol to use.  As per the Java SE 6 Cryptography
321       *                   Architecture document, the set of supported protocols
322       *                   should include at least "SSLv3", "TLSv1", "TLSv1.1", and
323       *                   "SSLv2Hello".  It must not be {@code null}.
324       *
325       * @return  The created SSL socket factory.
326       *
327       * @throws  GeneralSecurityException  If a problem occurs while creating or
328       *                                    initializing the SSL socket factory.
329       */
330      public SSLSocketFactory createSSLSocketFactory(final String protocol)
331             throws GeneralSecurityException
332      {
333        return createSSLContext(protocol).getSocketFactory();
334      }
335    
336    
337    
338      /**
339       * Creates an SSL socket factory with the configured key and trust managers.
340       *
341       * @param  protocol  The protocol to use.  As per the Java SE 6 Cryptography
342       *                   Architecture document, the set of supported protocols
343       *                   should include at least "SSLv3", "TLSv1", "TLSv1.1", and
344       *                   "SSLv2Hello".  It must not be {@code null}.
345       * @param  provider  The name of the provider to use for cryptographic
346       *                   operations.  It must not be {@code null}.
347       *
348       * @return  The created SSL socket factory.
349       *
350       * @throws  GeneralSecurityException  If a problem occurs while creating or
351       *                                    initializing the SSL socket factory.
352       */
353      public SSLSocketFactory createSSLSocketFactory(final String protocol,
354                                                     final String provider)
355             throws GeneralSecurityException
356      {
357        return createSSLContext(protocol, provider).getSocketFactory();
358      }
