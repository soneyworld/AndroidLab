040    /**
041     * This class defines a generic command line argument, which provides
042     * functionality applicable to all argument types.  Subclasses may enforce
043     * additional constraints or provide additional functionality.
044     */
045    @NotExtensible()
046    @Mutable()
047    @ThreadSafety(level=ThreadSafetyLevel.NOT_THREADSAFE)
048    public abstract class Argument
049           implements Serializable
050    {
051      /**
052       * The serial version UID for this serializable class.
053       */
054      private static final long serialVersionUID = -6938320885602903919L;
055    
056    
057    
058      // Indicates whether this argument should be excluded from usage information.
059      private boolean isHidden;
060    
061      // Indicates whether this argument has been registered with the argument
062      // parser.
063      private boolean isRegistered;
064    
065      // Indicates whether this argument is required to be present.
066      private final boolean isRequired;
067    
068      // Indicates whether this argument is used to display usage information.
069      private boolean isUsageArgument;
070    
071      // The short identifier for this argument, or an empty list if there are none.
072      private final ArrayList<Character> shortIdentifiers;
073    
074      // The maximum number of times this argument is allowed to be provided.
075      private int maxOccurrences;
076    
077      // The number of times this argument was included in the provided command line
078      // arguments.
079      private int numOccurrences;
080    
081      // The description for this argument.
082      private final String description;
083    
084      // The long identifier(s) for this argument, or an empty list if there are
085      // none.
086      private final ArrayList<String> longIdentifiers;
087    
088      // The value placeholder for this argument, or {@code null} if it does not
089      // take a value.
090      private final String valuePlaceholder;
091    
092    
093    
094      /**
095       * Creates a new argument with the provided information.
096       *
097       * @param  shortIdentifier   The short identifier for this argument.  It may
098       *                           not be {@code null} if the long identifier is
099       *                           {@code null}.
100       * @param  longIdentifier    The long identifier for this argument.  It may
101       *                           not be {@code null} if the short identifier is
102       *                           {@code null}.
103       * @param  isRequired        Indicates whether this argument is required to
104       *                           be provided.
105       * @param  maxOccurrences    The maximum number of times this argument may be
106       *                           provided on the command line.  A value less than
107       *                           or equal to zero indicates that it may be present
108       *                           any number of times.
109       * @param  valuePlaceholder  A placeholder to display in usage information to
110       *                           indicate that a value must be provided.  If this
111       *                           is {@code null}, then the argument will not be
112       *                           allowed to take a value.  If it is not
113       *                           {@code null}, then the argument will be required
114       *                           to take a value.
115       * @param  description       A human-readable description for this argument.
116       *                           It must not be {@code null}.
117       *
118       * @throws  ArgumentException  If there is a problem with the definition of
119       *                             this argument.
120       */
121      protected Argument(final Character shortIdentifier,
122                         final String longIdentifier,
123                         final boolean isRequired, final int maxOccurrences,
124                         final String valuePlaceholder, final String description)
125                throws ArgumentException
126      {
127        if (description == null)
128        {
129          throw new ArgumentException(ERR_ARG_DESCRIPTION_NULL.get());
130        }
131    
132        if ((shortIdentifier == null) && (longIdentifier == null))
133        {
134          throw new ArgumentException(ERR_ARG_NO_IDENTIFIERS.get());
135        }
136    
137        shortIdentifiers = new ArrayList<Character>(1);
138        if (shortIdentifier != null)
139        {
140          shortIdentifiers.add(shortIdentifier);
141        }
142    
143        longIdentifiers = new ArrayList<String>(1);
144        if (longIdentifier != null)
145        {
146          longIdentifiers.add(longIdentifier);
147        }
148    
149        this.isRequired       = isRequired;
150        this.valuePlaceholder = valuePlaceholder;
151        this.description      = description;
152    
153        if (maxOccurrences > 0)
154        {
155          this.maxOccurrences = maxOccurrences;
156        }
157        else
158        {
159          this.maxOccurrences = Integer.MAX_VALUE;
160        }
161    
162        numOccurrences  = 0;
163        isHidden        = false;
164        isRegistered    = false;
165        isUsageArgument = false;
166      }
167    
168    
169    
170      /**
171       * Indicates whether this argument has a short identifier.
172       *
173       * @return  {@code true} if it has a short identifier, or {@code false} if
174       *          not.
175       */
176      public final boolean hasShortIdentifier()
177      {
178        return (! shortIdentifiers.isEmpty());
179      }
180    
181    
182    
183      /**
184       * Retrieves the short identifier for this argument.  If there is more than
185       * one, then the first will be returned.
186       *
187       * @return  The short identifier for this argument, or {@code null} if none is
188       *          defined.
189       */
190      public final Character getShortIdentifier()
191      {
192        if (shortIdentifiers.isEmpty())
193        {
194          return null;
195        }
196        else
197        {
198          return shortIdentifiers.get(0);
199        }
200      }
201    
202    
203    
204      /**
205       * Retrieves the list of short identifiers for this argument.
206       *
207       * @return  The list of short identifiers for this argument, or an empty list
208       *          if there are none.
209       */
210      public final List<Character> getShortIdentifiers()
211      {
212        return Collections.unmodifiableList(shortIdentifiers);
213      }
214    
215    
216    
217      /**
218       * Adds the provided character to the set of short identifiers for this
219       * argument.  Note that this must be called before this argument is registered
220       * with the argument parser.
221       *
222       * @param  c  The character to add to the set of short identifiers for this
223       *            argument.  It must not be {@code null}.
224       *
225       * @throws  ArgumentException  If this argument is already registered with the
226       *                             argument parser.
227       */
228      public final void addShortIdentifier(final Character c)
229             throws ArgumentException
230      {
231        if (isRegistered)
232        {
233          throw new ArgumentException(ERR_ARG_ID_CHANGE_AFTER_REGISTERED.get(
234                                           getIdentifierString()));
235        }
236    
237        shortIdentifiers.add(c);
238      }
239    
240    
241    
242      /**
243       * Indicates whether this argument has a long identifier.
244       *
245       * @return  {@code true} if it has a long identifier, or {@code false} if
246       *          not.
247       */
248      public final boolean hasLongIdentifier()
249      {
250        return (! longIdentifiers.isEmpty());
251      }
252    
253    
254    
255      /**
256       * Retrieves the long identifier for this argument.  If it has multiple long
257       * identifiers, then the first will be returned.
258       *
259       * @return  The long identifier for this argument, or {@code null} if none is
260       *          defined.
261       */
262      public final String getLongIdentifier()
263      {
264        if (longIdentifiers.isEmpty())
265        {
266          return null;
267        }
268        else
269        {
270          return longIdentifiers.get(0);
271        }
272      }
273    
274    
275    
276      /**
277       * Retrieves the list of long identifiers for this argument.
278       *
279       * @return  The long identifier for this argument, or an empty list if there
280       *          are none.
281       */
282      public final List<String> getLongIdentifiers()
283      {
284        return Collections.unmodifiableList(longIdentifiers);
285      }
286    
287    
288    
289      /**
290       * Adds the provided string to the set of short identifiers for this argument.
291       * Note that this must be called before this argument is registered with the
292       * argument parser.
293       *
294       * @param  s  The string to add to the set of short identifiers for this
295       *            argument.  It must not be {@code null}.
296       *
297       * @throws  ArgumentException  If this argument is already registered with the
298       *                             argument parser.
299       */
300      public final void addLongIdentifier(final String s)
301             throws ArgumentException
302      {
303        if (isRegistered)
304        {
305          throw new ArgumentException(ERR_ARG_ID_CHANGE_AFTER_REGISTERED.get(
306                                           getIdentifierString()));
307        }
308    
309        longIdentifiers.add(s);
310      }

313    
314      /**
315       * Retrieves a string that may be used to identify this argument.  If a long
316       * identifier is defined, then the value returned will be two dashes followed
317       * by that string.  Otherwise, the value returned will be a single dash
318       * followed by the short identifier.
319       *
320       * @return  A string that may be used to identify this argument.
321       */
322      public final String getIdentifierString()
323      {
324        if (longIdentifiers.isEmpty())
325        {
326          return "-" + shortIdentifiers.get(0);
327        }
328        else
329        {
330          return "--" + longIdentifiers.get(0);
331        }
332      }

337       * Indicates whether this argument is required to be provided.
338       *
339       * @return  {@code true} if this argument is required to be provided, or
340       *          {@code false} if not.
341       */
342      public final boolean isRequired()
343      {
344        return isRequired;
345      }
346    
347    
348    
349      /**
350       * Retrieves the maximum number of times that this argument may be provided.
351       *
352       * @return  The maximum number of times that this argument may be provided.
353       */
354      public final int getMaxOccurrences()
355      {
356        return maxOccurrences;
357      }
358    
359    
360    
361      /**
362       * Specifies the maximum number of times that this argument may be provided.
363       *
364       * @param  maxOccurrences  The maximum number of times that this argument
365       *                         may be provided.  A value less than or equal to
366       *                         zero indicates that there should be no limit on the
367       *                         maximum number of occurrences.
368       */
369      public final void setMaxOccurrences(final int maxOccurrences)
370      {
371        if (maxOccurrences <= 0)
372        {
373          this.maxOccurrences = Integer.MAX_VALUE;
374        }
375        else
376        {
377          this.maxOccurrences = maxOccurrences;
378        }
379      }

383      /**
384       * Indicates whether this argument takes a value.
385       *
386       * @return  {@code true} if this argument takes a value, or {@code false} if
387       *          not.
388       */
389      public boolean takesValue()
390      {
391        return (valuePlaceholder != null);
392      }
393    
394    
395    
396      /**
397       * Retrieves the value placeholder string for this argument.
398       *
399       * @return  The value placeholder string for this argument, or {@code null} if
400       *          it does not take a value.
401       */
402      public final String getValuePlaceholder()
403      {
404        return valuePlaceholder;
405      }
406

409      /**
410       * Retrieves the description for this argument.
411       *
412       * @return  The description for this argument.
413       */
414      public final String getDescription()
415      {
416        return description;
417      }
418    
419    
420    
421      /**
422       * Indicates whether this argument should be excluded from usage information.
423       *
424       * @return  {@code true} if this argument should be excluded from usage
425       *          information, or {@code false} if not.
426       */
427      public final boolean isHidden()
428      {
429        return isHidden;
430      }

435       * Specifies whether this argument should be excluded from usage information.
436       *
437       * @param  isHidden  Specifies whether this argument should be excluded from
438       *                   usage information.

440      public final void setHidden(final boolean isHidden)
441      {
442        this.isHidden = isHidden;
443      }

447      /**
448       * Indicates whether this argument is intended to be used to trigger the
449       * display of usage information.  If a usage argument is provided on the
450       * command line, then the argument parser will not complain about missing
451       * required arguments or unresolved dependencies.
452       *
453       * @return  {@code true} if this argument is a usage argument, or
454       *          {@code false} if not.
455       */
456      public final boolean isUsageArgument()
457      {
458        return isUsageArgument;
459      }
460    
461    
462    
463      /**
464       * Specifies whether this argument should be considered a usage argument.
465       *
466       * @param  isUsageArgument  Specifies whether this argument should be
467       *                          considered a usage argument.
468       */
469      public final void setUsageArgument(final boolean isUsageArgument)
470      {
471        this.isUsageArgument = isUsageArgument;
472      }
473    
474    
475    
476      /**
477       * Indicates whether this argument was either included in the provided set of
478       * command line arguments or has a default value that can be used instead.
479       * This method should not be called until after the argument parser has
480       * processed the provided set of arguments.
481       *
482       * @return  {@code true} if this argument was included in the provided set of
483       *          command line arguments, or {@code false} if not.
484       */
485      public final boolean isPresent()
486      {
487        return ((numOccurrences > 0) || hasDefaultValue());
488      }
489    
490    
491    
492      /**
493       * Retrieves the number of times that this argument was included in the
494       * provided set of command line arguments.  This method should not be called
495       * until after the argument parser has processed the provided set of
496       * arguments.
497       *
498       * @return  The number of times that this argument was included in the
499       *          provided set of command line arguments.
500       */
501      public final int getNumOccurrences()
502      {
503        return numOccurrences;
504      }
505    
506    
507    
508      /**
509       * Increments the number of occurrences for this argument in the provided set
510       * of command line arguments.  This method should only be called by the
511       * argument parser.
512       *
513       * @throws  ArgumentException  If incrementing the number of occurrences would
514       *                             exceed the maximum allowed number.
515       */
516      final void incrementOccurrences()
517            throws ArgumentException
518      {
519        if (numOccurrences >= maxOccurrences)
520        {
521          throw new ArgumentException(ERR_ARG_MAX_OCCURRENCES_EXCEEDED.get(
522                                           getIdentifierString()));
523        }
524    
525        numOccurrences++;
526      }
527

530      /**
531       * Adds the provided value to the set of values for this argument.  This
532       * method should only be called by the argument parser.
533       *
534       * @param  valueString  The string representation of the value.
535       *
536       * @throws  ArgumentException  If the provided value is not acceptable, if
537       *                             this argument does not accept values, or if
538       *                             this argument already has the maximum allowed
539       *                             number of values.
540       */
541      protected abstract void addValue(final String valueString)
542                throws ArgumentException;
543    
544    
545    
546      /**
547       * Indicates whether this argument has one or more default values that will be
548       * used if it is not provided on the command line.
549       *
550       * @return  {@code true} if this argument has one or more default values, or
551       *          {@code false} if not.
552       */
553      protected abstract boolean hasDefaultValue();
554    
555    
556    
557      /**
558       * Indicates whether this argument has been registered with the argument
559       * parser.
560       *
561       * @return  {@code true} if this argument has been registered with the
562       *          argument parser, or {@code false} if not.
563       */
564      boolean isRegistered()
565      {
566        return isRegistered;
567      }
568    
569    
570    
571      /**
572       * Specifies that this argument has been registered with the argument parser.
573       * This method should only be called by the argument parser method used to
574       * register the argument.
575       *
576       * @throws  ArgumentException  If this argument has already been registered.
577       */
578      void setRegistered()
579           throws ArgumentException
580      {
581        if (isRegistered)
582        {
583          throw new ArgumentException(ERR_ARG_ALREADY_REGISTERED.get(
584                                           getIdentifierString()));
585        }
586    
587        isRegistered = true;
588      }
