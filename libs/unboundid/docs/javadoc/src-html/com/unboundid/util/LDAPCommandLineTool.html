075    /**
076     * This class provides a basis for developing command-line tools that
077     * communicate with an LDAP directory server.  It provides a common set of
078     * options for connecting and authenticating to a directory server, and then
079     * provides a mechanism for obtaining connections and connection pools to use
080     * when communicating with that server.
081     * <BR><BR>
082     * The arguments that this class supports include:
083     * <UL>
084     *   <LI>"-h {address}" or "--hostname {address}" -- Specifies the address of
085     *       the directory server.  If this isn't specified, then a default of
086     *       "localhost" will be used.</LI>
087     *   <LI>"-p {port}" or "--port {port}" -- Specifies the port number of the
088     *       directory server.  If this isn't specified, then a default port of 389
089     *       will be used.</LI>
090     *   <LI>"-D {bindDN}" or "--bindDN {bindDN}" -- Specifies the DN to use to bind
091     *       to the directory server using simple authentication.  If this isn't
092     *       specified, then simple authentication will not be performed.</LI>
093     *   <LI>"-w {password}" or "--bindPassword {password}" -- Specifies the
094     *       password to use when binding with simple authentication or a
095     *       password-based SASL mechanism.</LI>
096     *   <LI>"-j {path}" or "--bindPasswordFile {path}" -- Specifies the path to the
097     *       file containing the password to use when binding with simple
098     *       authentication or a password-based SASL mechanism.</LI>
099     *   <LI>"-Z" or "--useSSL" -- Indicates that the communication with the server
100     *       should be secured using SSL.</LI>
101     *   <LI>"-q" or "--useStartTLS" -- Indicates that the communication with the
102     *       server should be secured using StartTLS.</LI>
103     *   <LI>"-X" or "--trustAll" -- Indicates that the client should trust any
104     *       certificate that the server presents to it.</LI>
105     *   <LI>"-K {path}" or "--keyStorePath {path}" -- Specifies the path to the
106     *       key store to use to obtain client certificates.</LI>
107     *   <LI>"-W {password}" or "--keyStorePassword {password}" -- Specifies the
108     *       password to use to access the contents of the key store.</LI>
109     *   <LI>"-u {path}" or "--keyStorePasswordFile {path}" -- Specifies the path to
110     *       the file containing the password to use to access the contents of the
111     *       key store.</LI>
112     *   <LI>"-P {path}" or "--trustStorePath {path}" -- Specifies the path to the
113     *       trust store to use when determining whether to trust server
114     *       certificates.</LI>
115     *   <LI>"-T {password}" or "--trustStorePassword {password}" -- Specifies the
116     *       password to use to access the contents of the trust store.</LI>
117     *   <LI>"-U {path}" or "--trustStorePasswordFile {path}" -- Specifies the path
118     *       to the file containing the password to use to access the contents of
119     *       the trust store.</LI>
120     *   <LI>"-N {nickname}" or "--certNickname {nickname}" -- Specifies the
121     *       nickname of the client certificate to use when performing SSL client
122     *       authentication.</LI>
123     *   <LI>"-o {name=value}" or "--saslOption {name=value}" -- Specifies a SASL
124     *       option to use when performing SASL authentication.</LI>
125     * </UL>
126     * If SASL authentication is to be used, then a "mech" SASL option must be
127     * provided to specify the name of the SASL mechanism to use (e.g.,
128     * "--saslOption mech=EXTERNAL" indicates that the EXTERNAL mechanism should be
129     * used).  Depending on the SASL mechanism, additional SASL options may be
130     * required or optional.  They include:
131     * <UL>
132     *   <LI>
133     *     mech=ANONYMOUS
134     *     <UL>
135     *       <LI>Required SASL options:  </LI>
136     *       <LI>Optional SASL options:  trace</LI>
137     *     </UL>
138     *   </LI>
139     *   <LI>
140     *     mech=CRAM-MD5
141     *     <UL>
142     *       <LI>Required SASL options:  authID</LI>
143     *       <LI>Optional SASL options:  </LI>
144     *     </UL>
145     *   </LI>
146     *   <LI>
147     *     mech=DIGEST-MD5
148     *     <UL>
149     *       <LI>Required SASL options:  authID</LI>
150     *       <LI>Optional SASL options:  authzID, realm</LI>
151     *     </UL>
152     *   </LI>
153     *   <LI>
154     *     mech=EXTERNAL
155     *     <UL>
156     *       <LI>Required SASL options:  </LI>
157     *       <LI>Optional SASL options:  </LI>
158     *     </UL>
159     *   </LI>
160     *   <LI>
161     *     mech=GSSAPI
162     *     <UL>
163     *       <LI>Required SASL options:  authID</LI>
164     *       <LI>Optional SASL options:  authzID, realm, kdcAddress</LI>
165     *     </UL>
166     *   </LI>
167     *   <LI>
168     *     mech=PLAIN
169     *     <UL>
170     *       <LI>Required SASL options:  authID</LI>
171     *       <LI>Optional SASL options:  authzID</LI>
172     *     </UL>
173     *   </LI>
174     * </UL>
175     * <BR><BR>
176     * Note that in general, methods in this class are not threadsafe.  However, the
177     * {@link #getConnection()} and {@link #getConnectionPool(int,int)} methods may
178     * be invoked concurrently by multiple threads accessing the same instance only
179     * while that instance is in the process of invoking the
180     * {@link #doToolProcessing()} method.
181     */
182    @Extensible()

184    public abstract class LDAPCommandLineTool

487      /**
488       * {@inheritDoc}
489       */
490      @Override()
491      public final void doExtendedArgumentValidation()
492             throws ArgumentException
493      {
494        // If more than one hostname or port number was provided, then make sure
495        // that the same number of values were provided for each.
496        if ((host.getValues().size() > 1) || (port.getValues().size() > 1))
497        {
498          if (host.getValues().size() != port.getValues().size())
499          {
500            throw new ArgumentException(
501                 ERR_LDAP_TOOL_HOST_PORT_COUNT_MISMATCH.get(
502                      host.getLongIdentifier(), port.getLongIdentifier()));
503          }
504        }
505    
506    
507        // If any SASL options were provided, then make sure they are valid.
508        saslMechanism = null;
509        saslOptions = new LinkedHashMap<String,String>(10);
510        if (saslOption.isPresent())
511        {
512          for (final String s : saslOption.getValues())
513          {
514            final int equalPos = s.indexOf('=');
515            if (equalPos < 0)
516            {
517              throw new ArgumentException(
518                   ERR_LDAP_TOOL_MALFORMED_SASL_OPTION.get(s));
519            }
520    
521            final String optionName  = toLowerCase(s.substring(0, equalPos));
522            final String optionValue = s.substring(equalPos+1);
523            saslOptions.put(optionName, optionValue);
524          }
525    
526          final LinkedHashMap<String,String> optionsCopy =
527               new LinkedHashMap<String,String>(saslOptions);
528          final String mech = optionsCopy.remove("mech");
529          if (mech == null)
530          {
531            throw new ArgumentException(ERR_LDAP_TOOL_NO_SASL_MECH.get());
532          }
533    
534          saslMechanism = toLowerCase(mech);
535          final List<String> requiredOptions =
536               REQUIRED_SASL_OPTIONS.get(saslMechanism);
537          final List<String> optionalOptions =
538               OPTIONAL_SASL_OPTIONS.get(saslMechanism);
539          if (requiredOptions == null)
540          {
541            throw new ArgumentException(
542                 ERR_LDAP_TOOL_UNSUPPORTED_SASL_MECH.get(mech));
543          }
544    
545          for (final String s : requiredOptions)
546          {
547            if (optionsCopy.remove(s) == null)
548            {
549              throw new ArgumentException(
550                   ERR_LDAP_TOOL_MISSING_REQUIRED_SASL_OPTION.get(s, mech));
551            }
552          }
553    
554          for (final String s : optionalOptions)
555          {
556            optionsCopy.remove(s);
557          }
558    
559          if (! optionsCopy.isEmpty())
560          {
561            final String option = optionsCopy.keySet().iterator().next();
562            throw new ArgumentException(
563                 ERR_LDAP_TOOL_INVALID_SASL_OPTION.get(option, mech));
564          }
565        }
566    
567    
568        doExtendedNonLDAPArgumentValidation();
569      }

623      /**
624       * Retrieves a connection that may be used to communicate with the target
625       * directory server.
626       * <BR><BR>
627       * Note that this method is threadsafe and may be invoked by multiple threads
628       * accessing the same instance only while that instance is in the process of
629       * invoking the {@link #doToolProcessing} method.
630       *
631       * @return  A connection that may be used to communicate with the target
632       *          directory server.
633       *
634       * @throws  LDAPException  If a problem occurs while creating the connection.
635       */
636      @ThreadSafety(level=ThreadSafetyLevel.METHOD_THREADSAFE)
637      public final LDAPConnection getConnection()
638             throws LDAPException
639      {
640        if (serverSet == null)
641        {
642          serverSet   = createServerSet();
643          bindRequest = createBindRequest();
644        }
645    
646        final LDAPConnection connection = serverSet.getConnection();
647    
648        if (useStartTLS.isPresent())
649        {
650          try
651          {
652            final ExtendedResult extendedResult =
653                 connection.processExtendedOperation(
654                      new StartTLSExtendedRequest(startTLSContext));
655            if (! extendedResult.getResultCode().equals(ResultCode.SUCCESS))
656            {
657              throw new LDAPException(extendedResult.getResultCode(),
658                   ERR_LDAP_TOOL_START_TLS_FAILED.get(
659                        extendedResult.getDiagnosticMessage()));
660            }
661          }
662          catch (LDAPException le)
663          {
664            debugException(le);
665            connection.close();
666            throw le;
667          }
668        }
669    
670        try
671        {
672          if (bindRequest != null)
673          {
674            connection.bind(bindRequest);
675          }
676        }
677        catch (LDAPException le)
678        {
679          debugException(le);
680          connection.close();
681          throw le;
682        }
683    
684        return connection;
685      }

689      /**
690       * Retrieves a connection pool that may be used to communicate with the target
691       * directory server.
692       * <BR><BR>
693       * Note that this method is threadsafe and may be invoked by multiple threads
694       * accessing the same instance only while that instance is in the process of
695       * invoking the {@link #doToolProcessing} method.
696       *
697       * @param  initialConnections  The number of connections that should be
698       *                             initially established in the pool.
699       * @param  maxConnections      The maximum number of connections to maintain
700       *                             in the pool.
701       *
702       * @return  A connection that may be used to communicate with the target
703       *          directory server.
704       *
705       * @throws  LDAPException  If a problem occurs while creating the connection
706       *                         pool.
707       */
708      @ThreadSafety(level=ThreadSafetyLevel.METHOD_THREADSAFE)
709      public final LDAPConnectionPool getConnectionPool(
710                                           final int initialConnections,
711                                           final int maxConnections)
712                throws LDAPException
713      {
714        if (serverSet == null)
715        {
716          serverSet   = createServerSet();
717          bindRequest = createBindRequest();
718        }
719    
720        PostConnectProcessor postConnectProcessor = null;
721        if (useStartTLS.isPresent())
722        {
723          postConnectProcessor = new StartTLSPostConnectProcessor(startTLSContext);
724        }
725    
726        return new LDAPConnectionPool(serverSet, bindRequest, initialConnections,
727                                      maxConnections, postConnectProcessor);
728      }

732      /**
733       * Creates the server set to use when creating connections or connection
734       * pools.
735       *
736       * @return  The server set to use when creating connections or connection
737       *          pools.
738       *
739       * @throws  LDAPException  If a problem occurs while creating the server set.
740       */
741      private ServerSet createServerSet()
742              throws LDAPException
743      {
744        final SSLUtil sslUtil = createSSLUtil();
745    
746        SocketFactory socketFactory = null;
747        if (useSSL.isPresent())
748        {
749          try
750          {
751            socketFactory = sslUtil.createSSLSocketFactory();
752          }
753          catch (Exception e)
754          {
755            debugException(e);
756            throw new LDAPException(ResultCode.LOCAL_ERROR,
757                 ERR_LDAP_TOOL_CANNOT_CREATE_SSL_SOCKET_FACTORY.get(
758                      getExceptionMessage(e)), e);
759          }
760        }
761        else if (useStartTLS.isPresent())
762        {
763          try
764          {
765            startTLSContext = sslUtil.createSSLContext();
766          }
767          catch (Exception e)
768          {
769            debugException(e);
770            throw new LDAPException(ResultCode.LOCAL_ERROR,
771                 ERR_LDAP_TOOL_CANNOT_CREATE_SSL_CONTEXT.get(
772                      getExceptionMessage(e)), e);
773          }
774        }
775    
776        if (host.getValues().size() == 1)
777        {
778          return new SingleServerSet(host.getValue(), port.getValue(),
779                                     socketFactory, getConnectionOptions());
780        }
781        else
782        {
783          final List<String>  hostList = host.getValues();
784          final List<Integer> portList = port.getValues();
785    
786          final String[] hosts = new String[hostList.size()];
787          final int[]    ports = new int[hosts.length];
788    
789          for (int i=0; i < hosts.length; i++)
790          {
791            hosts[i] = hostList.get(i);
792            ports[i] = portList.get(i);
793          }
794    
795          return new RoundRobinServerSet(hosts, ports, socketFactory,
796                                         getConnectionOptions());
797        }
798      }

802      /**
803       * Creates the SSLUtil instance to use for secure communication.
804       *
805       * @return  The SSLUtil instance to use for secure communication, or
806       *          {@code null} if secure communication is not needed.
807       *
808       * @throws  LDAPException  If a problem occurs while creating the SSLUtil
809       *                         instance.
810       */
811      private SSLUtil createSSLUtil()
812              throws LDAPException
813      {
814        if (useSSL.isPresent() || useStartTLS.isPresent())
815        {
816          KeyManager keyManager = null;
817          if (keyStorePath.isPresent())
818          {
819            char[] pw = null;
820            if (keyStorePassword.isPresent())
821            {
822              pw = keyStorePassword.getValue().toCharArray();
823            }
824            else if (keyStorePasswordFile.isPresent())
825            {
826              try
827              {
828                pw = keyStorePasswordFile.getNonBlankFileLines().get(0).
829                          toCharArray();
830              }
831              catch (Exception e)
832              {
833                debugException(e);
834                throw new LDAPException(ResultCode.LOCAL_ERROR,
835                     ERR_LDAP_TOOL_CANNOT_READ_KEY_STORE_PASSWORD.get(
836                          getExceptionMessage(e)), e);
837              }
838            }
839    
840            try
841            {
842              keyManager = new KeyStoreKeyManager(keyStorePath.getValue(), pw, null,
843                                                  certificateNickname.getValue());
844            }
845            catch (Exception e)
846            {
847              debugException(e);
848              throw new LDAPException(ResultCode.LOCAL_ERROR,
849                   ERR_LDAP_TOOL_CANNOT_CREATE_KEY_MANAGER.get(
850                        getExceptionMessage(e)), e);
851            }
852          }
853    
854          TrustManager trustManager;
855          if (trustAll.isPresent())
856          {
857            trustManager = new TrustAllTrustManager(false);
858          }
859          else if (trustStorePath.isPresent())
860          {
861            char[] pw = null;
862            if (trustStorePassword.isPresent())
863            {
864              pw = trustStorePassword.getValue().toCharArray();
865            }
866            else if (trustStorePasswordFile.isPresent())
867            {
868              try
869              {
870                pw = trustStorePasswordFile.getNonBlankFileLines().get(0).
871                          toCharArray();
872              }
873              catch (Exception e)
874              {
875                debugException(e);
876                throw new LDAPException(ResultCode.LOCAL_ERROR,
877                     ERR_LDAP_TOOL_CANNOT_READ_TRUST_STORE_PASSWORD.get(
878                          getExceptionMessage(e)), e);
879              }
880            }
881    
882            trustManager = new TrustStoreTrustManager(trustStorePath.getValue(), pw,
883                                                      null, true);
884          }
885          else
886          {
887            trustManager = promptTrustManager.get();
888            if (trustManager == null)
889            {
890              final PromptTrustManager m = new PromptTrustManager();
891              promptTrustManager.compareAndSet(null, m);
892              trustManager = promptTrustManager.get();
893            }
894          }
895    
896          return new SSLUtil(keyManager, trustManager);
897        }
898        else
899        {
900          return null;
901        }
902      }

906      /**
907       * Creates the bind request to use to authenticate to the server.
908       *
909       * @return  The bind request to use to authenticate to the server, or
910       *          {@code null} if no bind should be performed.
911       *
912       * @throws  LDAPException  If a problem occurs while creating the bind
913       *                         request.
914       */
915      private BindRequest createBindRequest()
916              throws LDAPException
917      {
918        final String pw;
919        if (bindPassword.isPresent())
920        {
921          pw = bindPassword.getValue();
922        }
923        else if (bindPasswordFile.isPresent())
924        {
925          try
926          {
927            pw = bindPasswordFile.getNonBlankFileLines().get(0);
928          }
929          catch (Exception e)
930          {
931            debugException(e);
932            throw new LDAPException(ResultCode.LOCAL_ERROR,
933                 ERR_LDAP_TOOL_CANNOT_READ_BIND_PASSWORD.get(
934                      getExceptionMessage(e)), e);
935          }
936        }
937        else
938        {
939          pw = null;
940        }
941    
942        if (bindDN.isPresent())
943        {
944          return new SimpleBindRequest(bindDN.getValue(), pw);
945        }
946        else if (saslMechanism != null)
947        {
948          if (saslMechanism.equals(SASL_MECH_ANONYMOUS))
949          {
950            return new ANONYMOUSBindRequest(saslOptions.get(SASL_OPTION_TRACE));
951          }
952          else if (saslMechanism.equals(SASL_MECH_CRAM_MD5))
953          {
954            return new CRAMMD5BindRequest(saslOptions.get(SASL_OPTION_AUTH_ID), pw);
955          }
956          else if (saslMechanism.equals(SASL_MECH_DIGEST_MD5))
957          {
958            return new DIGESTMD5BindRequest(saslOptions.get(SASL_OPTION_AUTH_ID),
959                 saslOptions.get(SASL_OPTION_AUTHZ_ID), pw,
960                 saslOptions.get(SASL_OPTION_REALM));
961          }
962          else if (saslMechanism.equals(SASL_MECH_EXTERNAL))
963          {
964            return new EXTERNALBindRequest();
965          }
966          else if (saslMechanism.equals(SASL_MECH_GSSAPI))
967          {
968            return new GSSAPIBindRequest(saslOptions.get(SASL_OPTION_AUTH_ID),
969                 saslOptions.get(SASL_OPTION_AUTHZ_ID), pw,
970                 saslOptions.get(SASL_OPTION_REALM),
971                 saslOptions.get(SASL_OPTION_KDC_ADDRESS), null);
972          }
973          else if (saslMechanism.equals(SASL_MECH_PLAIN))
974          {
975            return new PLAINBindRequest(saslOptions.get(SASL_OPTION_AUTH_ID),
976                 saslOptions.get(SASL_OPTION_AUTHZ_ID), pw);
977          }
978          else
979          {
980            throw new LDAPException(ResultCode.NOT_SUPPORTED,
981                 ERR_LDAP_TOOL_UNSUPPORTED_SASL_MECH.get(saslMechanism));
982          }
983        }
984        else
985        {
986          return null;
987        }
988      }
