040    /**
041     * This class provides a mechanism for writing one or more ASN.1 elements into a
042     * byte string buffer.  It may be cleared and re-used any number of times, and
043     * the contents may be written to an {@code OutputStream} or {@code ByteBuffer},
044     * or copied to a byte array.  {@code ASN1Buffer} instances are not threadsafe
045     * and should not be accessed concurrently by multiple threads.
046     */
047    @Mutable()
048    @ThreadSafety(level=ThreadSafetyLevel.NOT_THREADSAFE)
049    public final class ASN1Buffer
050           implements Serializable
051    {

053       * The default maximum buffer size.
054       */
055      private static final int DEFAULT_MAX_BUFFER_SIZE = 1048576;

124      /**
125       * Creates a new instance of this ASN.1 buffer with an optional maximum
126       * retained size.  If a maximum size is defined, then this buffer may be used
127       * to hold elements larger than that, but when the buffer is cleared it will
128       * be shrunk to the maximum size.
129       *
130       * @param  maxBufferSize  The maximum buffer size that will be retained by
131       *                        this ASN.1 buffer.  A value less than or equal to
132       *                        zero indicates that no maximum size should be
133       *                        enforced.
134       */
135      public ASN1Buffer(final int maxBufferSize)
136      {
137        this.maxBufferSize = maxBufferSize;
138    
139        buffer = new ByteStringBuffer();
140      }

144      /**
145       * Clears the contents of this buffer.  If there are any outstanding sequences
146       * or sets that have been created but not closed, then they must no longer be
147       * used and any attempt to do so may yield unpredictable results.
148       */
149      public void clear()
150      {
151        buffer.clear();
152    
153        if ((maxBufferSize > 0) && (buffer.capacity() > maxBufferSize))
154        {
155          buffer.setCapacity(maxBufferSize);
156        }
157      }

258      /**
259       * Adds an integer element to this ASN.1 buffer using the provided BER type.
260       *
261       * @param  type      The BER type to use for the integer element.
262       * @param  intValue  The value to use for the integer element.
263       */
264      public void addInteger(final byte type, final int intValue)
265      {
266        buffer.append(type);
267    
268        if (intValue < 0)
269        {
270          if ((intValue & 0xFFFFFF80) == 0xFFFFFF80)
271          {
272            buffer.append((byte) 0x01);
273            buffer.append((byte) (intValue & 0xFF));
274          }
275          else if ((intValue & 0xFFFF8000) == 0xFFFF8000)
276          {
277            buffer.append((byte) 0x02);
278            buffer.append((byte) ((intValue >> 8) & 0xFF));
279            buffer.append((byte) (intValue & 0xFF));
280          }
281          else if ((intValue & 0xFF800000) == 0xFF800000)
282          {
283            buffer.append((byte) 0x03);
284            buffer.append((byte) ((intValue >> 16) & 0xFF));
285            buffer.append((byte) ((intValue >> 8) & 0xFF));
286            buffer.append((byte) (intValue & 0xFF));
287          }
288          else
289          {
290            buffer.append((byte) 0x04);
291            buffer.append((byte) ((intValue >> 24) & 0xFF));
292            buffer.append((byte) ((intValue >> 16) & 0xFF));
293            buffer.append((byte) ((intValue >> 8) & 0xFF));
294            buffer.append((byte) (intValue & 0xFF));
295          }
296        }
297        else
298        {
299          if ((intValue & 0x0000007F) == intValue)
300          {
301            buffer.append((byte) 0x01);
302            buffer.append((byte) (intValue & 0x7F));
303          }
304          else if ((intValue & 0x00007FFF) == intValue)
305          {
306            buffer.append((byte) 0x02);
307            buffer.append((byte) ((intValue >> 8) & 0x7F));
308            buffer.append((byte) (intValue & 0xFF));
309          }
310          else if ((intValue & 0x007FFFFF) == intValue)
311          {
312            buffer.append((byte) 0x03);
313            buffer.append((byte) ((intValue >> 16) & 0x7F));
314            buffer.append((byte) ((intValue >> 8) & 0xFF));
315            buffer.append((byte) (intValue & 0xFF));
316          }
317          else
318          {
319            buffer.append((byte) 0x04);
320            buffer.append((byte) ((intValue >> 24) & 0x7F));
321            buffer.append((byte) ((intValue >> 16) & 0xFF));
322            buffer.append((byte) ((intValue >> 8) & 0xFF));
323            buffer.append((byte) (intValue & 0xFF));
324          }
325        }
326      }

638      /**
639       * Adds an octet string element to this ASN.1 buffer using the provided BER
640       * type.
641       *
642       * @param  type   The BER type to use for the octet string element.
643       * @param  value  The value to use for the octet string element.
644       */
645      public void addOctetString(final byte type, final String value)
646      {
647        buffer.append(type);
648    
649        if (value == null)
650        {
651          buffer.append((byte) 0x00);
652        }
653        else
654        {
655          // We'll assume that the string contains only ASCII characters and
656          // therefore the number of bytes will equal the number of characters.
657          // However, save the position in case we're wrong and need to re-encode.
658          final int lengthStartPos = buffer.length();
659          ASN1Element.encodeLengthTo(value.length(), buffer);
660    
661          final int valueStartPos = buffer.length();
662          buffer.append(value);
663    
664          if (buffer.length() != (valueStartPos + value.length()))
665          {
666            final byte[] valueBytes = new byte[buffer.length() - valueStartPos];
667            System.arraycopy(buffer.getBackingArray(), valueStartPos, valueBytes, 0,
668                             valueBytes.length);
669    
670            buffer.setLength(lengthStartPos);
671            ASN1Element.encodeLengthTo(valueBytes.length, buffer);
672            buffer.append(valueBytes);
673          }
674        }
675      }

694      /**
695       * Begins adding elements to an ASN.1 sequence using the provided BER type.
696       *
697       * @param  type  The BER type to use for the sequence.
698       *
699       * @return  An object that may be used to indicate when the end of the
700       *          sequence has been reached.  Once all embedded sequence elements
701       *          have been added, then the {@link ASN1BufferSequence#end} method
702       *          MUST be called to ensure that the sequence is properly encoded.
703       */
704      public ASN1BufferSequence beginSequence(final byte type)
705      {

745      /**
746       * Ensures that the appropriate length is inserted into the internal buffer
747       * after all elements in a sequence or set have been added.
748       *
749       * @param  valueStartPos  The position in which the first value was added.
750       */
751      void endSequenceOrSet(final int valueStartPos)
752      {
753        final int length = buffer.length() - valueStartPos;
754        if (length == 0)
755        {
756          buffer.append((byte) 0x00);
757          return;
758        }
759    
760        if ((length & 0x7F) == length)
761        {
762          buffer.insert(valueStartPos, (byte) length);
763        }
764        else if ((length & 0xFF) == length)
765        {
766          buffer.insert(valueStartPos, MULTIBYTE_LENGTH_HEADER_PLUS_ONE);
767    
768          final byte[] backingArray = buffer.getBackingArray();
769          backingArray[valueStartPos+1] = (byte) (length & 0xFF);
770        }
771        else if ((length & 0xFFFF) == length)
772        {
773          buffer.insert(valueStartPos, MULTIBYTE_LENGTH_HEADER_PLUS_TWO);
774    
775          final byte[] backingArray = buffer.getBackingArray();
776          backingArray[valueStartPos+1] = (byte) ((length >> 8) & 0xFF);
777          backingArray[valueStartPos+2] = (byte) (length & 0xFF);
778        }
779        else if ((length & 0xFFFFFF) == length)
780        {
781          buffer.insert(valueStartPos, MULTIBYTE_LENGTH_HEADER_PLUS_THREE);
782    
783          final byte[] backingArray = buffer.getBackingArray();
784          backingArray[valueStartPos+1] = (byte) ((length >> 16) & 0xFF);
785          backingArray[valueStartPos+2] = (byte) ((length >> 8) & 0xFF);
786          backingArray[valueStartPos+3] = (byte) (length & 0xFF);
787        }
788        else
789        {
790          buffer.insert(valueStartPos, MULTIBYTE_LENGTH_HEADER_PLUS_FOUR);
791    
792          final byte[] backingArray = buffer.getBackingArray();
793          backingArray[valueStartPos+1] = (byte) ((length >> 24) & 0xFF);
794          backingArray[valueStartPos+2] = (byte) ((length >> 16) & 0xFF);
795          backingArray[valueStartPos+3] = (byte) ((length >> 8) & 0xFF);
796          backingArray[valueStartPos+4] = (byte) (length & 0xFF);
797        }
798      }
