050    /**
051     * This class implements the processing necessary to perform an LDAPv3 modify DN
052     * operation, which can be used to rename and/or move an entry or subtree in the
053     * directory.  A modify DN request contains the DN of the target entry, the new
054     * RDN to use for that entry, and a flag which indicates whether to remove the
055     * current RDN attribute value(s) from the entry.  It may optionally contain a
056     * new superior DN, which will cause the entry to be moved below that new parent
057     * entry.
058     * <BR><BR>
059     * Note that some directory servers may not support all possible uses of the
060     * modify DN operation.  In particular, some servers may not support the use of
061     * a new superior DN, especially if it may cause the entry to be moved to a
062     * different database or another server.  Also, some servers may not support
063     * renaming or moving non-leaf entries (i.e., entries that have one or more
064     * subordinates).
065     * <BR><BR>
066     * {@code ModifyDNRequest} objects are mutable and therefore can be altered and
067     * re-used for multiple requests.  Note, however, that {@code ModifyDNRequest}
068     * objects are not threadsafe and therefore a single {@code ModifyDNRequest}
069     * object instance should not be used to process multiple requests at the same
070     * time.
071     * <BR><BR>
072     * <H2>Example</H2>
073     * The following example demonstrates the process for performing a modify DN
074     * operation.  In this case, it will rename "ou=People,dc=example,dc=com" to
075     * "ou=Users,dc=example,dc=com".  It will not move the entry below a new parent.
076     * <PRE>
077     *   ModifyDNRequest modifyDNRequest =
078     *        new ModifyDNRequest("ou=People,dc=example,dc=com", "ou=Users", true);
079     *
080     *   try
081     *   {
082     *     LDAPResult modifyDNResult = connection.modifyDN(modifyDNRequest);
083     *
084     *     System.out.println("The entry was renamed successfully.");
085     *   }
086     *   catch (LDAPException le)
087     *   {
088     *     System.err.println("The modify DN operation failed.");
089     *   }
090     * </PRE>
091     */

589      /**
590       * Sends this modify DN request to the directory server over the provided
591       * connection and returns the associated response.
592       *
593       * @param  connection  The connection to use to communicate with the directory
594       *                     server.
595       * @param  depth       The current referral depth for this request.  It should
596       *                     always be one for the initial request, and should only
597       *                     be incremented when following referrals.
598       *
599       * @return  An LDAP result object that provides information about the result
600       *          of the modify DN processing.
601       *
602       * @throws  LDAPException  If a problem occurs while sending the request or
603       *                         reading the response.
604       */
605      @Override()
606      protected LDAPResult process(final LDAPConnection connection, final int depth)
607                throws LDAPException
608      {
609        if (connection.synchronousMode())
610        {
611          return processSync(connection, depth);
612        }
613    
614        final long requestTime = System.nanoTime();
615        processAsync(connection, null);
616    
617        try
618        {
619          // Wait for and process the response.
620          final LDAPResponse response;
621          try
622          {
623            final long responseTimeout = getResponseTimeoutMillis(connection);
624            if (responseTimeout > 0)
625            {
626              response = responseQueue.poll(responseTimeout, TimeUnit.MILLISECONDS);
627            }
628            else
629            {
630              response = responseQueue.take();
631            }
632          }
633          catch (InterruptedException ie)
634          {
635            debugException(ie);
636            throw new LDAPException(ResultCode.LOCAL_ERROR,
637                 ERR_MODDN_INTERRUPTED.get(connection.getHostPort()), ie);
638          }
639    
640          return handleResponse(connection, response, requestTime, depth);
641        }
642        finally
643        {
644          connection.deregisterResponseAcceptor(messageID);
645        }
646      }

650      /**
651       * Sends this modify DN request to the directory server over the provided
652       * connection and returns the message ID for the request.
653       *
654       * @param  connection      The connection to use to communicate with the
655       *                         directory server.
656       * @param  resultListener  The async result listener that is to be notified
657       *                         when the response is received.  It may be
658       *                         {@code null} only if the result is to be processed
659       *                         by this class.
660       *
661       * @return  The LDAP message ID for the modify DN request that was sent to the
662       *          server.
663       *
664       * @throws  LDAPException  If a problem occurs while sending the request.
665       */
666      int processAsync(final LDAPConnection connection,
667                       final AsyncResultListener resultListener)
668          throws LDAPException
669      {
670        // Create the LDAP message.
671        messageID = connection.nextMessageID();
672        final LDAPMessage message = new LDAPMessage(messageID, this, getControls());
673    
674    
675        // If the provided async result listener is {@code null}, then we'll use
676        // this class as the message acceptor.  Otherwise, create an async helper
677        // and use it as the message acceptor.
678        if (resultListener == null)
679        {
680          connection.registerResponseAcceptor(messageID, this);
681        }
682        else
683        {
684          final AsyncHelper helper = new AsyncHelper(connection,
685               LDAPMessage.PROTOCOL_OP_TYPE_MODIFY_DN_RESPONSE, resultListener,
686               getIntermediateResponseListener());
687          connection.registerResponseAcceptor(messageID, helper);
688        }
689    
690    
691        // Send the request to the server.
692        try
693        {
694          debugLDAPRequest(this);
695          connection.getConnectionStatistics().incrementNumModifyDNRequests();
696          connection.sendMessage(message);
697          return messageID;
698        }
699        catch (LDAPException le)
700        {
701          debugException(le);
702    
703          connection.deregisterResponseAcceptor(messageID);
704          throw le;
705        }
706      }

710      /**
711       * Processes this modify DN operation in synchronous mode, in which the same
712       * thread will send the request and read the response.
713       *
714       * @param  connection  The connection to use to communicate with the directory
715       *                     server.
716       * @param  depth       The current referral depth for this request.  It should
717       *                     always be one for the initial request, and should only
718       *                     be incremented when following referrals.
719       *
720       * @return  An LDAP result object that provides information about the result
721       *          of the modify DN processing.
722       *
723       * @throws  LDAPException  If a problem occurs while sending the request or
724       *                         reading the response.
725       */
726      private LDAPResult processSync(final LDAPConnection connection,
727                                     final int depth)
728              throws LDAPException
729      {
730        // Create the LDAP message.
731        messageID = connection.nextMessageID();
732        final LDAPMessage message =
733             new LDAPMessage(messageID,  this, getControls());
734    
735    
736        // Set the appropriate timeout on the socket.
737        try
738        {
739          connection.getConnectionInternals().getSocket().setSoTimeout(
740               (int) getResponseTimeoutMillis(connection));
741        }
742        catch (Exception e)
743        {
744          debugException(e);
745        }
746    
747    
748        // Send the request to the server.
749        final long requestTime = System.nanoTime();
750        debugLDAPRequest(this);
751        connection.getConnectionStatistics().incrementNumModifyDNRequests();
752        connection.sendMessage(message);
753    
754        final LDAPResponse response = connection.readResponse(messageID);
755        return handleResponse(connection, response, requestTime, depth);
756      }

760      /**
761       * Performs the necessary processing for handling a response.
762       *
763       * @param  connection   The connection used to read the response.
764       * @param  response     The response to be processed.
765       * @param  requestTime  The time the request was sent to the server.
766       * @param  depth        The current referral depth for this request.  It
767       *                      should always be one for the initial request, and
768       *                      should only be incremented when following referrals.
769       *
770       * @return  The modify DN result.
771       *
772       * @throws  LDAPException  If a problem occurs.
773       */
774      private LDAPResult handleResponse(final LDAPConnection connection,
775                                        final LDAPResponse response,
776                                        final long requestTime, final int depth)
777              throws LDAPException
778      {
779        if (response == null)
780        {
781          final long waitTime = nanosToMillis(System.nanoTime() - requestTime);
782          throw new LDAPException(ResultCode.TIMEOUT,
783               ERR_MODIFY_DN_CLIENT_TIMEOUT.get(waitTime,
784                    connection.getHostPort()));
785        }
786    
787        connection.getConnectionStatistics().incrementNumModifyDNResponses(
788             System.nanoTime() - requestTime);
789        if (response instanceof ConnectionClosedResponse)
790        {
791          // The connection was closed while waiting for the response.
792          final ConnectionClosedResponse ccr = (ConnectionClosedResponse) response;
793          final String message = ccr.getMessage();
794          if (message == null)
795          {
796            throw new LDAPException(ccr.getResultCode(),
797                 ERR_CONN_CLOSED_WAITING_FOR_MODIFY_DN_RESPONSE.get(
798                      connection.getHostPort(), toString()));
799          }
800          else
801          {
802            throw new LDAPException(ccr.getResultCode(),
803                 ERR_CONN_CLOSED_WAITING_FOR_MODIFY_DN_RESPONSE_WITH_MESSAGE.get(
804                      connection.getHostPort(), toString(), message));
805          }
806        }
807    
808        final LDAPResult result = (LDAPResult) response;
809        if ((result.getResultCode().equals(ResultCode.REFERRAL)) &&
810            followReferrals(connection))
811        {
812          if (depth >= connection.getConnectionOptions().getReferralHopLimit())
813          {
814            return new LDAPResult(messageID, ResultCode.REFERRAL_LIMIT_EXCEEDED,
815                                  ERR_TOO_MANY_REFERRALS.get(),
816                                  result.getMatchedDN(), result.getReferralURLs(),
817                                  result.getResponseControls());
818          }
819    
820          return followReferral(result, connection, depth);
821        }
822        else
823        {
824          return result;
825        }
826      }

830      /**
831       * Attempts to follow a referral to perform a modify DN operation in the
832       * target server.
833       *
834       * @param  referralResult  The LDAP result object containing information about
835       *                         the referral to follow.
836       * @param  connection      The connection on which the referral was received.
837       * @param  depth           The number of referrals followed in the course of
838       *                         processing this request.
839       *
840       * @return  The result of attempting to process the modify DN operation by
841       *          following the referral.
842       *
843       * @throws  LDAPException  If a problem occurs while attempting to establish
844       *                         the referral connection, sending the request, or
845       *                         reading the result.
846       */
847      private LDAPResult followReferral(final LDAPResult referralResult,
848                                        final LDAPConnection connection,
849                                        final int depth)
850              throws LDAPException
851      {
852        for (final String urlString : referralResult.getReferralURLs())
853        {
854          try
855          {
856            final LDAPURL referralURL = new LDAPURL(urlString);
857            final String host = referralURL.getHost();
858    
859            if (host == null)
860            {
861              // We can't handle a referral in which there is no host.
862              continue;
863            }
864    
865            final ModifyDNRequest modifyDNRequest;
866            if (referralURL.baseDNProvided())
867            {
868              modifyDNRequest =
869                   new ModifyDNRequest(referralURL.getBaseDN().toString(),
870                                       newRDN, deleteOldRDN, newSuperiorDN,
871                                       getControls());
872            }
873            else
874            {
875              modifyDNRequest = this;
876            }
877    
878            final LDAPConnection referralConn =
879                 connection.getReferralConnection(referralURL, connection);
880            try
881            {
882              return modifyDNRequest.process(referralConn, depth+1);
883            }
884            finally
885            {
886              referralConn.setDisconnectInfo(DisconnectType.REFERRAL, null, null);
887              referralConn.close();
888            }
889          }
890          catch (LDAPException le)
891          {
892            debugException(le);
893          }
894        }
895    
896        // If we've gotten here, then we could not follow any of the referral URLs,
897        // so we'll just return the original referral result.
898        return referralResult;
899      }

1000      {
1001        buffer.append("ModifyDNRequest(dn='");
1002        buffer.append(dn);
1003        buffer.append("', newRDN='");
1004        buffer.append(newRDN);
1005        buffer.append("', deleteOldRDN=");

1011          buffer.append(newSuperiorDN);
1012          buffer.append('\'');
1013        }
1014    
1015        final Control[] controls = getControls();
1016        if (controls.length > 0)
1017        {
1018          buffer.append(", controls={");
1019          for (int i=0; i < controls.length; i++)
1020          {
1021            if (i > 0)
1022            {
1023              buffer.append(", ");
1024            }
1025    
1026            buffer.append(controls[i]);
1027          }
1028          buffer.append('}');
1029        }
1030    
1031        buffer.append(')');
1032      }
