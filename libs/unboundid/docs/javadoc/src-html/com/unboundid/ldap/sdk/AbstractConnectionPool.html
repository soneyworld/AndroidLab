106      /**
107       * Releases the provided connection back to the pool after an exception has
108       * been encountered while processing an operation on that connection.  The
109       * connection pool health check instance associated with this pool will be
110       * used to determine whether the provided connection is still valid and will
111       * either release it back for use in processing other operations on the
112       * connection or will terminate the connection and create a new one to take
113       * its place.
114       *
115       * @param  connection  The connection to be evaluated and released back to the
116       *                     pool or replaced with a new connection.
117       * @param  exception   The exception caught while processing an operation on
118       *                     the connection.
119       */
120      public void releaseConnectionAfterException(final LDAPConnection connection,
121                                                  final LDAPException exception)
122      {
123        final LDAPConnectionPoolHealthCheck healthCheck = getHealthCheck();
124    
125        try
126        {
127          healthCheck.ensureConnectionValidAfterException(connection, exception);
128          releaseConnection(connection);
129        }
130        catch (LDAPException le)
131        {
132          debugException(le);
133          releaseDefunctConnection(connection);
134        }
135      }

2355       * Processes multiple requests in the order they are provided over a single
2356       * connection from this pool.
2357       *
2358       * @param  requests         The list of requests to be processed.  It must not
2359       *                          be {@code null} or empty.
2360       * @param  continueOnError  Indicates whether to attempt to process subsequent
2361       *                          requests if any of the operations does not
2362       *                          complete successfully.
2363       *
2364       * @return  The set of results from the requests that were processed.  The
2365       *          order of result objects will correspond to the order of the
2366       *          request objects, although the list of results may contain fewer
2367       *          elements than the list of requests if an error occurred during
2368       *          processing and {@code continueOnError} is {@code false}.
2369       *
2370       * @throws  LDAPException  If a problem occurs while trying to obtain a
2371       *                         connection to use for the requests.
2372       */
2373      public List<LDAPResult> processRequests(final List<LDAPRequest> requests,
2374                                              final boolean continueOnError)
2375             throws LDAPException
2376      {
2377        ensureNotNull(requests);
2378        ensureFalse(requests.isEmpty(),
2379             "LDAPConnectionPool.processRequests.requests must not be empty.");
2380    
2381        final LDAPConnection conn;
2382        try
2383        {
2384          conn = getConnection();
2385        }
2386        catch (LDAPException le)
2387        {
2388          debugException(le);
2389          throw new LDAPSearchException(le);
2390        }
2391    
2392        final ArrayList<LDAPResult> results =
2393             new ArrayList<LDAPResult>(requests.size());
2394        boolean isDefunct = false;
2395    
2396        try
2397        {
2398    requestLoop:
2399          for (final LDAPRequest request : requests)
2400          {
2401            try
2402            {
2403              final LDAPResult result = request.process(conn, 1);
2404              results.add(result);
2405              switch (result.getResultCode().intValue())
2406              {
2407                case ResultCode.SUCCESS_INT_VALUE:
2408                case ResultCode.COMPARE_FALSE_INT_VALUE:
2409                case ResultCode.COMPARE_TRUE_INT_VALUE:
2410                case ResultCode.NO_OPERATION_INT_VALUE:
2411                  // These will be considered successful operations.
2412                  break;
2413    
2414                default:
2415                  // Anything else will be considered a failure.
2416                  if (! ResultCode.isConnectionUsable(result.getResultCode()))
2417                  {
2418                    isDefunct = true;
2419                  }
2420    
2421                  if (! continueOnError)
2422                  {
2423                    break requestLoop;
2424                  }
2425                  break;
2426              }
2427            }
2428            catch (LDAPException le)
2429            {
2430              debugException(le);
2431              results.add(new LDAPResult(request.getLastMessageID(),
2432                                         le.getResultCode(), le.getMessage(),
2433                                         le.getMatchedDN(), le.getReferralURLs(),
2434                                         le.getResponseControls()));
2435    
2436              if (! ResultCode.isConnectionUsable(le.getResultCode()))
2437              {
2438                isDefunct = true;
2439              }
2440    
2441              if (! continueOnError)
2442              {
2443                break;
2444              }
2445            }
2446          }
2447        }
2448        finally
2449        {
2450          if (isDefunct)
2451          {
2452            releaseDefunctConnection(conn);
2453          }
2454          else
2455          {
2456            releaseConnection(conn);
2457          }
2458        }
2459    
2460        return results;
2461      }
