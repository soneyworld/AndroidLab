036    /**
037     * This class provides an implementation of the authorization identity bind
038     * request control as described in
039     * <A HREF="http://www.ietf.org/rfc/rfc3829.txt">RFC 3829</A>.  It may be
040     * included in a bind request to request that the server include the
041     * authorization identity associated with the client connection in the bind
042     * response message, in the form of an
043     * {@link AuthorizationIdentityResponseControl}.
044     * <BR><BR>
045     * The authorization identity request control is similar to the "Who Am I?"
046     * extended request as implemented in the
047     * {@link com.unboundid.ldap.sdk.extensions.WhoAmIExtendedRequest} class.  The
048     * primary difference between them is that the "Who Am I?" extended request can
049     * be used at any time but requires a separate operation, while the
050     * authorization identity request control can be included only with a bind
051     * request but does not require a separate operation.
052     * <BR><BR>
053     * <H2>Example</H2>
054     * The following example demonstrates the use of the authorization identity
055     * request and response controls.  It authenticates to the directory server and
056     * attempts to retrieve the authorization identity from the response:
057     * <PRE>
058     *   String authzID = null;
059     *   BindRequest bindRequest =
060     *        new SimpleBindRequest("uid=john.doe,ou=People,dc=example,dc=com",
061     *                              "password",
062     *                              new AuthorizationIdentityRequestControl());
063     *
064     *   BindResult bindResult = connection.bind(bindRequest);
065     *   for (Control c : bindResult.getResponseControls())
066     *   {
067     *     if (c instanceof AuthorizationIdentityResponseControl)
068     *     {
069     *       AuthorizationIdentityResponseControl authzResponseControl =
070     *            (AuthorizationIdentityResponseControl) c;
071     *       authzID = authzResponseControl.getAuthorizationID();
072     *       break;
073     *     }
074     *   }
075     * </PRE>
076     */
077    @NotMutable()
078    @ThreadSafety(level=ThreadSafetyLevel.COMPLETELY_THREADSAFE)
079    public final class AuthorizationIdentityRequestControl
080           extends Control
081    {
082      /**
083       * The OID (2.16.840.1.113730.3.4.16) for the authorization identity request
084       * control.
085       */
086      public static final String AUTHORIZATION_IDENTITY_REQUEST_OID =
087           "2.16.840.1.113730.3.4.16";
088    
089    
090    
091      /**
092       * The serial version UID for this serializable class.
093       */
094      private static final long serialVersionUID = -4059607155175828138L;
095    
096    
097    
098      /**
099       * Creates a new authorization identity request control.  The control will not
100       * be marked critical.
101       */
102      public AuthorizationIdentityRequestControl()
103      {
104        super(AUTHORIZATION_IDENTITY_REQUEST_OID, false, null);
105      }
106    
107    
108    
109      /**
110       * Creates a new authorization identity request control.
111       *
112       * @param  isCritical  Indicates whether the control should be marked
113       *                     critical.
114       */
115      public AuthorizationIdentityRequestControl(final boolean isCritical)
116      {
117        super(AUTHORIZATION_IDENTITY_REQUEST_OID, isCritical, null);
118      }
119    
120    
121    
122      /**
123       * Creates a new authorization identity request control which is decoded from
124       * the provided generic control.
125       *
126       * @param  control  The generic control to be decoded as an authorization
127       *                  identity request control.
128       *
129       * @throws  LDAPException  If the provided control cannot be decoded as an
130       *                         authorization identity request control.
131       */
132      public AuthorizationIdentityRequestControl(final Control control)
133             throws LDAPException
134      {
135        super(control);
136    
137        if (control.hasValue())
138        {
139          throw new LDAPException(ResultCode.DECODING_ERROR,
140                                  ERR_AUTHZID_REQUEST_HAS_VALUE.get());
141        }
142      }
143    
144    
145    
146      /**
147       * {@inheritDoc}
148       */
149      @Override()
150      public String getControlName()
151      {
152        return INFO_CONTROL_NAME_AUTHZID_REQUEST.get();
153      }
154    
155    
156    
157      /**
158       * {@inheritDoc}
159       */
160      @Override()
161      public void toString(final StringBuilder buffer)
162      {
163        buffer.append("AuthorizationIdentityRequestControl(isCritical=");
164        buffer.append(isCritical());
165        buffer.append(')');
166      }
167    }
