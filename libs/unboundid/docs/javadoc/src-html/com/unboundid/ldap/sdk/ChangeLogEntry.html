047    /**
048     * This class provides a data structure for representing a changelog entry as
049     * described in draft-good-ldap-changelog.  Changelog entries provide
050     * information about a change (add, delete, modify, or modify DN) operation
051     * that was processed in the directory server.  Changelog entries may be
052     * parsed from entries, and they may be converted to LDIF change records or
053     * processed as LDAP operations.
054     */
055    @NotMutable()
056    @ThreadSafety(level=ThreadSafetyLevel.COMPLETELY_THREADSAFE)
057    public final class ChangeLogEntry
058           extends ReadOnlyEntry
059    {
060      /**
061       * The name of the attribute that contains the change number that identifies
062       * the change and the order it was processed in the server.
063       */
064      public static final String ATTR_CHANGE_NUMBER = "changeNumber";
065    
066    
067    
068      /**
069       * The name of the attribute that contains the DN of the entry targeted by
070       * the change.
071       */
072      public static final String ATTR_TARGET_DN = "targetDN";
073    
074

077       * The name of the attribute that contains the type of change made to the

085       * The name of the attribute used to hold a list of changes.  For an add
086       * operation, this will be an LDIF representation of the attributes that make
087       * up the entry.  For a modify operation, this will be an LDIF representation
088       * of the changes to the target entry.
089       */
090      public static final String ATTR_CHANGES = "changes";

112       * The name of the attribute used to hold the new superior DN for a modify DN

161      /**
162       * Creates a new changelog entry from the provided entry.
163       *
164       * @param  entry  The entry from which to create this changelog entry.
165       *
166       * @throws  LDAPException  If the provided entry cannot be parsed as a
167       *                         changelog entry.
168       */
169      public ChangeLogEntry(final Entry entry)
170             throws LDAPException
171      {
172        super(entry);
173    
174    
175        final Attribute changeNumberAttr = entry.getAttribute(ATTR_CHANGE_NUMBER);
176        if ((changeNumberAttr == null) || (! changeNumberAttr.hasValue()))
177        {
178          throw new LDAPException(ResultCode.DECODING_ERROR,
179                                  ERR_CHANGELOG_NO_CHANGE_NUMBER.get());
180        }
181    
182        try
183        {
184          changeNumber = Long.parseLong(changeNumberAttr.getValue());
185        }
186        catch (NumberFormatException nfe)
187        {
188          throw new LDAPException(ResultCode.DECODING_ERROR,
189               ERR_CHANGELOG_INVALID_CHANGE_NUMBER.get(changeNumberAttr.getValue()),
190               nfe);
191        }
192    
193    
194        final Attribute targetDNAttr = entry.getAttribute(ATTR_TARGET_DN);
195        if ((targetDNAttr == null) || (! targetDNAttr.hasValue()))
196        {
197          throw new LDAPException(ResultCode.DECODING_ERROR,
198                                  ERR_CHANGELOG_NO_TARGET_DN.get());
199        }
200        targetDN = targetDNAttr.getValue();
201    
202    
203        final Attribute changeTypeAttr = entry.getAttribute(ATTR_CHANGE_TYPE);
204        if ((changeTypeAttr == null) || (! changeTypeAttr.hasValue()))
205        {
206          throw new LDAPException(ResultCode.DECODING_ERROR,
207                                  ERR_CHANGELOG_NO_CHANGE_TYPE.get());
208        }
209        changeType = ChangeType.forName(changeTypeAttr.getValue());
210        if (changeType == null)
211        {
212          throw new LDAPException(ResultCode.DECODING_ERROR,
213               ERR_CHANGELOG_INVALID_CHANGE_TYPE.get(changeTypeAttr.getValue()));
214        }
215    
216    
217        switch (changeType)
218        {
219          case ADD:
220            attributes    = parseAddAttributeList(entry, targetDN);
221            modifications = null;
222            newRDN        = null;
223            deleteOldRDN  = false;
224            newSuperior   = null;
225            break;
226    
227          case DELETE:
228            attributes    = parseDeletedAttributeList(entry, targetDN);
229            modifications = null;
230            newRDN        = null;
231            deleteOldRDN  = false;
232            newSuperior   = null;
233            break;
234    
235          case MODIFY:
236            attributes    = null;
237            modifications = parseModificationList(entry, targetDN, false);
238            newRDN        = null;
239            deleteOldRDN  = false;
240            newSuperior   = null;
241            break;
242    
243          case MODIFY_DN:
244            attributes    = null;
245            modifications = parseModificationList(entry, targetDN, true);
246            newSuperior   = getAttributeValue(ATTR_NEW_SUPERIOR);
247    
248            final Attribute newRDNAttr = getAttribute(ATTR_NEW_RDN);
249            if ((newRDNAttr == null) || (! newRDNAttr.hasValue()))
250            {
251              throw new LDAPException(ResultCode.DECODING_ERROR,
252                                      ERR_CHANGELOG_MISSING_NEW_RDN.get());
253            }
254            newRDN = newRDNAttr.getValue();
255    
256            final Attribute deleteOldRDNAttr = getAttribute(ATTR_DELETE_OLD_RDN);
257            if ((deleteOldRDNAttr == null) || (! deleteOldRDNAttr.hasValue()))
258            {
259              throw new LDAPException(ResultCode.DECODING_ERROR,
260                                      ERR_CHANGELOG_MISSING_DELETE_OLD_RDN.get());
261            }
262            final String delOldRDNStr = toLowerCase(deleteOldRDNAttr.getValue());
263            if (delOldRDNStr.equals("true"))
264            {
265              deleteOldRDN = true;
266            }
267            else if (delOldRDNStr.equals("false"))
268            {
269              deleteOldRDN = false;
270            }
271            else
272            {
273              throw new LDAPException(ResultCode.DECODING_ERROR,
274                   ERR_CHANGELOG_MISSING_DELETE_OLD_RDN.get(delOldRDNStr));
275            }
276            break;
277    
278          default:
279            // This should never happen.
280            throw new LDAPException(ResultCode.DECODING_ERROR,
281                 ERR_CHANGELOG_INVALID_CHANGE_TYPE.get(changeTypeAttr.getValue()));
282        }
283      }

287      /**
288       * Parses the attribute list from a changelog entry representing an add
289       * operation.
290       *
291       * @param  entry     The entry containing the data to parse.
292       * @param  targetDN  The DN of the target entry.
293       *
294       * @return  The parsed attribute list.
295       *
296       * @throws  LDAPException  If an error occurs while parsing the attribute
297       *                         list.
298       */
299      private static List<Attribute> parseAddAttributeList(final Entry entry,
300                                                           final String targetDN)
301              throws LDAPException
302      {
303        final Attribute changesAttr = entry.getAttribute(ATTR_CHANGES);
304        if ((changesAttr == null) || (! changesAttr.hasValue()))
305        {
306          throw new LDAPException(ResultCode.DECODING_ERROR,
307                                  ERR_CHANGELOG_MISSING_CHANGES.get());
308        }
309    
310        final ArrayList<String> ldifLines = new ArrayList<String>();
311        ldifLines.add("dn: " + targetDN);
312    
313        final StringTokenizer tokenizer =
314             new StringTokenizer(changesAttr.getValue(), "\r\n");
315        while (tokenizer.hasMoreTokens())
316        {
317          ldifLines.add(tokenizer.nextToken());
318        }
319    
320        final String[] lineArray = new String[ldifLines.size()];
321        ldifLines.toArray(lineArray);
322    
323        try
324        {
325          final Entry e = LDIFReader.decodeEntry(lineArray);
326          return Collections.unmodifiableList(
327                      new ArrayList<Attribute>(e.getAttributes()));
328        }
329        catch (LDIFException le)
330        {
331          throw new LDAPException(ResultCode.DECODING_ERROR,
332               ERR_CHANGELOG_CANNOT_PARSE_CHANGES.get(getExceptionMessage(le)), le);
333        }
334      }

338      /**
339       * Parses the list of deleted attributes from a changelog entry representing a
340       * delete operation.  The attribute is optional, so it may not be present at
341       * all, and there are two different encodings that we need to handle.  One
342       * encoding is the same as is used for the add attribute list, and the second
343       * is similar to the encoding used for the list of changes, except that it
344       * ends with a NULL byte (0x00).
345       *
346       * @param  entry     The entry containing the data to parse.
347       * @param  targetDN  The DN of the target entry.
348       *
349       * @return  The parsed deleted attribute list, or {@code null} if the
350       *          changelog entry does not include a deleted attribute list.
351       *
352       * @throws  LDAPException  If an error occurs while parsing the deleted
353       *                         attribute list.
354       */
355      private static List<Attribute> parseDeletedAttributeList(final Entry entry,
356                                          final String targetDN)
357              throws LDAPException
358      {
359        final Attribute deletedEntryAttrs =
360             entry.getAttribute(ATTR_DELETED_ENTRY_ATTRS);
361        if ((deletedEntryAttrs == null) || (! deletedEntryAttrs.hasValue()))
362        {
363          return null;
364        }
365    
366        final byte[] valueBytes = deletedEntryAttrs.getValueByteArray();
367        if ((valueBytes.length > 0) && (valueBytes[valueBytes.length-1] == 0x00))
368        {
369          final String valueStr = new String(valueBytes, 0, valueBytes.length-2);
370    
371          final ArrayList<String> ldifLines = new ArrayList<String>();
372          ldifLines.add("dn: " + targetDN);
373          ldifLines.add("changetype: modify");
374    
375          final StringTokenizer tokenizer = new StringTokenizer(valueStr, "\r\n");
376          while (tokenizer.hasMoreTokens())
377          {
378            ldifLines.add(tokenizer.nextToken());
379          }
380    
381          final String[] lineArray = new String[ldifLines.size()];
382          ldifLines.toArray(lineArray);
383    
384          try
385          {
386    
387            final LDIFModifyChangeRecord changeRecord =
388                 (LDIFModifyChangeRecord) LDIFReader.decodeChangeRecord(lineArray);
389            final Modification[] mods = changeRecord.getModifications();
390            final ArrayList<Attribute> attrs =
391                 new ArrayList<Attribute>(mods.length);
392            for (final Modification m : mods)
393            {
394              if (! m.getModificationType().equals(ModificationType.DELETE))
395              {
396                throw new LDAPException(ResultCode.DECODING_ERROR,
397                     ERR_CHANGELOG_INVALID_DELENTRYATTRS_MOD_TYPE.get(
398                          ATTR_DELETED_ENTRY_ATTRS));
399              }
400    
401              attrs.add(m.getAttribute());
402            }
403    
404            return Collections.unmodifiableList(attrs);
405          }
406          catch (LDIFException le)
407          {
408            throw new LDAPException(ResultCode.DECODING_ERROR,
409                 ERR_CHANGELOG_INVALID_DELENTRYATTRS_MODS.get(
410                      ATTR_DELETED_ENTRY_ATTRS, getExceptionMessage(le)), le);
411          }
412        }
413        else
414        {
415          final ArrayList<String> ldifLines = new ArrayList<String>();
416          ldifLines.add("dn: " + targetDN);
417    
418          final StringTokenizer tokenizer =
419               new StringTokenizer(deletedEntryAttrs.getValue(), "\r\n");
420          while (tokenizer.hasMoreTokens())
421          {
422            ldifLines.add(tokenizer.nextToken());
423          }
424    
425          final String[] lineArray = new String[ldifLines.size()];
426          ldifLines.toArray(lineArray);
427    
428          try
429          {
430            final Entry e = LDIFReader.decodeEntry(lineArray);
431            return Collections.unmodifiableList(
432                        new ArrayList<Attribute>(e.getAttributes()));
433          }
434          catch (LDIFException le)
435          {
436            throw new LDAPException(ResultCode.DECODING_ERROR,
437                 ERR_CHANGELOG_CANNOT_PARSE_DELENTRYATTRS.get(
438                      ATTR_DELETED_ENTRY_ATTRS, getExceptionMessage(le)), le);
439          }
440        }
441      }

445      /**
446       * Parses the modification list from a changelog entry representing a modify
447       * operation.
448       *
449       * @param  entry         The entry containing the data to parse.
450       * @param  targetDN      The DN of the target entry.
451       * @param  allowMissing  Indicates whether it is acceptable for the changes
452       *                       attribute to be absent from the entry.
453       *
454       * @return  The parsed modification list, or {@code null} if the changelog
455       *          entry does not include any modifications and modifications are
456       *          not required.
457       *
458       * @throws  LDAPException  If an error occurs while parsing the modification
459       *                         list.
460       */
461      private static List<Modification> parseModificationList(
462                                             final Entry entry,
463                                             final String targetDN,
464                                             final boolean allowMissing)
465              throws LDAPException
466      {
467        final Attribute changesAttr = entry.getAttribute(ATTR_CHANGES);
468        if ((changesAttr == null) || (! changesAttr.hasValue()))
469        {
470          if (allowMissing)
471          {
472            return null;
473          }
474          else
475          {
476            throw new LDAPException(ResultCode.DECODING_ERROR,
477                                    ERR_CHANGELOG_MISSING_CHANGES.get());
478          }
479        }
480    
481    
482        final ArrayList<String> ldifLines = new ArrayList<String>();
483        ldifLines.add("dn: " + targetDN);
484        ldifLines.add("changetype: modify");
485    
486        // Even though it's a violation of the specification in
487        // draft-good-ldap-changelog, it appears that some servers (e.g., Sun DSEE)
488        // may terminate the changes value with a null character (\u0000).  If that
489        // is the case, then we'll need to strip it off before trying to parse it.
490        final StringTokenizer tokenizer;
491        final byte[] valueBytes = changesAttr.getValueByteArray();
492        if ((valueBytes.length > 0) && (valueBytes[valueBytes.length-1] == 0x00))
493        {
494          final String fullValue = changesAttr.getValue();
495          final String realValue = fullValue.substring(0, fullValue.length()-2);
496          tokenizer = new StringTokenizer(realValue, "\r\n");
497        }
498        else
499        {
500          tokenizer = new StringTokenizer(changesAttr.getValue(), "\r\n");
501        }
502    
503        while (tokenizer.hasMoreTokens())
504        {
505          ldifLines.add(tokenizer.nextToken());
506        }
507    
508        final String[] lineArray = new String[ldifLines.size()];
509        ldifLines.toArray(lineArray);
510    
511        try
512        {
513          final LDIFModifyChangeRecord changeRecord =
514               (LDIFModifyChangeRecord) LDIFReader.decodeChangeRecord(lineArray);
515          return Collections.unmodifiableList(
516                      Arrays.asList(changeRecord.getModifications()));
517        }
518        catch (LDIFException le)
519        {
520          throw new LDAPException(ResultCode.DECODING_ERROR,
521               ERR_CHANGELOG_CANNOT_PARSE_CHANGES.get(getExceptionMessage(le)), le);
522        }
523      }

668      /**
669       * Retrieves an LDIF change record that is analogous to the operation
670       * represented by this changelog entry.
671       *
672       * @return  An LDIF change record that is analogous to the operation
673       *          represented by this changelog entry.
674       */
675      public LDIFChangeRecord toLDIFChangeRecord()
676      {
677        switch (changeType)
678        {
679          case ADD:
680            return new LDIFAddChangeRecord(targetDN, attributes);
681    
682          case DELETE:
683            return new LDIFDeleteChangeRecord(targetDN);
684    
685          case MODIFY:
686            return new LDIFModifyChangeRecord(targetDN, modifications);
687    
688          case MODIFY_DN:
689            return new LDIFModifyDNChangeRecord(targetDN, newRDN, deleteOldRDN,
690                                                newSuperior);
691    
692          default:
693            // This should never happen.
694            return null;
695        }
696      }

700      /**
701       * Processes the operation represented by this changelog entry using the
702       * provided LDAP connection.
703       *
704       * @param  connection  The connection (or connection pool) to use to process
705       *                     the operation.
706       *
707       * @return  The result of processing the operation.
708       *
709       * @throws  LDAPException  If the operation could not be processed
710       *                         successfully.
711       */
712      public LDAPResult processChange(final LDAPInterface connection)
713             throws LDAPException
714      {
715        switch (changeType)
716        {
717          case ADD:
718            return connection.add(targetDN, attributes);
719    
720          case DELETE:
721            return connection.delete(targetDN);
722    
723          case MODIFY:
724            return connection.modify(targetDN, modifications);
725    
726          case MODIFY_DN:
727            return connection.modifyDN(targetDN, newRDN, deleteOldRDN, newSuperior);
728    
729          default:
730            // This should never happen.
731            return null;
732        }
733      }
