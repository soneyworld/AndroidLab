048    /**
049     * This class provides an implementation of a generic response protocol op.
050     * It must be subclassed by classes providing implementations for each
051     * operation type.
052     */
053    @InternalUseOnly()
054    @NotExtensible()
055    @ThreadSafety(level=ThreadSafetyLevel.COMPLETELY_THREADSAFE)
056    public abstract class GenericResponseProtocolOp
057           implements ProtocolOp
058    {
059      /**
060       * The BER type for the referral URLs elements.
061       */
062      public static final byte TYPE_REFERRALS = (byte) 0xA3;
063    
064    
065    
066      /**
067       * The serial version UID for this serializable class.
068       */
069      private static final long serialVersionUID = 3837308973105414874L;
070    
071    
072    
073      // The BER type for this response.
074      private final byte type;
075    
076      // The result code for this response.
077      private final int resultCode;
078    
079      // The referral URLs for this response.
080      private final List<String> referralURLs;
081    
082      // The diagnostic message for this response.
083      private final String diagnosticMessage;
084    
085      // The matched DN for this response.Static
086      private final String matchedDN;
087    
088    
089    
090      /**
091       * Creates a new instance of this response with the provided information.
092       *
093       * @param  type               The BER type for this response.
094       * @param  resultCode         The result code for this response.
095       * @param  matchedDN          The matched DN for this result, if available.
096       * @param  diagnosticMessage  The diagnostic message for this response, if
097       *                            available.
098       * @param  referralURLs       The list of referral URLs for this response, if
099       *                            available.
100       */
101      protected GenericResponseProtocolOp(final byte type, final int resultCode,
102                                        final String matchedDN,
103                                        final String diagnosticMessage,
104                                        final List<String> referralURLs)
105      {
106        this.type              = type;
107        this.resultCode        = resultCode;
108        this.matchedDN         = matchedDN;
109        this.diagnosticMessage = diagnosticMessage;
110    
111        if (referralURLs == null)
112        {
113          this.referralURLs = Collections.emptyList();
114        }
115        else
116        {
117          this.referralURLs = Collections.unmodifiableList(referralURLs);
118        }
119      }
120    
121    
122    
123      /**
124       * Creates a new response read from the provided ASN.1 stream reader.
125       *
126       * @param  reader  The ASN.1 stream reader from which to read the response.
127       *
128       * @throws  LDAPException  If a problem occurs while reading or parsing the
129       *                         response.
130       */
131      protected GenericResponseProtocolOp(final ASN1StreamReader reader)
132                throws LDAPException
133      {
134        try
135        {
136          type = (byte) reader.peek();
137          final ASN1StreamReaderSequence opSequence = reader.beginSequence();
138          resultCode = reader.readEnumerated();
139    
140          String s = reader.readString();
141          ensureNotNull(s);
142          if (s.length() == 0)
143          {
144            matchedDN = null;
145          }
146          else
147          {
148            matchedDN = s;
149          }
150    
151          s = reader.readString();
152          ensureNotNull(s);
153          if (s.length() == 0)
154          {
155            diagnosticMessage = null;
156          }
157          else
158          {
159            diagnosticMessage = s;
160          }
161    
162          if (opSequence.hasMoreElements())
163          {
164            final ArrayList<String> refs = new ArrayList<String>(1);
165            final ASN1StreamReaderSequence refSequence = reader.beginSequence();
166            while (refSequence.hasMoreElements())
167            {
168              refs.add(reader.readString());
169            }
170            referralURLs = Collections.unmodifiableList(refs);
171          }
172          else
173          {
174            referralURLs = Collections.emptyList();
175          }
176        }
177        catch (Exception e)
178        {
179          debugException(e);
180          throw new LDAPException(ResultCode.DECODING_ERROR,
181               ERR_RESPONSE_CANNOT_DECODE.get(getExceptionMessage(e)), e);
182        }
183      }
184    
185    
186    
187      /**
188       * Retrieves the result code for this response.
189       *
190       * @return  The result code for this response.
191       */
192      public final int getResultCode()
193      {
194        return resultCode;
195      }
196    
197    
198    
199      /**
200       * Retrieves the matched DN for this response, if any.
201       *
202       * @return  The matched DN for this response, or {@code null} if there is
203       *          no matched DN.
204       */
205      public final String getMatchedDN()
206      {
207        return matchedDN;
208      }
209    
210    
211    
212      /**
213       * Retrieves the diagnostic message for this response, if any.
214       *
215       * @return  The diagnostic message for this response, or {@code null} if there
216       *          is no diagnostic message.
217       */
218      public final String getDiagnosticMessage()
219      {
220        return diagnosticMessage;
221      }
222    
223    
224    
225      /**
226       * Retrieves the list of referral URLs for this response.
227       *
228       * @return  The list of referral URLs for this response, or an empty list
229       *          if there are no referral URLs.
230       */
231      public final List<String> getReferralURLs()
232      {
233        return referralURLs;
234      }
235    
236    
237    
238      /**
239       * {@inheritDoc}
240       */
241      public byte getProtocolOpType()
242      {
243        return type;
244      }
245    
246    
247    
248      /**
249       * {@inheritDoc}
250       */
251      public final void writeTo(final ASN1Buffer buffer)
252      {
253        final ASN1BufferSequence opSequence = buffer.beginSequence(type);
254        buffer.addEnumerated(resultCode);
255        buffer.addOctetString(matchedDN);
256        buffer.addOctetString(diagnosticMessage);
257    
258        if (! referralURLs.isEmpty())
259        {
260          final ASN1BufferSequence refSequence =
261               buffer.beginSequence(TYPE_REFERRALS);
262          for (final String s : referralURLs)
263          {
264            buffer.addOctetString(s);
265          }
266          refSequence.end();
267        }
268        opSequence.end();
269      }
270    
271    
272    
273      /**
274       * Retrieves a string representation of this protocol op.
275       *
276       * @return  A string representation of this protocol op.
277       */
278      @Override()
279      public final String toString()
280      {
281        final StringBuilder buffer = new StringBuilder();
282        toString(buffer);
283        return buffer.toString();
284      }
285    
286    
287    
288      /**
289       * {@inheritDoc}
290       */
291      public final void toString(final StringBuilder buffer)
292      {
293        buffer.append("ResponseProtocolOp(type=");
294        toHex(type, buffer);
295        buffer.append(", resultCode=");
296        buffer.append(resultCode);
297    
298        if (matchedDN != null)
299        {
300          buffer.append(", matchedDN='");
301          buffer.append(matchedDN);
302          buffer.append('\'');
303        }
304    
305        if (diagnosticMessage != null)
306        {
307          buffer.append(", diagnosticMessage='");
308          buffer.append(diagnosticMessage);
309          buffer.append('\'');
310        }
311    
312        if (! referralURLs.isEmpty())
313        {
314          buffer.append(", referralURLs={");
315    
316          final Iterator<String> iterator = referralURLs.iterator();
317          while (iterator.hasNext())
318          {
319            buffer.append('\'');
320            buffer.append(iterator.next());
321            buffer.append('\'');
322            if (iterator.hasNext())
323            {
324              buffer.append(',');
325            }
326          }
327    
328          buffer.append('}');
329        }
330        buffer.append(')');
331      }
