060    /**
061     * This class provides a tool that can be used to test authentication processing
062     * in an LDAP directory server using multiple threads.  Each authentication will
063     * consist of two operations:  a search to find the target entry followed by a
064     * bind to verify the credentials for that user.  The search will use the given
065     * base DN and filter, either or both of which may be a value pattern as
066     * described in the {@link ValuePattern} class.  This makes it possible to
067     * search over a range of entries rather than repeatedly performing searches
068     * with the same base DN and filter.
069     * <BR><BR>
070     * Some of the APIs demonstrated by this example include:
071     * <UL>
072     *   <LI>Argument Parsing (from the {@code com.unboundid.util.args}
073     *       package)</LI>
074     *   <LI>LDAP Command-Line Tool (from the {@code com.unboundid.util}
075     *       package)</LI>
076     *   <LI>LDAP Communication (from the {@code com.unboundid.ldap.sdk}
077     *       package)</LI>
078     *   <LI>Value Patterns (from the {@code com.unboundid.util} package)</LI>
079     * </UL>
080     * Each search must match exactly one entry, and this tool will then attempt to
081     * authenticate as the user associated with that entry.  It supports simple
082     * authentication, as well as the CRAM-MD5, DIGEST-MD5, and PLAIN SASL
083     * mechanisms.
084     * <BR><BR>
085     * All of the necessary information is provided using command line arguments.
086     * Supported arguments include those allowed by the {@link LDAPCommandLineTool}
087     * class, as well as the following additional arguments:
088     * <UL>
089     *   <LI>"-b {baseDN}" or "--baseDN {baseDN}" -- specifies the base DN to use
090     *       for the searches.  This must be provided.  It may be a simple DN, or it
091     *       may be a value pattern to express a range of base DNs.</LI>
092     *   <LI>"-s {scope}" or "--scope {scope}" -- specifies the scope to use for the
093     *       search.  The scope value should be one of "base", "one", "sub", or
094     *       "subord".  If this isn't specified, then a scope of "sub" will be
095     *       used.</LI>
096     *   <LI>"-f {filter}" or "--filter {filter}" -- specifies the filter to use for
097     *       the searches.  This must be provided.  It may be a simple filter, or it
098     *       may be a value pattern to express a range of filters.</LI>
099     *   <LI>"-A {name}" or "--attribute {name}" -- specifies the name of an
100     *       attribute that should be included in entries returned from the server.
101     *       If this is not provided, then all user attributes will be requested.
102     *       This may include special tokens that the server may interpret, like
103     *       "1.1" to indicate that no attributes should be returned, "*", for all
104     *       user attributes, or "+" for all operational attributes.  Multiple
105     *       attributes may be requested with multiple instances of this
106     *       argument.</LI>
107     *   <LI>"-C {password}" or "--credentials {password}" -- specifies the password
108     *       to use when authenticating users identified by the searches.</LI>
109     *   <LI>"-a {authType}" or "--authType {authType}" -- specifies the type of
110     *       authentication to attempt.  Supported values include "SIMPLE",
111     *       "CRAM-MD5", "DIGEST-MD5", and "PLAIN".
112     *   <LI>"-t {num}" or "--numThreads {num}" -- specifies the number of
113     *       concurrent threads to use when performing the authentication
114     *       processing.  If this is not provided, then a default of one thread will
115     *       be used.</LI>
116     *   <LI>"-i {sec}" or "--intervalDuration {sec}" -- specifies the length of
117     *       time in seconds between lines out output.  If this is not provided,
118     *       then a default interval duration of five seconds will be used.</LI>
119     *   <LI>"-I {num}" or "--numIntervals {num}" -- specifies the maximum number of
120     *       intervals for which to run.  If this is not provided, then it will
121     *       run forever.</LI>
122     *   <LI>"-r {auths-per-second}" or "--ratePerSecond {auths-per-second}" --
123     *       specifies the target number of authorizations to perform per second.
124     *       It is still necessary to specify a sufficient number of threads for
125     *       achieving this rate.  If this option is not provided, then the tool
126     *       will run at the maximum rate for the specified number of threads.</LI>
127     *   <LI>"--warmUpIntervals {num}" -- specifies the number of intervals to
128     *       complete before beginning overall statistics collection.</LI>
129     *   <LI>"--timestampFormat {format}" -- specifies the format to use for
130     *       timestamps included before each output line.  The format may be one of
131     *       "none" (for no timestamps), "with-date" (to include both the date and
132     *       the time), or "without-date" (to include only time time).</LI>
133     *   <LI>"--suppressErrorResultCodes" -- Indicates that information about the
134     *       result codes for failed operations should not be displayed.</LI>
135     *   <LI>"-c" or "--csv" -- Generate output in CSV format rather than a
136     *       display-friendly format.</LI>
137     * </UL>
138     */

462      /**
463       * Performs the actual processing for this tool.  In this case, it gets a
464       * connection to the directory server and uses it to perform the requested
465       * searches.
466       *
467       * @return  The result code for the processing that was performed.
468       */
469      @Override()
470      public ResultCode doToolProcessing()
471      {
472        // Convert the search scope from a string to an integer.
473        final SearchScope scope;
474        if (scopeStr.getValue().equalsIgnoreCase("base"))
475        {
476          scope = SearchScope.BASE;
477        }
478        else if (scopeStr.getValue().equalsIgnoreCase("one"))
479        {
480          scope = SearchScope.ONE;
481        }
482        else if (scopeStr.getValue().equalsIgnoreCase("subord"))
483        {
484          scope = SearchScope.SUBORDINATE_SUBTREE;
485        }
486        else
487        {
488          scope = SearchScope.SUB;
489        }
490    
491    
492        // Create value patterns for the base DN and filter.
493        final ValuePattern dnPattern;
494        try
495        {
496          dnPattern = new ValuePattern(baseDN.getValue());
497        }
498        catch (ParseException pe)
499        {
500          err("Unable to parse the base DN value pattern:  ", pe.getMessage());
501          return ResultCode.PARAM_ERROR;
502        }
503    
504        final ValuePattern filterPattern;
505        try
506        {
507          filterPattern = new ValuePattern(filter.getValue());
508        }
509        catch (ParseException pe)
510        {
511          err("Unable to parse the filter pattern:  ", pe.getMessage());
512          return ResultCode.PARAM_ERROR;
513        }
514    
515    
516        // Get the attributes to return.
517        final String[] attrs;
518        if (attributes.isPresent())
519        {
520          final List<String> attrList = attributes.getValues();
521          attrs = new String[attrList.size()];
522          attrList.toArray(attrs);
523        }
524        else
525        {
526          attrs = new String[0];
527        }
528    
529    
530        // If the --ratePerSecond option was specified, then limit the rate
531        // accordingly.
532        FixedRateBarrier fixedRateBarrier = null;
533        if (ratePerSecond.isPresent())
534        {
535          final int intervalSeconds = collectionInterval.getValue();
536          final int ratePerInterval = ratePerSecond.getValue() * intervalSeconds;
537    
538          fixedRateBarrier =
539               new FixedRateBarrier(1000L * intervalSeconds, ratePerInterval);
540        }
541    
542    
543        // Determine whether to include timestamps in the output and if so what
544        // format should be used for them.
545        final boolean includeTimestamp;
546        final String timeFormat;
547        if (timestampFormat.getValue().equalsIgnoreCase("with-date"))
548        {
549          includeTimestamp = true;
550          timeFormat       = "dd/MM/yyyy HH:mm:ss";
551        }
552        else if (timestampFormat.getValue().equalsIgnoreCase("without-date"))
553        {
554          includeTimestamp = true;
555          timeFormat       = "HH:mm:ss";
556        }
557        else
558        {
559          includeTimestamp = false;
560          timeFormat       = null;
561        }
562    
563    
564        // Determine whether any warm-up intervals should be run.
565        final long totalIntervals;
566        final boolean warmUp;
567        int remainingWarmUpIntervals = warmUpIntervals.getValue();
568        if (remainingWarmUpIntervals > 0)
569        {
570          warmUp = true;
571          totalIntervals = 0L + numIntervals.getValue() + remainingWarmUpIntervals;
572        }
573        else
574        {
575          warmUp = true;
576          totalIntervals = 0L + numIntervals.getValue();
577        }
578    
579    
580        // Create the table that will be used to format the output.
581        final OutputFormat outputFormat;
582        if (csvFormat.isPresent())
583        {
584          outputFormat = OutputFormat.CSV;
585        }
586        else
587        {
588          outputFormat = OutputFormat.COLUMNS;
589        }
590    
591        final ColumnFormatter formatter = new ColumnFormatter(includeTimestamp,
592             timeFormat, outputFormat, " ",
593             new FormattableColumn(12, HorizontalAlignment.RIGHT, "Recent",
594                      "Auths/Sec"),
595             new FormattableColumn(12, HorizontalAlignment.RIGHT, "Recent",
596                      "Avg Dur ms"),
597             new FormattableColumn(12, HorizontalAlignment.RIGHT, "Recent",
598                      "Errors/Sec"),
599             new FormattableColumn(12, HorizontalAlignment.RIGHT, "Overall",
600                      "Auths/Sec"),
601             new FormattableColumn(12, HorizontalAlignment.RIGHT, "Overall",
602                      "Avg Dur ms"));
603    
604    
605        // Create values to use for statistics collection.
606        final AtomicLong        authCounter   = new AtomicLong(0L);
607        final AtomicLong        errorCounter  = new AtomicLong(0L);
608        final AtomicLong        authDurations = new AtomicLong(0L);
609        final ResultCodeCounter rcCounter     = new ResultCodeCounter();
610    
611    
612        // Determine the length of each interval in milliseconds.
613        final long intervalMillis = 1000L * collectionInterval.getValue();
614    
615    
616        // Create the threads to use for the searches.
617        final CyclicBarrier barrier = new CyclicBarrier(numThreads.getValue() + 1);
618        final AuthRateThread[] threads = new AuthRateThread[numThreads.getValue()];
619        for (int i=0; i < threads.length; i++)
620        {
621          final LDAPConnection searchConnection;
622          final LDAPConnection bindConnection;
623          try
624          {
625            searchConnection = getConnection();
626            bindConnection   = getConnection();
627          }
628          catch (LDAPException le)
629          {
630            err("Unable to connect to the directory server:  ",
631                getExceptionMessage(le));
632            return le.getResultCode();
633          }
634    
635          threads[i] = new AuthRateThread(i, searchConnection, bindConnection,
636               dnPattern, scope, filterPattern, attrs, userPassword.getValue(),
637               authType.getValue(), barrier, authCounter, authDurations,
638               errorCounter, rcCounter, fixedRateBarrier);
639          threads[i].start();
640        }
641    
642    
643        // Display the table header.
644        for (final String headerLine : formatter.getHeaderLines(true))
645        {
646          out(headerLine);
647        }
648    
649    
650        // Indicate that the threads can start running.
651        try
652        {
653          barrier.await();
654        } catch (Exception e) {}
655        long overallStartTime = System.nanoTime();
656        long nextIntervalStartTime = System.currentTimeMillis() + intervalMillis;
657    
658    
659        boolean setOverallStartTime = false;
660        long    lastDuration        = 0L;
661        long    lastNumErrors       = 0L;
662        long    lastNumAuths        = 0L;
663        long    lastEndTime         = System.nanoTime();
664        for (long i=0; i < totalIntervals; i++)
665        {
666          final long startTimeMillis = System.currentTimeMillis();
667          final long sleepTimeMillis = nextIntervalStartTime - startTimeMillis;
668          nextIntervalStartTime += intervalMillis;
669          try
670          {
671            if (sleepTimeMillis > 0)
672            {
673              Thread.sleep(sleepTimeMillis);
674            }
675          } catch (Exception e) {}
676    
677          final long endTime          = System.nanoTime();
678          final long intervalDuration = endTime - lastEndTime;
679    
680          final long numAuths;
681          final long numErrors;
682          final long totalDuration;
683          if (warmUp && (remainingWarmUpIntervals > 0))
684          {
685            numAuths      = authCounter.getAndSet(0L);
686            numErrors     = errorCounter.getAndSet(0L);
687            totalDuration = authDurations.getAndSet(0L);
688          }
689          else
690          {
691            numAuths      = authCounter.get();
692            numErrors     = errorCounter.get();
693            totalDuration = authDurations.get();
694          }
695    
696          final long recentNumAuths  = numAuths - lastNumAuths;
697          final long recentNumErrors = numErrors - lastNumErrors;
698          final long recentDuration = totalDuration - lastDuration;
699    
700          final double numSeconds = intervalDuration / 1000000000.0D;
701          final double recentAuthRate = recentNumAuths / numSeconds;
702          final double recentErrorRate  = recentNumErrors / numSeconds;
703          final double recentAvgDuration =
704               1.0D * recentDuration / recentNumAuths / 1000000;
705    
706          if (warmUp && (remainingWarmUpIntervals > 0))
707          {
708            out(formatter.formatRow(recentAuthRate, recentAvgDuration,
709                 recentErrorRate, "warming up", "warming up"));
710    
711            remainingWarmUpIntervals--;
712            if (remainingWarmUpIntervals == 0)
713            {
714              out("Warm-up completed.  Beginning overall statistics collection.");
715              setOverallStartTime = true;
716            }
717          }
718          else
719          {
720            if (setOverallStartTime)
721            {
722              overallStartTime    = lastEndTime;
723              setOverallStartTime = false;
724            }
725    
726            final double numOverallSeconds =
727                 (endTime - overallStartTime) / 1000000000.0D;
728            final double overallAuthRate = numAuths / numOverallSeconds;
729            final double overallAvgDuration =
730                 1.0D * totalDuration / numAuths / 1000000;
731    
732            out(formatter.formatRow(recentAuthRate, recentAvgDuration,
733                 recentErrorRate, overallAuthRate, overallAvgDuration));
734    
735            lastNumAuths    = numAuths;
736            lastNumErrors   = numErrors;
737            lastDuration    = totalDuration;
738          }
739    
740          final List<ObjectPair<ResultCode,Long>> rcCounts =
741               rcCounter.getCounts(true);
742          if ((! suppressErrorsArgument.isPresent()) && (! rcCounts.isEmpty()))
743          {
744            err("\tError Results:");
745            for (final ObjectPair<ResultCode,Long> p : rcCounts)
746            {
747              err("\t", p.getFirst().getName(), ":  ", p.getSecond());
748            }
749          }
750    
751          lastEndTime = endTime;
752        }
753    
754    
755        // Stop all of the threads.
756        ResultCode resultCode = ResultCode.SUCCESS;
757        for (final AuthRateThread t : threads)
758        {
759          final ResultCode r = t.stopRunning();
760          if (resultCode == ResultCode.SUCCESS)
761          {
762            resultCode = r;
763          }
764        }
765    
766        return resultCode;
767      }
