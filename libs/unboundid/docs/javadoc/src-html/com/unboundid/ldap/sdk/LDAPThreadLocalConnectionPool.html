039    /**
040     * This class provides an implementation of an LDAP connection pool which
041     * maintains a dedicated connection for each thread using the connection pool.
042     * Connections will be created on an on-demand basis, so that if a thread
043     * attempts to use this connection pool for the first time then a new connection
044     * will be created by that thread.  This implementation eliminates the need to
045     * determine how best to size the connection pool, and it can eliminate
046     * contention among threads when trying to access a shared set of connections.
047     * All connections will be properly closed when the connection pool itself is
048     * closed, but if any thread which had previously used the connection pool stops
049     * running before the connection pool is closed, then the connection associated
050     * with that thread will also be closed by the Java finalizer.
051     * <BR><BR>
052     * If a thread obtains a connection to this connection pool, then that
053     * connection should not be made available to any other thread.  Similarly, if
054     * a thread attempts to check out multiple connections from the pool, then the
055     * same connection instance will be returned each time.
056     * <BR><BR>
057     * The capabilities offered by this class are generally the same as those
058     * provided by the {@link LDAPConnectionPool} class, as is the manner in which
059     * applications should interact with it.  See the class-level documentation for
060     * the {@code LDAPConnectionPool} class for additional information and examples.
061     * <BR><BR>
062     * One difference between this connection pool implementation and that provided
063     * by the {@link LDAPConnectionPool} class is that this implementation does not
064     * currently support periodic background health checks.  You can define health
065     * checks that will be invoked when a new connection is created, just before it
066     * is checked out for use, just after it is released, and if an error occurs
067     * while using the connection, but it will not maintain a separate background
068     * thread
069     */
070    @ThreadSafety(level=ThreadSafetyLevel.COMPLETELY_THREADSAFE)
071    public final class LDAPThreadLocalConnectionPool
072           extends AbstractConnectionPool
073    {
074      /**
075       * The default health check interval for this connection pool, which is set to
076       * 60000 milliseconds (60 seconds).
077       */
078      private static final long DEFAULT_HEALTH_CHECK_INTERVAL = 60000L;

154      /**
155       * Creates a new LDAP thread-local connection pool in which all connections
156       * will be clones of the provided connection.
157       *
158       * @param  connection            The connection to use to provide the template
159       *                               for the other connections to be created.
160       *                               This connection will be included in the pool.
161       *                               It must not be {@code null}, and it must be
162       *                               established to the target server.  It does
163       *                               not necessarily need to be authenticated if
164       *                               all connections in the pool are to be
165       *                               unauthenticated.
166       * @param  postConnectProcessor  A processor that should be used to perform
167       *                               any post-connect processing for connections
168       *                               in this pool.  It may be {@code null} if no
169       *                               special processing is needed.  Note that this
170       *                               processing will not be invoked on the
171       *                               provided connection that will be used as the
172       *                               first connection in the pool.
173       *
174       * @throws  LDAPException  If the provided connection cannot be used to
175       *                         initialize the pool.  If this is thrown, then all
176       *                         connections associated with the pool (including the
177       *                         one provided as an argument) will be closed.
178       */
179      public LDAPThreadLocalConnectionPool(final LDAPConnection connection,
180                  final PostConnectProcessor postConnectProcessor)
181             throws LDAPException
182      {
183        ensureNotNull(connection);
184    
185        this.postConnectProcessor = postConnectProcessor;
186    
187        healthCheck         = new LDAPConnectionPoolHealthCheck();
188        healthCheckInterval = DEFAULT_HEALTH_CHECK_INTERVAL;
189        poolStatistics      = new LDAPConnectionPoolStatistics(this);
190        connectionPoolName  = null;
191    
192        if (! connection.isConnected())
193        {
194          throw new LDAPException(ResultCode.PARAM_ERROR,
195                                  ERR_POOL_CONN_NOT_ESTABLISHED.get());
196        }
197    
198    
199        serverSet = new SingleServerSet(connection.getConnectedAddress(),
200                                        connection.getConnectedPort(),
201                                        connection.getLastUsedSocketFactory(),
202                                        connection.getConnectionOptions());
203        bindRequest = connection.getLastBindRequest();
204    
205        connections = new ConcurrentHashMap<Thread,LDAPConnection>();
206        connections.put(Thread.currentThread(), connection);
207    
208        lastExpiredDisconnectTime = 0L;
209        maxConnectionAge          = 0L;
210        closed                    = false;
211        minDisconnectInterval     = 0L;
212    
213        healthCheckThread = new LDAPConnectionPoolHealthCheckThread(this);
214        healthCheckThread.start();
215      }

284      /**
285       * Creates a new LDAP connection for use in this pool.
286       *
287       * @return  A new connection created for use in this pool.
288       *
289       * @throws  LDAPException  If a problem occurs while attempting to establish
290       *                         the connection.  If a connection had been created,
291       *                         it will be closed.
292       */
293      private LDAPConnection createConnection()
294              throws LDAPException
295      {
296        final LDAPConnection c = serverSet.getConnection(healthCheck);
297        c.setConnectionPool(this);
298    
299        // Auto-reconnect must be disabled for pooled connections, so turn it off
300        // if the associated connection options have it enabled for some reason.
301        LDAPConnectionOptions opts = c.getConnectionOptions();
302        if (opts.autoReconnect())
303        {
304          opts = opts.duplicate();
305          opts.setAutoReconnect(false);
306          c.setConnectionOptions(opts);
307        }
308    
309        if (postConnectProcessor != null)
310        {
311          try
312          {
313            postConnectProcessor.processPreAuthenticatedConnection(c);
314          }
315          catch (Exception e)
316          {
317            debugException(e);
318    
319            try
320            {
321              poolStatistics.incrementNumFailedConnectionAttempts();
322              c.setDisconnectInfo(DisconnectType.POOL_CREATION_FAILURE, null, e);
323              c.terminate(null);
324            }
325            catch (Exception e2)
326            {
327              debugException(e2);
328            }
329    
330            if (e instanceof LDAPException)
331            {
332              throw ((LDAPException) e);
333            }
334            else
335            {
336              throw new LDAPException(ResultCode.CONNECT_ERROR,
337                   ERR_POOL_POST_CONNECT_ERROR.get(getExceptionMessage(e)), e);
338            }
339          }
340        }
341    
342        try
343        {
344          if (bindRequest != null)
345          {
346            c.bind(bindRequest.duplicate());
347          }
348        }
349        catch (Exception e)
350        {
351          debugException(e);
352          try
353          {
354            poolStatistics.incrementNumFailedConnectionAttempts();
355            c.setDisconnectInfo(DisconnectType.BIND_FAILED, null, e);
356            c.terminate(null);
357          }
358          catch (Exception e2)
359          {
360            debugException(e2);
361          }
362    
363          if (e instanceof LDAPException)
364          {
365            throw ((LDAPException) e);
366          }
367          else
368          {
369            throw new LDAPException(ResultCode.CONNECT_ERROR,
370                 ERR_POOL_CONNECT_ERROR.get(getExceptionMessage(e)), e);
371          }
372        }
373    
374        if (postConnectProcessor != null)
375        {
376          try
377          {
378            postConnectProcessor.processPostAuthenticatedConnection(c);
379          }
380          catch (Exception e)
381          {
382            debugException(e);
383            try
384            {
385              poolStatistics.incrementNumFailedConnectionAttempts();
386              c.setDisconnectInfo(DisconnectType.POOL_CREATION_FAILURE, null, e);
387              c.terminate(null);
388            }
389            catch (Exception e2)
390            {
391              debugException(e2);
392            }
393    
394            if (e instanceof LDAPException)
395            {
396              throw ((LDAPException) e);
397            }
398            else
399            {
400              throw new LDAPException(ResultCode.CONNECT_ERROR,
401                   ERR_POOL_POST_CONNECT_ERROR.get(getExceptionMessage(e)), e);
402            }
403          }
404        }
405    
406        c.setConnectionPoolName(connectionPoolName);
407        poolStatistics.incrementNumSuccessfulConnectionAttempts();
408        return c;
409      }

413      /**
414       * {@inheritDoc}
415       */
416      @Override()
417      public void close()
418      {
419        closed = true;
420        healthCheckThread.stopRunning();
421    
422        final Iterator<Map.Entry<Thread,LDAPConnection>> iterator =
423             connections.entrySet().iterator();
424        while (iterator.hasNext())
425        {
426          final LDAPConnection conn = iterator.next().getValue();
427          iterator.remove();
428    
429          conn.setDisconnectInfo(DisconnectType.POOL_CLOSED, null, null);
430          conn.terminate(null);
431        }
432      }

447      /**
448       * {@inheritDoc}
449       */
450      @Override()
451      public LDAPConnection getConnection()
452             throws LDAPException
453      {
454        final Thread t = Thread.currentThread();
455        LDAPConnection conn = connections.get(t);
456    
457        if (closed)
458        {
459          if (conn != null)
460          {
461            conn.terminate(null);
462            connections.remove(t);
463          }
464    
465          poolStatistics.incrementNumFailedCheckouts();
466          throw new LDAPException(ResultCode.CONNECT_ERROR,
467                                  ERR_POOL_CLOSED.get());
468        }
469    
470        boolean created = false;
471        if (conn == null)
472        {
473          conn = createConnection();
474          connections.put(t, conn);
475          created = true;
476        }
477    
478        try
479        {
480          healthCheck.ensureConnectionValidForCheckout(conn);
481          if (created)
482          {
483            poolStatistics.incrementNumSuccessfulCheckoutsNewConnection();
484          }
485          else
486          {
487            poolStatistics.incrementNumSuccessfulCheckoutsWithoutWaiting();
488          }
489          return conn;
490        }
491        catch (LDAPException le)
492        {
493          debugException(le);
494    
495          conn.close(null);
496          connections.remove(t);
497    
498          if (created)
499          {
500            poolStatistics.incrementNumFailedCheckouts();
501            throw le;
502          }
503        }
504    
505        try
506        {
507          conn = createConnection();
508          healthCheck.ensureConnectionValidForCheckout(conn);
509          connections.put(t, conn);
510          poolStatistics.incrementNumSuccessfulCheckoutsNewConnection();
511          return conn;
512        }
513        catch (LDAPException le)
514        {
515          debugException(le);
516    
517          poolStatistics.incrementNumFailedCheckouts();
518    
519          if (conn != null)
520          {
521            conn.terminate(null);
522          }
523    
524          throw le;
525        }
526      }

530      /**
531       * {@inheritDoc}
532       */
533      @Override()
534      public void releaseConnection(final LDAPConnection connection)
535      {
536        if (connection == null)
537        {
538          return;
539        }
540    
541        connection.setConnectionPoolName(connectionPoolName);
542        if (connectionIsExpired(connection))
543        {
544          try
545          {
546            final LDAPConnection newConnection = createConnection();
547            connections.put(Thread.currentThread(), newConnection);
548    
549            connection.setDisconnectInfo(DisconnectType.POOLED_CONNECTION_EXPIRED,
550                 null, null);
551            connection.terminate(null);
552            poolStatistics.incrementNumConnectionsClosedExpired();
553            lastExpiredDisconnectTime = System.currentTimeMillis();
554          }
555          catch (final LDAPException le)
556          {
557            debugException(le);
558          }
559        }
560    
561        try
562        {
563          healthCheck.ensureConnectionValidForRelease(connection);
564        }
565        catch (LDAPException le)
566        {
567          releaseDefunctConnection(connection);
568          return;
569        }
570    
571        poolStatistics.incrementNumReleasedValid();
572    
573        if (closed)
574        {
575          close();
576        }
577      }

599      /**
600       * Performs the real work of terminating a defunct connection and replacing it
601       * with a new connection if possible.
602       *
603       * @param  connection  The defunct connection to be replaced.
604       */
605      private void handleDefunctConnection(final LDAPConnection connection)
606      {
607        final Thread t = Thread.currentThread();
608    
609        connection.setDisconnectInfo(DisconnectType.POOLED_CONNECTION_DEFUNCT, null,
610                                     null);
611        connection.terminate(null);
612        connections.remove(t);
613    
614        if (closed)
615        {
616          return;
617        }
618    
619        try
620        {
621          final LDAPConnection conn = createConnection();
622          connections.put(t, conn);
623        }
624        catch (LDAPException le)
625        {
626          debugException(le);
627        }
628      }

632      /**
633       * Indicates whether the provided connection should be considered expired.
634       *
635       * @param  connection  The connection for which to make the determination.
636       *
637       * @return  {@code true} if the provided connection should be considered
638       *          expired, or {@code false} if not.
639       */
640      private boolean connectionIsExpired(final LDAPConnection connection)
641      {
642        // If connection expiration is not enabled, then there is nothing to do.
643        if (maxConnectionAge <= 0L)
644        {
645          return false;
646        }
647    
648        // If there is a minimum disconnect interval, then make sure that we have
649        // not closed another expired connection too recently.
650        final long currentTime = System.currentTimeMillis();
651        if ((currentTime - lastExpiredDisconnectTime) < minDisconnectInterval)
652        {
653          return false;
654        }
655    
656        // Get the age of the connection and see if it is expired.
657        final long connectionAge = currentTime - connection.getConnectTime();
658        return (connectionAge > maxConnectionAge);
659      }

820      /**
821       * {@inheritDoc}
822       */
823      @Override()
824      protected void doHealthCheck()
825      {
826        final Iterator<Map.Entry<Thread,LDAPConnection>> iterator =
827             connections.entrySet().iterator();
828        while (iterator.hasNext())
829        {
830          final Map.Entry<Thread,LDAPConnection> e = iterator.next();
831          final Thread                           t = e.getKey();
832          final LDAPConnection                   c = e.getValue();
833    
834          if (! t.isAlive())
835          {
836            c.setDisconnectInfo(DisconnectType.POOLED_CONNECTION_UNNEEDED, null,
837                                null);
838            c.terminate(null);
839            iterator.remove();
840          }
841        }
842      }
