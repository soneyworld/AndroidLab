021    package com.unboundid.ldap.protocol;
022    
023    
024    
025    import com.unboundid.asn1.ASN1Buffer;
026    import com.unboundid.asn1.ASN1StreamReader;
027    import com.unboundid.ldap.sdk.LDAPException;
028    import com.unboundid.ldap.sdk.ResultCode;
029    import com.unboundid.util.NotMutable;
030    import com.unboundid.util.InternalUseOnly;
031    import com.unboundid.util.ThreadSafety;
032    import com.unboundid.util.ThreadSafetyLevel;
033    
034    import static com.unboundid.ldap.protocol.ProtocolMessages.*;
035    import static com.unboundid.util.Debug.*;
036    import static com.unboundid.util.StaticUtils.*;
037    
038    
039    
040    /**
041     * This class provides an implementation of an LDAP unbind request protocol op.
042     */
043    @InternalUseOnly()
044    @NotMutable()
045    @ThreadSafety(level=ThreadSafetyLevel.COMPLETELY_THREADSAFE)
046    public final class UnbindRequestProtocolOp
047           implements ProtocolOp
048    {
049      /**
050       * The serial version UID for this serializable class.
051       */
052      private static final long serialVersionUID = 1703200292192488474L;
053    
054    
055    
056      /**
057       * Creates a new unbind request protocol op.
058       */
059      public UnbindRequestProtocolOp()
060      {
061        // No implementation required.
062      }
063    
064    
065    
066      /**
067       * Creates a new unbind request protocol op read from the provided ASN.1
068       * stream reader.
069       *
070       * @param  reader  The ASN.1 stream reader from which to read the unbind
071       *                 request protocol op.
072       *
073       * @throws  LDAPException  If a problem occurs while reading or parsing the
074       *                         unbind request.
075       */
076      UnbindRequestProtocolOp(final ASN1StreamReader reader)
077           throws LDAPException
078      {
079        try
080        {
081          reader.readNull();
082        }
083        catch (Exception e)
084        {
085          debugException(e);
086    
087          throw new LDAPException(ResultCode.DECODING_ERROR,
088               ERR_UNBIND_REQUEST_CANNOT_DECODE.get(getExceptionMessage(e)), e);
089        }
090      }
091    
092    
093    
094      /**
095       * {@inheritDoc}
096       */
097      public byte getProtocolOpType()
098      {
099        return LDAPMessage.PROTOCOL_OP_TYPE_UNBIND_REQUEST;
100      }
101    
102    
103    
104      /**
105       * {@inheritDoc}
106       */
107      public void writeTo(final ASN1Buffer buffer)
108      {
109        buffer.addNull(LDAPMessage.PROTOCOL_OP_TYPE_UNBIND_REQUEST);
110      }
111    
112    
113    
114      /**
115       * Retrieves a string representation of this protocol op.
116       *
117       * @return  A string representation of this protocol op.
118       */
119      @Override()
120      public String toString()
121      {
122        final StringBuilder buffer = new StringBuilder();
123        toString(buffer);
124        return buffer.toString();
125      }
126    
127    
128    
129      /**
130       * {@inheritDoc}
131       */
132      public void toString(final StringBuilder buffer)
133      {
134        buffer.append("UnbindRequestProtocolOp()");
135      }
136    }
