051    /**
052     * This class provides a data structure for holding information about an LDAP
053     * modification, which describes a change to apply to an attribute.  A
054     * modification includes the following elements:
055     * <UL>
056     *   <LI>A modification type, which describes the type of change to apply.</LI>
057     *   <LI>An attribute name, which specifies which attribute should be
058     *       updated.</LI>
059     *   <LI>An optional set of values to use for the modification.</LI>
060     * </UL>
061     */
062    @NotMutable()
063    @ThreadSafety(level=ThreadSafetyLevel.COMPLETELY_THREADSAFE)
064    public final class Modification
065           implements Serializable
066    {

069       * values.

351      /**
352       * Writes an ASN.1-encoded representation of this modification to the provided
353       * ASN.1 buffer.
354       *
355       * @param  buffer  The ASN.1 buffer to which the encoded representation should
356       *                 be written.
357       */
358      public void writeTo(final ASN1Buffer buffer)
359      {
360        final ASN1BufferSequence modSequence = buffer.beginSequence();
361        buffer.addEnumerated(modificationType.intValue());
362    
363        final ASN1BufferSequence attrSequence = buffer.beginSequence();
364        buffer.addOctetString(attributeName);
365    
366        final ASN1BufferSet valueSet = buffer.beginSet();
367        for (final ASN1OctetString v : values)
368        {
369          buffer.addElement(v);
370        }
371        valueSet.end();
372        attrSequence.end();
373        modSequence.end();
374      }

379       * Encodes this modification to an ASN.1 sequence suitable for use in the LDAP
380       * protocol.
381       *
382       * @return  An ASN.1 sequence containing the encoded value.
383       */
384      public ASN1Sequence encode()
385      {
386        final ASN1Element[] attrElements =
387        {
388          new ASN1OctetString(attributeName),
389          new ASN1Set(values)
390        };
391    
392        final ASN1Element[] modificationElements =
393        {
394          new ASN1Enumerated(modificationType.intValue()),
395          new ASN1Sequence(attrElements)
396        };
397    
398        return new ASN1Sequence(modificationElements);
399      }

403      /**
404       * Reads and decodes an LDAP modification from the provided ASN.1 stream
405       * reader.
406       *
407       * @param  reader  The ASN.1 stream reader from which to read the
408       *                 modification.
409       *
410       * @return  The decoded modification.
411       *
412       * @throws  LDAPException  If a problem occurs while trying to read or decode
413       *                         the modification.
414       */
415      public static Modification readFrom(final ASN1StreamReader reader)
416             throws LDAPException
417      {
418        try
419        {
420          ensureNotNull(reader.beginSequence());
421          final ModificationType modType =
422               ModificationType.valueOf(reader.readEnumerated());
423    
424          ensureNotNull(reader.beginSequence());
425          final String attrName = reader.readString();
426    
427          final ArrayList<ASN1OctetString> valueList =
428               new ArrayList<ASN1OctetString>(5);
429          final ASN1StreamReaderSet valueSet = reader.beginSet();
430          while (valueSet.hasMoreElements())
431          {
432            valueList.add(new ASN1OctetString(reader.readBytes()));
433          }
434    
435          final ASN1OctetString[] values = new ASN1OctetString[valueList.size()];
436          valueList.toArray(values);
437    
438          return new Modification(modType, attrName, values);
439        }
440        catch (Exception e)
441        {
442          debugException(e);
443          throw new LDAPException(ResultCode.DECODING_ERROR,
444               ERR_MOD_CANNOT_DECODE.get(getExceptionMessage(e)), e);
445        }
446      }

450      /**
451       * Decodes the provided ASN.1 sequence as an LDAP modification.
452       *
453       * @param  modificationSequence  The ASN.1 sequence to decode as an LDAP
454       *                               modification.  It must not be {@code null}.
455       *
456       * @return  The decoded LDAP modification.
457       *
458       * @throws  LDAPException  If a problem occurs while trying to decode the
459       *                         provided ASN.1 sequence as an LDAP modification.
460       */
461      public static Modification decode(final ASN1Sequence modificationSequence)
462             throws LDAPException
463      {
464        ensureNotNull(modificationSequence);
465    
466        final ASN1Element[] modificationElements = modificationSequence.elements();
467        if (modificationElements.length != 2)
468        {
469          throw new LDAPException(ResultCode.DECODING_ERROR,
470                                  ERR_MOD_DECODE_INVALID_ELEMENT_COUNT.get(
471                                       modificationElements.length));
472        }
473    
474        final int modType;
475        try
476        {
477          final ASN1Enumerated typeEnumerated =
478               ASN1Enumerated.decodeAsEnumerated(modificationElements[0]);
479          modType = typeEnumerated.intValue();
480        }
481        catch (final ASN1Exception ae)
482        {
483          debugException(ae);
484          throw new LDAPException(ResultCode.DECODING_ERROR,
485               ERR_MOD_DECODE_CANNOT_PARSE_MOD_TYPE.get(getExceptionMessage(ae)),
486               ae);
487        }
488    
489        final ASN1Sequence attrSequence;
490        try
491        {
492          attrSequence = ASN1Sequence.decodeAsSequence(modificationElements[1]);
493        }
494        catch (final ASN1Exception ae)
495        {
496          debugException(ae);
497          throw new LDAPException(ResultCode.DECODING_ERROR,
498               ERR_MOD_DECODE_CANNOT_PARSE_ATTR.get(getExceptionMessage(ae)), ae);
499        }
500    
501        final ASN1Element[] attrElements = attrSequence.elements();
502        if (attrElements.length != 2)
503        {
504          throw new LDAPException(ResultCode.DECODING_ERROR,
505                                  ERR_MOD_DECODE_INVALID_ATTR_ELEMENT_COUNT.get(
506                                       attrElements.length));
507        }
508    
509        final String attrName =
510             ASN1OctetString.decodeAsOctetString(attrElements[0]).stringValue();
511    
512        final ASN1Set valueSet;
513        try
514        {
515          valueSet = ASN1Set.decodeAsSet(attrElements[1]);
516        }
517        catch (final ASN1Exception ae)
518        {
519          debugException(ae);
520          throw new LDAPException(ResultCode.DECODING_ERROR,
521                                  ERR_MOD_DECODE_CANNOT_PARSE_ATTR_VALUE_SET.get(
522                                       getExceptionMessage(ae)), ae);
523        }
524    
525        final ASN1Element[] valueElements = valueSet.elements();
526        final ASN1OctetString[] values = new ASN1OctetString[valueElements.length];
527        for (int i=0; i < values.length; i++)
528        {
529          values[i] = ASN1OctetString.decodeAsOctetString(valueElements[i]);
530        }
531    
532        return new Modification(ModificationType.valueOf(modType), attrName,
533                                values);
534      }

539       * Calculates a hash code for this LDAP modification.
540       *
541       * @return  The generated hash code for this LDAP modification.
542       */
543      @Override()
544      public int hashCode()
545      {
546        int hashCode = modificationType.intValue() +
547                       toLowerCase(attributeName).hashCode();
548    
549        for (final ASN1OctetString value : values)
550        {
551          hashCode += value.hashCode();
552        }
553    
554        return hashCode;
555      }

559      /**
560       * Indicates whether the provided object is equal to this LDAP modification.
561       * The provided object will only be considered equal if it is an LDAP
562       * modification with the same modification type, attribute name, and set of
563       * values as this LDAP modification.
564       *
565       * @param  o  The object for which to make the determination.
566       *
567       * @return  {@code true} if the provided object is equal to this modification,
568       *          or {@code false} if not.
569       */
570      @Override()
571      public boolean equals(final Object o)
572      {
573        if (o == null)
574        {
575          return false;
576        }
577    
578        if (o == this)
579        {
580          return true;
581        }
582    
583        if (! (o instanceof Modification))
584        {
585          return false;
586        }
587    
588        final Modification mod = (Modification) o;
589        if (modificationType != mod.modificationType)
590        {
591          return false;
592        }
593    
594        if (! attributeName.equalsIgnoreCase(mod.attributeName))
595        {
596          return false;
597        }
598    
599        if (values.length != mod.values.length)
600        {
601          return false;
602        }
603    
604        // Look at the values using a byte-for-byte matching.
605        for (final ASN1OctetString value : values)
606        {
607          boolean found = false;
608          for (int j = 0; j < mod.values.length; j++)
609          {
610            if (value.equalsIgnoreType(mod.values[j]))
611            {
612              found = true;
613              break;
614            }
615          }
616    
617          if (!found)
618          {
619            return false;
620          }
621        }
622    
623        // If we've gotten here, then we can consider the object equal to this LDAP
624        // modification.
625        return true;
626      }

645      /**
646       * Appends a string representation of this LDAP modification to the provided
647       * buffer.
648       *
649       * @param  buffer  The buffer to which to append the string representation of
650       *                 this LDAP modification.
651       */
652      public void toString(final StringBuilder buffer)
653      {
654        buffer.append("LDAPModification(type=");
655    
656        switch (modificationType.intValue())
657        {
658          case 0:
659            buffer.append("add");
660            break;
661          case 1:
662            buffer.append("delete");
663            break;
664          case 2:
665            buffer.append("replace");
666            break;
667          case 3:
668            buffer.append("increment");
669            break;
670          default:
671            buffer.append(modificationType);
672            break;
673        }
674    
675        buffer.append(", attr=");
676        buffer.append(attributeName);
677    
678        buffer.append(", values={");
679        if (values.length > 0)
680        {
681          buffer.append('\'');
682    
683          for (int i=0; i < values.length; i++)
684          {
685            if (i > 0)
686            {
687              buffer.append("', '");
688            }
689    
690            buffer.append(values[i].stringValue());
691          }
692    
693          buffer.append('\'');
694        }
695    
696        buffer.append("})");
697      }
