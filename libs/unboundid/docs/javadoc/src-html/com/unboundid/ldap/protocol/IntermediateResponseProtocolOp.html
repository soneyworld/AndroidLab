021    package com.unboundid.ldap.protocol;
022    
023    
024    
025    import com.unboundid.asn1.ASN1Buffer;
026    import com.unboundid.asn1.ASN1BufferSequence;
027    import com.unboundid.asn1.ASN1OctetString;
028    import com.unboundid.asn1.ASN1StreamReader;
029    import com.unboundid.asn1.ASN1StreamReaderSequence;
030    import com.unboundid.ldap.sdk.LDAPException;
031    import com.unboundid.ldap.sdk.ResultCode;
032    import com.unboundid.util.NotMutable;
033    import com.unboundid.util.InternalUseOnly;
034    import com.unboundid.util.ThreadSafety;
035    import com.unboundid.util.ThreadSafetyLevel;
036    
037    import static com.unboundid.ldap.protocol.ProtocolMessages.*;
038    import static com.unboundid.util.Debug.*;
039    import static com.unboundid.util.StaticUtils.*;
040    
041    
042    
043    /**
044     * This class provides an implementation of an LDAP intermediate response
045     * protocol op.
046     */
047    @InternalUseOnly()
048    @NotMutable()
049    @ThreadSafety(level=ThreadSafetyLevel.COMPLETELY_THREADSAFE)
050    public final class IntermediateResponseProtocolOp
051           implements ProtocolOp
052    {
053      /**
054       * The BER type for the OID element.
055       */
056      public static final byte TYPE_OID = (byte) 0x80;
057    
058    
059    
060      /**
061       * The BER type for the value element.
062       */
063      public static final byte TYPE_VALUE = (byte) 0x81;
064    
065    
066    
067      /**
068       * The serial version UID for this serializable class.
069       */
070      private static final long serialVersionUID = 118549806265654465L;
071    
072    
073    
074      // The value for this intermediate response.
075      private final ASN1OctetString value;
076    
077      // The OID for this intermediate response.
078      private final String oid;
079    
080    
081    
082      /**
083       * Creates a new intermediate response protocol op with the provided
084       * information.
085       *
086       * @param  oid    The OID for this intermediate response, or {@code null} if
087       *                there should not be an OID.
088       * @param  value  The value for this intermediate response, or {@code null} if
089       *                there should not be a value.
090       */
091      public IntermediateResponseProtocolOp(final String oid,
092                                            final ASN1OctetString value)
093      {
094        this.oid = oid;
095    
096        if (value == null)
097        {
098          this.value = null;
099        }
100        else
101        {
102          this.value = new ASN1OctetString(TYPE_VALUE, value.getValue());
103        }
104      }
105    
106    
107    
108      /**
109       * Creates a new intermediate response protocol op read from the provided
110       * ASN.1 stream reader.
111       *
112       * @param  reader  The ASN.1 stream reader from which to read the intermediate
113       *                 response protocol op.
114       *
115       * @throws  LDAPException  If a problem occurs while reading or parsing the
116       *                         intermediate response.
117       */
118      IntermediateResponseProtocolOp(final ASN1StreamReader reader)
119           throws LDAPException
120      {
121        try
122        {
123          final ASN1StreamReaderSequence opSequence = reader.beginSequence();
124    
125          String o = null;
126          ASN1OctetString v = null;
127          while (opSequence.hasMoreElements())
128          {
129            final byte type = (byte) reader.peek();
130            if (type == TYPE_OID)
131            {
132              o = reader.readString();
133            }
134            else if (type == TYPE_VALUE)
135            {
136              v = new ASN1OctetString(type, reader.readBytes());
137            }
138            else
139            {
140              throw new LDAPException(ResultCode.DECODING_ERROR,
141                   ERR_INTERMEDIATE_RESPONSE_INVALID_ELEMENT.get(toHex(type)));
142            }
143          }
144    
145          oid = o;
146          value = v;
147        }
148        catch (LDAPException le)
149        {
150          debugException(le);
151          throw le;
152        }
153        catch (Exception e)
154        {
155          debugException(e);
156    
157          throw new LDAPException(ResultCode.DECODING_ERROR,
158               ERR_INTERMEDIATE_RESPONSE_CANNOT_DECODE.get(getExceptionMessage(e)),
159               e);
160        }
161      }
162    
163    
164    
165      /**
166       * Retrieves the OID for this intermediate response, if any.
167       *
168       * @return  The OID for this intermediate response, or {@code null} if there
169       *          is no response OID.
170       */
171      public String getOID()
172      {
173        return oid;
174      }
175    
176    
177    
178      /**
179       * Retrieves the value for this intermediate response, if any.
180       *
181       * @return  The value for this intermediate response, or {@code null} if there
182       *          is no response value.
183       */
184      public ASN1OctetString getValue()
185      {
186        return value;
187      }
188    
189    
190    
191      /**
192       * {@inheritDoc}
193       */
194      public byte getProtocolOpType()
195      {
196        return LDAPMessage.PROTOCOL_OP_TYPE_INTERMEDIATE_RESPONSE;
197      }
198    
199    
200    
201      /**
202       * {@inheritDoc}
203       */
204      public void writeTo(final ASN1Buffer buffer)
205      {
206        final ASN1BufferSequence opSequence = buffer.beginSequence(
207             LDAPMessage.PROTOCOL_OP_TYPE_INTERMEDIATE_RESPONSE);
208    
209        if (oid != null)
210        {
211          buffer.addOctetString(TYPE_OID, oid);
212        }
213    
214        if (value != null)
215        {
216          buffer.addElement(value);
217        }
218    
219        opSequence.end();
220      }
221    
222    
223    
224      /**
225       * Retrieves a string representation of this protocol op.
226       *
227       * @return  A string representation of this protocol op.
228       */
229      @Override()
230      public String toString()
231      {
232        final StringBuilder buffer = new StringBuilder();
233        toString(buffer);
234        return buffer.toString();
235      }
236    
237    
238    
239      /**
240       * {@inheritDoc}
241       */
242      public void toString(final StringBuilder buffer)
243      {
244        buffer.append("IntermediateResponseProtocolOp(");
245    
246        if (oid != null)
247        {
248          buffer.append("oid='");
249          buffer.append(oid);
250          buffer.append('\'');
251        }
252    
253        buffer.append(')');
254      }
255    }
