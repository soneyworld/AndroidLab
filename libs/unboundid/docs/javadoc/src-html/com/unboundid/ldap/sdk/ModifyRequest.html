056    /**
057     * This class implements the processing necessary to perform an LDAPv3 modify
058     * operation, which can be used to update an entry in the directory server.  A
059     * modify request contains the DN of the entry to modify, as well as one or more
060     * changes to apply to that entry.  See the {@link Modification} class for more
061     * information about the types of modifications that may be processed.
062     * <BR><BR>
063     * A modify request can be created with a DN and set of modifications, but it
064     * can also be as a list of the lines that comprise the LDIF representation of
065     * the modification as described in
066     * <A HREF="http://www.ietf.org/rfc/rfc2849.txt">RFC 2849</A>.  For example, the
067     * following code demonstrates creating a modify request from the LDIF
068     * representation of the modification:
069     * <PRE>
070     *   ModifyRequest modifyRequest = new ModifyRequest(
071     *     "dn: dc=example,dc=com",
072     *     "changetype: modify",
073     *     "replace: description",
074     *     "description: This is the new description.");
075     * </PRE>
076     * <BR><BR>
077     * {@code ModifyRequest} objects are mutable and therefore can be altered and
078     * re-used for multiple requests.  Note, however, that {@code ModifyRequest}
079     * objects are not threadsafe and therefore a single {@code ModifyRequest}
080     * object instance should not be used to process multiple requests at the same
081     * time.
082     */
083    @Mutable()
084    @ThreadSafety(level=ThreadSafetyLevel.NOT_THREADSAFE)
085    public final class ModifyRequest
086           extends UpdatableLDAPRequest
087           implements ReadOnlyModifyRequest, ResponseAcceptor, ProtocolOp

393      /**
394       * Creates a new modify request from the provided LDIF representation of the
395       * changes.
396       *
397       * @param  ldifModificationLines  The lines that comprise an LDIF
398       *                                representation of a modify change record.
399       *                                It must not be {@code null} or empty.
400       *
401       * @throws  LDIFException  If the provided set of lines cannot be parsed as an
402       *                         LDIF modify change record.
403       */
404      public ModifyRequest(final String... ldifModificationLines)
405             throws LDIFException
406      {
407        super(null);
408    
409        final LDIFChangeRecord changeRecord =
410             LDIFReader.decodeChangeRecord(ldifModificationLines);
411        if (! (changeRecord instanceof LDIFModifyChangeRecord))
412        {
413          throw new LDIFException(ERR_MODIFY_INVALID_LDIF.get(), 0, false,
414                                  ldifModificationLines, null);
415        }
416    
417        final LDIFModifyChangeRecord modifyRecord =
418             (LDIFModifyChangeRecord) changeRecord;
419        final ModifyRequest r = modifyRecord.toModifyRequest();
420    
421        dn            = r.dn;
422        modifications = r.modifications;
423      }

573      /**
574       * {@inheritDoc}
575       */
576      public void writeTo(final ASN1Buffer writer)
577      {
578        final ASN1BufferSequence requestSequence =
579             writer.beginSequence(LDAPMessage.PROTOCOL_OP_TYPE_MODIFY_REQUEST);
580        writer.addOctetString(dn);
581    
582        final ASN1BufferSequence modSequence = writer.beginSequence();
583        for (final Modification m : modifications)
584        {
585          m.writeTo(writer);
586        }
587        modSequence.end();
588        requestSequence.end();
589      }

593      /**
594       * Encodes the modify request protocol op to an ASN.1 element.
595       *
596       * @return  The ASN.1 element with the encoded modify request protocol op.
597       */
598      ASN1Element encodeProtocolOp()
599      {
600        final ASN1Element[] modElements = new ASN1Element[modifications.size()];
601        for (int i=0; i < modElements.length; i++)
602        {
603          modElements[i] = modifications.get(i).encode();
604        }
605    
606        final ASN1Element[] protocolOpElements =
607        {
608          new ASN1OctetString(dn),
609          new ASN1Sequence(modElements)
610        };
611    
612    
613    
614        return new ASN1Sequence(LDAPMessage.PROTOCOL_OP_TYPE_MODIFY_REQUEST,
615                                protocolOpElements);
616      }

620      /**
621       * Sends this modify request to the directory server over the provided
622       * connection and returns the associated response.
623       *
624       * @param  connection  The connection to use to communicate with the directory
625       *                     server.
626       * @param  depth       The current referral depth for this request.  It should
627       *                     always be one for the initial request, and should only
628       *                     be incremented when following referrals.
629       *
630       * @return  An LDAP result object that provides information about the result
631       *          of the modify processing.
632       *
633       * @throws  LDAPException  If a problem occurs while sending the request or
634       *                         reading the response.
635       */
636      @Override()
637      protected LDAPResult process(final LDAPConnection connection, final int depth)
638                throws LDAPException
639      {
640        if (connection.synchronousMode())
641        {
642          return processSync(connection, depth);
643        }
644    
645        final long requestTime = System.nanoTime();
646        processAsync(connection, null);
647    
648        try
649        {
650          // Wait for and process the response.
651          final LDAPResponse response;
652          try
653          {
654            final long responseTimeout = getResponseTimeoutMillis(connection);
655            if (responseTimeout > 0)
656            {
657              response = responseQueue.poll(responseTimeout, TimeUnit.MILLISECONDS);
658            }
659            else
660            {
661              response = responseQueue.take();
662            }
663          }
664          catch (InterruptedException ie)
665          {
666            debugException(ie);
667            throw new LDAPException(ResultCode.LOCAL_ERROR,
668                 ERR_MODIFY_INTERRUPTED.get(connection.getHostPort()), ie);
669          }
670    
671          return handleResponse(connection, response, requestTime, depth);
672        }
673        finally
674        {
675          connection.deregisterResponseAcceptor(messageID);
676        }
677      }

681      /**
682       * Sends this modify request to the directory server over the provided
683       * connection and returns the message ID for the request.
684       *
685       * @param  connection      The connection to use to communicate with the
686       *                         directory server.
687       * @param  resultListener  The async result listener that is to be notified
688       *                         when the response is received.  It may be
689       *                         {@code null} only if the result is to be processed
690       *                         by this class.
691       *
692       * @return  The LDAP message ID for the modify request that was sent to the
693       *          server.
694       *
695       * @throws  LDAPException  If a problem occurs while sending the request.
696       */
697      int processAsync(final LDAPConnection connection,
698                       final AsyncResultListener resultListener)
699          throws LDAPException
700      {
701        // Create the LDAP message.
702        messageID = connection.nextMessageID();
703        final LDAPMessage message = new LDAPMessage(messageID, this, getControls());
704    
705    
706        // If the provided async result listener is {@code null}, then we'll use
707        // this class as the message acceptor.  Otherwise, create an async helper
708        // and use it as the message acceptor.
709        if (resultListener == null)
710        {
711          connection.registerResponseAcceptor(messageID, this);
712        }
713        else
714        {
715          final AsyncHelper helper = new AsyncHelper(connection,
716               LDAPMessage.PROTOCOL_OP_TYPE_MODIFY_RESPONSE, resultListener,
717               getIntermediateResponseListener());
718          connection.registerResponseAcceptor(messageID, helper);
719        }
720    
721    
722        // Send the request to the server.
723        try
724        {
725          debugLDAPRequest(this);
726          connection.getConnectionStatistics().incrementNumModifyRequests();
727          connection.sendMessage(message);
728          return messageID;
729        }
730        catch (LDAPException le)
731        {
732          debugException(le);
733    
734          connection.deregisterResponseAcceptor(messageID);
735          throw le;
736        }
737      }

741      /**
742       * Processes this modify operation in synchronous mode, in which the same
743       * thread will send the request and read the response.
744       *
745       * @param  connection  The connection to use to communicate with the directory
746       *                     server.
747       * @param  depth       The current referral depth for this request.  It should
748       *                     always be one for the initial request, and should only
749       *                     be incremented when following referrals.
750       *
751       * @return  An LDAP result object that provides information about the result
752       *          of the modify processing.
753       *
754       * @throws  LDAPException  If a problem occurs while sending the request or
755       *                         reading the response.
756       */
757      private LDAPResult processSync(final LDAPConnection connection,
758                                     final int depth)
759              throws LDAPException
760      {
761        // Create the LDAP message.
762        messageID = connection.nextMessageID();
763        final LDAPMessage message =
764             new LDAPMessage(messageID,  this, getControls());
765    
766    
767        // Set the appropriate timeout on the socket.
768        try
769        {
770          connection.getConnectionInternals().getSocket().setSoTimeout(
771               (int) getResponseTimeoutMillis(connection));
772        }
773        catch (Exception e)
774        {
775          debugException(e);
776        }
777    
778    
779        // Send the request to the server.
780        final long requestTime = System.nanoTime();
781        debugLDAPRequest(this);
782        connection.getConnectionStatistics().incrementNumModifyRequests();
783        connection.sendMessage(message);
784    
785        final LDAPResponse response = connection.readResponse(messageID);
786        return handleResponse(connection, response, requestTime, depth);
787      }

791      /**
792       * Performs the necessary processing for handling a response.
793       *
794       * @param  connection   The connection used to read the response.
795       * @param  response     The response to be processed.
796       * @param  requestTime  The time the request was sent to the server.
797       * @param  depth        The current referral depth for this request.  It
798       *                      should always be one for the initial request, and
799       *                      should only be incremented when following referrals.
800       *
801       * @return  The modify result.
802       *
803       * @throws  LDAPException  If a problem occurs.
804       */
805      private LDAPResult handleResponse(final LDAPConnection connection,
806                                        final LDAPResponse response,
807                                        final long requestTime, final int depth)
808              throws LDAPException
809      {
810        if (response == null)
811        {
812          final long waitTime = nanosToMillis(System.nanoTime() - requestTime);
813          throw new LDAPException(ResultCode.TIMEOUT,
814               ERR_MODIFY_CLIENT_TIMEOUT.get(waitTime, connection.getHostPort()));
815        }
816    
817        connection.getConnectionStatistics().incrementNumModifyResponses(
818             System.nanoTime() - requestTime);
819        if (response instanceof ConnectionClosedResponse)
820        {
821          // The connection was closed while waiting for the response.
822          final ConnectionClosedResponse ccr = (ConnectionClosedResponse) response;
823          final String message = ccr.getMessage();
824          if (message == null)
825          {
826            throw new LDAPException(ccr.getResultCode(),
827                 ERR_CONN_CLOSED_WAITING_FOR_MODIFY_RESPONSE.get(
828                      connection.getHostPort(), toString()));
829          }
830          else
831          {
832            throw new LDAPException(ccr.getResultCode(),
833                 ERR_CONN_CLOSED_WAITING_FOR_MODIFY_RESPONSE_WITH_MESSAGE.get(
834                      connection.getHostPort(), toString(), message));
835          }
836        }
837    
838        final LDAPResult result = (LDAPResult) response;
839        if ((result.getResultCode().equals(ResultCode.REFERRAL)) &&
840            followReferrals(connection))
841        {
842          if (depth >= connection.getConnectionOptions().getReferralHopLimit())
843          {
844            return new LDAPResult(messageID, ResultCode.REFERRAL_LIMIT_EXCEEDED,
845                                  ERR_TOO_MANY_REFERRALS.get(),
846                                  result.getMatchedDN(), result.getReferralURLs(),
847                                  result.getResponseControls());
848          }
849    
850          return followReferral(result, connection, depth);
851        }
852        else
853        {
854          return result;
855        }
856      }

860      /**
861       * Attempts to follow a referral to perform a modify operation in the target
862       * server.
863       *
864       * @param  referralResult  The LDAP result object containing information about
865       *                         the referral to follow.
866       * @param  connection      The connection on which the referral was received.
867       * @param  depth           The number of referrals followed in the course of
868       *                         processing this request.
869       *
870       * @return  The result of attempting to process the modify operation by
871       *          following the referral.
872       *
873       * @throws  LDAPException  If a problem occurs while attempting to establish
874       *                         the referral connection, sending the request, or
875       *                         reading the result.
876       */
877      private LDAPResult followReferral(final LDAPResult referralResult,
878                                        final LDAPConnection connection,
879                                        final int depth)
880              throws LDAPException
881      {
882        for (final String urlString : referralResult.getReferralURLs())
883        {
884          try
885          {
886            final LDAPURL referralURL = new LDAPURL(urlString);
887            final String host = referralURL.getHost();
888    
889            if (host == null)
890            {
891              // We can't handle a referral in which there is no host.
892              continue;
893            }
894    
895            final ModifyRequest modifyRequest;
896            if (referralURL.baseDNProvided())
897            {
898              modifyRequest = new ModifyRequest(referralURL.getBaseDN(),
899                                                modifications, getControls());
900            }
901            else
902            {
903              modifyRequest = this;
904            }
905    
906            final LDAPConnection referralConn =
907                 connection.getReferralConnection(referralURL, connection);
908            try
909            {
910              return modifyRequest.process(referralConn, depth+1);
911            }
912            finally
913            {
914              referralConn.setDisconnectInfo(DisconnectType.REFERRAL, null, null);
915              referralConn.close();
916            }
917          }
918          catch (LDAPException le)
919          {
920            debugException(le);
921          }
922        }
923    
924        // If we've gotten here, then we could not follow any of the referral URLs,
925        // so we'll just return the original referral result.
926        return referralResult;
927      }

1023      /**
1024       * {@inheritDoc}
1025       */
1026      @Override()
1027      public void toString(final StringBuilder buffer)
1028      {
1029        buffer.append("ModifyRequest(dn='");
1030        buffer.append(dn);
1031        buffer.append("', mods={");
1032        for (int i=0; i < modifications.size(); i++)
1033        {
1034          final Modification m = modifications.get(i);
1035    
1036          if (i > 0)
1037          {
1038            buffer.append(", ");
1039          }
1040    
1041          switch (m.getModificationType().intValue())
1042          {
1043            case 0:
1044              buffer.append("ADD ");
1045              break;
1046    
1047            case 1:
1048              buffer.append("DELETE ");
1049              break;
1050    
1051            case 2:
1052              buffer.append("REPLACE ");
1053              break;
1054    
1055            case 3:
1056              buffer.append("INCREMENT ");
1057              break;
1058          }
1059    
1060          buffer.append(m.getAttributeName());
1061        }
1062        buffer.append('}');
1063    
1064        final Control[] controls = getControls();
1065        if (controls.length > 0)
1066        {
1067          buffer.append(", controls={");
1068          for (int i=0; i < controls.length; i++)
1069          {
1070            if (i > 0)
1071            {
1072              buffer.append(", ");
1073            }
1074    
1075            buffer.append(controls[i]);
1076          }
1077          buffer.append('}');
1078        }
1079    
1080        buffer.append(')');
1081      }
