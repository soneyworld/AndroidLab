048    /**
049     * This class implements the processing necessary to perform an LDAPv3 compare
050     * operation, which may be used to determine whether a specified entry contains
051     * a given attribute value.  Compare requests include the DN of the target
052     * entry, the name of the target attribute, and the value for which to make the
053     * determination.  It may also include a set of controls to send to the server.
054     * <BR><BR>
055     * The assertion value may be specified as either a string or a byte array.  If
056     * it is specified as a byte array, then it may represent either a binary or a
057     * string value.  If a string value is provided as a byte array, then it should
058     * use the UTF-8 encoding for that value.
059     * <BR><BR>
060     * {@code CompareRequest} objects are mutable and therefore can be altered and
061     * re-used for multiple requests.  Note, however, that {@code CompareRequest}
062     * objects are not threadsafe and therefore a single {@code CompareRequest}
063     * object instance should not be used to process multiple requests at the same
064     * time.
065     * <BR><BR>
066     * <H2>Example</H2>
067     * The following example demonstrates the process for performing a compare
068     * operation:
069     * <PRE>
070     *   CompareRequest compareRequest =
071     *        new CompareRequest("dc=example,dc=com", "description", "test");
072     *
073     *   try
074     *   {
075     *     CompareResult compareResult = connection.compare(compareRequest);
076     *
077     *     // The compare operation didn't throw an exception, so we can try to
078     *     // determine whether the compare matched.
079     *     if (compareResult.compareMatched())
080     *     {
081     *       System.out.println("The entry does have a description value of test");
082     *     }
083     *     else
084     *     {
085     *       System.out.println("The entry does not have a description value of " +
086     *                          "test");
087     *     }
088     *   }
089     *   catch (LDAPException le)
090     *   {
091     *     System.err.println("The compare operation failed.");
092     *   }
093     * </PRE>
094     */
095    @Mutable()
096    @ThreadSafety(level=ThreadSafetyLevel.NOT_THREADSAFE)
097    public final class CompareRequest
098           extends UpdatableLDAPRequest
099           implements ReadOnlyCompareRequest, ResponseAcceptor, ProtocolOp

550      /**
551       * Sends this delete request to the directory server over the provided
552       * connection and returns the associated response.
553       *
554       * @param  connection  The connection to use to communicate with the directory
555       *                     server.
556       * @param  depth       The current referral depth for this request.  It should
557       *                     always be one for the initial request, and should only
558       *                     be incremented when following referrals.
559       *
560       * @return  An LDAP result object that provides information about the result
561       *          of the delete processing.
562       *
563       * @throws  LDAPException  If a problem occurs while sending the request or
564       *                         reading the response.
565       */
566      @Override()
567      protected CompareResult process(final LDAPConnection connection,
568                                      final int depth)
569                throws LDAPException
570      {
571        if (connection.synchronousMode())
572        {
573          return processSync(connection, depth);
574        }
575    
576        final long requestTime = System.nanoTime();
577        processAsync(connection, null);
578    
579        try
580        {
581          // Wait for and process the response.
582          final LDAPResponse response;
583          try
584          {
585            final long responseTimeout = getResponseTimeoutMillis(connection);
586            if (responseTimeout > 0)
587            {
588              response = responseQueue.poll(responseTimeout, TimeUnit.MILLISECONDS);
589            }
590            else
591            {
592              response = responseQueue.take();
593            }
594          }
595          catch (InterruptedException ie)
596          {
597            debugException(ie);
598            throw new LDAPException(ResultCode.LOCAL_ERROR,
599                 ERR_COMPARE_INTERRUPTED.get(connection.getHostPort()), ie);
600          }
601    
602          return handleResponse(connection, response,  requestTime, depth);
603        }
604        finally
605        {
606          connection.deregisterResponseAcceptor(messageID);
607        }
608      }

612      /**
613       * Sends this compare request to the directory server over the provided
614       * connection and returns the message ID for the request.
615       *
616       * @param  connection      The connection to use to communicate with the
617       *                         directory server.
618       * @param  resultListener  The async result listener that is to be notified
619       *                         when the response is received.  It may be
620       *                         {@code null} only if the result is to be processed
621       *                         by this class.
622       *
623       * @return  The LDAP message ID for the compare request that was sent to the
624       *          server.
625       *
626       * @throws  LDAPException  If a problem occurs while sending the request.
627       */
628      int processAsync(final LDAPConnection connection,
629                       final AsyncCompareResultListener resultListener)
630          throws LDAPException
631      {
632        // Create the LDAP message.
633        messageID = connection.nextMessageID();
634        final LDAPMessage message = new LDAPMessage(messageID, this, getControls());
635    
636    
637        // If the provided async result listener is {@code null}, then we'll use
638        // this class as the message acceptor.  Otherwise, create an async helper
639        // and use it as the message acceptor.
640        if (resultListener == null)
641        {
642          connection.registerResponseAcceptor(messageID, this);
643        }
644        else
645        {
646          final AsyncCompareHelper helper = new AsyncCompareHelper(connection,
647               resultListener, getIntermediateResponseListener());
648          connection.registerResponseAcceptor(messageID, helper);
649        }
650    
651    
652        // Send the request to the server.
653        try
654        {
655          debugLDAPRequest(this);
656          connection.getConnectionStatistics().incrementNumCompareRequests();
657          connection.sendMessage(message);
658          return messageID;
659        }
660        catch (LDAPException le)
661        {
662          debugException(le);
663    
664          connection.deregisterResponseAcceptor(messageID);
665          throw le;
666        }
667      }

671      /**
672       * Processes this compare operation in synchronous mode, in which the same
673       * thread will send the request and read the response.
674       *
675       * @param  connection  The connection to use to communicate with the directory
676       *                     server.
677       * @param  depth       The current referral depth for this request.  It should
678       *                     always be one for the initial request, and should only
679       *                     be incremented when following referrals.
680       *
681       * @return  An LDAP result object that provides information about the result
682       *          of the compare processing.
683       *
684       * @throws  LDAPException  If a problem occurs while sending the request or
685       *                         reading the response.
686       */
687      private CompareResult processSync(final LDAPConnection connection,
688                                        final int depth)
689              throws LDAPException
690      {
691        // Create the LDAP message.
692        messageID = connection.nextMessageID();
693        final LDAPMessage message =
694             new LDAPMessage(messageID,  this, getControls());
695    
696    
697        // Set the appropriate timeout on the socket.
698        try
699        {
700          connection.getConnectionInternals().getSocket().setSoTimeout(
701               (int) getResponseTimeoutMillis(connection));
702        }
703        catch (Exception e)
704        {
705          debugException(e);
706        }
707    
708    
709        // Send the request to the server.
710        final long requestTime = System.nanoTime();
711        debugLDAPRequest(this);
712        connection.getConnectionStatistics().incrementNumCompareRequests();
713        connection.sendMessage(message);
714    
715        final LDAPResponse response = connection.readResponse(messageID);
716        return handleResponse(connection, response, requestTime, depth);
717      }

721      /**
722       * Performs the necessary processing for handling a response.
723       *
724       * @param  connection   The connection used to read the response.
725       * @param  response     The response to be processed.
726       * @param  requestTime  The time the request was sent to the server.
727       * @param  depth        The current referral depth for this request.  It
728       *                      should always be one for the initial request, and
729       *                      should only be incremented when following referrals.
730       *
731       * @return  The compare result.
732       *
733       * @throws  LDAPException  If a problem occurs.
734       */
735      private CompareResult handleResponse(final LDAPConnection connection,
736                                           final LDAPResponse response,
737                                           final long requestTime, final int depth)
738              throws LDAPException
739      {
740        if (response == null)
741        {
742          final long waitTime = nanosToMillis(System.nanoTime() - requestTime);
743          throw new LDAPException(ResultCode.TIMEOUT,
744               ERR_COMPARE_CLIENT_TIMEOUT.get(waitTime, connection.getHostPort()));
745        }
746    
747        connection.getConnectionStatistics().incrementNumCompareResponses(
748             System.nanoTime() - requestTime);
749        if (response instanceof ConnectionClosedResponse)
750        {
751          // The connection was closed while waiting for the response.
752          final ConnectionClosedResponse ccr = (ConnectionClosedResponse) response;
753          final String message = ccr.getMessage();
754          if (message == null)
755          {
756            throw new LDAPException(ccr.getResultCode(),
757                 ERR_CONN_CLOSED_WAITING_FOR_COMPARE_RESPONSE.get(
758                      connection.getHostPort(), toString()));
759          }
760          else
761          {
762            throw new LDAPException(ccr.getResultCode(),
763                 ERR_CONN_CLOSED_WAITING_FOR_COMPARE_RESPONSE_WITH_MESSAGE.get(
764                      connection.getHostPort(), toString(), message));
765          }
766        }
767    
768        final CompareResult result = (CompareResult) response;
769        if ((result.getResultCode().equals(ResultCode.REFERRAL)) &&
770            followReferrals(connection))
771        {
772          if (depth >= connection.getConnectionOptions().getReferralHopLimit())
773          {
774            return new CompareResult(messageID,
775                                     ResultCode.REFERRAL_LIMIT_EXCEEDED,
776                                     ERR_TOO_MANY_REFERRALS.get(),
777                                     result.getMatchedDN(),
778                                     result.getReferralURLs(),
779                                     result.getResponseControls());
780          }
781    
782          return followReferral(result, connection, depth);
783        }
784        else
785        {
786          return result;
787        }
788      }

792      /**
793       * Attempts to follow a referral to perform a compare operation in the target
794       * server.
795       *
796       * @param  referralResult  The LDAP result object containing information about
797       *                         the referral to follow.
798       * @param  connection      The connection on which the referral was received.
799       * @param  depth           The number of referrals followed in the course of
800       *                         processing this request.
801       *
802       * @return  The result of attempting to process the compare operation by
803       *          following the referral.
804       *
805       * @throws  LDAPException  If a problem occurs while attempting to establish
806       *                         the referral connection, sending the request, or
807       *                         reading the result.
808       */
809      private CompareResult followReferral(final CompareResult referralResult,
810                                           final LDAPConnection connection,
811                                           final int depth)
812              throws LDAPException
813      {
814        for (final String urlString : referralResult.getReferralURLs())
815        {
816          try
817          {
818            final LDAPURL referralURL = new LDAPURL(urlString);
819            final String host = referralURL.getHost();
820    
821            if (host == null)
822            {
823              // We can't handle a referral in which there is no host.
824              continue;
825            }
826    
827            final CompareRequest compareRequest;
828            if (referralURL.baseDNProvided())
829            {
830              compareRequest = new CompareRequest(referralURL.getBaseDN(),
831                                                  attributeName, assertionValue,
832                                                  getControls());
833            }
834            else
835            {
836              compareRequest = this;
837            }
838    
839            final LDAPConnection referralConn =
840                 connection.getReferralConnection(referralURL, connection);
841            try
842            {
843              return compareRequest.process(referralConn, depth+1);
844            }
845            finally
846            {
847              referralConn.setDisconnectInfo(DisconnectType.REFERRAL, null, null);
848              referralConn.close();
849            }
850          }
851          catch (LDAPException le)
852          {
853            debugException(le);
854          }
855        }
856    
857        // If we've gotten here, then we could not follow any of the referral URLs,
858        // so we'll just return the original referral result.
859        return referralResult;
860      }

909      public CompareRequest duplicate(final Control[] controls)
910      {
911        final CompareRequest r = new CompareRequest(dn, attributeName,
912             assertionValue.getValue(), controls);
913    
914        if (followReferralsInternal() != null)
915        {
916          r.setFollowReferrals(followReferralsInternal());
917        }
918    
919        r.setResponseTimeoutMillis(getResponseTimeoutMillis(null));
920    
921        return r;
922      }
