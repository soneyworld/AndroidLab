060    /**
061     * This class provides a tool that can be used to perform repeated modifications
062     * in an LDAP directory server using multiple threads.  It can help provide an
063     * estimate of the modify performance that a directory server is able to
064     * achieve.  The target entry DN may be a value pattern as described in the
065     * {@link ValuePattern} class.  This makes it possible to modify a range of
066     * entries rather than repeatedly updating the same entry.
067     * <BR><BR>
068     * Some of the APIs demonstrated by this example include:
069     * <UL>
070     *   <LI>Argument Parsing (from the {@code com.unboundid.util.args}
071     *       package)</LI>
072     *   <LI>LDAP Command-Line Tool (from the {@code com.unboundid.util}
073     *       package)</LI>
074     *   <LI>LDAP Communication (from the {@code com.unboundid.ldap.sdk}
075     *       package)</LI>
076     *   <LI>Value Patterns (from the {@code com.unboundid.util} package)</LI>
077     * </UL>
078     * <BR><BR>
079     * All of the necessary information is provided using command line arguments.
080     * Supported arguments include those allowed by the {@link LDAPCommandLineTool}
081     * class, as well as the following additional arguments:
082     * <UL>
083     *   <LI>"-b {entryDN}" or "--targetDN {baseDN}" -- specifies the DN of the
084     *       entry to be modified.  This must be provided.  It may be a simple DN,
085     *       or it may be a value pattern to express a range of entry DNs.</LI>
086     *   <LI>"-A {name}" or "--attribute {name}" -- specifies the name of the
087     *       attribute to modify.  Multiple attributes may be modified by providing
088     *       multiple instances of this argument.  At least one attribute must be
089     *       provided.</LI>
090     *   <LI>"-l {num}" or "--valueLength {num}" -- specifies the length in bytes to
091     *       use for the values of the target attributes.  If this is not provided,
092     *       then a default length of 10 bytes will be used.</LI>
093     *   <LI>"-C {chars}" or "--characterSet {chars}" -- specifies the set of
094     *       characters that will be used to generate the values to use for the
095     *       target attributes.  It should only include ASCII characters.  Values
096     *       will be generated from randomly-selected characters from this set.  If
097     *       this is not provided, then a default set of lowercase alphabetic
098     *       characters will be used.</LI>
099     *   <LI>"-t {num}" or "--numThreads {num}" -- specifies the number of
100     *       concurrent threads to use when performing the modifications.  If this
101     *       is not provided, then a default of one thread will be used.</LI>
102     *   <LI>"-i {sec}" or "--intervalDuration {sec}" -- specifies the length of
103     *       time in seconds between lines out output.  If this is not provided,
104     *       then a default interval duration of five seconds will be used.</LI>
105     *   <LI>"-I {num}" or "--numIntervals {num}" -- specifies the maximum number of
106     *       intervals for which to run.  If this is not provided, then it will
107     *       run forever.</LI>
108     *   <LI>"-r {modifies-per-second}" or "--ratePerSecond {modifies-per-second}"
109     *       -- specifies the target number of modifies to perform per second.  It
110     *       is still necessary to specify a sufficient number of threads for
111     *       achieving this rate.  If this option is not provided, then the tool
112     *       will run at the maximum rate for the specified number of threads.</LI>
113     *   <LI>"--warmUpIntervals {num}" -- specifies the number of intervals to
114     *       complete before beginning overall statistics collection.</LI>
115     *   <LI>"--timestampFormat {format}" -- specifies the format to use for
116     *       timestamps included before each output line.  The format may be one of
117     *       "none" (for no timestamps), "with-date" (to include both the date and
118     *       the time), or "without-date" (to include only time time).</LI>
119     *   <LI>"-Y {authzID}" or "--proxyAs {authzID}" -- Use the proxied
120     *       authorization v2 control to request that the operation be processed
121     *       using an alternate authorization identity.  In this case, the bind DN
122     *       should be that of a user that has permission to use this control.  The
123     *       authorization identity may be a value pattern.</LI>
124     *   <LI>"--suppressErrorResultCodes" -- Indicates that information about the
125     *       result codes for failed operations should not be displayed.</LI>
126     *   <LI>"-c" or "--csv" -- Generate output in CSV format rather than a
127     *       display-friendly format.</LI>
128     * </UL>
129     */

431      /**
432       * Performs the actual processing for this tool.  In this case, it gets a
433       * connection to the directory server and uses it to perform the requested
434       * modifications.
435       *
436       * @return  The result code for the processing that was performed.
437       */
438      @Override()
439      public ResultCode doToolProcessing()
440      {
441        // Create the value patterns for the target entry DN and proxied
442        // authorization identities.
443        final ValuePattern dnPattern;
444        try
445        {
446          dnPattern = new ValuePattern(entryDN.getValue());
447        }
448        catch (ParseException pe)
449        {
450          err("Unable to parse the entry DN value pattern:  ", pe.getMessage());
451          return ResultCode.PARAM_ERROR;
452        }
453    
454        final ValuePattern authzIDPattern;
455        if (proxyAs.isPresent())
456        {
457          try
458          {
459            authzIDPattern = new ValuePattern(proxyAs.getValue());
460          }
461          catch (ParseException pe)
462          {
463            err("Unable to parse the proxied authorization pattern:  ",
464                pe.getMessage());
465            return ResultCode.PARAM_ERROR;
466          }
467        }
468        else
469        {
470          authzIDPattern = null;
471        }
472    
473    
474        // Get the names of the attributes to modify.
475        final String[] attrs = new String[attribute.getValues().size()];
476        attribute.getValues().toArray(attrs);
477    
478    
479        // Get the character set as a byte array.
480        final byte[] charSet = getBytes(characterSet.getValue());
481    
482    
483        // If the --ratePerSecond option was specified, then limit the rate
484        // accordingly.
485        FixedRateBarrier fixedRateBarrier = null;
486        if (ratePerSecond.isPresent())
487        {
488          final int intervalSeconds = collectionInterval.getValue();
489          final int ratePerInterval = ratePerSecond.getValue() * intervalSeconds;
490    
491          fixedRateBarrier =
492               new FixedRateBarrier(1000L * intervalSeconds, ratePerInterval);
493        }
494    
495    
496        // Determine whether to include timestamps in the output and if so what
497        // format should be used for them.
498        final boolean includeTimestamp;
499        final String timeFormat;
500        if (timestampFormat.getValue().equalsIgnoreCase("with-date"))
501        {
502          includeTimestamp = true;
503          timeFormat       = "dd/MM/yyyy HH:mm:ss";
504        }
505        else if (timestampFormat.getValue().equalsIgnoreCase("without-date"))
506        {
507          includeTimestamp = true;
508          timeFormat       = "HH:mm:ss";
509        }
510        else
511        {
512          includeTimestamp = false;
513          timeFormat       = null;
514        }
515    
516    
517        // Determine whether any warm-up intervals should be run.
518        final long totalIntervals;
519        final boolean warmUp;
520        int remainingWarmUpIntervals = warmUpIntervals.getValue();
521        if (remainingWarmUpIntervals > 0)
522        {
523          warmUp = true;
524          totalIntervals = 0L + numIntervals.getValue() + remainingWarmUpIntervals;
525        }
526        else
527        {
528          warmUp = true;
529          totalIntervals = 0L + numIntervals.getValue();
530        }
531    
532    
533        // Create the table that will be used to format the output.
534        final OutputFormat outputFormat;
535        if (csvFormat.isPresent())
536        {
537          outputFormat = OutputFormat.CSV;
538        }
539        else
540        {
541          outputFormat = OutputFormat.COLUMNS;
542        }
543    
544        final ColumnFormatter formatter = new ColumnFormatter(includeTimestamp,
545             timeFormat, outputFormat, " ",
546             new FormattableColumn(12, HorizontalAlignment.RIGHT, "Recent",
547                      "Mods/Sec"),
548             new FormattableColumn(12, HorizontalAlignment.RIGHT, "Recent",
549                      "Avg Dur ms"),
550             new FormattableColumn(12, HorizontalAlignment.RIGHT, "Recent",
551                      "Errors/Sec"),
552             new FormattableColumn(12, HorizontalAlignment.RIGHT, "Overall",
553                      "Mods/Sec"),
554             new FormattableColumn(12, HorizontalAlignment.RIGHT, "Overall",
555                      "Avg Dur ms"));
556    
557    
558        // Create values to use for statistics collection.
559        final AtomicLong        modCounter   = new AtomicLong(0L);
560        final AtomicLong        errorCounter = new AtomicLong(0L);
561        final AtomicLong        modDurations = new AtomicLong(0L);
562        final ResultCodeCounter rcCounter    = new ResultCodeCounter();
563    
564    
565        // Determine the length of each interval in milliseconds.
566        final long intervalMillis = 1000L * collectionInterval.getValue();
567    
568    
569        // Create a random number generator to use for seeding the per-thread
570        // generators.
571        final Random random = new Random();
572    
573    
574        // Create the threads to use for the modifications.
575        final CyclicBarrier barrier = new CyclicBarrier(numThreads.getValue() + 1);
576        final ModRateThread[] threads = new ModRateThread[numThreads.getValue()];
577        for (int i=0; i < threads.length; i++)
578        {
579          final LDAPConnection connection;
580          try
581          {
582            connection = getConnection();
583          }
584          catch (LDAPException le)
585          {
586            err("Unable to connect to the directory server:  ",
587                getExceptionMessage(le));
588            return le.getResultCode();
589          }
590    
591          threads[i] = new ModRateThread(i, connection, dnPattern, attrs, charSet,
592               valueLength.getValue(), authzIDPattern, random.nextLong(), barrier,
593               modCounter, modDurations, errorCounter, rcCounter, fixedRateBarrier);
594          threads[i].start();
595        }
596    
597    
598        // Display the table header.
599        for (final String headerLine : formatter.getHeaderLines(true))
600        {
601          out(headerLine);
602        }
603    
604    
605        // Indicate that the threads can start running.
606        try
607        {
608          barrier.await();
609        } catch (Exception e) {}
610        long overallStartTime = System.nanoTime();
611        long nextIntervalStartTime = System.currentTimeMillis() + intervalMillis;
612    
613    
614        boolean setOverallStartTime = false;
615        long    lastDuration        = 0L;
616        long    lastNumErrors       = 0L;
617        long    lastNumMods         = 0L;
618        long    lastEndTime         = System.nanoTime();
619        for (long i=0; i < totalIntervals; i++)
620        {
621          final long startTimeMillis = System.currentTimeMillis();
622          final long sleepTimeMillis = nextIntervalStartTime - startTimeMillis;
623          nextIntervalStartTime += intervalMillis;
624          try
625          {
626            if (sleepTimeMillis > 0)
627            {
628              Thread.sleep(sleepTimeMillis);
629            }
630          } catch (Exception e) {}
631    
632          final long endTime          = System.nanoTime();
633          final long intervalDuration = endTime - lastEndTime;
634    
635          final long numMods;
636          final long numErrors;
637          final long totalDuration;
638          if (warmUp && (remainingWarmUpIntervals > 0))
639          {
640            numMods       = modCounter.getAndSet(0L);
641            numErrors     = errorCounter.getAndSet(0L);
642            totalDuration = modDurations.getAndSet(0L);
643          }
644          else
645          {
646            numMods       = modCounter.get();
647            numErrors     = errorCounter.get();
648            totalDuration = modDurations.get();
649          }
650    
651          final long recentNumMods = numMods - lastNumMods;
652          final long recentNumErrors = numErrors - lastNumErrors;
653          final long recentDuration = totalDuration - lastDuration;
654    
655          final double numSeconds = intervalDuration / 1000000000.0D;
656          final double recentModRate = recentNumMods / numSeconds;
657          final double recentErrorRate  = recentNumErrors / numSeconds;
658          final double recentAvgDuration =
659                      1.0D * recentDuration / recentNumMods / 1000000;
660    
661          if (warmUp && (remainingWarmUpIntervals > 0))
662          {
663            out(formatter.formatRow(recentModRate, recentAvgDuration,
664                 recentErrorRate, "warming up", "warming up"));
665    
666            remainingWarmUpIntervals--;
667            if (remainingWarmUpIntervals == 0)
668            {
669              out("Warm-up completed.  Beginning overall statistics collection.");
670              setOverallStartTime = true;
671            }
672          }
673          else
674          {
675            if (setOverallStartTime)
676            {
677              overallStartTime    = lastEndTime;
678              setOverallStartTime = false;
679            }
680    
681            final double numOverallSeconds =
682                 (endTime - overallStartTime) / 1000000000.0D;
683            final double overallAuthRate = numMods / numOverallSeconds;
684            final double overallAvgDuration =
685                 1.0D * totalDuration / numMods / 1000000;
686    
687            out(formatter.formatRow(recentModRate, recentAvgDuration,
688                 recentErrorRate, overallAuthRate, overallAvgDuration));
689    
690            lastNumMods     = numMods;
691            lastNumErrors   = numErrors;
692            lastDuration    = totalDuration;
693          }
694    
695          final List<ObjectPair<ResultCode,Long>> rcCounts =
696               rcCounter.getCounts(true);
697          if ((! suppressErrorsArgument.isPresent()) && (! rcCounts.isEmpty()))
698          {
699            err("\tError Results:");
700            for (final ObjectPair<ResultCode,Long> p : rcCounts)
701            {
702              err("\t", p.getFirst().getName(), ":  ", p.getSecond());
703            }
704          }
705    
706          lastEndTime = endTime;
707        }
708    
709    
710        // Stop all of the threads.
711        ResultCode resultCode = ResultCode.SUCCESS;
712        for (final ModRateThread t : threads)
713        {
714          final ResultCode r = t.stopRunning();
715          if (resultCode == ResultCode.SUCCESS)
716          {
717            resultCode = r;
718          }
719        }
720    
721        return resultCode;
722      }
