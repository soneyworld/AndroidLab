053    /**
054     * This class provides a mechanism for validating entries against a schema.  It
055     * provides the ability to customize the types of validation to perform, and can
056     * collect information about the entries that fail validation to provide a
057     * summary of the problems encountered.
058     * <BR><BR>
059     * The types of validation that may be performed for each entry include:
060     * <UL>
061     *   <LI>Ensure that the entry has a valid DN.</LI>
062     *   <LI>Ensure that the entry has exactly one structural object class.</LI>
063     *   <LI>Ensure that all of the object classes for the entry are defined in the
064     *       schema.</LI>
065     *   <LI>Ensure that all of the auxiliary classes for the entry are allowed by
066     *       the DIT content rule for the entry's structural object class (if such a
067     *        DIT content rule is defined).</LI>
068     *   <LI>Ensure that all attributes contained in the entry are defined in the
069     *       schema.</LI>
070     *   <LI>Ensure that all attributes required by the entry's object classes or
071     *       DIT content rule (if defined) are present in the entry.</LI>
072     *   <LI>Ensure that all of the user attributes contained in the entry are
073     *       allowed by the entry's object classes or DIT content rule (if
074     *       defined).</LI>
075     *   <LI>Ensure that all attribute values conform to the requirements of the
076     *       associated attribute syntax.</LI>
077     *   <LI>Ensure that all attributes with multiple values are defined as
078     *       multi-valued in the associated schema.</LI>
079     *   <LI>If there is a name form associated with the entry's structural object
080     *       class, then ensure that the entry's RDN satisfies its constraints.</LI>
081     * </UL>
082     * All of these forms of validation will be performed by default, but individual
083     * types of validation may be enabled or disabled.
084     * <BR><BR>
085     * This class will not make any attempt to validate compliance with DIT
086     * structure rules, nor will it check the OBSOLETE field for any of the schema
087     * elements.  In addition, attempts to validate whether attribute values
088     * conform to the syntax for the associated attribute type may only be
089     * completely accurate for syntaxes supported by the LDAP SDK.
090     * <BR><BR>
091     * This class is largely threadsafe, and the {@link EntryValidator#entryIsValid}
092     * is designed so that it can be invoked concurrently by multiple threads.
093     * Note, however, that it is not recommended that the any of the other methods
094     * in this class be used while any threads are running the {@code entryIsValid}
095     * method because changing the configuration or attempting to retrieve retrieve
096     * information may yield inaccurate or inconsistent results.
097     */

568      /**
569       * Indicates whether the provided entry passes all of the enabled types of
570       * validation.
571       *
572       * @param  entry           The entry to be examined.   It must not be
573       *                         {@code null}.
574       * @param  invalidReasons  A list to which messages may be added which provide
575       *                         information about why the entry is invalid.  It may
576       *                         be {@code null} if this information is not needed.
577       *
578       * @return  {@code true} if the entry conforms to all of the enabled forms of
579       *          validation, or {@code false} if the entry fails at least one of
580       *          the tests.
581       */
582      public boolean entryIsValid(final Entry entry,
583                                  final List<String> invalidReasons)
584      {
585        ensureNotNull(entry);
586    
587        boolean entryValid = true;
588        entriesExamined.incrementAndGet();
589    
590        // Get the parsed DN for the entry.
591        RDN rdn = null;
592        try
593        {
594          rdn = entry.getParsedDN().getRDN();
595        }
596        catch (LDAPException le)
597        {
598          debugException(le);
599          if (checkMalformedDNs)
600          {
601            entryValid = false;
602            malformedDNs.incrementAndGet();
603            if (invalidReasons != null)
604            {
605              invalidReasons.add(ERR_ENTRY_MALFORMED_DN.get(
606                   getExceptionMessage(le)));
607            }
608          }
609        }
610    
611        // Get the object class descriptions for the object classes in the entry.
612        final HashSet<ObjectClassDefinition> ocSet =
613             new HashSet<ObjectClassDefinition>();
614        final boolean missingOC =
615             (! getObjectClasses(entry, ocSet, invalidReasons));
616        if (missingOC)
617        {
618          entryValid = false;
619        }
620    
621        // If the entry was not missing any object classes, then get the structural
622        // class for the entry and use it to get the associated DIT content rule and
623        // name form.
624        DITContentRuleDefinition ditContentRule = null;
625        NameFormDefinition nameForm = null;
626        if (! missingOC)
627        {
628          final AtomicReference<ObjectClassDefinition> ref =
629               new AtomicReference<ObjectClassDefinition>(null);
630          entryValid &= getStructuralClass(ocSet, ref, invalidReasons);
631          final ObjectClassDefinition structuralClass = ref.get();
632          if (structuralClass != null)
633          {
634            ditContentRule = schema.getDITContentRule(structuralClass.getOID());
635            nameForm =
636                 schema.getNameFormByObjectClass(structuralClass.getNameOrOID());
637          }
638        }
639    
640        // If we should check for missing required attributes, then do so.
641        HashSet<AttributeTypeDefinition> requiredAttrs = null;
642        if (checkMissingAttributes || checkProhibitedAttributes)
643        {
644          requiredAttrs = getRequiredAttributes(ocSet, ditContentRule);
645          if (checkMissingAttributes)
646          {
647            entryValid &= checkForMissingAttributes(entry, rdn, requiredAttrs,
648                                                    invalidReasons);
649          }
650        }
651    
652        // Iterate through all of the attributes in the entry.  Make sure that they
653        // are all defined in the schema, that they are allowed to be present in the
654        // entry, that their values conform to the associated syntax, and that any
655        // single-valued attributes have only one value.
656        HashSet<AttributeTypeDefinition> optionalAttrs = null;
657        if (checkProhibitedAttributes)
658        {
659          optionalAttrs =
660               getOptionalAttributes(ocSet, ditContentRule, requiredAttrs);
661        }
662        for (final Attribute a : entry.getAttributes())
663        {
664          entryValid &=
665               checkAttribute(a, requiredAttrs, optionalAttrs, invalidReasons);
666        }
667    
668        // If there is a DIT content rule, then check to ensure that all of the
669        // auxiliary object classes are allowed.
670        if (checkProhibitedObjectClasses && (ditContentRule != null))
671        {
672          entryValid &=
673               checkAuxiliaryClasses(ocSet, ditContentRule, invalidReasons);
674        }
675    
676        // Check the entry's RDN to ensure that all attributes are defined in the
677        // schema, allowed to be present, and comply with the name form.
678        if (rdn != null)
679        {
680          entryValid &= checkRDN(rdn, requiredAttrs, optionalAttrs, nameForm,
681                                 invalidReasons);
682        }
683    
684        if (! entryValid)
685        {
686          invalidEntries.incrementAndGet();
687        }
688    
689        return entryValid;
690      }

694      /**
695       * Gets the object classes for the entry, including any that weren't
696       * explicitly included but should be because they were superior to classes
697       * that were included.
698       *
699       * @param  entry           The entry to examine.
700       * @param  ocSet           The set into which the object class definitions
701       *                         should be placed.
702       * @param  invalidReasons  A list to which messages may be added which provide
703       *                         information about why the entry is invalid.  It may
704       *                         be {@code null} if this information is not needed.
705       *
706       * @return  {@code true} if the entry passed all validation processing
707       *          performed by this method, or {@code false} if there were any
708       *          failures.
709       */
710      private boolean getObjectClasses(final Entry entry,
711                                       final HashSet<ObjectClassDefinition> ocSet,
712                                       final List<String> invalidReasons)
713      {
714        final String[] ocValues = entry.getObjectClassValues();
715        if ((ocValues == null) || (ocValues.length == 0))
716        {
717          noObjectClasses.incrementAndGet();
718          if (invalidReasons != null)
719          {
720            invalidReasons.add(ERR_ENTRY_NO_OCS.get());
721          }
722          return false;
723        }
724    
725        boolean entryValid = true;
726        final HashSet<String> missingOCs = new HashSet<String>(ocValues.length);
727        for (final String ocName : entry.getObjectClassValues())
728        {
729          final ObjectClassDefinition d = schema.getObjectClass(ocName);
730          if (d == null)
731          {
732            if (checkUndefinedObjectClasses)
733            {
734              entryValid = false;
735              missingOCs.add(toLowerCase(ocName));
736              updateCount(ocName, undefinedObjectClasses);
737              if (invalidReasons != null)
738              {
739                invalidReasons.add(ERR_ENTRY_UNDEFINED_OC.get(ocName));
740              }
741            }
742          }
743          else
744          {
745            ocSet.add(d);
746          }
747        }
748    
749        for (final ObjectClassDefinition d : ocSet)
750        {
751          entryValid &= addSuperiorClasses(d, ocSet, missingOCs, invalidReasons);
752        }
753    
754        return entryValid;
755      }

796                if (invalidReasons != null)
797                {
798                  invalidReasons.add(ERR_ENTRY_UNDEFINED_SUP_OC.get(
799                       d.getNameOrOID(), ocName));
800                }
801              }
802            }
803          }
804          else
805          {
806            ocSet.add(supOC);
807            entryValid &=
808                 addSuperiorClasses(supOC, ocSet, missingOCNames, invalidReasons);
809          }
810        }

817      /**
818       * Retrieves the structural object class from the set of provided object
819       * classes.
820       *
821       * @param  ocSet            The set of object class definitions for the entry.
822       * @param  structuralClass  The reference that will be updated with the
823       *                          entry's structural object class.
824       * @param  invalidReasons   A list to which messages may be added which
825       *                          provide provide information about why the entry is
826       *                          invalid.  It may be {@code null} if this
827       *                          information is not needed.
828       *
829       * @return  {@code true} if the entry passes all validation checks performed
830       *          by this method, or {@code false} if not.
831       */
832      private boolean getStructuralClass(final HashSet<ObjectClassDefinition> ocSet,
833                   final AtomicReference<ObjectClassDefinition> structuralClass,
834                   final List<String> invalidReasons)
835      {
836        final HashSet<ObjectClassDefinition> ocCopy =
837             new HashSet<ObjectClassDefinition>(ocSet);
838        for (final ObjectClassDefinition d : ocSet)
839        {
840          final ObjectClassType t = d.getObjectClassType(schema);
841          if (t == ObjectClassType.STRUCTURAL)
842          {
843            ocCopy.removeAll(d.getSuperiorClasses(schema, true));
844          }
845          else if (t == ObjectClassType.AUXILIARY)
846          {
847            ocCopy.remove(d);
848            ocCopy.removeAll(d.getSuperiorClasses(schema, true));
849          }
850        }
851    
852        // Iterate through the set of remaining classes and strip out any
853        // abstract classes.
854        boolean entryValid = true;
855        Iterator<ObjectClassDefinition> iterator = ocCopy.iterator();
856        while (iterator.hasNext())
857        {
858          final ObjectClassDefinition d = iterator.next();
859          if (d.getObjectClassType(schema) == ObjectClassType.ABSTRACT)
860          {
861            if (checkProhibitedObjectClasses)
862            {
863              entryValid = false;
864              updateCount(d.getNameOrOID(), prohibitedObjectClasses);
865              if (invalidReasons != null)
866              {
867                invalidReasons.add(ERR_ENTRY_INVALID_ABSTRACT_CLASS.get(
868                     d.getNameOrOID()));
869              }
870            }
871            iterator.remove();
872          }
873        }
874    
875        switch (ocCopy.size())
876        {
877          case 0:
878            if (checkStructuralObjectClasses)
879            {
880              entryValid = false;
881              noStructuralClass.incrementAndGet();
882              if (invalidReasons != null)
883              {
884                invalidReasons.add(ERR_ENTRY_NO_STRUCTURAL_CLASS.get());
885              }
886            }
887            break;
888    
889          case 1:
890            structuralClass.set(ocCopy.iterator().next());
891            break;
892    
893          default:
894            if (checkStructuralObjectClasses)
895            {
896              entryValid = false;
897              multipleStructuralClasses.incrementAndGet();
898              if (invalidReasons != null)
899              {
900                final StringBuilder ocList = new StringBuilder();
901                iterator = ocCopy.iterator();
902                while (iterator.hasNext())
903                {
904                  ocList.append(iterator.next().getNameOrOID());
905                  if (iterator.hasNext())
906                  {
907                    ocList.append(", ");
908                  }
909                }
910                invalidReasons.add(
911                     ERR_ENTRY_MULTIPLE_STRUCTURAL_CLASSES.get(ocList));
912              }
913            }
914            break;
915        }
916    
917        return entryValid;
918      }

1024       * Checks the provided entry to determine whether it is missing any required
1025       * attributes.
1026       *
1027       * @param  entry           The entry to examine.
1028       * @param  rdn             The RDN for the entry, if available.
1029       * @param  requiredAttrs   The set of attribute types which are required to be
1030       *                         included in the entry.
1031       * @param  invalidReasons  A list to which messages may be added which provide
1032       *                         information about why the entry is invalid.  It may
1033       *                         be {@code null} if this information is not needed.
1034       *
1035       * @return  {@code true} if the entry has all required attributes, or
1036       *          {@code false} if not.
1037       */
1038      private boolean checkForMissingAttributes(final Entry entry, final RDN rdn,
1039                           final HashSet<AttributeTypeDefinition> requiredAttrs,
1040                           final List<String> invalidReasons)
1041      {

1043

1045        {

1047          for (final String s : d.getNames())
1048          {

1050            {
1051              found = true;
1052              break;
1053            }
1054          }
1055    
1056          if (! found)
1057          {
1058            if (! (entry.hasAttribute(d.getOID()) ||
1059                   ((rdn != null) && (rdn.hasAttribute(d.getOID())))))
1060            {
1061              entryValid = false;
1062              updateCount(d.getNameOrOID(), missingAttributes);
1063              if (invalidReasons != null)
1064              {
1065                invalidReasons.add(ERR_ENTRY_MISSING_REQUIRED_ATTR.get(
1066                     d.getNameOrOID()));
1067              }
1068            }
1069          }
1070        }
1071    
1072        return entryValid;
1073      }

1077      /**
1078       * Checks the provided attribute to determine whether it appears to be valid.
1079       *
1080       * @param  attr            The attribute to examine.
1081       * @param  requiredAttrs   The set of attribute types which are required to be
1082       *                         included in the entry.
1083       * @param  optionalAttrs   The set of attribute types which may optionally be
1084       *                         included in the entry.
1085       * @param  invalidReasons  A list to which messages may be added which provide
1086       *                         information about why the entry is invalid.  It may
1087       *                         be {@code null} if this information is not needed.
1088       *
1089       * @return  {@code true} if the attribute passed all of the checks and appears
1090       *          to be valid, or {@code false} if it failed any of the checks.
1091       */
1092      private boolean checkAttribute(final Attribute attr,
1093                           final HashSet<AttributeTypeDefinition> requiredAttrs,
1094                           final HashSet<AttributeTypeDefinition> optionalAttrs,
1095                           final List<String> invalidReasons)
1096      {
1097        boolean entryValid = true;
1098    
1099        final AttributeTypeDefinition d =
1100             schema.getAttributeType(attr.getBaseName());
1101        if (d == null)
1102        {
1103          if (checkUndefinedAttributes)
1104          {
1105            entryValid = false;
1106            updateCount(attr.getBaseName(), undefinedAttributes);
1107            if (invalidReasons != null)
1108            {
1109              invalidReasons.add(ERR_ENTRY_UNDEFINED_ATTR.get(attr.getBaseName()));
1110            }
1111          }
1112    
1113          return entryValid;
1114        }
1115    
1116        if (checkProhibitedAttributes && (! d.isOperational()))
1117        {
1118          if (! (requiredAttrs.contains(d) || optionalAttrs.contains(d)))
1119          {
1120            entryValid = false;
1121            updateCount(d.getNameOrOID(), prohibitedAttributes);
1122            if (invalidReasons != null)
1123            {
1124              invalidReasons.add(ERR_ENTRY_ATTR_NOT_ALLOWED.get(d.getNameOrOID()));
1125            }
1126          }
1127        }
1128    
1129        final ASN1OctetString[] rawValues = attr.getRawValues();
1130        if (checkSingleValuedAttributes && d.isSingleValued() &&
1131            (rawValues.length > 1))
1132        {
1133          entryValid = false;
1134          updateCount(d.getNameOrOID(), singleValueViolations);
1135          if (invalidReasons != null)
1136          {
1137            invalidReasons.add(
1138                 ERR_ENTRY_ATTR_HAS_MULTIPLE_VALUES.get(d.getNameOrOID()));
1139          }
1140        }
1141    
1142        if (checkAttributeSyntax)
1143        {
1144          final MatchingRule r =
1145               MatchingRule.selectEqualityMatchingRule(d.getNameOrOID(), schema);
1146          for (final ASN1OctetString v : rawValues)
1147          {
1148            try
1149            {
1150              r.normalize(v);
1151            }
1152            catch (LDAPException le)
1153            {
1154              debugException(le);
1155              entryValid = false;
1156              updateCount(d.getNameOrOID(), attributesViolatingSyntax);
1157              if (invalidReasons != null)
1158              {
1159                invalidReasons.add(ERR_ENTRY_ATTR_INVALID_SYNTAX.get(
1160                     v.stringValue(), d.getNameOrOID(), getExceptionMessage(le)));
1161              }
1162            }
1163          }
1164        }
1165    
1166        return entryValid;
1167      }

1222      /**
1223       * Ensures that the provided RDN is acceptable.  It will ensure that all
1224       * attributes are defined in the schema and allowed for the entry, and that
1225       * the entry optionally conforms to the associated name form.
1226       *
1227       * @param  rdn             The RDN to examine.
1228       * @param  requiredAttrs   The set of attribute types which are required to be
1229       *                         included in the entry.
1230       * @param  optionalAttrs   The set of attribute types which may optionally be
1231       *                         included in the entry.
1232       * @param  nameForm        The name for to use to make the determination, if
1233       *                         defined.
1234       * @param  invalidReasons  A list to which messages may be added which provide
1235       *                         information about why the entry is invalid.  It may
1236       *                         be {@code null} if this information is not needed.
1237       *
1238       * @return  {@code true} if the entry passes all checks performed by this
1239       *          method, or {@code false} if not.
1240       */
1241      private boolean checkRDN(final RDN rdn,
1242                               final HashSet<AttributeTypeDefinition> requiredAttrs,
1243                               final HashSet<AttributeTypeDefinition> optionalAttrs,
1244                               final NameFormDefinition nameForm,
1245                               final List<String> invalidReasons)
1246      {
1247        final HashSet<AttributeTypeDefinition> nfReqAttrs =
1248             new HashSet<AttributeTypeDefinition>();
1249        final HashSet<AttributeTypeDefinition> nfAllowedAttrs =
1250             new HashSet<AttributeTypeDefinition>();
1251        if (nameForm != null)
1252        {
1253          for (final String s : nameForm.getRequiredAttributes())
1254          {
1255            final AttributeTypeDefinition d = schema.getAttributeType(s);
1256            if (d != null)
1257            {
1258              nfReqAttrs.add(d);
1259            }
1260          }
1261    
1262          nfAllowedAttrs.addAll(nfReqAttrs);
1263          for (final String s : nameForm.getOptionalAttributes())
1264          {
1265            final AttributeTypeDefinition d = schema.getAttributeType(s);
1266            if (d != null)
1267            {
1268              nfAllowedAttrs.add(d);
1269            }
1270          }
1271        }
1272    
1273        boolean entryValid = true;
1274        for (final String s : rdn.getAttributeNames())
1275        {
1276          final AttributeTypeDefinition d = schema.getAttributeType(s);
1277          if (d == null)
1278          {
1279            if (checkUndefinedAttributes)
1280            {
1281              entryValid = false;
1282              updateCount(s, undefinedAttributes);
1283              if (invalidReasons != null)
1284              {
1285                invalidReasons.add(ERR_ENTRY_RDN_ATTR_NOT_DEFINED.get(s));
1286              }
1287            }
1288          }
1289          else
1290          {
1291            if (checkProhibitedAttributes &&
1292                (! (requiredAttrs.contains(d) || optionalAttrs.contains(d))))
1293            {
1294              entryValid = false;
1295              updateCount(d.getNameOrOID(), prohibitedAttributes);
1296              if (invalidReasons != null)
1297              {
1298                invalidReasons.add(ERR_ENTRY_RDN_ATTR_NOT_ALLOWED_IN_ENTRY.get(
1299                     d.getNameOrOID()));
1300              }
1301            }
1302    
1303            if (checkNameForms && (nameForm != null))
1304            {
1305              if (! nfReqAttrs.remove(d))
1306              {
1307                if (! nfAllowedAttrs.contains(d))
1308                {
1309                  if (entryValid)
1310                  {
1311                    entryValid = false;
1312                    nameFormViolations.incrementAndGet();
1313                  }
1314                  if (invalidReasons != null)
1315                  {
1316                    invalidReasons.add(ERR_ENTRY_RDN_ATTR_NOT_ALLOWED_BY_NF.get(s));
1317                  }
1318                }
1319              }
1320            }
1321          }
1322        }
1323    
1324        if (checkNameForms && (! nfReqAttrs.isEmpty()))
1325        {
1326          if (entryValid)
1327          {
1328            entryValid = false;
1329            nameFormViolations.incrementAndGet();
1330          }
1331          if (invalidReasons != null)
1332          {
1333            for (final AttributeTypeDefinition d : nfReqAttrs)
1334            {
1335              invalidReasons.add(ERR_ENTRY_RDN_MISSING_REQUIRED_ATTR.get(
1336                   d.getNameOrOID()));
1337            }
1338          }
1339        }
1340    
1341        return entryValid;
1342      }
