040    /**
041     * This class provides a data structure for interacting with LDAP URLs.  It may
042     * be used to encode and decode URLs, as well as access the various elements
043     * that they contain.  Note that this implementation currently does not support
044     * the use of extensions in an LDAP URL.
045     * <BR><BR>
046     * The components that may be included in an LDAP URL include:
047     * <UL>
048     *   <LI>Scheme -- This specifies the protocol to use when communicating with
049     *       the server.  The official LDAP URL specification only allows a scheme
050     *       of "{@code ldap}", but this implementation also supports the use of the
051     *       "{@code ldaps}" scheme to indicate that clients should attempt to
052     *       perform SSL-based communication with the target server (LDAPS) rather
053     *       than unencrypted LDAP.</LI>
054     *   <LI>Host -- This specifies the address of the directory server to which the
055     *       URL refers.  If no host is provided, then it is expected that the
056     *       client has some prior knowledge of the host (it often implies the same
057     *       server from which the URL was retrieved).</LI>
058     *   <LI>Port -- This specifies the port of the directory server to which the
059     *       URL refers.  If no host or port is provided, then it is assumed that
060     *       the client has some prior knowledge of the instance to use (it often
061     *       implies the same instance from which the URL was retrieved).  If a host
062     *       is provided without a port, then it should be assumed that the standard
063     *       LDAP port of 389 should be used (or the standard LDAPS port of 636 if
064     *       the scheme is "{@code ldaps}").</LI>
065     *   <LI>Base DN -- This specifies the base DN for the URL.  If no base DN is
066     *       provided, then a default of the null DN should be assumed.</LI>
067     *   <LI>Requested attributes -- This specifies the set of requested attributes
068     *       for the URL.  If no attributes are specified, then the behavior should
069     *       be the same as if no attributes had been provided for a search request
070     *       (i.e., all user attributes should be included).
071     *       <BR><BR>
072     *       In the string representation of an LDAP URL, the names of the requested
073     *       attributes (if more than one is provided) should be separated by
074     *       commas.</LI>
075     *   <LI>Scope -- This specifies the scope for the URL.  It should be one of the
076     *       standard scope values as defined in the {@link SearchRequest}
077     *       class.  If no scope is provided, then it should be assumed that a
078     *       scope of {@link SearchScope#BASE} should be used.
079     *       <BR><BR>
080     *       In the string representation, the names of the scope values that are
081     *       allowed include:
082     *       <UL>
083     *         <LI>base -- Equivalent to {@link SearchScope#BASE}.</LI>
084     *         <LI>one -- Equivalent to {@link SearchScope#ONE}.</LI>
085     *         <LI>sub -- Equivalent to {@link SearchScope#SUB}.</LI>
086     *         <LI>subordinates -- Equivalent to
087     *             {@link SearchScope#SUBORDINATE_SUBTREE}.</LI>
088     *       </UL></LI>
089     *   <LI>Filter -- This specifies the filter for the URL.  If no filter is
090     *       provided, then a default of "{@code (objectClass=*)}" should be
091     *       assumed.</LI>
092     * </UL>
093     * An LDAP URL encapsulates many of the properties of a search request, and in
094     * fact the {@link LDAPURL#toSearchRequest} method may be used  to create a
095     * {@link SearchRequest} object from an LDAP URL.
096     * <BR><BR>
097     * See <A HREF="http://www.ietf.org/rfc/rfc4516.txt">RFC 4516</A> for a complete
098     * description of the LDAP URL syntax.  Some examples of LDAP URLs include:
099     * <UL>
100     *   <LI>{@code ldap://} -- This is the smallest possible LDAP URL that can be
101     *       represented.  The default values will be used for all components other
102     *       than the scheme.</LI>
103     *   <LI>{@code
104     *        ldap://server.example.com:1234/dc=example,dc=com?cn,sn?sub?(uid=john)}
105     *       -- This is an example of a URL containing all of the elements.  The
106     *       scheme is "{@code ldap}", the host is "{@code server.example.com}",
107     *       the port is "{@code 1234}", the base DN is "{@code dc=example,dc=com}",
108     *       the requested attributes are "{@code cn}" and "{@code sn}", the scope
109     *       is "{@code sub}" (which indicates a subtree scope equivalent to
110     *       {@link SearchScope#SUB}), and a filter of
111     *       "{@code (uid=john)}".</LI>
112     * </UL>
113     */

216      /**
217       * Creates a new LDAP URL from the provided string representation.
218       *
219       * @param  urlString  The string representation for this LDAP URL.  It must
220       *                    not be {@code null}.
221       *
222       * @throws  LDAPException  If the provided URL string cannot be parsed as an
223       *                         LDAP URL.
224       */
225      public LDAPURL(final String urlString)
226             throws LDAPException
227      {
228        ensureNotNull(urlString);
229    
230        this.urlString = urlString;
231    
232    
233        // Find the location of the first colon.  It should mark the end of the
234        // scheme.
235        final int colonPos = urlString.indexOf("://");
236        if (colonPos < 0)
237        {
238          throw new LDAPException(ResultCode.DECODING_ERROR,
239                                  ERR_LDAPURL_NO_COLON_SLASHES.get());
240        }
241    
242        scheme = toLowerCase(urlString.substring(0, colonPos));
243        final int defaultPort;
244        if (scheme.equals("ldap"))
245        {
246          defaultPort = DEFAULT_LDAP_PORT;
247        }
248        else if (scheme.equals("ldaps"))
249        {
250          defaultPort = DEFAULT_LDAPS_PORT;
251        }
252        else
253        {
254          throw new LDAPException(ResultCode.DECODING_ERROR,
255                                  ERR_LDAPURL_INVALID_SCHEME.get(scheme));
256        }
257    
258    
259        // Look for the first slash after the "://".  It will designate the end of
260        // the hostport section.
261        final int slashPos = urlString.indexOf('/', colonPos+3);
262        if (slashPos < 0)
263        {
264          // This is fine.  It just means that the URL won't have a base DN,
265          // attribute list, scope, or filter, and that the rest of the value is
266          // the hostport element.
267          baseDN             = DEFAULT_BASE_DN;
268          baseDNProvided     = false;
269          attributes         = DEFAULT_ATTRIBUTES;
270          attributesProvided = false;
271          scope              = DEFAULT_SCOPE;
272          scopeProvided      = false;
273          filter             = DEFAULT_FILTER;
274          filterProvided     = false;
275    
276          final String hostPort = urlString.substring(colonPos+3);
277          final StringBuilder hostBuffer = new StringBuilder(hostPort.length());
278          final int portValue = decodeHostPort(hostPort, hostBuffer);
279          if (portValue < 0)
280          {
281            port         = defaultPort;
282            portProvided = false;
283          }
284          else
285          {
286            port         = portValue;
287            portProvided = true;
288          }
289    
290          if (hostBuffer.length() == 0)
291          {
292            host = null;
293          }
294          else
295          {
296            host = hostBuffer.toString();
297          }
298          return;
299        }
300    
301        final String hostPort = urlString.substring(colonPos+3, slashPos);
302        final StringBuilder hostBuffer = new StringBuilder(hostPort.length());
303        final int portValue = decodeHostPort(hostPort, hostBuffer);
304        if (portValue < 0)
305        {
306          port         = defaultPort;
307          portProvided = false;
308        }
309        else
310        {
311          port         = portValue;
312          portProvided = true;
313        }
314    
315        if (hostBuffer.length() == 0)
316        {
317          host = null;
318        }
319        else
320        {
321          host = hostBuffer.toString();
322        }
323    
324    
325        // Look for the first question mark after the slash.  It will designate the
326        // end of the base DN.
327        final int questionMarkPos = urlString.indexOf('?', slashPos+1);
328        if (questionMarkPos < 0)
329        {
330          // This is fine.  It just means that the URL won't have an attribute list,
331          // scope, or filter, and that the rest of the value is the base DN.
332          attributes         = DEFAULT_ATTRIBUTES;
333          attributesProvided = false;
334          scope              = DEFAULT_SCOPE;
335          scopeProvided      = false;
336          filter             = DEFAULT_FILTER;
337          filterProvided     = false;
338    
339          baseDN = new DN(percentDecode(urlString.substring(slashPos+1)));
340          baseDNProvided = (! baseDN.isNullDN());
341          return;
342        }
343    
344        baseDN = new DN(percentDecode(urlString.substring(slashPos+1,
345                                                          questionMarkPos)));
346        baseDNProvided = (! baseDN.isNullDN());
347    
348    
349        // Look for the next question mark.  It will designate the end of the
350        // attribute list.
351        final int questionMark2Pos = urlString.indexOf('?', questionMarkPos+1);
352        if (questionMark2Pos < 0)
353        {
354          // This is fine.  It just means that the URL won't have a scope or filter,
355          // and that the rest of the value is the attribute list.
356          scope          = DEFAULT_SCOPE;
357          scopeProvided  = false;
358          filter         = DEFAULT_FILTER;
359          filterProvided = false;
360    
361          attributes = decodeAttributes(urlString.substring(questionMarkPos+1));
362          attributesProvided = (attributes.length > 0);
363          return;
364        }
365    
366        attributes = decodeAttributes(urlString.substring(questionMarkPos+1,
367                                                          questionMark2Pos));
368        attributesProvided = (attributes.length > 0);
369    
370    
371        // Look for the next question mark.  It will designate the end of the scope.
372        final int questionMark3Pos = urlString.indexOf('?', questionMark2Pos+1);
373        if (questionMark3Pos < 0)
374        {
375          // This is fine.  It just means that the URL won't have a filter, and that
376          // the rest of the value is the scope.
377          filter         = DEFAULT_FILTER;
378          filterProvided = false;
379    
380          final String scopeStr =
381               toLowerCase(urlString.substring(questionMark2Pos+1));
382          if (scopeStr.length() == 0)
383          {
384            scope         = SearchScope.BASE;
385            scopeProvided = false;
386          }
387          else if (scopeStr.equals("base"))
388          {
389            scope         = SearchScope.BASE;
390            scopeProvided = true;
391          }
392          else if (scopeStr.equals("one"))
393          {
394            scope         = SearchScope.ONE;
395            scopeProvided = true;
396          }
397          else if (scopeStr.equals("sub"))
398          {
399            scope         = SearchScope.SUB;
400            scopeProvided = true;
401          }
402          else if (scopeStr.equals("subord") || scopeStr.equals("subordinates"))
403          {
404            scope         = SearchScope.SUBORDINATE_SUBTREE;
405            scopeProvided = true;
406          }
407          else
408          {
409            throw new LDAPException(ResultCode.DECODING_ERROR,
410                                    ERR_LDAPURL_INVALID_SCOPE.get(scopeStr));
411          }
412          return;
413        }
414    
415        final String scopeStr =
416             toLowerCase(urlString.substring(questionMark2Pos+1, questionMark3Pos));
417        if (scopeStr.length() == 0)
418        {
419          scope         = SearchScope.BASE;
420          scopeProvided = false;
421        }
422        else if (scopeStr.equals("base"))
423        {
424          scope         = SearchScope.BASE;
425          scopeProvided = true;
426        }
427        else if (scopeStr.equals("one"))
428        {
429          scope         = SearchScope.ONE;
430          scopeProvided = true;
431        }
432        else if (scopeStr.equals("sub"))
433        {
434          scope         = SearchScope.SUB;
435          scopeProvided = true;
436        }
437            else if (scopeStr.equals("subord") || scopeStr.equals("subordinates"))
438        {
439          scope         = SearchScope.SUBORDINATE_SUBTREE;
440          scopeProvided = true;
441        }
442        else
443        {
444          throw new LDAPException(ResultCode.DECODING_ERROR,
445                                  ERR_LDAPURL_INVALID_SCOPE.get(scopeStr));
446        }
447    
448    
449        // The remainder of the value must be the filter.
450        final String filterStr =
451             percentDecode(urlString.substring(questionMark3Pos+1));
452        if (filterStr.length() == 0)
453        {
454          filter = DEFAULT_FILTER;
455          filterProvided = false;
456        }
457        else
458        {
459          filter = Filter.create(filterStr);
460          filterProvided = true;
461        }
462      }

466      /**
467       * Creates a new LDAP URL with the provided information.
468       *
469       * @param  scheme      The scheme for this LDAP URL.  It must not be
470       *                     {@code null} and must be either "ldap" or "ldaps".
471       * @param  host        The host for this LDAP URL.  It may be {@code null} if
472       *                     no host is to be included.
473       * @param  port        The port for this LDAP URL.  It may be {@code null} if
474       *                     no port is to be included.  If it is provided, it must
475       *                     be between 1 and 65535, inclusive.
476       * @param  baseDN      The base DN for this LDAP URL.  It may be {@code null}
477       *                     if no base DN is to be included.
478       * @param  attributes  The set of requested attributes for this LDAP URL.  It
479       *                     may be {@code null} or empty if no attribute list is to
480       *                     be included.
481       * @param  scope       The scope for this LDAP URL.  It may be {@code null} if
482       *                     no scope is to be included.  Otherwise, it must be a
483       *                     value between zero and three, inclusive.
484       * @param  filter      The filter for this LDAP URL.  It may be {@code null}
485       *                     if no filter is to be included.
486       *
487       * @throws  LDAPException  If there is a problem with any of the provided
488       *                         arguments.
489       */
490      public LDAPURL(final String scheme, final String host, final Integer port,
491                     final DN baseDN, final String[] attributes,
492                     final SearchScope scope, final Filter filter)
493             throws LDAPException
494      {
495        ensureNotNull(scheme);
496    
497        final StringBuilder buffer = new StringBuilder();
498    
499        this.scheme = toLowerCase(scheme);
500        final int defaultPort;
501        if (scheme.equals("ldap"))
502        {
503          defaultPort = DEFAULT_LDAP_PORT;
504        }
505        else if (scheme.equals("ldaps"))
506        {
507          defaultPort = DEFAULT_LDAPS_PORT;
508        }
509        else
510        {
511          throw new LDAPException(ResultCode.DECODING_ERROR,
512                                  ERR_LDAPURL_INVALID_SCHEME.get(scheme));
513        }
514    
515        buffer.append(scheme);
516        buffer.append("://");
517    
518        if ((host == null) || (host.length() == 0))
519        {
520          this.host = null;
521        }
522        else
523        {
524          this.host = host;
525          buffer.append(host);
526        }
527    
528        if (port == null)
529        {
530          this.port = defaultPort;
531          portProvided = false;
532        }
533        else
534        {
535          this.port = port;
536          portProvided = true;
537          buffer.append(':');
538          buffer.append(port);
539    
540          if ((port < 1) || (port > 65535))
541          {
542            throw new LDAPException(ResultCode.PARAM_ERROR,
543                                    ERR_LDAPURL_INVALID_PORT.get(port));
544          }
545        }
546    
547        buffer.append('/');
548        if (baseDN == null)
549        {
550          this.baseDN = DEFAULT_BASE_DN;
551          baseDNProvided = false;
552        }
553        else
554        {
555          this.baseDN = baseDN;
556          baseDNProvided = true;
557          percentEncode(baseDN.toString(), buffer);
558        }
559    
560        buffer.append('?');
561        if ((attributes == null) || (attributes.length == 0))
562        {
563          this.attributes = DEFAULT_ATTRIBUTES;
564          attributesProvided = false;
565        }
566        else
567        {
568          this.attributes = attributes;
569          attributesProvided = true;
570    
571          for (int i=0; i < attributes.length; i++)
572          {
573            if (i > 0)
574            {
575              buffer.append(',');
576            }
577            buffer.append(attributes[i]);
578          }
579        }
580    
581        buffer.append('?');
582        if (scope == null)
583        {
584          this.scope = DEFAULT_SCOPE;
585          scopeProvided = false;
586        }
587        else
588        {
589          switch (scope.intValue())
590          {
591            case 0:
592              this.scope = scope;
593              scopeProvided = true;
594              buffer.append("base");
595              break;
596            case 1:
597              this.scope = scope;
598              scopeProvided = true;
599              buffer.append("one");
600              break;
601            case 2:
602              this.scope = scope;
603              scopeProvided = true;
604              buffer.append("sub");
605              break;
606            case 3:
607              this.scope = scope;
608              scopeProvided = true;
609              buffer.append("subordinates");
610              break;
611            default:
612              throw new LDAPException(ResultCode.PARAM_ERROR,
613                                      ERR_LDAPURL_INVALID_SCOPE_VALUE.get(scope));
614          }
615        }
616    
617        buffer.append('?');
618        if (filter == null)
619        {
620          this.filter = DEFAULT_FILTER;
621          filterProvided = false;
622        }
623        else
624        {
625          this.filter = filter;
626          filterProvided = true;
627          percentEncode(filter.toString(), buffer);
628        }
629    
630        urlString = buffer.toString();
631      }

635      /**
636       * Decodes the provided string as a host and optional port number.
637       *
638       * @param  hostPort    The string to be decoded.
639       * @param  hostBuffer  The buffer to which the decoded host address will be
640       *                     appended.
641       *
642       * @return  The port number decoded from the provided string, or -1 if there
643       *          was no port number.
644       *
645       * @throws  LDAPException  If the provided string cannot be decoded as a
646       *                         hostport element.
647       */
648      private static int decodeHostPort(final String hostPort,
649                                        final StringBuilder hostBuffer)
650              throws LDAPException
651      {
652        final int length = hostPort.length();
653        if (length == 0)
654        {
655          // It's an empty string, so we'll just use the defaults.
656          return -1;
657        }
658    
659        if (hostPort.charAt(0) == '[')
660        {
661          // It starts with a square bracket, which means that the address is an
662          // IPv6 literal address.  Find the closing bracket, and the address
663          // will be inside them.
664          final int closingBracketPos = hostPort.indexOf(']');
665          if (closingBracketPos < 0)
666          {
667            throw new LDAPException(ResultCode.DECODING_ERROR,
668                                    ERR_LDAPURL_IPV6_HOST_MISSING_BRACKET.get());
669          }
670    
671          hostBuffer.append(hostPort.substring(1, closingBracketPos).trim());
672          if (hostBuffer.length() == 0)
673          {
674            throw new LDAPException(ResultCode.DECODING_ERROR,
675                                    ERR_LDAPURL_IPV6_HOST_EMPTY.get());
676          }
677    
678          // The closing bracket must either be the end of the hostport element
679          // (in which case we'll use the default port), or it must be followed by
680          // a colon and an integer (which will be the port).
681          if (closingBracketPos == (length - 1))
682          {
683            return -1;
684          }
685          else
686          {
687            if (hostPort.charAt(closingBracketPos+1) != ':')
688            {
689              throw new LDAPException(ResultCode.DECODING_ERROR,
690                                      ERR_LDAPURL_IPV6_HOST_UNEXPECTED_CHAR.get(
691                                           hostPort.charAt(closingBracketPos+1)));
692            }
693            else
694            {
695              try
696              {
697                final int decodedPort =
698                     Integer.parseInt(hostPort.substring(closingBracketPos+2));
699                if ((decodedPort >= 1) && (decodedPort <= 65535))
700                {
701                  return decodedPort;
702                }
703                else
704                {
705                  throw new LDAPException(ResultCode.DECODING_ERROR,
706                                          ERR_LDAPURL_INVALID_PORT.get(
707                                               decodedPort));
708                }
709              }
710              catch (NumberFormatException nfe)
711              {
712                debugException(nfe);
713                throw new LDAPException(ResultCode.DECODING_ERROR,
714                                        ERR_LDAPURL_PORT_NOT_INT.get(hostPort),
715                                        nfe);
716              }
717            }
718          }
719        }
720    
721    
722        // If we've gotten here, then the address is either a resolvable name or an
723        // IPv4 address.  If there is a colon in the string, then it will separate
724        // the address from the port.  Otherwise, the remaining value will be the
725        // address and we'll use the default port.
726        final int colonPos = hostPort.indexOf(':');
727        if (colonPos < 0)
728        {
729          hostBuffer.append(hostPort);
730          return -1;
731        }
732        else
733        {
734          try
735          {
736            final int decodedPort =
737                 Integer.parseInt(hostPort.substring(colonPos+1));
738            if ((decodedPort >= 1) && (decodedPort <= 65535))
739            {
740              hostBuffer.append(hostPort.substring(0, colonPos));
741              return decodedPort;
742            }
743            else
744            {
745              throw new LDAPException(ResultCode.DECODING_ERROR,
746                                      ERR_LDAPURL_INVALID_PORT.get(decodedPort));
747            }
748          }
749          catch (NumberFormatException nfe)
750          {
751            debugException(nfe);
752            throw new LDAPException(ResultCode.DECODING_ERROR,
753                                    ERR_LDAPURL_PORT_NOT_INT.get(hostPort), nfe);
754          }
755        }
756      }

760      /**
761       * Decodes the contents of the provided string as an attribute list.
762       *
763       * @param  s  The string to decode as an attribute list.
764       *
765       * @return  The array of decoded attribute names.
766       *
767       * @throws  LDAPException  If an error occurred while attempting to decode the
768       *                         attribute list.
769       */
770      private static String[] decodeAttributes(final String s)
771              throws LDAPException
772      {
773        final int length = s.length();
774        if (length == 0)
775        {
776          return DEFAULT_ATTRIBUTES;
777        }
778    
779        final ArrayList<String> attrList = new ArrayList<String>();
780        int startPos = 0;
781        while (startPos < length)
782        {
783          final int commaPos = s.indexOf(',', startPos);
784          if (commaPos < 0)
785          {
786            // There are no more commas, so there can only be one attribute left.
787            final String attrName = s.substring(startPos).trim();
788            if (attrName.length() == 0)
789            {
790              // This is only acceptable if the attribute list is empty (there was
791              // probably a space in the attribute list string, which is technically
792              // not allowed, but we'll accept it).  If the attribute list is not
793              // empty, then there were two consecutive commas, which is not
794              // allowed.
795              if (attrList.isEmpty())
796              {
797                return DEFAULT_ATTRIBUTES;
798              }
799              else
800              {
801                throw new LDAPException(ResultCode.DECODING_ERROR,
802                                        ERR_LDAPURL_ATTRLIST_ENDS_WITH_COMMA.get());
803              }
804            }
805            else
806            {
807              attrList.add(attrName);
808              break;
809            }
810          }
811          else
812          {
813            final String attrName = s.substring(startPos, commaPos).trim();
814            if (attrName.length() == 0)
815            {
816              throw new LDAPException(ResultCode.DECODING_ERROR,
817                                      ERR_LDAPURL_ATTRLIST_EMPTY_ATTRIBUTE.get());
818            }
819            else
820            {
821              attrList.add(attrName);
822              startPos = commaPos+1;
823              if (startPos >= length)
824              {
825                throw new LDAPException(ResultCode.DECODING_ERROR,
826                                        ERR_LDAPURL_ATTRLIST_ENDS_WITH_COMMA.get());
827              }
828            }
829          }
830        }
831    
832        final String[] attributes = new String[attrList.size()];
833        attrList.toArray(attributes);
834        return attributes;
835      }

839      /**
840       * Decodes any percent-encoded values that may be contained in the provided
841       * string.
842       *
843       * @param  s  The string to be decoded.
844       *
845       * @return  The percent-decoded form of the provided string.
846       *
847       * @throws  LDAPException  If a problem occurs while attempting to decode the
848       *                         provided string.
849       */
850      public static String percentDecode(final String s)
851              throws LDAPException
852      {
853        // First, see if there are any percent characters at all in the provided
854        // string.  If not, then just return the string as-is.
855        int firstPercentPos = -1;
856        final int length = s.length();
857        for (int i=0; i < length; i++)
858        {
859          if (s.charAt(i) == '%')
860          {
861            firstPercentPos = i;
862            break;
863          }
864        }
865    
866        if (firstPercentPos < 0)
867        {
868          return s;
869        }
870    
871        int pos = firstPercentPos;
872        final StringBuilder buffer = new StringBuilder(2 * length);
873        buffer.append(s.substring(0, firstPercentPos));
874    
875        while (pos < length)
876        {
877          final char c = s.charAt(pos++);
878          if (c == '%')
879          {
880            if (pos >= length)
881            {
882              throw new LDAPException(ResultCode.DECODING_ERROR,
883                                      ERR_LDAPURL_HEX_STRING_TOO_SHORT.get(s));
884            }
885    
886    
887            final ByteBuffer byteBuffer = ByteBuffer.allocate(length - pos);
888            while (pos < length)
889            {
890              byte b;
891              switch (s.charAt(pos++))
892              {
893                case '0':
894                  b = 0x00;
895                  break;
896                case '1':
897                  b = 0x10;
898                  break;
899                case '2':
900                  b = 0x20;
901                  break;
902                case '3':
903                  b = 0x30;
904                  break;
905                case '4':
906                  b = 0x40;
907                  break;
908                case '5':
909                  b = 0x50;
910                  break;
911                case '6':
912                  b = 0x60;
913                  break;
914                case '7':
915                  b = 0x70;
916                  break;
917                case '8':
918                  b = (byte) 0x80;
919                  break;
920                case '9':
921                  b = (byte) 0x90;
922                  break;
923                case 'a':
924                case 'A':
925                  b = (byte) 0xA0;
926                  break;
927                case 'b':
928                case 'B':
929                  b = (byte) 0xB0;
930                  break;
931                case 'c':
932                case 'C':
933                  b = (byte) 0xC0;
934                  break;
935                case 'd':
936                case 'D':
937                  b = (byte) 0xD0;
938                  break;
939                case 'e':
940                case 'E':
941                  b = (byte) 0xE0;
942                  break;
943                case 'f':
944                case 'F':
945                  b = (byte) 0xF0;
946                  break;
947                default:
948                  throw new LDAPException(ResultCode.DECODING_ERROR,
949                                          ERR_LDAPURL_INVALID_HEX_CHAR.get(
950                                               s.charAt(pos-1)));
951              }
952    
953              if (pos >= length)
954              {
955                throw new LDAPException(ResultCode.DECODING_ERROR,
956                                        ERR_LDAPURL_HEX_STRING_TOO_SHORT.get(s));
957              }
958    
959              switch (s.charAt(pos++))
960              {
961                case '0':
962                  b |= 0x00;
963                  break;
964                case '1':
965                  b |= 0x01;
966                  break;
967                case '2':
968                  b |= 0x02;
969                  break;
970                case '3':
971                  b |= 0x03;
972                  break;
973                case '4':
974                  b |= 0x04;
975                  break;
976                case '5':
977                  b |= 0x05;
978                  break;
979                case '6':
980                  b |= 0x06;
981                  break;
982                case '7':
983                  b |= 0x07;
984                  break;
985                case '8':
986                  b |= 0x08;
987                  break;
988                case '9':
989                  b |= 0x09;
990                  break;
991                case 'a':
992                case 'A':
993                  b |= 0x0A;
994                  break;
995                case 'b':
996                case 'B':
997                  b |= 0x0B;
998                  break;
999                case 'c':
1000                case 'C':
1001                  b |= 0x0C;
1002                  break;
1003                case 'd':
1004                case 'D':
1005                  b |= 0x0D;
1006                  break;
1007                case 'e':
1008                case 'E':
1009                  b |= 0x0E;
1010                  break;
1011                case 'f':
1012                case 'F':
1013                  b |= 0x0F;
1014                  break;
1015                default:
1016                  throw new LDAPException(ResultCode.DECODING_ERROR,
1017                                          ERR_LDAPURL_INVALID_HEX_CHAR.get(
1018                                               s.charAt(pos-1)));
1019              }
1020    
1021              byteBuffer.put(b);
1022              if ((pos < length) && (s.charAt(pos) != '%'))
1023              {
1024                break;
1025              }
1026            }
1027    
1028            byteBuffer.flip();
1029            final byte[] byteArray = new byte[byteBuffer.limit()];
1030            byteBuffer.get(byteArray);
1031    
1032            buffer.append(toUTF8String(byteArray));
1033          }
1034          else
1035          {
1036            buffer.append(c);
1037          }
1038        }
1039    
1040        return buffer.toString();
1041      }

1415      /**
1416       * Appends a normalized string representation of this LDAP URL to the provided
1417       * buffer.
1418       *
1419       * @param  buffer  The buffer to which to append the normalized string
1420       *                 representation of this LDAP URL.
1421       */
1422      public void toNormalizedString(final StringBuilder buffer)
1423      {
1424        buffer.append(scheme);
1425        buffer.append("://");
1426    
1427        if (host != null)
1428        {
1429          if (host.indexOf(':') >= 0)
1430          {
1431            buffer.append('[');
1432            buffer.append(toLowerCase(host));
1433            buffer.append(']');
1434          }
1435          else
1436          {
1437            buffer.append(toLowerCase(host));
1438          }
1439        }
1440    
1441        buffer.append(':');
1442        buffer.append(port);
1443        buffer.append('/');
1444        percentEncode(baseDN.toNormalizedString(), buffer);
1445        buffer.append('?');
1446    
1447        for (int i=0; i < attributes.length; i++)
1448        {
1449          if (i > 0)
1450          {
1451            buffer.append(',');
1452          }
1453    
1454          buffer.append(toLowerCase(attributes[i]));
1455        }
1456    
1457        buffer.append('?');
1458        switch (scope.intValue())
1459        {
1460          case 0:  // BASE
1461            buffer.append("base");
1462            break;
1463          case 1:  // ONE
1464            buffer.append("one");
1465            break;
1466          case 2:  // SUB
1467            buffer.append("sub");
1468            break;
1469          case 3:  // SUBORDINATE_SUBTREE
1470            buffer.append("subordinates");
1471            break;
1472        }
1473    
1474        buffer.append('?');
1475        percentEncode(filter.toNormalizedString(), buffer);
1476      }
