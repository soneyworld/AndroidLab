057    /**
058     * This class provides a data structure for holding information about an LDAP
059     * entry.  An entry contains a distinguished name (DN) and a set of attributes.
060     * An entry can be created from these components, and it can also be created
061     * from its LDIF representation as described in
062     * <A HREF="http://www.ietf.org/rfc/rfc2849.txt">RFC 2849</A>.  For example:
063     * <BR><BR>
064     * <PRE>
065     *   Entry entry = new Entry(
066     *     "dn: dc=example,dc=com",
067     *     "objectClass: top",
068     *     "objectClass: domain",
069     *     "dc: example");
070     * </PRE>
071     * <BR><BR>
072     * This class also provides methods for retrieving the LDIF representation of
073     * an entry, either as a single string or as an array of strings that make up
074     * the LDIF lines.
075     * <BR><BR>
076     * The {@link Entry#diff} method may be used to obtain the set of differences
077     * between two entries, and to retrieve a list of {@link Modification} objects
078     * that can be used to modify one entry so that it contains the same set of
079     * data as another.  The {@link Entry#applyModifications} method may be used to
080     * apply a set of modifications to an entry.
081     * <BR><BR>
082     * Entry objects are mutable, and the DN, set of attributes, and individual
083     * attribute values can be altered.
084     */

143       * Creates a new entry with the provided DN and set of attributes.
144       *
145       * @param  dn          The DN for this entry.  It must not be {@code null}.
146       * @param  attributes  The set of attributes for this entry.  It must not be
147       *                     {@code null}.
148       */

150      {
151        ensureNotNull(dn, attributes);
152    
153        this.dn = dn;
154    
155        this.attributes = new LinkedHashMap<String,Attribute>(attributes.length);
156        for (final Attribute a : attributes)
157        {
158          final String name = toLowerCase(a.getName());
159          final Attribute attr = this.attributes.get(name);
160          if (attr == null)
161          {
162            this.attributes.put(name, a);
163          }
164          else
165          {
166            this.attributes.put(name, Attribute.mergeAttributes(attr, a));
167          }
168        }
169      }

600      /**
601       * Retrieves the list of attributes with the given base name and all of the
602       * specified options.
603       *
604       * @param  baseName  The base name (without any options) for the attribute to
605       *                   retrieve.  It must not be {@code null}.
606       * @param  options   The set of options that should be included in the
607       *                   attributes that are returned.  It may be empty or
608       *                   {@code null} if all attributes with the specified base
609       *                   name should be returned, regardless of the options that
610       *                   they contain (if any).
611       *
612       * @return  The list of attributes with the given base name and all of the
613       *          specified options.  It may be empty if there are no attributes
614       *          with the specified base name and set of options.
615       */
616      public final List<Attribute> getAttributesWithOptions(final String baseName,
617                                        final Set<String> options)
618      {
619        ensureNotNull(baseName);
620    
621        final ArrayList<Attribute> attrList = new ArrayList<Attribute>(10);
622    
623        for (final Attribute a : attributes.values())
624        {
625          if (a.getBaseName().equalsIgnoreCase(baseName))
626          {
627            if ((options == null) || options.isEmpty())
628            {
629              attrList.add(a);
630            }
631            else
632            {
633              boolean allFound = true;
634              for (final String option : options)
635              {
636                if (! a.hasOption(option))
637                {
638                  allFound = false;
639                  break;
640                }
641              }
642    
643              if (allFound)
644              {
645                attrList.add(a);
646              }
647            }
648          }
649        }
650    
651        return Collections.unmodifiableList(attrList);
652      }

1424       *
1425       * @param  sourceEntry  The source entry for which the set of modifications
1426       *                      should be generated.
1427       * @param  targetEntry  The target entry, which is what the source entry
1428       *                      should look like if the returned modifications are
1429       *                      applied.
1430       * @param  ignoreRDN    Indicates whether to ignore differences in the RDNs
1431       *                      of the provided entries.  If this is {@code false},
1432       *                      then the resulting set of modifications may include
1433       *                      changes to the RDN attribute.  If it is {@code true},
1434       *                      then differences in the entry DNs will be ignored.
1435       * @param  reversible   Indicates whether to generate the diff in reversible
1436       *                      form.  In reversible form, only the ADD or DELETE
1437       *                      modification types will be used so that source entry
1438       *                      could be reconstructed from the target and the
1439       *                      resulting modifications.  In non-reversible form, only
1440       *                      the REPLACE modification type will be used.  Attempts
1441       *                      to apply the modifications obtained when using
1442       *                      reversible form are more likely to fail if the entry
1443       *                      has been modified since the source and target forms
1444       *                      were obtained.
1445       * @param  attributes   The set of attributes to be compared.  If this is
1446       *                      {@code null} or empty, then all attributes will be
1447       *                      compared.
1448       *
1449       * @return  A set of modifications that can be applied to the source entry in
1450       *          order to make it match the target entry.
1451       */
1452      public static List<Modification> diff(final Entry sourceEntry,
1453                                            final Entry targetEntry,
1454                                            final boolean ignoreRDN,
1455                                            final boolean reversible,
1456                                            final String... attributes)
1457      {
1458        HashSet<String> compareAttrs = null;
1459        if ((attributes != null) && (attributes.length > 0))
1460        {
1461          compareAttrs = new HashSet<String>(attributes.length);
1462          for (final String s : attributes)
1463          {
1464            compareAttrs.add(toLowerCase(s));
1465          }
1466        }
1467    
1468        final LinkedHashMap<String,Attribute> sourceOnlyAttrs =
1469             new LinkedHashMap<String,Attribute>();
1470        final LinkedHashMap<String,Attribute> targetOnlyAttrs =
1471             new LinkedHashMap<String,Attribute>();
1472        final LinkedHashMap<String,Attribute> commonAttrs =
1473             new LinkedHashMap<String,Attribute>();
1474    
1475        for (final Map.Entry<String,Attribute> e :
1476             sourceEntry.attributes.entrySet())
1477        {
1478          final String lowerName = toLowerCase(e.getKey());
1479          if ((compareAttrs != null) && (! compareAttrs.contains(lowerName)))
1480          {
1481            continue;
1482          }
1483    
1484          sourceOnlyAttrs.put(lowerName, e.getValue());
1485          commonAttrs.put(lowerName, e.getValue());
1486        }
1487    
1488        for (final Map.Entry<String,Attribute> e :
1489             targetEntry.attributes.entrySet())
1490        {
1491          final String lowerName = toLowerCase(e.getKey());
1492          if ((compareAttrs != null) && (! compareAttrs.contains(lowerName)))
1493          {
1494            continue;
1495          }
1496    
1497    
1498          if (sourceOnlyAttrs.remove(lowerName) == null)
1499          {
1500            // It wasn't in the set of source attributes, so it must be a

1531          }

1537        {
1538          if (reversible)
1539          {
1540            ASN1OctetString[] values = a.getRawValues();

1542            {
1543              final ArrayList<ASN1OctetString> newValues =
1544                   new ArrayList<ASN1OctetString>(values.length);
1545              for (final ASN1OctetString value : values)
1546              {
1547                if (! sourceRDN.hasAttributeValue(a.getName(), value.getValue()))
1548                {
1549                  newValues.add(value);
1550                }
1551              }
1552

1554              {
1555                continue;
1556              }

1560                newValues.toArray(values);
1561              }
1562            }
1563

1565                 values));
1566          }
1567          else
1568          {
1569            mods.add(new Modification(ModificationType.REPLACE, a.getName()));
1570          }
1571        }
1572

1574        {

1576          if ((targetRDN != null) && (targetRDN.hasAttribute(a.getName())))
1577          {
1578            final ArrayList<ASN1OctetString> newValues =
1579                 new ArrayList<ASN1OctetString>(values.length);

1601            mods.add(new Modification(ModificationType.ADD, a.getName(), values));

1607          }

1611        {
1612          final Attribute targetAttr =
1613               targetEntry.getAttribute(sourceAttr.getName());

1615          {
1616            continue;
1617          }

1619          if (reversible ||
1620              ((targetRDN != null) && targetRDN.hasAttribute(targetAttr.getName())))
1621          {
1622            final ArrayList<ASN1OctetString> addValues =
1623                 new ArrayList<ASN1OctetString>();

1625                 new ArrayList<ASN1OctetString>();

1631                    (! sourceRDN.hasAttributeValue(sourceAttr.getName(),
1632                                                   value.getValue())))
1633                {
1634                  delValues.add(value);
1635                }
1636              }
1637            }
1638    
1639            for (final ASN1OctetString value : targetAttr.getRawValues())
1640            {
1641              if (! sourceAttr.hasValue(value))
1642              {
1643                if ((targetRDN == null) ||
1644                    (! targetRDN.hasAttributeValue(targetAttr.getName(),
1645                                                   value.getValue())))
1646                {
1647                  addValues.add(value);
1648                }
1649              }
1650            }
1651

1653            {

1655                   new ASN1OctetString[addValues.size()];

1657                   addValues.toArray(addArray)));

1671                 targetAttr.getName(), targetAttr.getRawValues()));

1713       * Creates a duplicate of the provided entry with the given set of
1714       * modifications applied to it.
1715       *
1716       * @param  entry          The entry to be modified.  It must not be
1717       *                        {@code null}.
1718       * @param  lenient        Indicates whether to exhibit a lenient behavior for
1719       *                        the modifications, which will cause it to ignore
1720       *                        problems like trying to add values that already
1721       *                        exist or to remove nonexistent attributes or values.
1722       * @param  modifications  The set of modifications to apply to the entry.  It
1723       *                        must not be {@code null} or empty.
1724       *
1725       * @return  An updated version of the entry with the requested modifications
1726       *          applied.
1727       *
1728       * @throws  LDAPException  If a problem occurs while attempting to apply the
1729       *                         modifications.
1730       */
1731      public static Entry applyModifications(final Entry entry,
1732                                             final boolean lenient,
1733                                             final List<Modification> modifications)
1734             throws LDAPException
1735      {
1736        ensureNotNull(entry, modifications);
1737        ensureFalse(modifications.isEmpty());
1738    
1739        final Entry e = entry.duplicate();
1740        final ArrayList<String> errors =
1741             new ArrayList<String>(modifications.size());
1742    
1743        for (final Modification m : modifications)
1744        {
1745          final String   name   = m.getAttributeName();
1746          final byte[][] values = m.getValueByteArrays();
1747          switch (m.getModificationType().intValue())
1748          {
1749            case ModificationType.ADD_INT_VALUE:
1750              if (lenient)
1751              {
1752                e.addAttribute(m.getAttribute());
1753              }
1754              else
1755              {
1756                if (values.length == 0)
1757                {
1758                  errors.add(ERR_ENTRY_APPLY_MODS_ADD_NO_VALUES.get(name));
1759                }
1760    
1761                for (int i=0; i < values.length; i++)
1762                {
1763                  if (! e.addAttribute(name, values[i]))
1764                  {
1765                    errors.add(ERR_ENTRY_APPLY_MODS_ADD_EXISTING.get(
1766                         m.getValues()[i], name));
1767                  }
1768                }
1769              }
1770              break;
1771    
1772            case ModificationType.DELETE_INT_VALUE:
1773              if (values.length == 0)
1774              {
1775                final boolean removed = e.removeAttribute(name);
1776                if (! (lenient || removed))
1777                {
1778                  errors.add(ERR_ENTRY_APPLY_MODS_DELETE_NONEXISTENT_ATTR.get(
1779                       name));
1780                }
1781              }
1782              else
1783              {
1784                for (int i=0; i < values.length; i++)
1785                {
1786                  final boolean removed = e.removeAttributeValue(name, values[i]);
1787                  if (! (lenient || removed))
1788                  {
1789                    errors.add(ERR_ENTRY_APPLY_MODS_DELETE_NONEXISTENT_VALUE.get(
1790                         m.getValues()[i], name));
1791                  }
1792                }
1793              }
1794              break;
1795    
1796            case ModificationType.REPLACE_INT_VALUE:
1797              if (values.length == 0)
1798              {
1799                e.removeAttribute(name);
1800              }
1801              else
1802              {
1803                e.setAttribute(m.getAttribute());
1804              }
1805              break;
1806    
1807            case ModificationType.INCREMENT_INT_VALUE:
1808              final Attribute a = e.getAttribute(name);
1809              if ((a == null) || (! a.hasValue()))
1810              {
1811                errors.add(ERR_ENTRY_APPLY_MODS_INCREMENT_NO_SUCH_ATTR.get(name));
1812                continue;
1813              }
1814    
1815              if (a.size() > 1)
1816              {
1817                errors.add(ERR_ENTRY_APPLY_MODS_INCREMENT_NOT_SINGLE_VALUED.get(
1818                     name));
1819                continue;
1820              }
1821    
1822              final BigInteger currentValue;
1823              try
1824              {
1825                currentValue = new BigInteger(a.getValue());
1826              }
1827              catch (NumberFormatException nfe)
1828              {
1829                debugException(nfe);
1830                errors.add(
1831                     ERR_ENTRY_APPLY_MODS_INCREMENT_ENTRY_VALUE_NOT_INTEGER.get(
1832                          name, a.getValue()));
1833                continue;
1834              }
1835    
1836              if (values.length == 0)
1837              {
1838                errors.add(ERR_ENTRY_APPLY_MODS_INCREMENT_NO_MOD_VALUES.get(name));
1839                continue;
1840              }
1841              else if (values.length > 1)
1842              {
1843                errors.add(ERR_ENTRY_APPLY_MODS_INCREMENT_MULTIPLE_MOD_VALUES.get(
1844                     name));
1845                continue;
1846              }
1847    
1848              final BigInteger incrementValue;
1849              final String incrementValueStr = m.getValues()[0];
1850              try
1851              {
1852                incrementValue = new BigInteger(incrementValueStr);
1853              }
1854              catch (NumberFormatException nfe)
1855              {
1856                debugException(nfe);
1857                errors.add(ERR_ENTRY_APPLY_MODS_INCREMENT_MOD_VALUE_NOT_INTEGER.get(
1858                     name, incrementValueStr));
1859                continue;
1860              }
1861    
1862              final BigInteger newValue = currentValue.add(incrementValue);
1863              e.setAttribute(name, newValue.toString());
1864              break;
1865    
1866            default:
1867              errors.add(ERR_ENTRY_APPLY_MODS_UNKNOWN_TYPE.get(
1868                   String.valueOf(m.getModificationType())));
1869              break;
1870          }
1871        }
1872    
1873        if (errors.isEmpty())
1874        {
1875          return e;
1876        }
1877    
1878        throw new LDAPException(ResultCode.CONSTRAINT_VIOLATION,
1879             ERR_ENTRY_APPLY_MODS_FAILURE.get(e.getDN(),
1880                  concatenateStrings(errors)));
1881      }
