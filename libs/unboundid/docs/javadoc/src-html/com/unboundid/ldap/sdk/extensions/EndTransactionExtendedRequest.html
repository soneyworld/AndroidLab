045    /**
046     * This class provides an implementation of the end transaction extended
047     * request as defined in RFC 5805.  It may be used to either commit or abort a
048     * transaction that was created using the start transaction request.  See the
049     * documentation for the {@link StartTransactionExtendedRequest} class for an
050     * example of processing an LDAP transaction.
051     */
052    @NotMutable()
053    @ThreadSafety(level=ThreadSafetyLevel.NOT_THREADSAFE)
054    public final class EndTransactionExtendedRequest
055           extends ExtendedRequest
056    {
057      /**
058       * The OID for the end transaction extended request.
059       */
060      public static final String END_TRANSACTION_REQUEST_OID = "1.3.6.1.1.21.3";
061    
062    
063    
064      /**
065       * The serial version UID for this serializable class.
066       */
067      private static final long serialVersionUID = -7135468264026410702L;
068    
069    
070    
071      // The transaction ID for the associated transaction.
072      private final ASN1OctetString transactionID;
073    
074      // Indicates whether to commit or abort the associated transaction.
075      private final boolean commit;
076    
077    
078    
079      /**
080       * Creates a new end transaction extended request with the provided
081       * information.
082       *
083       * @param  transactionID  The transaction ID for the transaction to commit or
084       *                        abort.  It must not be {@code null}.
085       * @param  commit         {@code true} if the transaction should be committed,
086       *                        or {@code false} if the transaction should be
087       *                        aborted.
088       * @param  controls       The set of controls to include in the request.
089       */
090      public EndTransactionExtendedRequest(final ASN1OctetString transactionID,
091                                           final boolean commit,
092                                           final Control... controls)
093      {
094        super(END_TRANSACTION_REQUEST_OID, encodeValue(transactionID, commit),
095              controls);
096    
097        this.transactionID = transactionID;
098        this.commit        = commit;
099      }
100    
101    
102    
103      /**
104       * Creates a new end transaction extended request from the provided generic
105       * extended request.
106       *
107       * @param  extendedRequest  The generic extended request to use to create this
108       *                          end transaction extended request.
109       *
110       * @throws  LDAPException  If a problem occurs while decoding the request.
111       */
112      public EndTransactionExtendedRequest(final ExtendedRequest extendedRequest)
113             throws LDAPException
114      {
115        super(extendedRequest);
116    
117        final ASN1OctetString value = extendedRequest.getValue();
118        if (value == null)
119        {
120          throw new LDAPException(ResultCode.DECODING_ERROR,
121               ERR_END_TXN_REQUEST_NO_VALUE.get());
122        }
123    
124        try
125        {
126          final ASN1Element valueElement = ASN1Element.decode(value.getValue());
127          final ASN1Element[] elements =
128               ASN1Sequence.decodeAsSequence(valueElement).elements();
129          if (elements.length == 1)
130          {
131            commit        = true;
132            transactionID = ASN1OctetString.decodeAsOctetString(elements[0]);
133          }
134          else
135          {
136            commit        = ASN1Boolean.decodeAsBoolean(elements[0]).booleanValue();
137            transactionID = ASN1OctetString.decodeAsOctetString(elements[1]);
138          }
139        }
140        catch (Exception e)
141        {
142          debugException(e);
143          throw new LDAPException(ResultCode.DECODING_ERROR,
144               ERR_END_TXN_REQUEST_CANNOT_DECODE.get(e), e);
145        }
146      }
147    
148    
149    
150      /**
151       * Generates the value to include in this extended request.
152       *
153       * @param  transactionID  The transaction ID for the transaction to commit or
154       *                        abort.  It must not be {@code null}.
155       * @param  commit         {@code true} if the transaction should be committed,
156       *                        or {@code false} if the transaction should be
157       *                        aborted.
158       *
159       * @return  The ASN.1 octet string containing the encoded request value.
160       */
161      private static ASN1OctetString
162           encodeValue(final ASN1OctetString transactionID,
163                       final boolean commit)
164      {
165        ensureNotNull(transactionID);
166    
167        final ASN1Element[] valueElements;
168        if (commit)
169        {
170          valueElements = new ASN1Element[]
171          {
172            transactionID
173          };
174        }
175        else
176        {
177          valueElements = new ASN1Element[]
178          {
179            new ASN1Boolean(commit),
180            transactionID
181          };
182        }
183    
184        return new ASN1OctetString(new ASN1Sequence(valueElements).encode());
185      }
186    
187    
188    
189      /**
190       * Retrieves the transaction ID for the transaction to commit or abort.
191       *
192       * @return  The transaction ID for the transaction to commit or abort.
193       */
194      public ASN1OctetString getTransactionID()
195      {
196        return transactionID;
197      }
198    
199    
200    
201      /**
202       * Indicates whether the transaction should be committed or aborted.
203       *
204       * @return  {@code true} if the transaction should be committed, or
205       *          {@code false} if it should be aborted.
206       */
207      public boolean commit()
208      {
209        return commit;
210      }
211    
212    
213    
214      /**
215       * {@inheritDoc}
216       */
217      @Override()
218      public EndTransactionExtendedResult process(final LDAPConnection connection,
219                                                  final int depth)
220             throws LDAPException
221      {
222        final ExtendedResult extendedResponse = super.process(connection, depth);
223        return new EndTransactionExtendedResult(extendedResponse);
224      }
225    
226    
227    
228      /**
229       * {@inheritDoc}
230       */
231      @Override()
232      public EndTransactionExtendedRequest duplicate()
233      {
234        return duplicate(getControls());
235      }
236    
237    
238    
239      /**
240       * {@inheritDoc}
241       */
242      @Override()
243      public EndTransactionExtendedRequest duplicate(final Control[] controls)
244      {
245        final EndTransactionExtendedRequest r =
246             new EndTransactionExtendedRequest(transactionID, commit, controls);
247        r.setResponseTimeoutMillis(getResponseTimeoutMillis(null));
248        return r;
249      }
250    
251    
252    
253      /**
254       * {@inheritDoc}
255       */
256      @Override()
257      public String getExtendedRequestName()
258      {
259        return INFO_EXTENDED_REQUEST_NAME_END_TXN.get();
260      }
261    
262    
263    
264      /**
265       * {@inheritDoc}
266       */
267      @Override()
268      public void toString(final StringBuilder buffer)
269      {
270        buffer.append("EndTransactionExtendedRequest(transactionID='");
271        buffer.append(transactionID.stringValue());
272        buffer.append("', commit=");
273        buffer.append(commit);
274    
275        final Control[] controls = getControls();
276        if (controls.length > 0)
277        {
278          buffer.append("controls={");
279          for (int i=0; i < controls.length; i++)
280          {
281            if (i > 0)
282            {
283              buffer.append(", ");
284            }
285    
286            buffer.append(controls[i]);
287          }
288          buffer.append('}');
289        }
290    
291        buffer.append(')');
292      }
