113      /**
114       * Creates a new attribute type from the provided string representation.
115       *
116       * @param  s  The string representation of the attribute type to create, using
117       *            the syntax described in RFC 4512 section 4.1.2.  It must not be
118       *            {@code null}.
119       *
120       * @throws  LDAPException  If the provided string cannot be decoded as an
121       *                         attribute type definition.
122       */
123      public AttributeTypeDefinition(final String s)
124             throws LDAPException
125      {
126        ensureNotNull(s);
127    
128        attributeTypeString = s.trim();
129    
130        // The first character must be an opening parenthesis.
131        final int length = attributeTypeString.length();
132        if (length == 0)
133        {
134          throw new LDAPException(ResultCode.DECODING_ERROR,
135                                  ERR_ATTRTYPE_DECODE_EMPTY.get());
136        }
137        else if (attributeTypeString.charAt(0) != '(')
138        {
139          throw new LDAPException(ResultCode.DECODING_ERROR,
140                                  ERR_ATTRTYPE_DECODE_NO_OPENING_PAREN.get(
141                                       attributeTypeString));
142        }
143    
144    
145        // Skip over any spaces until we reach the start of the OID, then read the
146        // OID until we find the next space.
147        int pos = skipSpaces(attributeTypeString, 1, length);
148    
149        StringBuilder buffer = new StringBuilder();
150        pos = readOID(attributeTypeString, pos, length, buffer);
151        oid = buffer.toString();
152    
153    
154        // Technically, attribute type elements are supposed to appear in a specific
155        // order, but we'll be lenient and allow remaining elements to come in any
156        // order.
157        final ArrayList<String> nameList = new ArrayList<String>(1);
158        AttributeUsage       attrUsage   = null;
159        Boolean              collective  = null;
160        Boolean              noUserMod   = null;
161        Boolean              obsolete    = null;
162        Boolean              singleValue = null;
163        final Map<String,String[]> exts  = new LinkedHashMap<String,String[]>();
164        String               descr       = null;
165        String               eqRule      = null;
166        String               ordRule     = null;
167        String               subRule     = null;
168        String               supType     = null;
169        String               synOID      = null;
170    
171        while (true)
172        {
173          // Skip over any spaces until we find the next element.
174          pos = skipSpaces(attributeTypeString, pos, length);
175    
176          // Read until we find the next space or the end of the string.  Use that
177          // token to figure out what to do next.
178          final int tokenStartPos = pos;
179          while ((pos < length) && (attributeTypeString.charAt(pos) != ' '))
180          {
181            pos++;
182          }
183    
184          final String token = attributeTypeString.substring(tokenStartPos, pos);
185          final String lowerToken = toLowerCase(token);
186          if (lowerToken.equals(")"))
187          {
188            // This indicates that we're at the end of the value.  There should not
189            // be any more closing characters.
190            if (pos < length)
191            {
192              throw new LDAPException(ResultCode.DECODING_ERROR,
193                                      ERR_ATTRTYPE_DECODE_CLOSE_NOT_AT_END.get(
194                                           attributeTypeString));
195            }
196            break;
197          }
198          else if (lowerToken.equals("name"))
199          {
200            if (nameList.isEmpty())
201            {
202              pos = skipSpaces(attributeTypeString, pos, length);
203              pos = readQDStrings(attributeTypeString, pos, length, nameList);
204            }
205            else
206            {
207              throw new LDAPException(ResultCode.DECODING_ERROR,
208                                      ERR_ATTRTYPE_DECODE_MULTIPLE_ELEMENTS.get(
209                                           attributeTypeString, "NAME"));
210            }
211          }
212          else if (lowerToken.equals("desc"))
213          {
214            if (descr == null)
215            {
216              pos = skipSpaces(attributeTypeString, pos, length);
217    
218              buffer = new StringBuilder();
219              pos = readQDString(attributeTypeString, pos, length, buffer);
220              descr = buffer.toString();
221            }
222            else
223            {
224              throw new LDAPException(ResultCode.DECODING_ERROR,
225                                      ERR_ATTRTYPE_DECODE_MULTIPLE_ELEMENTS.get(
226                                           attributeTypeString, "DESC"));
227            }
228          }
229          else if (lowerToken.equals("obsolete"))
230          {
231            if (obsolete == null)
232            {
233              obsolete = true;
234            }
235            else
236            {
237              throw new LDAPException(ResultCode.DECODING_ERROR,
238                                      ERR_ATTRTYPE_DECODE_MULTIPLE_ELEMENTS.get(
239                                           attributeTypeString, "OBSOLETE"));
240            }
241          }
242          else if (lowerToken.equals("sup"))
243          {
244            if (supType == null)
245            {
246              pos = skipSpaces(attributeTypeString, pos, length);
247    
248              buffer = new StringBuilder();
249              pos = readOID(attributeTypeString, pos, length, buffer);
250              supType = buffer.toString();
251            }
252            else
253            {
254              throw new LDAPException(ResultCode.DECODING_ERROR,
255                                      ERR_ATTRTYPE_DECODE_MULTIPLE_ELEMENTS.get(
256                                           attributeTypeString, "SUP"));
257            }
258          }
259          else if (lowerToken.equals("equality"))
260          {
261            if (eqRule == null)
262            {
263              pos = skipSpaces(attributeTypeString, pos, length);
264    
265              buffer = new StringBuilder();
266              pos = readOID(attributeTypeString, pos, length, buffer);
267              eqRule = buffer.toString();
268            }
269            else
270            {
271              throw new LDAPException(ResultCode.DECODING_ERROR,
272                                      ERR_ATTRTYPE_DECODE_MULTIPLE_ELEMENTS.get(
273                                           attributeTypeString, "EQUALITY"));
274            }
275          }
276          else if (lowerToken.equals("ordering"))
277          {
278            if (ordRule == null)
279            {
280              pos = skipSpaces(attributeTypeString, pos, length);
281    
282              buffer = new StringBuilder();
283              pos = readOID(attributeTypeString, pos, length, buffer);
284              ordRule = buffer.toString();
285            }
286            else
287            {
288              throw new LDAPException(ResultCode.DECODING_ERROR,
289                                      ERR_ATTRTYPE_DECODE_MULTIPLE_ELEMENTS.get(
290                                           attributeTypeString, "ORDERING"));
291            }
292          }
293          else if (lowerToken.equals("substr"))
294          {
295            if (subRule == null)
296            {
297              pos = skipSpaces(attributeTypeString, pos, length);
298    
299              buffer = new StringBuilder();
300              pos = readOID(attributeTypeString, pos, length, buffer);
301              subRule = buffer.toString();
302            }
303            else
304            {
305              throw new LDAPException(ResultCode.DECODING_ERROR,
306                                      ERR_ATTRTYPE_DECODE_MULTIPLE_ELEMENTS.get(
307                                           attributeTypeString, "SUBSTR"));
308            }
309          }
310          else if (lowerToken.equals("syntax"))
311          {
312            if (synOID == null)
313            {
314              pos = skipSpaces(attributeTypeString, pos, length);
315    
316              buffer = new StringBuilder();
317              pos = readOID(attributeTypeString, pos, length, buffer);
318              synOID = buffer.toString();
319            }
320            else
321            {
322              throw new LDAPException(ResultCode.DECODING_ERROR,
323                                      ERR_ATTRTYPE_DECODE_MULTIPLE_ELEMENTS.get(
324                                           attributeTypeString, "SYNTAX"));
325            }
326          }
327          else if (lowerToken.equals("single-value"))
328          {
329            if (singleValue == null)
330            {
331              singleValue = true;
332            }
333            else
334            {
335              throw new LDAPException(ResultCode.DECODING_ERROR,
336                                      ERR_ATTRTYPE_DECODE_MULTIPLE_ELEMENTS.get(
337                                           attributeTypeString, "SINGLE-VALUE"));
338            }
339          }
340          else if (lowerToken.equals("collective"))
341          {
342            if (collective == null)
343            {
344              collective = true;
345            }
346            else
347            {
348              throw new LDAPException(ResultCode.DECODING_ERROR,
349                                      ERR_ATTRTYPE_DECODE_MULTIPLE_ELEMENTS.get(
350                                           attributeTypeString, "COLLECTIVE"));
351            }
352          }
353          else if (lowerToken.equals("no-user-modification"))
354          {
355            if (noUserMod == null)
356            {
357              noUserMod = true;
358            }
359            else
360            {
361              throw new LDAPException(ResultCode.DECODING_ERROR,
362                                      ERR_ATTRTYPE_DECODE_MULTIPLE_ELEMENTS.get(
363                                           attributeTypeString,
364                                           "NO-USER-MODIFICATION"));
365            }
366          }
367          else if (lowerToken.equals("usage"))
368          {
369            if (attrUsage == null)
370            {
371              pos = skipSpaces(attributeTypeString, pos, length);
372    
373              buffer = new StringBuilder();
374              pos = readOID(attributeTypeString, pos, length, buffer);
375    
376              final String usageStr = toLowerCase(buffer.toString());
377              if (usageStr.equals("userapplications"))
378              {
379                attrUsage = AttributeUsage.USER_APPLICATIONS;
380              }
381              else if (usageStr.equals("directoryoperation"))
382              {
383                attrUsage = AttributeUsage.DIRECTORY_OPERATION;
384              }
385              else if (usageStr.equals("distributedoperation"))
386              {
387                attrUsage = AttributeUsage.DISTRIBUTED_OPERATION;
388              }
389              else if (usageStr.equals("dsaoperation"))
390              {
391                attrUsage = AttributeUsage.DSA_OPERATION;
392              }
393              else
394              {
395                throw new LDAPException(ResultCode.DECODING_ERROR,
396                                        ERR_ATTRTYPE_DECODE_INVALID_USAGE.get(
397                                             attributeTypeString, usageStr));
398              }
399            }
400            else
401            {
402              throw new LDAPException(ResultCode.DECODING_ERROR,
403                                      ERR_ATTRTYPE_DECODE_MULTIPLE_ELEMENTS.get(
404                                           attributeTypeString, "USAGE"));
405            }
406          }
407          else if (lowerToken.startsWith("x-"))
408          {
409            pos = skipSpaces(attributeTypeString, pos, length);
410    
411            final ArrayList<String> valueList = new ArrayList<String>();
412            pos = readQDStrings(attributeTypeString, pos, length, valueList);
413    
414            final String[] values = new String[valueList.size()];
415            valueList.toArray(values);
416    
417            if (exts.containsKey(token))
418            {
419              throw new LDAPException(ResultCode.DECODING_ERROR,
420                                      ERR_ATTRTYPE_DECODE_DUP_EXT.get(
421                                           attributeTypeString, token));
422            }
423    
424            exts.put(token, values);
425          }
426          else
427          {
428            throw new LDAPException(ResultCode.DECODING_ERROR,
429                                    ERR_ATTRTYPE_DECODE_UNEXPECTED_TOKEN.get(
430                                         attributeTypeString, token));
431          }
432        }
433    
434        description           = descr;
435        equalityMatchingRule  = eqRule;
436        orderingMatchingRule  = ordRule;
437        substringMatchingRule = subRule;
438        superiorType          = supType;
439        syntaxOID             = synOID;
440    
441        names = new String[nameList.size()];
442        nameList.toArray(names);
443    
444        isObsolete           = (obsolete != null);
445        isSingleValued       = (singleValue != null);
446        isCollective         = (collective != null);
447        isNoUserModification = (noUserMod != null);
448    
449        if (attrUsage == null)
450        {
451          usage = AttributeUsage.USER_APPLICATIONS;
452        }
453        else
454        {
455          usage = attrUsage;
456        }
457    
458        extensions = Collections.unmodifiableMap(exts);
459      }

463      /**
464       * Creates a new attribute type with the provided information.
465       *
466       * @param  oid                    The OID for this attribute type.  It must
467       *                                not be {@code null}.
468       * @param  names                  The set of names for this attribute type.
469       *                                It may be {@code null} or empty if the
470       *                                attribute type should only be referenced by
471       *                                OID.
472       * @param  description            The description for this attribute type.  It
473       *                                may be {@code null} if there is no
474       *                                description.
475       * @param  isObsolete             Indicates whether this attribute type is
476       *                                declared obsolete.
477       * @param  superiorType           The name or OID of the superior attribute
478       *                                type.  It may be {@code null} if there is no
479       *                                superior type.
480       * @param  equalityMatchingRule   The name or OID of the equality matching
481       *                                rule for this attribute type.  It may be
482       *                                {@code null} if a default rule is to be
483       *                                inherited.
484       * @param  orderingMatchingRule   The name or OID of the ordering matching
485       *                                rule for this attribute type.  It may be
486       *                                {@code null} if a default rule is to be
487       *                                inherited.
488       * @param  substringMatchingRule  The name or OID of the substring matching
489       *                                rule for this attribute type.  It may be
490       *                                {@code null} if a default rule is to be
491       *                                inherited.
492       * @param  syntaxOID              The syntax OID for this attribute type.  It
493       *                                may be {@code null} if a default syntax is
494       *                                to be inherited.
495       * @param  isSingleValued         Indicates whether attributes of this type
496       *                                are only allowed to have a single value.
497       * @param  isCollective           Indicates whether this attribute type should
498       *                                be considered collective.
499       * @param  isNoUserModification   Indicates whether clients should be allowed
500       *                                to modify attributes of this type.
501       * @param  usage                  The attribute usage for this attribute type.
502       *                                It may be {@code null} if the default usage
503       *                                of userApplications is to be used.
504       * @param  extensions             The set of extensions for this attribute
505       *                                type.  It may be {@code null} or empty if
506       *                                there should not be any extensions.
507       */
508      public AttributeTypeDefinition(final String oid, final String[] names,
509                                     final String description,
510                                     final boolean isObsolete,
511                                     final String superiorType,
512                                     final String equalityMatchingRule,
513                                     final String orderingMatchingRule,
514                                     final String substringMatchingRule,
515                                     final String syntaxOID,
516                                     final boolean isSingleValued,
517                                     final boolean isCollective,
518                                     final boolean isNoUserModification,
519                                     final AttributeUsage usage,
520                                     final Map<String,String[]> extensions)
521      {
522        ensureNotNull(oid);
523    
524        this.oid                   = oid;
525        this.description           = description;
526        this.isObsolete            = isObsolete;
527        this.superiorType          = superiorType;
528        this.equalityMatchingRule  = equalityMatchingRule;
529        this.orderingMatchingRule  = orderingMatchingRule;
530        this.substringMatchingRule = substringMatchingRule;
531        this.syntaxOID             = syntaxOID;
532        this.isSingleValued        = isSingleValued;
533        this.isCollective          = isCollective;
534        this.isNoUserModification  = isNoUserModification;
535    
536        if (names == null)
537        {
538          this.names = NO_STRINGS;
539        }
540        else
541        {
542          this.names = names;
543        }
544    
545        if (usage == null)
546        {
547          this.usage = AttributeUsage.USER_APPLICATIONS;
548        }
549        else
550        {
551          this.usage = usage;
552        }
553    
554        if (extensions == null)
555        {
556          this.extensions = Collections.emptyMap();
557        }
558        else
559        {
560          this.extensions = Collections.unmodifiableMap(extensions);
561        }
562    
563        final StringBuilder buffer = new StringBuilder();
564        createDefinitionString(buffer);
565        attributeTypeString = buffer.toString();
566      }

570      /**
571       * Constructs a string representation of this attribute type definition in the
572       * provided buffer.
573       *
574       * @param  buffer  The buffer in which to construct a string representation of
575       *                 this attribute type definition.
576       */
577      private void createDefinitionString(final StringBuilder buffer)
578      {
579        buffer.append("( ");
580        buffer.append(oid);
581    
582        if (names.length == 1)
583        {
584          buffer.append(" NAME '");
585          buffer.append(names[0]);
586          buffer.append('\'');
587        }
588        else if (names.length > 1)
589        {
590          buffer.append(" NAME (");
591          for (final String name : names)
592          {
593            buffer.append(" '");
594            buffer.append(name);
595            buffer.append('\'');
596          }
597          buffer.append(" )");
598        }
599    
600        if (description != null)
601        {
602          buffer.append(" DESC '");
603          encodeValue(description, buffer);
604          buffer.append('\'');
605        }
606    
607        if (isObsolete)
608        {
609          buffer.append(" OBSOLETE");
610        }
611    
612        if (superiorType != null)
613        {
614          buffer.append(" SUP ");
615          buffer.append(superiorType);
616        }
617    
618        if (equalityMatchingRule != null)
619        {
620          buffer.append(" EQUALITY ");
621          buffer.append(equalityMatchingRule);
622        }
623    
624        if (orderingMatchingRule != null)
625        {
626          buffer.append(" ORDERING ");
627          buffer.append(orderingMatchingRule);
628        }
629    
630        if (substringMatchingRule != null)
631        {
632          buffer.append(" SUBSTR ");
633          buffer.append(substringMatchingRule);
634        }
635    
636        if (syntaxOID != null)
637        {
638          buffer.append(" SYNTAX ");
639          buffer.append(syntaxOID);
640        }
641    
642        if (isSingleValued)
643        {
644          buffer.append(" SINGLE-VALUE");
645        }
646    
647        if (isCollective)
648        {
649          buffer.append(" COLLECTIVE");
650        }
651    
652        if (isNoUserModification)
653        {
654          buffer.append(" NO-USER-MODIFICATION");
655        }
656    
657        buffer.append(" USAGE ");
658        buffer.append(usage.getName());
659    
660        for (final Map.Entry<String,String[]> e : extensions.entrySet())
661        {
662          final String   name   = e.getKey();
663          final String[] values = e.getValue();
664          if (values.length == 1)
665          {
666            buffer.append(' ');
667            buffer.append(name);
668            buffer.append(" '");
669            encodeValue(values[0], buffer);
670            buffer.append('\'');
671          }
672          else
673          {
674            buffer.append(' ');
675            buffer.append(name);
676            buffer.append(" (");
677            for (final String value : values)
678            {
679              buffer.append(" '");
680              encodeValue(value, buffer);
681              buffer.append('\'');
682            }
683            buffer.append(" )");
684          }
685        }
686    
687        buffer.append(" )");
688      }

717      /**
718       * Retrieves the primary name that can be used to reference this attribute
719       * type.  If one or more names are defined, then the first name will be used.
720       * Otherwise, the OID will be returned.
721       *
722       * @return  The primary name that can be used to reference this attribute
723       *          type.
724       */
725      public String getNameOrOID()
726      {
727        if (names.length == 0)
728        {
729          return oid;
730        }
731        else
732        {
733          return names[0];
734        }
735      }

739      /**
740       * Indicates whether the provided string matches the OID or any of the names
741       * for this attribute type.
742       *
743       * @param  s  The string for which to make the determination.  It must not be
744       *            {@code null}.
745       *
746       * @return  {@code true} if the provided string matches the OID or any of the
747       *          names for this attribute type, or {@code false} if not.
748       */
749      public boolean hasNameOrOID(final String s)
750      {
751        for (final String name : names)
752        {
753          if (s.equalsIgnoreCase(name))
754          {
755            return true;
756          }
757        }
758    
759        return s.equalsIgnoreCase(oid);
760      }
