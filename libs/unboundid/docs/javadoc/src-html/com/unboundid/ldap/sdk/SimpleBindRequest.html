047     * bind operation, which authenticates using a bind DN and password.

174                               final Control... controls)
175      {
176        super(controls);
177    
178        if (bindDN == null)

184          this.bindDN = new ASN1OctetString(bindDN);
185        }
186    
187        if (password == null)
188        {
189          this.password = NO_PASSWORD;
190        }
191        else
192        {
193          this.password = new ASN1OctetString(CRED_TYPE_SIMPLE, password);
194        }
195      }

365      /**
366       * {@inheritDoc}
367       */
368      @Override()
369      protected BindResult process(final LDAPConnection connection, final int depth)
370                throws LDAPException
371      {
372        if (connection.synchronousMode())
373        {
374          return processSync(connection);
375        }
376    
377        // See if a bind DN was provided without a password.  If that is the case
378        // and this should not be allowed, then throw an exception.
379        if ((bindDN.getValue().length > 0) && (password.getValue().length == 0) &&
380            connection.getConnectionOptions().bindWithDNRequiresPassword())
381        {
382          final LDAPException le = new LDAPException(ResultCode.PARAM_ERROR,
383               ERR_SIMPLE_BIND_DN_WITHOUT_PASSWORD.get());
384          debugCodingError(le);
385          throw le;
386        }
387    
388    
389        // Create the LDAP message.
390        messageID = connection.nextMessageID();
391        final LDAPMessage message = new LDAPMessage(messageID, this, getControls());
392    
393    
394        // Register with the connection reader to be notified of responses for the
395        // request that we've created.
396        connection.registerResponseAcceptor(messageID, this);
397    
398    
399        try
400        {
401          // Send the request to the server.
402          debugLDAPRequest(this);
403          final long requestTime = System.nanoTime();
404          connection.getConnectionStatistics().incrementNumBindRequests();
405          connection.sendMessage(message);
406    
407          // Wait for and process the response.
408          final LDAPResponse response;
409          try
410          {
411            final long responseTimeout = getResponseTimeoutMillis(connection);
412            if (responseTimeout > 0)
413            {
414              response = responseQueue.poll(responseTimeout, TimeUnit.MILLISECONDS);
415            }
416            else
417            {
418              response = responseQueue.take();
419            }
420          }
421          catch (InterruptedException ie)
422          {
423            debugException(ie);
424            throw new LDAPException(ResultCode.LOCAL_ERROR,
425                 ERR_BIND_INTERRUPTED.get(connection.getHostPort()), ie);
426          }
427    
428          return handleResponse(connection, response, requestTime);
429        }
430        finally
431        {
432          connection.deregisterResponseAcceptor(messageID);
433        }
434      }

438      /**
439       * Processes this bind operation in synchronous mode, in which the same
440       * thread will send the request and read the response.
441       *
442       * @param  connection  The connection to use to communicate with the directory
443       *                     server.
444       *
445       * @return  An LDAP result object that provides information about the result
446       *          of the bind processing.
447       *
448       * @throws  LDAPException  If a problem occurs while sending the request or
449       *                         reading the response.
450       */
451      private BindResult processSync(final LDAPConnection connection)
452              throws LDAPException
453      {
454        // Create the LDAP message.
455        messageID = connection.nextMessageID();
456        final LDAPMessage message =
457             new LDAPMessage(messageID,  this, getControls());
458    
459    
460        // Set the appropriate timeout on the socket.
461        try
462        {
463          connection.getConnectionInternals().getSocket().setSoTimeout(
464               (int) getResponseTimeoutMillis(connection));
465        }
466        catch (Exception e)
467        {
468          debugException(e);
469        }
470    
471    
472        // Send the request to the server.
473        final long requestTime = System.nanoTime();
474        debugLDAPRequest(this);
475        connection.getConnectionStatistics().incrementNumBindRequests();
476        connection.sendMessage(message);
477    
478        final LDAPResponse response = connection.readResponse(messageID);
479        return handleResponse(connection, response, requestTime);
480      }

484      /**
485       * Performs the necessary processing for handling a response.
486       *
487       * @param  connection   The connection used to read the response.
488       * @param  response     The response to be processed.
489       * @param  requestTime  The time the request was sent to the server.
490       *
491       * @return  The bind result.
492       *
493       * @throws  LDAPException  If a problem occurs.
494       */
495      private BindResult handleResponse(final LDAPConnection connection,
496                                        final LDAPResponse response,
497                                        final long requestTime)
498              throws LDAPException
499      {
500        if (response == null)
501        {
502          final long waitTime = nanosToMillis(System.nanoTime() - requestTime);
503          throw new LDAPException(ResultCode.TIMEOUT,
504               ERR_BIND_CLIENT_TIMEOUT.get(waitTime, connection.getHostPort()));
505        }
506    
507        connection.getConnectionStatistics().incrementNumBindResponses(
508             System.nanoTime() - requestTime);
509        if (response instanceof ConnectionClosedResponse)
510        {
511          // The connection was closed while waiting for the response.
512          final ConnectionClosedResponse ccr = (ConnectionClosedResponse) response;
513          final String message = ccr.getMessage();
514          if (message == null)
515          {
516            throw new LDAPException(ccr.getResultCode(),
517                 ERR_CONN_CLOSED_WAITING_FOR_BIND_RESPONSE.get(
518                      connection.getHostPort(), toString()));
519          }
520          else
521          {
522            throw new LDAPException(ccr.getResultCode(),
523                 ERR_CONN_CLOSED_WAITING_FOR_BIND_RESPONSE_WITH_MESSAGE.get(
524                      connection.getHostPort(), toString(), message));
525          }
526        }
527    
528        return (BindResult) response;
529      }

612      /**
613       * {@inheritDoc}
614       */
615      @Override()
616      public void toString(final StringBuilder buffer)
617      {
618        buffer.append("SimpleBindRequest(dn='");
619        buffer.append(bindDN);
620        buffer.append('\'');
621    
622        final Control[] controls = getControls();
623        if (controls.length > 0)
624        {
625          buffer.append(", controls={");
626          for (int i=0; i < controls.length; i++)
627          {
628            if (i > 0)
629            {
630              buffer.append(", ");
631            }
632    
633            buffer.append(controls[i]);
634          }
635          buffer.append('}');
636        }
637    
638        buffer.append(')');
639      }
