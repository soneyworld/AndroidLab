056    /**
057     * This class provides a simple tool that can be used to search an LDAP
058     * directory server.  Some of the APIs demonstrated by this example include:
059     * <UL>
060     *   <LI>Argument Parsing (from the {@code com.unboundid.util.args}
061     *       package)</LI>
062     *   <LI>LDAP Command-Line Tool (from the {@code com.unboundid.util}
063     *       package)</LI>
064     *   <LI>LDAP Communication (from the {@code com.unboundid.ldap.sdk}
065     *       package)</LI>
066     * </UL>
067     * <BR><BR>
068     * All of the necessary information is provided using
069     * command line arguments.  Supported arguments include those allowed by the
070     * {@link LDAPCommandLineTool} class, as well as the following additional
071     * arguments:
072     * <UL>
073     *   <LI>"-b {baseDN}" or "--baseDN {baseDN}" -- specifies the base DN to use
074     *       for the search.  This must be provided.</LI>
075     *   <LI>"-s {scope}" or "--scope {scope}" -- specifies the scope to use for the
076     *       search.  The scope value should be one of "base", "one", "sub", or
077     *       "subord".  If this isn't specified, then a scope of "sub" will be
078     *       used.</LI>
079     *   <LI>"-R" or "--followReferrals" -- indicates that the tool should follow
080     *       any referrals encountered while searching.</LI>
081     *   <LI>"-t" or "--terse" -- indicates that the tool should generate minimal
082     *       output beyond the search results.</LI>
083     *   <LI>"-i {millis}" or "--repeatIntervalMillis {millis}" -- indicates that
084     *       the search should be periodically repeated with the specified delay
085     *       (in milliseconds) between requests.</LI>
086     *   <LI>"-n {count}" or "--numSearches {count}" -- specifies the total number
087     *       of times that the search should be performed.  This may only be used in
088     *       conjunction with the "--repeatIntervalMillis" argument.  If
089     *       "--repeatIntervalMillis" is used without "--numSearches", then the
090     *       searches will continue to be repeated until the tool is
091     *       interrupted.</LI>
092     * </UL>
093     * In addition, after the above named arguments are provided, a set of one or
094     * more unnamed trailing arguments must be given.  The first argument should be
095     * the string representation of the filter to use for the search.  If there are
096     * any additional trailing arguments, then they will be interpreted as the
097     * attributes to return in matching entries.  If no attribute names are given,
098     * then the server should return all user attributes in matching entries.
099     * <BR><BR>
100     * Note that this class implements the SearchResultListener interface, which
101     * will be notified whenever a search result entry or reference is returned from
102     * the server.  Whenever an entry is received, it will simply be printed
103     * displayed in LDIF.
104     */
105    @ThreadSafety(level=ThreadSafetyLevel.NOT_THREADSAFE)
106    public final class LDAPSearch
107           extends LDAPCommandLineTool
108           implements SearchResultListener

265      /**
266       * Adds the arguments used by this program that aren't already provided by the
267       * generic {@code LDAPCommandLineTool} framework.
268       *
269       * @param  parser  The argument parser to which the arguments should be added.
270       *
271       * @throws  ArgumentException  If a problem occurs while adding the arguments.
272       */
273      @Override()
274      public void addNonLDAPArguments(final ArgumentParser parser)
275             throws ArgumentException
276      {
277        this.parser = parser;
278    
279        String description = "The base DN to use for the search.  This must be " +
280                             "provided.";
281        baseDN = new DNArgument('b', "baseDN", true, 1, "{dn}", description);
282        parser.addArgument(baseDN);
283    
284    
285        description = "The scope to use for the search.  It should be 'base', " +
286                      "'one', 'sub', or 'subord'.  If this is not provided, then " +
287                      "a default scope of 'sub' will be used.";
288        final LinkedHashSet<String> allowedScopes = new LinkedHashSet<String>(4);
289        allowedScopes.add("base");
290        allowedScopes.add("one");
291        allowedScopes.add("sub");
292        allowedScopes.add("subord");
293        scopeStr = new StringArgument('s', "scope", false, 1, "{scope}",
294                                      description, allowedScopes, "sub");
295        parser.addArgument(scopeStr);
296    
297    
298        description = "Follow any referrals encountered during processing.";
299        followReferrals = new BooleanArgument('R', "followReferrals", description);
300        parser.addArgument(followReferrals);
301    
302    
303        description = "Generate terse output with minimal additional information.";
304        terseMode = new BooleanArgument('t', "terse", description);
305        parser.addArgument(terseMode);
306    
307    
308        description = "Specifies the length of time in milliseconds to sleep " +
309                      "before repeating the same search.  If this is not " +
310                      "provided, then the search will only be performed once.";
311        repeatIntervalMillis = new IntegerArgument('i', "repeatIntervalMillis",
312                                                   false, 1, "{millis}",
313                                                   description, 0,
314                                                   Integer.MAX_VALUE);
315        parser.addArgument(repeatIntervalMillis);
316    
317    
318        description = "Specifies the number of times that the search should be " +
319                      "performed.  If this argument is present, then the " +
320                      "--repeatIntervalMillis argument must also be provided to " +
321                      "specify the length of time between searches.  If " +
322                      "--repeatIntervalMillis is used without --numSearches, " +
323                      "then the search will be repeated until the tool is " +
324                      "interrupted.";
325        numSearches = new IntegerArgument('n', "numSearches", false, 1, "{count}",
326                                          description, 1, Integer.MAX_VALUE);
327        parser.addArgument(numSearches);
328        parser.addDependentArgumentSet(numSearches, repeatIntervalMillis);
329      }

333      /**
334       * Performs the actual processing for this tool.  In this case, it gets a
335       * connection to the directory server and uses it to perform the requested
336       * search.
337       *
338       * @return  The result code for the processing that was performed.
339       */
340      @Override()
341      public ResultCode doToolProcessing()
342      {
343        // Make sure that at least one trailing argument was provided, which will be
344        // the filter.  If there were any other arguments, then they will be the
345        // attributes to return.
346        final List<String> trailingArguments = parser.getTrailingArguments();
347        if (trailingArguments.isEmpty())
348        {
349          err("No search filter was provided.");
350          err();
351          err(parser.getUsageString(79));
352          return ResultCode.PARAM_ERROR;
353        }
354    
355        final Filter filter;
356        try
357        {
358          filter = Filter.create(trailingArguments.get(0));
359        }
360        catch (LDAPException le)
361        {
362          err("Invalid search filter:  ", le.getMessage());
363          return le.getResultCode();
364        }
365    
366        final String[] attributesToReturn;
367        if (trailingArguments.size() > 1)
368        {
369          attributesToReturn = new String[trailingArguments.size() - 1];
370          for (int i=1; i < trailingArguments.size(); i++)
371          {
372            attributesToReturn[i-1] = trailingArguments.get(i);
373          }
374        }
375        else
376        {
377          attributesToReturn = new String[0];
378        }
379    
380    
381        // Convert the search scope from a string to an integer.
382        final SearchScope scope;
383        if (scopeStr.getValue().equalsIgnoreCase("base"))
384        {
385          scope = SearchScope.BASE;
386        }
387        else if (scopeStr.getValue().equalsIgnoreCase("one"))
388        {
389          scope = SearchScope.ONE;
390        }
391        else if (scopeStr.getValue().equalsIgnoreCase("subord"))
392        {
393          scope = SearchScope.SUBORDINATE_SUBTREE;
394        }
395        else
396        {
397          scope = SearchScope.SUB;
398        }
399    
400    
401    
402        // Get the connection to the directory server.
403        final LDAPConnection connection;
404        try
405        {
406          connection = getConnection();
407          if (! terseMode.isPresent())
408          {
409            out("# Connected to ", connection.getConnectedAddress(), ':',
410                 connection.getConnectedPort());
411          }
412        }
413        catch (LDAPException le)
414        {
415          err("Error connecting to the directory server:  ", le.getMessage());
416          return le.getResultCode();
417        }
418    
419    
420        // Create a search request with the appropriate information and process it
421        // in the server.  Note that in this case, we're creating a search result
422        // listener to handle the results since there could potentially be a lot of
423        // them.
424        final SearchRequest searchRequest =
425             new SearchRequest(this, baseDN.getStringValue(), scope,
426                               DereferencePolicy.NEVER, 0, 0, false, filter,
427                               attributesToReturn);
428        searchRequest.setFollowReferrals(followReferrals.isPresent());
429    
430    
431        final boolean infinite;
432        final int numIterations;
433        if (repeatIntervalMillis.isPresent())
434        {
435          repeat = true;
436    
437          if (numSearches.isPresent())
438          {
439            infinite      = false;
440            numIterations = numSearches.getValue();
441          }
442          else
443          {
444            infinite      = true;
445            numIterations = Integer.MAX_VALUE;
446          }
447        }
448        else
449        {
450          infinite      = false;
451          repeat        = false;
452          numIterations = 1;
453        }
454    
455        ResultCode resultCode = ResultCode.SUCCESS;
456        long lastSearchTime = System.currentTimeMillis();
457        final WakeableSleeper sleeper = new WakeableSleeper();
458        for (int i=0; (infinite || (i < numIterations)); i++)
459        {
460          if (repeat && (i > 0))
461          {
462            final long sleepTime =
463                 (lastSearchTime + repeatIntervalMillis.getValue()) -
464                 System.currentTimeMillis();
465            if (sleepTime > 0)
466            {
467              sleeper.sleep(sleepTime);
468            }
469            lastSearchTime = System.currentTimeMillis();
470          }
471    
472          try
473          {
474            final SearchResult searchResult = connection.search(searchRequest);
475            if ((! repeat) && (! terseMode.isPresent()))
476            {
477              out("# The search operation was processed successfully.");
478              out("# Entries returned:  ", searchResult.getEntryCount());
479              out("# References returned:  ", searchResult.getReferenceCount());
480            }
481          }
482          catch (LDAPException le)
483          {
484            err("An error occurred while processing the search:  ",
485                 le.getMessage());
486            err("Result Code:  ", le.getResultCode().intValue(), " (",
487                 le.getResultCode().getName(), ')');
488            if (le.getMatchedDN() != null)
489            {
490              err("Matched DN:  ", le.getMatchedDN());
491            }
492    
493            if (le.getReferralURLs() != null)
494            {
495              for (final String url : le.getReferralURLs())
496              {
497                err("Referral URL:  ", url);
498              }
499            }
500    
501            if (resultCode == ResultCode.SUCCESS)
502            {
503              resultCode = le.getResultCode();
504            }
505    
506            if (! le.getResultCode().isConnectionUsable())
507            {
508              break;
509            }
510          }
511        }
512    
513    
514        // Close the connection to the directory server and exit.
515        connection.close();
516        if (! terseMode.isPresent())
517        {
518          out();
519          out("# Disconnected from the server");
520        }
521        return resultCode;
522      }

526      /**
527       * Indicates that the provided search result entry was returned from the
528       * associated search operation.
529       *
530       * @param  entry  The entry that was returned from the search.
531       */
532      public void searchEntryReturned(final SearchResultEntry entry)
533      {
534        if (repeat)
535        {
536          out("# ", DATE_FORMAT.format(new Date()));
537        }
538    
539        out(entry.toLDIFString());
540      }
