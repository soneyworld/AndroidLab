240      /**
241       * Creates a new schema object by decoding the information in the provided
242       * entry.
243       *
244       * @param  schemaEntry  The schema entry to decode.
245       */
246      public Schema(final Entry schemaEntry)
247      {
248        // Decode the attribute syntaxes from the schema entry.
249        String[] defs = schemaEntry.getAttributeValues(ATTR_ATTRIBUTE_SYNTAX);
250        if (defs == null)
251        {
252          asMap = Collections.emptyMap();
253          asSet = Collections.emptySet();
254        }
255        else
256        {
257          final LinkedHashMap<String,AttributeSyntaxDefinition> m =
258               new LinkedHashMap<String,AttributeSyntaxDefinition>(defs.length);
259          final LinkedHashSet<AttributeSyntaxDefinition> s =
260               new LinkedHashSet<AttributeSyntaxDefinition>(defs.length);
261    
262          for (final String def : defs)
263          {
264            try
265            {
266              final AttributeSyntaxDefinition as =
267                   new AttributeSyntaxDefinition(def);
268              s.add(as);
269              m.put(toLowerCase(as.getOID()), as);
270            }
271            catch (final LDAPException le)
272            {
273              debugException(le);
274            }
275          }
276    
277          asMap = Collections.unmodifiableMap(m);
278          asSet = Collections.unmodifiableSet(s);
279        }
280    
281    
282        // Decode the attribute types from the schema entry.
283        defs = schemaEntry.getAttributeValues(ATTR_ATTRIBUTE_TYPE);
284        if (defs == null)
285        {
286          atMap            = Collections.emptyMap();
287          atSet            = Collections.emptySet();
288          operationalATSet = Collections.emptySet();
289          userATSet        = Collections.emptySet();
290        }
291        else
292        {
293          final LinkedHashMap<String,AttributeTypeDefinition> m =
294               new LinkedHashMap<String,AttributeTypeDefinition>(2*defs.length);
295          final LinkedHashSet<AttributeTypeDefinition> s =
296               new LinkedHashSet<AttributeTypeDefinition>(defs.length);
297          final LinkedHashSet<AttributeTypeDefinition> sUser =
298               new LinkedHashSet<AttributeTypeDefinition>(defs.length);
299          final LinkedHashSet<AttributeTypeDefinition> sOperational =
300               new LinkedHashSet<AttributeTypeDefinition>(defs.length);
301    
302          for (final String def : defs)
303          {
304            try
305            {
306              final AttributeTypeDefinition at = new AttributeTypeDefinition(def);
307              s.add(at);
308              m.put(toLowerCase(at.getOID()), at);
309              for (final String name : at.getNames())
310              {
311                m.put(toLowerCase(name), at);
312              }
313    
314              if (at.isOperational())
315              {
316                sOperational.add(at);
317              }
318              else
319              {
320                sUser.add(at);
321              }
322            }
323            catch (final LDAPException le)
324            {
325              debugException(le);
326            }
327          }
328    
329          atMap            = Collections.unmodifiableMap(m);
330          atSet            = Collections.unmodifiableSet(s);
331          operationalATSet = Collections.unmodifiableSet(sOperational);
332          userATSet        = Collections.unmodifiableSet(sUser);
333        }
334    
335    
336        // Decode the DIT content rules from the schema entry.
337        defs = schemaEntry.getAttributeValues(ATTR_DIT_CONTENT_RULE);
338        if (defs == null)
339        {
340          dcrMap = Collections.emptyMap();
341          dcrSet = Collections.emptySet();
342        }
343        else
344        {
345          final LinkedHashMap<String,DITContentRuleDefinition> m =
346               new LinkedHashMap<String,DITContentRuleDefinition>(2*defs.length);
347          final LinkedHashSet<DITContentRuleDefinition> s =
348               new LinkedHashSet<DITContentRuleDefinition>(defs.length);
349    
350          for (final String def : defs)
351          {
352            try
353            {
354              final DITContentRuleDefinition dcr =
355                   new DITContentRuleDefinition(def);
356              s.add(dcr);
357              m.put(toLowerCase(dcr.getOID()), dcr);
358              for (final String name : dcr.getNames())
359              {
360                m.put(toLowerCase(name), dcr);
361              }
362            }
363            catch (final LDAPException le)
364            {
365              debugException(le);
366            }
367          }
368    
369          dcrMap = Collections.unmodifiableMap(m);
370          dcrSet = Collections.unmodifiableSet(s);
371        }
372    
373    
374        // Decode the DIT structure rules from the schema entry.
375        defs = schemaEntry.getAttributeValues(ATTR_DIT_STRUCTURE_RULE);
376        if (defs == null)
377        {
378          dsrMapByID       = Collections.emptyMap();
379          dsrMapByName     = Collections.emptyMap();
380          dsrMapByNameForm = Collections.emptyMap();
381          dsrSet           = Collections.emptySet();
382        }
383        else
384        {
385          final LinkedHashMap<Integer,DITStructureRuleDefinition> mID =
386               new LinkedHashMap<Integer,DITStructureRuleDefinition>(defs.length);
387          final LinkedHashMap<String,DITStructureRuleDefinition> mN =
388               new LinkedHashMap<String,DITStructureRuleDefinition>(defs.length);
389          final LinkedHashMap<String,DITStructureRuleDefinition> mNF =
390               new LinkedHashMap<String,DITStructureRuleDefinition>(defs.length);
391          final LinkedHashSet<DITStructureRuleDefinition> s =
392               new LinkedHashSet<DITStructureRuleDefinition>(defs.length);
393    
394          for (final String def : defs)
395          {
396            try
397            {
398              final DITStructureRuleDefinition dsr =
399                   new DITStructureRuleDefinition(def);
400              s.add(dsr);
401              mID.put(dsr.getRuleID(), dsr);
402              mNF.put(toLowerCase(dsr.getNameFormID()), dsr);
403              for (final String name : dsr.getNames())
404              {
405                mN.put(toLowerCase(name), dsr);
406              }
407            }
408            catch (final LDAPException le)
409            {
410              debugException(le);
411            }
412          }
413    
414          dsrMapByID       = Collections.unmodifiableMap(mID);
415          dsrMapByName     = Collections.unmodifiableMap(mN);
416          dsrMapByNameForm = Collections.unmodifiableMap(mNF);
417          dsrSet           = Collections.unmodifiableSet(s);
418        }
419    
420    
421        // Decode the matching rules from the schema entry.
422        defs = schemaEntry.getAttributeValues(ATTR_MATCHING_RULE);
423        if (defs == null)
424        {
425          mrMap = Collections.emptyMap();
426          mrSet = Collections.emptySet();
427        }
428        else
429        {
430          final LinkedHashMap<String,MatchingRuleDefinition> m =
431               new LinkedHashMap<String,MatchingRuleDefinition>(2*defs.length);
432          final LinkedHashSet<MatchingRuleDefinition> s =
433               new LinkedHashSet<MatchingRuleDefinition>(defs.length);
434    
435          for (final String def : defs)
436          {
437            try
438            {
439              final MatchingRuleDefinition mr = new MatchingRuleDefinition(def);
440              s.add(mr);
441              m.put(toLowerCase(mr.getOID()), mr);
442              for (final String name : mr.getNames())
443              {
444                m.put(toLowerCase(name), mr);
445              }
446            }
447            catch (final LDAPException le)
448            {
449              debugException(le);
450            }
451          }
452    
453          mrMap = Collections.unmodifiableMap(m);
454          mrSet = Collections.unmodifiableSet(s);
455        }
456    
457    
458        // Decode the matching rule uses from the schema entry.
459        defs = schemaEntry.getAttributeValues(ATTR_MATCHING_RULE_USE);
460        if (defs == null)
461        {
462          mruMap = Collections.emptyMap();
463          mruSet = Collections.emptySet();
464        }
465        else
466        {
467          final LinkedHashMap<String,MatchingRuleUseDefinition> m =
468               new LinkedHashMap<String,MatchingRuleUseDefinition>(2*defs.length);
469          final LinkedHashSet<MatchingRuleUseDefinition> s =
470               new LinkedHashSet<MatchingRuleUseDefinition>(defs.length);
471    
472          for (final String def : defs)
473          {
474            try
475            {
476              final MatchingRuleUseDefinition mru =
477                   new MatchingRuleUseDefinition(def);
478              s.add(mru);
479              m.put(toLowerCase(mru.getOID()), mru);
480              for (final String name : mru.getNames())
481              {
482                m.put(toLowerCase(name), mru);
483              }
484            }
485            catch (final LDAPException le)
486            {
487              debugException(le);
488            }
489          }
490    
491          mruMap = Collections.unmodifiableMap(m);
492          mruSet = Collections.unmodifiableSet(s);
493        }
494    
495    
496        // Decode the name forms from the schema entry.
497        defs = schemaEntry.getAttributeValues(ATTR_NAME_FORM);
498        if (defs == null)
499        {
500          nfMapByName = Collections.emptyMap();
501          nfMapByOC   = Collections.emptyMap();
502          nfSet       = Collections.emptySet();
503        }
504        else
505        {
506          final LinkedHashMap<String,NameFormDefinition> mN =
507               new LinkedHashMap<String,NameFormDefinition>(2*defs.length);
508          final LinkedHashMap<String,NameFormDefinition> mOC =
509               new LinkedHashMap<String,NameFormDefinition>(defs.length);
510          final LinkedHashSet<NameFormDefinition> s =
511               new LinkedHashSet<NameFormDefinition>(defs.length);
512    
513          for (final String def : defs)
514          {
515            try
516            {
517              final NameFormDefinition nf = new NameFormDefinition(def);
518              s.add(nf);
519              mOC.put(toLowerCase(nf.getStructuralClass()), nf);
520              mN.put(toLowerCase(nf.getOID()), nf);
521              for (final String name : nf.getNames())
522              {
523                mN.put(toLowerCase(name), nf);
524              }
525            }
526            catch (final LDAPException le)
527            {
528              debugException(le);
529            }
530          }
531    
532          nfMapByName = Collections.unmodifiableMap(mN);
533          nfMapByOC   = Collections.unmodifiableMap(mOC);
534          nfSet       = Collections.unmodifiableSet(s);
535        }
536    
537    
538        // Decode the object classes from the schema entry.
539        defs = schemaEntry.getAttributeValues(ATTR_OBJECT_CLASS);
540        if (defs == null)
541        {
542          ocMap           = Collections.emptyMap();
543          ocSet           = Collections.emptySet();
544          abstractOCSet   = Collections.emptySet();
545          auxiliaryOCSet  = Collections.emptySet();
546          structuralOCSet = Collections.emptySet();
547        }
548        else
549        {
550          final LinkedHashMap<String,ObjectClassDefinition> m =
551               new LinkedHashMap<String,ObjectClassDefinition>(2*defs.length);
552          final LinkedHashSet<ObjectClassDefinition> s =
553               new LinkedHashSet<ObjectClassDefinition>(defs.length);
554          final LinkedHashSet<ObjectClassDefinition> sAbstract =
555               new LinkedHashSet<ObjectClassDefinition>(defs.length);
556          final LinkedHashSet<ObjectClassDefinition> sAuxiliary =
557               new LinkedHashSet<ObjectClassDefinition>(defs.length);
558          final LinkedHashSet<ObjectClassDefinition> sStructural =
559               new LinkedHashSet<ObjectClassDefinition>(defs.length);
560    
561          for (final String def : defs)
562          {
563            try
564            {
565              final ObjectClassDefinition oc = new ObjectClassDefinition(def);
566              s.add(oc);
567              m.put(toLowerCase(oc.getOID()), oc);
568              for (final String name : oc.getNames())
569              {
570                m.put(toLowerCase(name), oc);
571              }
572    
573              switch (getOCType(oc, m))
574              {
575                case ABSTRACT:
576                  sAbstract.add(oc);
577                  break;
578                case AUXILIARY:
579                  sAuxiliary.add(oc);
580                  break;
581                case STRUCTURAL:
582                  sStructural.add(oc);
583                  break;
584              }
585            }
586            catch (final LDAPException le)
587            {
588              debugException(le);
589            }
590          }
591    
592          ocMap           = Collections.unmodifiableMap(m);
593          ocSet           = Collections.unmodifiableSet(s);
594          abstractOCSet   = Collections.unmodifiableSet(sAbstract);
595          auxiliaryOCSet  = Collections.unmodifiableSet(sAuxiliary);
596          structuralOCSet = Collections.unmodifiableSet(sStructural);
597        }
598      }

602      /**
603       * Retrieves the directory server schema over the provided connection.  The
604       * root DSE will first be retrieved in order to get its subschemaSubentry DN,
605       * and then that entry will be retrieved from the server and its contents
606       * decoded as schema elements.  This should be sufficient for directories that
607       * only provide a single schema, but for directories with multiple schemas it
608       * may be necessary to specify the DN of an entry for which to retrieve the
609       * subschema subentry.
610       *
611       * @param  connection  The connection to use in order to retrieve the server
612       *                     schema.  It must not be {@code null}.
613       *
614       * @return  A decoded representation of the server schema.
615       *
616       * @throws  LDAPException  If a problem occurs while obtaining the server
617       *                         schema.
618       */
619      public static Schema getSchema(final LDAPConnection connection)
620             throws LDAPException
621      {
622        return getSchema(connection, "");
623      }

627      /**
628       * Retrieves the directory server schema that governs the specified entry.
629       * In some servers, different portions of the DIT may be served by different
630       * schemas, and in such cases it will be necessary to provide the DN of the
631       * target entry in order to ensure that the appropriate schema which governs
632       * that entry is returned.  For servers that support only a single schema,
633       * any entry DN (including that of the root DSE) should be sufficient.
634       *
635       * @param  connection  The connection to use in order to retrieve the server
636       *                     schema.  It must not be {@code null}.
637       * @param  entryDN     The DN of the entry for which to retrieve the governing
638       *                     schema.  It may be {@code null} or an empty string in
639       *                     order to retrieve the schema that governs the server's
640       *                     root DSE.
641       *
642       * @return  A decoded representation of the server schema, or {@code null} if
643       *          it is not available for some reason (e.g., the client does not
644       *          have permission to read the server schema).
645       *
646       * @throws  LDAPException  If a problem occurs while obtaining the server
647       *                         schema.
648       */
649      public static Schema getSchema(final LDAPConnection connection,
650                                     final String entryDN)
651             throws LDAPException
652      {
653        ensureNotNull(connection);
654    
655        final String subschemaSubentryDN;
656        if (entryDN == null)
657        {
658          subschemaSubentryDN = getSubschemaSubentryDN(connection, "");
659        }
660        else
661        {
662          subschemaSubentryDN = getSubschemaSubentryDN(connection, entryDN);
663        }
664    
665        if (subschemaSubentryDN == null)
666        {
667          return null;
668        }
669    
670        final Entry schemaEntry =
671             connection.getEntry(subschemaSubentryDN, SCHEMA_REQUEST_ATTRS);
672        if (schemaEntry == null)
673        {
674          return null;
675        }
676    
677        return new Schema(schemaEntry);
678      }

682      /**
683       * Reads schema information from one or more files containing the schema
684       * represented in LDIF form, with the definitions represented in the form
685       * described in section 4.1 of RFC 4512.  Each file should contain a single
686       * entry.
687       *
688       * @param  schemaFiles  The paths to the LDIF files containing the schema
689       *                      information to be read.  At least one file must be
690       *                      specified.  If multiple files are specified, then they
691       *                      will be processed in the order in which they have been
692       *                      listed.
693       *
694       * @return  The schema read from the specified schema files, or {@code null}
695       *          if none of the files contains any LDIF data to be read.
696       *
697       * @throws  IOException  If a problem occurs while attempting to read from
698       *                       any of the specified files.
699       *
700       * @throws  LDIFException  If a problem occurs while attempting to parse the
701       *                         contents of any of the schema files.
702       */
703      public static Schema getSchema(final String... schemaFiles)
704             throws IOException, LDIFException
705      {
706        ensureNotNull(schemaFiles);
707        ensureFalse(schemaFiles.length == 0);
708    
709        final ArrayList<File> files = new ArrayList<File>(schemaFiles.length);
710        for (final String s : schemaFiles)
711        {
712          files.add(new File(s));
713        }
714    
715        return getSchema(files);
716      }

722       * represented in LDIF form, with the definitions represented in the form
723       * described in section 4.1 of RFC 4512.  Each file should contain a single
724       * entry.
725       *
726       * @param  schemaFiles  The paths to the LDIF files containing the schema
727       *                      information to be read.  At least one file must be
728       *                      specified.  If multiple files are specified, then they
729       *                      will be processed in the order in which they have been

752      /**
753       * Reads schema information from one or more files containing the schema
754       * represented in LDIF form, with the definitions represented in the form
755       * described in section 4.1 of RFC 4512.  Each file should contain a single
756       * entry.
757       *
758       * @param  schemaFiles  The paths to the LDIF files containing the schema
759       *                      information to be read.  At least one file must be
760       *                      specified.  If multiple files are specified, then they
761       *                      will be processed in the order in which they have been
762       *                      listed.
763       *
764       * @return  The schema read from the specified schema files, or {@code null}
765       *          if none of the files contains any LDIF data to be read.
766       *
767       * @throws  IOException  If a problem occurs while attempting to read from
768       *                       any of the specified files.
769       *
770       * @throws  LDIFException  If a problem occurs while attempting to parse the
771       *                         contents of any of the schema files.
772       */
773      public static Schema getSchema(final List<File> schemaFiles)
774             throws IOException, LDIFException
775      {
776        ensureNotNull(schemaFiles);
777        ensureFalse(schemaFiles.isEmpty());
778    
779        Entry schemaEntry = null;
780        for (final File f : schemaFiles)
781        {
782          final LDIFReader ldifReader = new LDIFReader(f);
783    
784          try
785          {
786            final Entry e = ldifReader.readEntry();
787            if (e == null)
788            {
789              continue;
790            }
791    
792            if (schemaEntry == null)
793            {
794              schemaEntry = e;
795            }
796            else
797            {
798              for (final Attribute a : e.getAttributes())
799              {
800                schemaEntry.addAttribute(a);
801              }
802            }
803          }
804          finally
805          {
806            ldifReader.close();
807          }
808        }
809    
810        if (schemaEntry == null)
811        {
812          return null;
813        }
814    
815        return new Schema(schemaEntry);
816      }

820      /**
821       * Retrieves the object class type for the specified object class, recursively
822       * checking its parents as needed.
823       *
824       * @param  oc  The object class definition for which to make the
825       *             determination.
826       * @param  m   The map of defined object classes.
827       *
828       * @return  The object class type for the object class.
829       */
830      private static ObjectClassType getOCType(final ObjectClassDefinition oc,
831                                          final Map<String,ObjectClassDefinition> m)
832      {
833        ObjectClassType t = oc.getObjectClassType();
834        if (t != null)
835        {
836          return t;
837        }
838    
839        for (final String s : oc.getSuperiorClasses())
840        {
841          final ObjectClassDefinition d = m.get(toLowerCase(s));
842          if (d != null)
843          {
844            t = getOCType(d, m);
845            if (t != null)
846            {
847              return t;
848            }
849          }
850        }
851    
852        return ObjectClassType.STRUCTURAL;
853      }
