039    /**
040     * This class defines the API for an LDAP matching rule, which may be used to
041     * determine whether two values are equal to each other, and to normalize values
042     * so that they may be more easily compared.
043     */
044    @Extensible()
045    @ThreadSafety(level=ThreadSafetyLevel.INTERFACE_THREADSAFE)
046    public abstract class MatchingRule
047           implements Serializable
048    {
049      /**
050       * The substring element type used for subInitial substring assertion
051       * components.
052       */
053      public static final byte SUBSTRING_TYPE_SUBINITIAL = (byte) 0x80;
054    
055    
056    
057      /**
058       * The substring element type used for subAny substring assertion components.
059       */
060      public static final byte SUBSTRING_TYPE_SUBANY = (byte) 0x81;
061    
062    
063    
064      /**
065       * The substring element type used for subFinal substring assertion
066       * components.
067       */
068      public static final byte SUBSTRING_TYPE_SUBFINAL = (byte) 0x82;

161      /**
162       * Indicates whether the provided values are equal to each other, according to
163       * the constraints of this matching rule.
164       *
165       * @param  value1  The first value for which to make the determination.
166       * @param  value2  The second value for which to make the determination.
167       *
168       * @return  {@code true} if the provided values are considered equal, or
169       *          {@code false} if not.
170       *
171       * @throws  LDAPException  If a problem occurs while making the determination,
172       *                         or if this matching rule does not support equality
173       *                         matching.
174       */
175      public abstract boolean valuesMatch(final ASN1OctetString value1,
176                                          final ASN1OctetString value2)
177             throws LDAPException;
178    
179    
180    
181      /**
182       * Indicates whether the provided value matches the given substring assertion,
183       * according to the constraints of this matching rule.
184       *
185       * @param  value       The value for which to make the determination.
186       * @param  subInitial  The subInitial portion of the substring assertion, or
187       *                     {@code null} if there is no subInitial element.
188       * @param  subAny      The subAny elements of the substring assertion, or
189       *                     {@code null} if there are no subAny elements.
190       * @param  subFinal    The subFinal portion of the substring assertion, or
191       *                     {@code null} if there is no subFinal element.
192       *
193       * @return  {@code true} if the provided value matches the substring
194       *          assertion, or {@code false} if not.
195       *
196       * @throws  LDAPException  If a problem occurs while making the determination,
197       *                         or if this matching rule does not support substring
198       *                         matching.
199       */
200      public abstract boolean matchesSubstring(final ASN1OctetString value,
201                                               final ASN1OctetString subInitial,
202                                               final ASN1OctetString[] subAny,
203                                               final ASN1OctetString subFinal)
204             throws LDAPException;
205

209       * Compares the provided values to determine their relative order in a sorted
210       * list.
211       *
212       * @param  value1  The first value to compare.
213       * @param  value2  The second value to compare.
214       *
215       * @return  A negative value if {@code value1} should come before
216       *          {@code value2} in a sorted list, a positive value if
217       *          {@code value1} should come after {@code value2} in a sorted list,
218       *          or zero if the values are equal or there is no distinction between
219       *          their orders in a sorted list.
220       *
221       * @throws  LDAPException  If a problem occurs while making the determination,
222       *                         or if this matching rule does not support ordering
223       *                         matching.
224       */
225      public abstract int compareValues(final ASN1OctetString value1,

227             throws LDAPException;

291      /**
292       * Attempts to select the appropriate matching rule to use for equality
293       * matching against the specified attribute.  If an appropriate matching rule
294       * cannot be determined, then the default equality matching rule will be
295       * selected.
296       *
297       * @param  attrName  The name of the attribute to examine in the provided
298       *                   schema.  It may be {@code null} if the matching rule
299       *                   should be selected using the matching rule ID.
300       * @param  ruleID    The OID of the desired matching rule.  It may be
301       *                   {@code null} if the matching rule should be selected only
302       *                   using the attribute name.  If a rule ID is provided, then
303       *                   it will be the only criteria used to select the matching
304       *                   rule.
305       * @param  schema    The schema to examine to make the appropriate
306       *                   determination.  If this is {@code null} and no rule ID
307       *                   was provided, then the default equality matching rule
308       *                   will be selected.
309       *
310       * @return  The selected matching rule.
311       */
312      public static MatchingRule selectEqualityMatchingRule(final String attrName,
313                                      final String ruleID, final Schema schema)
314      {
315        if (ruleID != null)
316        {
317          return selectEqualityMatchingRule(ruleID);
318        }
319    
320        if ((attrName == null) || (schema == null))
321        {
322          return getDefaultEqualityMatchingRule();
323        }
324    
325        final AttributeTypeDefinition attrType = schema.getAttributeType(attrName);
326        if (attrType == null)
327        {
328          return getDefaultEqualityMatchingRule();
329        }
330    
331        final String mrName = attrType.getEqualityMatchingRule(schema);
332        if (mrName != null)
333        {
334          return selectEqualityMatchingRule(mrName);
335        }
336    
337        final String syntaxOID = attrType.getBaseSyntaxOID(schema);
338        if (syntaxOID != null)
339        {
340          return selectMatchingRuleForSyntax(syntaxOID);
341        }
342    
343        return getDefaultEqualityMatchingRule();
344      }

348      /**
349       * Attempts to select the appropriate matching rule to use for equality
350       * matching using the specified matching rule.  If an appropriate matching
351       * rule cannot be determined, then the default equality matching rule will be
352       * selected.
353       *
354       * @param  ruleID  The name or OID of the desired matching rule.
355       *
356       * @return  The selected matching rule.
357       */
358      public static MatchingRule selectEqualityMatchingRule(final String ruleID)
359      {
360        if ((ruleID == null) || (ruleID.length() == 0))
361        {
362          return getDefaultEqualityMatchingRule();
363        }
364    
365        final String lowerName = toLowerCase(ruleID);
366        if (lowerName.equals(BooleanMatchingRule.LOWER_EQUALITY_RULE_NAME) ||
367            lowerName.equals(BooleanMatchingRule.EQUALITY_RULE_OID))
368        {
369          return BooleanMatchingRule.getInstance();
370        }
371        else if (lowerName.equals(
372                      CaseExactStringMatchingRule.LOWER_EQUALITY_RULE_NAME) ||
373                 lowerName.equals(CaseExactStringMatchingRule.EQUALITY_RULE_OID) ||
374                 lowerName.equals("caseexactia5match") ||
375                 lowerName.equals("1.3.6.1.4.1.1466.109.114.1"))
376        {
377          return CaseExactStringMatchingRule.getInstance();
378        }
379        else if (lowerName.equals(
380                      CaseIgnoreListMatchingRule.LOWER_EQUALITY_RULE_NAME) ||
381                 lowerName.equals(CaseIgnoreListMatchingRule.EQUALITY_RULE_OID))
382        {
383          return CaseIgnoreListMatchingRule.getInstance();
384        }
385        else if (lowerName.equals(
386                      CaseIgnoreStringMatchingRule.LOWER_EQUALITY_RULE_NAME) ||
387                 lowerName.equals(CaseIgnoreStringMatchingRule.EQUALITY_RULE_OID) ||
388                 lowerName.equals("caseignoreia5match") ||
389                 lowerName.equals("1.3.6.1.4.1.1466.109.114.2"))
390        {
391          return CaseIgnoreStringMatchingRule.getInstance();
392        }
393        else if (lowerName.equals(
394                      DistinguishedNameMatchingRule.LOWER_EQUALITY_RULE_NAME) ||
395                 lowerName.equals(
396                      DistinguishedNameMatchingRule.EQUALITY_RULE_OID) ||
397                 lowerName.equals("uniquemembermatch") ||
398                 lowerName.equals("2.5.13.23"))
399        {
400          // NOTE -- Technically uniqueMember should use a name and optional UID
401          // matching rule, but the SDK doesn't currently provide one and the
402          // distinguished name matching rule should be sufficient the vast
403          // majority of the time.
404          return DistinguishedNameMatchingRule.getInstance();
405        }
406        else if (lowerName.equals(
407                      GeneralizedTimeMatchingRule.LOWER_EQUALITY_RULE_NAME) ||
408                 lowerName.equals(GeneralizedTimeMatchingRule.EQUALITY_RULE_OID))
409        {
410          return GeneralizedTimeMatchingRule.getInstance();
411        }
412        else if (lowerName.equals(IntegerMatchingRule.LOWER_EQUALITY_RULE_NAME) ||
413                 lowerName.equals(IntegerMatchingRule.EQUALITY_RULE_OID))
414        {
415          return IntegerMatchingRule.getInstance();
416        }
417        else if (lowerName.equals(
418                      NumericStringMatchingRule.LOWER_EQUALITY_RULE_NAME) ||
419                 lowerName.equals(NumericStringMatchingRule.EQUALITY_RULE_OID))
420        {
421          return NumericStringMatchingRule.getInstance();
422        }
423        else if (lowerName.equals(
424                      OctetStringMatchingRule.LOWER_EQUALITY_RULE_NAME) ||
425                 lowerName.equals(OctetStringMatchingRule.EQUALITY_RULE_OID))
426        {
427          return OctetStringMatchingRule.getInstance();
428        }
429        else if (lowerName.equals(
430                      TelephoneNumberMatchingRule.LOWER_EQUALITY_RULE_NAME) ||
431                 lowerName.equals(TelephoneNumberMatchingRule.EQUALITY_RULE_OID))
432        {
433          return TelephoneNumberMatchingRule.getInstance();
434        }
435        else
436        {
437          return getDefaultEqualityMatchingRule();
438        }
439      }
