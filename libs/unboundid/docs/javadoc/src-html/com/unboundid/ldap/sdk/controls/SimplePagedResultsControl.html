043    /**
044     * This class provides an implementation of the simple paged results control as
045     * defined in <A HREF="http://www.ietf.org/rfc/rfc2696.txt">RFC 2696</A>.  It
046     * allows the client to iterate through a potentially large set of search
047     * results in subsets of a specified number of entries (i.e., "pages").
048     * <BR><BR>
049     * The same control encoding is used for both the request control sent by
050     * clients and the response control returned by the server.  It may contain
051     * two elements:
052     * <UL>
053     *   <LI>Size -- In a request control, this provides the requested page size,
054     *       which is the maximum number of entries that the server should return
055     *       in the next iteration of the search.  In a response control, it is an
056     *       estimate of the total number of entries that match the search
057     *       criteria.</LI>
058     *   <LI>Cookie -- A token which is used by the server to keep track of its
059     *       position in the set of search results.  The first request sent by the
060     *       client should not include a cookie, and the last response sent by the
061     *       server should not include a cookie.  For all other intermediate search
062     *       requests and responses,  the server will include a cookie value in its
063     *       response that the client should include in its next request.</LI>
064     * </UL>
065     * When the client wishes to use the paged results control, the first search
066     * request should include a version of the paged results request control that
067     * was created with a requested page size but no cookie.  The corresponding
068     * response from the server will include a version of the paged results control
069     * that may include an estimate of the total number of matching entries, and
070     * may also include a cookie.  The client should include this cookie in the
071     * next request (with the same set of search criteria) to retrieve the next page
072     * of results.  This process should continue until the response control returned
073     * by the server does not include a cookie, which indicates that the end of the
074     * result set has been reached.
075     * <BR><BR>
076     * Note that the simple paged results control is similar to the
077     * {@link VirtualListViewRequestControl} in that both allow the client to
078     * request that only a portion of the result set be returned at any one time.
079     * However, there are significant differences between them, including:
080     * <UL>
081     *   <LI>In order to use the virtual list view request control, it is also
082     *       necessary to use the {@link ServerSideSortRequestControl} to ensure
083     *       that the entries are sorted.  This is not a requirement for the
084     *       simple paged results control.</LI>
085     *   <LI>The simple paged results control may only be used to iterate
086     *       sequentially through the set of search results.  The virtual list view
087     *       control can retrieve pages out of order, can retrieve overlapping
088     *       pages, and can re-request pages that it had already retrieved.</LI>
089     * </UL>
090     * <H2>Example</H2>
091     * The following example demonstrates the use of the simple paged results
092     * control.  It will iterate through all users in the "Sales" department,
093     * retrieving up to 10 entries at a time:
094     * <PRE>
095     *   SearchRequest searchRequest =
096     *        new SearchRequest("dc=example,dc=com", SearchScope.SUB,"(ou=Sales)");
097     *   ASN1OctetString cookie = null;
098     *   do
099     *   {
100     *     searchRequest.setControls(
101     *          new Control[] { new SimplePagedResultsControl(10, cookie) });
102     *     SearchResult searchResult = connection.search(searchRequest);
103     *
104     *     // Do something with the entries that are returned.
105     *
106     *     cookie = null;
107     *     for (Control c : searchResult.getResponseControls())
108     *     {
109     *       if (c instanceof SimplePagedResultsControl)
110     *       {
111     *         cookie = ((SimplePagedResultsControl) c).getCookie();
112     *       }
113     *     }
114     *   } while (cookie != null);
115     * </PRE>
116     */
117    @NotMutable()
118    @ThreadSafety(level=ThreadSafetyLevel.COMPLETELY_THREADSAFE)
119    public final class SimplePagedResultsControl
120           extends Control
121           implements DecodeableControl
122    {
123      /**
124       * The OID (1.2.840.113556.1.4.319) for the paged results control.
125       */
126      public static final String PAGED_RESULTS_OID = "1.2.840.113556.1.4.319";
127    
128    
129    
130      /**
131       * The serial version UID for this serializable class.
132       */
133      private static final long serialVersionUID = 2186787148024999291L;
134    
135    
136    
137      // The encoded cookie returned from the server (for a response control) or
138      // that should be included in the next request to the server (for a request
139      // control).
140      private final ASN1OctetString cookie;
141    
142      // The maximum requested page size (for a request control), or the estimated
143      // total result set size (for a response control).
144      private final int size;
145    
146    
147    
148      /**
149       * Creates a new empty control instance that is intended to be used only for
150       * decoding controls via the {@code DecodeableControl} interface.
151       */
152      SimplePagedResultsControl()
153      {
154        size   = 0;
155        cookie = new ASN1OctetString();
156      }
157    
158    
159    
160      /**
161       * Creates a new paged results control with the specified page size.  This
162       * version of the constructor should only be used when creating the first
163       * search as part of the set of paged results.  Subsequent searches to
164       * retrieve additional pages should use the response control returned by the
165       * server in their next request, until the response control returned by the
166       * server does not include a cookie.
167       *
168       * @param  pageSize  The maximum number of entries that the server should
169       *                   return in the first page.
170       */
171      public SimplePagedResultsControl(final int pageSize)
172      {
173        super(PAGED_RESULTS_OID, false, encodeValue(pageSize, null));
174    
175        size   = pageSize;
176        cookie = new ASN1OctetString();
177      }
178    
179    
180    
181      /**
182       * Creates a new paged results control with the specified page size.  This
183       * version of the constructor should only be used when creating the first
184       * search as part of the set of paged results.  Subsequent searches to
185       * retrieve additional pages should use the response control returned by the
186       * server in their next request, until the response control returned by the
187       * server does not include a cookie.
188       *
189       * @param  pageSize    The maximum number of entries that the server should
190       *                     return in the first page.
191       * @param  isCritical  Indicates whether this control should be marked
192       *                     critical.
193       */
194      public SimplePagedResultsControl(final int pageSize, final boolean isCritical)
195      {
196        super(PAGED_RESULTS_OID, isCritical, encodeValue(pageSize, null));
197    
198        size   = pageSize;
199        cookie = new ASN1OctetString();
200      }
201    
202    
203    
204      /**
205       * Creates a new paged results control with the specified page size and the
206       * provided cookie.  This version of the constructor should be used to
207       * continue iterating through an existing set of results, but potentially
208       * using a different page size.
209       *
210       * @param  pageSize  The maximum number of entries that the server should
211       *                   return in the next page of the results.
212       * @param  cookie    The cookie provided by the server after returning the
213       *                   previous page of results, or {@code null} if this request
214       *                   will retrieve the first page of results.
215       */
216      public SimplePagedResultsControl(final int pageSize,
217                                       final ASN1OctetString cookie)
218      {
219        super(PAGED_RESULTS_OID, false, encodeValue(pageSize, cookie));
220    
221        size = pageSize;
222    
223        if (cookie == null)
224        {
225          this.cookie = new ASN1OctetString();
226        }
227        else
228        {
229          this.cookie = cookie;
230        }
231      }
232    
233    
234    
235      /**
236       * Creates a new paged results control with the specified page size and the
237       * provided cookie.  This version of the constructor should be used to
238       * continue iterating through an existing set of results, but potentially
239       * using a different page size.
240       *
241       * @param  pageSize    The maximum number of entries that the server should
242       *                     return in the first page.
243       * @param  cookie      The cookie provided by the server after returning the
244       *                     previous page of results, or {@code null} if this
245       *                     request will retrieve the first page of results.
246       * @param  isCritical  Indicates whether this control should be marked
247       *                     critical.
248       */
249      public SimplePagedResultsControl(final int pageSize,
250                                       final ASN1OctetString cookie,
251                                       final boolean isCritical)
252      {
253        super(PAGED_RESULTS_OID, isCritical, encodeValue(pageSize, cookie));
254    
255        size = pageSize;
256    
257        if (cookie == null)
258        {
259          this.cookie = new ASN1OctetString();
260        }
261        else
262        {
263          this.cookie = cookie;
264        }
265      }

269      /**
270       * Creates a new paged results control from the control with the provided set
271       * of information.  This should be used to decode the paged results response
272       * control returned by the server with a page of results.
273       *
274       * @param  oid         The OID for the control.
275       * @param  isCritical  Indicates whether the control should be marked
276       *                     critical.
277       * @param  value       The encoded value for the control.  This may be
278       *                     {@code null} if no value was provided.
279       *
280       * @throws  LDAPException  If the provided control cannot be decoded as a
281       *                         simple paged results control.
282       */
283      public SimplePagedResultsControl(final String oid, final boolean isCritical,
284                                       final ASN1OctetString value)
285             throws LDAPException
286      {
287        super(oid, isCritical, value);
288    
289        if (value == null)
290        {
291          throw new LDAPException(ResultCode.DECODING_ERROR,
292                                  ERR_PAGED_RESULTS_NO_VALUE.get());
293        }
294    
295        final ASN1Sequence valueSequence;
296        try
297        {
298          final ASN1Element valueElement = ASN1Element.decode(value.getValue());
299          valueSequence = ASN1Sequence.decodeAsSequence(valueElement);
300        }
301        catch (final ASN1Exception ae)
302        {
303          debugException(ae);
304          throw new LDAPException(ResultCode.DECODING_ERROR,
305                                  ERR_PAGED_RESULTS_VALUE_NOT_SEQUENCE.get(ae), ae);
306        }
307    
308        final ASN1Element[] valueElements = valueSequence.elements();
309        if (valueElements.length != 2)
310        {
311          throw new LDAPException(ResultCode.DECODING_ERROR,
312                                  ERR_PAGED_RESULTS_INVALID_ELEMENT_COUNT.get(
313                                       valueElements.length));
314        }
315    
316        try
317        {
318          size = ASN1Integer.decodeAsInteger(valueElements[0]).intValue();
319        }
320        catch (final ASN1Exception ae)
321        {
322          debugException(ae);
323          throw new LDAPException(ResultCode.DECODING_ERROR,
324                                  ERR_PAGED_RESULTS_FIRST_NOT_INTEGER.get(ae), ae);
325        }
326    
327        cookie = ASN1OctetString.decodeAsOctetString(valueElements[1]);
328      }
329

332      /**
333       * {@inheritDoc}
334       */
335      public SimplePagedResultsControl
336                  decodeControl(final String oid, final boolean isCritical,
337                                final ASN1OctetString value)
338             throws LDAPException
339      {
340        return new SimplePagedResultsControl(oid, isCritical, value);
341      }

345      /**
346       * Encodes the provided information into an octet string that can be used as
347       * the value for this control.
348       *
349       * @param  pageSize  The maximum number of entries that the server should
350       *                   return in the next page of the results.
351       * @param  cookie    The cookie provided by the server after returning the
352       *                   previous page of results, or {@code null} if this request
353       *                   will retrieve the first page of results.
354       *
355       * @return  An ASN.1 octet string that can be used as the value for this
356       *          control.
357       */
358      private static ASN1OctetString encodeValue(final int pageSize,
359                                                 final ASN1OctetString cookie)
360      {
361        final ASN1Element[] valueElements;
362        if (cookie == null)
363        {
364          valueElements = new ASN1Element[]
365          {
366            new ASN1Integer(pageSize),
367            new ASN1OctetString()
368          };
369        }
370        else
371        {
372          valueElements = new ASN1Element[]
373          {
374            new ASN1Integer(pageSize),
375            cookie
376          };
377        }
378    
379        return new ASN1OctetString(new ASN1Sequence(valueElements).encode());
380      }

384      /**
385       * Retrieves the size for this paged results control.  For a request control,
386       * it may be used to specify the number of entries that should be included in
387       * the next page of results.  For a response control, it may be used to
388       * specify the estimated number of entries in the complete result set.
389       *
390       * @return  The size for this paged results control.
391       */
392      public int getSize()
393      {
394        return size;
395      }
396    
397    
398    
399      /**
400       * Retrieves the cookie for this control, which may be used in a subsequent
401       * request to resume reading entries from the next page of results.  The
402       * value should have a length of zero when used to retrieve the first page of
403       * results for a given search, and also in the response from the server when
404       * there are no more entries to send.  It should be non-empty for all other
405       * conditions.
406       *
407       * @return  The cookie for this control, or {@code null} if there is none.
408       */
409      public ASN1OctetString getCookie()
410      {
411        return cookie;
412      }
413    
414    
415    
416      /**
417       * Indicates whether there are more results to return as part of this search.
418       *
419       * @return  {@code true} if there are more results to return, or
420       *          {@code false} if not.
421       */
422      public boolean moreResultsToReturn()
423      {
424        return (cookie.getValue().length > 0);
425      }

447        buffer.append(size);

449        buffer.append(isCritical());
450        buffer.append(')');
