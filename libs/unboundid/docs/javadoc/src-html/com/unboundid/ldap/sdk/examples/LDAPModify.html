046    /**
047     * This class provides a simple tool that can be used to perform add, delete,
048     * modify, and modify DN operations against an LDAP directory server.  The
049     * changes to apply can be read either from standard input or from an LDIF file.
050     * <BR><BR>
051     * Some of the APIs demonstrated by this example include:
052     * <UL>
053     *   <LI>Argument Parsing (from the {@code com.unboundid.util.args}
054     *       package)</LI>
055     *   <LI>LDAP Command-Line Tool (from the {@code com.unboundid.util}
056     *       package)</LI>
057     *   <LI>LDIF Processing (from the {@code com.unboundid.ldif} package)</LI>
058     * </UL>
059     * <BR><BR>
060     * The behavior of this utility is controlled by command line arguments.
061     * Supported arguments include those allowed by the {@link LDAPCommandLineTool}
062     * class, as well as the following additional arguments:
063     * <UL>
064     *   <LI>"-f {path}" or "--ldifFile {path}" -- specifies the path to the LDIF
065     *       file containing the changes to apply.  If this is not provided, then
066     *       changes will be read from standard input.</LI>
067     *   <LI>"-a" or "--defaultAdd" -- indicates that any LDIF records encountered
068     *       that do not include a changetype should be treated as add change
069     *       records.  If this is not provided, then such records will be
070     *       rejected.</LI>
071     *   <LI>"-c" or "--continueOnError" -- indicates that processing should
072     *       continue if an error occurs while processing an earlier change.  If
073     *       this is not provided, then the command will exit on the first error
074     *       that occurs.</LI>
075     * </UL>
076     */
077    @ThreadSafety(level=ThreadSafetyLevel.NOT_THREADSAFE)
078    public final class LDAPModify
079           extends LDAPCommandLineTool
080           implements Serializable
081    {
082      /**
083       * The serial version UID for this serializable class.
084       */

088    
089      // Indicates whether processing should continue even if an error has occurred.
090      private BooleanArgument continueOnError;
091    
092      // Indicates whether LDIF records without a changetype should be considered
093      // add records.
094      private BooleanArgument defaultAdd;
095    
096      // The LDIF file to be processed.
097      private FileArgument ldifFile;

101      /**
102       * Parse the provided command line arguments and make the appropriate set of
103       * changes.
104       *
105       * @param  args  The command line arguments provided to this program.
106       */
107      public static void main(final String[] args)
108      {
109        final ResultCode resultCode = main(args, System.out, System.err);
110        if (resultCode != ResultCode.SUCCESS)
111        {
112          System.exit(resultCode.intValue());
113        }
114      }

121       *
122       * @param  args       The command line arguments provided to this program.
123       * @param  outStream  The output stream to which standard out should be
124       *                    written.  It may be {@code null} if output should be
125       *                    suppressed.
126       * @param  errStream  The output stream to which standard error should be
127       *                    written.  It may be {@code null} if error messages
128       *                    should be suppressed.
129       *
130       * @return  A result code indicating whether the processing was successful.
131       */
132      public static ResultCode main(final String[] args,
133                                    final OutputStream outStream,
134                                    final OutputStream errStream)
135      {
136        final LDAPModify ldapModify = new LDAPModify(outStream, errStream);
137        return ldapModify.runTool(args);
138      }

186      /**
187       * Adds the arguments used by this program that aren't already provided by the
188       * generic {@code LDAPCommandLineTool} framework.
189       *
190       * @param  parser  The argument parser to which the arguments should be added.
191       *
192       * @throws  ArgumentException  If a problem occurs while adding the arguments.
193       */
194      @Override()
195      public void addNonLDAPArguments(final ArgumentParser parser)
196             throws ArgumentException
197      {
198        String description = "Treat LDIF records that do not contain a " +
199                             "changetype as add records.";
200        defaultAdd = new BooleanArgument('a', "defaultAdd", description);
201        parser.addArgument(defaultAdd);
202    
203    
204        description = "Attempt to continue processing additional changes if " +
205                      "an error occurs.";
206        continueOnError = new BooleanArgument('c', "continueOnError",
207                                              description);
208        parser.addArgument(continueOnError);
209    
210    
211        description = "The path to the LDIF file containing the changes.  If " +
212                      "this is not provided, then the changes will be read from " +
213                      "standard input.";
214        ldifFile = new FileArgument('f', "ldifFile", false, 1, "{path}",
215                                    description, true, false, true, false);
216        parser.addArgument(ldifFile);
217      }

221      /**
222       * Performs the actual processing for this tool.  In this case, it gets a
223       * connection to the directory server and uses it to perform the requested
224       * operations.
225       *
226       * @return  The result code for the processing that was performed.
227       */
228      @Override()
229      public ResultCode doToolProcessing()
230      {
231        // Set up the LDIF reader that will be used to read the changes to apply.
232        final LDIFReader ldifReader;
233        try
234        {
235          if (ldifFile.isPresent())
236          {
237            // An LDIF file was specified on the command line, so we will use it.
238            ldifReader = new LDIFReader(ldifFile.getValue());
239          }
240          else
241          {
242            // No LDIF file was specified, so we will read from standard input.
243            ldifReader = new LDIFReader(System.in);
244          }
245        }
246        catch (IOException ioe)
247        {
248          err("I/O error creating the LDIF reader:  ", ioe.getMessage());
249          return ResultCode.LOCAL_ERROR;
250        }
251    
252    
253        // Get the connection to the directory server.
254        final LDAPConnection connection;
255        try
256        {
257          connection = getConnection();
258          out("Connected to ", connection.getConnectedAddress(), ':',
259              connection.getConnectedPort());
260        }
261        catch (LDAPException le)
262        {
263          err("Error connecting to the directory server:  ", le.getMessage());
264          return le.getResultCode();
265        }
266    
267    
268        // Attempt to process and apply the changes to the server.
269        ResultCode resultCode = ResultCode.SUCCESS;
270        while (true)
271        {
272          // Read the next change to process.
273          final LDIFChangeRecord changeRecord;
274          try
275          {
276            changeRecord = ldifReader.readChangeRecord(defaultAdd.isPresent());
277          }
278          catch (LDIFException le)
279          {
280            err("Malformed change record:  ", le.getMessage());
281            if (! le.mayContinueReading())
282            {
283              err("Unable to continue processing the LDIF content.");
284              resultCode = ResultCode.DECODING_ERROR;
285              break;
286            }
287            else if (! continueOnError.isPresent())
288            {
289              resultCode = ResultCode.DECODING_ERROR;
290              break;
291            }
292            else
293            {
294              // We can try to keep processing, so do so.
295              continue;
296            }
297          }
298          catch (IOException ioe)
299          {
300            err("I/O error encountered while reading a change record:  ",
301                ioe.getMessage());
302            resultCode = ResultCode.LOCAL_ERROR;
303            break;
304          }
305    
306    
307          // If the change record was null, then it means there are no more changes
308          // to be processed.
309          if (changeRecord == null)
310          {
311            break;
312          }
313    
314    
315          // Apply the target change to the server.
316          try
317          {
318            out("Processing ", changeRecord.getChangeType().toString(),
319                " operation for ", changeRecord.getDN());
320            changeRecord.processChange(connection);
321            out("Success");
322            out();
323          }
324          catch (LDAPException le)
325          {
326            err("Error:  ", le.getMessage());
327            err("Result Code:  ", le.getResultCode().intValue(), " (",
328                le.getResultCode().getName(), ')');
329            if (le.getMatchedDN() != null)
330            {
331              err("Matched DN:  ", le.getMatchedDN());
332            }
333    
334            if (le.getReferralURLs() != null)
335            {
336              for (final String url : le.getReferralURLs())
337              {
338                err("Referral URL:  ", url);
339              }
340            }
341    
342            err();
343            if (! continueOnError.isPresent())
344            {
345              resultCode = le.getResultCode();
346              break;
347            }
348          }
349        }
350    
351    
352        // Close the connection to the directory server and exit.
353        connection.close();
354        out("Disconnected from the server");
355        return resultCode;
356      }

375          "--ldifFile", "changes.ldif"
376        };
377        String description =
378             "Attempt to apply the add, delete, modify, and/or modify DN " +
379             "operations contained in the 'changes.ldif' file against the " +
380             "specified directory server.";

382    
383        args = new String[]
384        {
385          "--hostname", "server.example.com",
386          "--port", "389",
387          "--bindDN", "uid=admin,dc=example,dc=com",
388          "--bindPassword", "password",
389          "--continueOnError",
390          "--defaultAdd"
391        };
392        description =
393             "Establish a connection to the specified directory server and then " +
394             "wait for information about the add, delete, modify, and/or modify " +
395             "DN operations to perform to be provided via standard input.  If " +
396             "any invalid operations are requested, then the tool will display " +
397             "an error message but will continue running.  Any LDIF record " +
398             "provided which does not include a 'changeType' line will be " +
399             "treated as an add request.";
400        examples.put(args, description);
