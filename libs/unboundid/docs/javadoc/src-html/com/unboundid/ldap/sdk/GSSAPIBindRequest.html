056    /**
057     * This class provides a SASL GSSAPI bind request implementation as described in
058     * <A HREF="http://www.ietf.org/rfc/rfc4752.txt">RFC 4752</A>.  It provides the
059     * ability to authenticate to a directory server using Kerberos V, which can
060     * serve as a kind of single sign-on mechanism that may be shared across
061     * client applications that support Kerberos.  At present, this implementation
062     * may only be used for authentication, as it does not yet offer support for
063     * integrity or confidentiality.
064     * <BR><BR>
065     * This class uses the Java Authentication and Authorization Service (JAAS)
066     * behind the scenes to perform all Kerberos processing.  This framework
067     * requires a configuration file to indicate the underlying mechanism to be
068     * used.  It is possible for clients to explicitly specify the path to the
069     * configuration file that should be used, but if none is given then a default
070     * file will be created and used.  This default file should be sufficient for
071     * Sun-provided JVMs, but a custom file may be required for JVMs provided by
072     * other vendors.
073     * <BR><BR>
074     * Elements included in a GSSAPI bind request include:
075     * <UL>
076     *   <LI>Authentication ID -- A string which identifies the user that is
077     *       attempting to authenticate.  It should be the user's Kerberos
078     *       principal.</LI>
079     *   <LI>Authorization ID -- An optional string which specifies an alternate
080     *       authorization identity that should be used for subsequent operations
081     *       requested on the connection.  Like the authentication ID, the
082     *       authorization ID should be a Kerberos principal.</LI>
083     *   <LI>KDC Address -- An optional string which specifies the IP address or
084     *       resolvable name for the Kerberos key distribution center.  If this is
085     *       not provided, an attempt will be made to determine the appropriate
086     *       value from the system configuration.</LI>
087     *   <LI>Realm -- An optional string which specifies the realm into which the
088     *       user should authenticate.  If this is not provided, an attempt will be
089     *       made to determine the appropriate value from the system
090     *       configuration</LI>
091     *   <LI>Password -- The clear-text password for the target user in the Kerberos
092     *       realm.</LI>
093     * </UL>
094     * <H2>Example</H2>
095     * The following example demonstrates the process for performing a GSSAPI bind
096     * against a directory server with a username of "john.doe" and a password
097     * of "password":
098     * <PRE>
099     *   GSSAPIBindRequest bindRequest =
100     *        new GSSAPIBindRequest("john.doe@EXAMPLE.COM", "password");
101     *   try
102     *   {
103     *     BindResult bindResult = connection.bind(bindRequest);
104     *     // If we get here, then the bind was successful.
105     *   }
106     *   catch (LDAPException le)
107     *   {
108     *     // The bind failed for some reason.
109     *   }
110     * </PRE>
111     */

615      /**
616       * Retrieves the path to the default JAAS configuration file that will be used
617       * if no file was explicitly provided.  A new file may be created if
618       * necessary.
619       *
620       * @return  The path to the default JAAS configuration file that will be used
621       *          if no file was explicitly provided.
622       *
623       * @throws  LDAPException  If an error occurs while attempting to create the
624       *                         configuration file.
625       */
626      private static String getDefaultConfigFilePath()
627              throws LDAPException
628      {
629        try
630        {
631          String value = DEFAULT_CONFIG_FILE_PATH.get();
632          if (value == null)
633          {
634            final File f =
635                 File.createTempFile("GSSAPIBindRequest-JAAS-Config-", ".conf");
636            f.deleteOnExit();
637            final PrintWriter w = new PrintWriter(new FileWriter(f));
638            try
639            {
640              w.println(JAAS_CLIENT_NAME + " {");
641              w.println("  com.sun.security.auth.module.Krb5LoginModule required " +
642                        "client=true useTicketCache=true;");
643              w.println("};");
644            }
645            finally
646            {
647              w.close();
648            }
649    
650            if (! DEFAULT_CONFIG_FILE_PATH.compareAndSet(null, f.getAbsolutePath()))
651            {
652              f.delete();
653            }
654    
655            value = DEFAULT_CONFIG_FILE_PATH.get();
656          }
657    
658          return value;
659        }
660        catch (Exception e)
661        {
662          debugException(e);
663    
664          throw new LDAPException(ResultCode.LOCAL_ERROR,
665               ERR_GSSAPI_CANNOT_CREATE_JAAS_CONFIG.get(getExceptionMessage(e)), e);
666        }
667      }

671      /**
672       * Sends this bind request to the target server over the provided connection
673       * and returns the corresponding response.
674       *
675       * @param  connection  The connection to use to send this bind request to the
676       *                     server and read the associated response.
677       * @param  depth       The current referral depth for this request.  It should
678       *                     always be one for the initial request, and should only
679       *                     be incremented when following referrals.
680       *
681       * @return  The bind response read from the server.
682       *
683       * @throws  LDAPException  If a problem occurs while sending the request or
684       *                         reading the response.
685       */
686      @Override()
687      protected BindResult process(final LDAPConnection connection, final int depth)
688                throws LDAPException
689      {
690        if (! conn.compareAndSet(null, connection))
691        {
692          throw new LDAPException(ResultCode.LOCAL_ERROR,
693                         ERR_GSSAPI_MULTIPLE_CONCURRENT_REQUESTS.get());
694        }
695    
696        System.setProperty(PROPERTY_CONFIG_FILE, configFilePath);
697        System.setProperty(PROPERTY_SUBJECT_CREDS_ONLY, "true");
698    
699        if (kdcAddress != null)
700        {
701          System.setProperty(PROPERTY_KDC_ADDRESS, kdcAddress);
702        }
703    
704        if (realm != null)
705        {
706          System.setProperty(PROPERTY_REALM, realm);
707        }
708    
709        try
710        {
711          final LoginContext context;
712          try
713          {
714            context = new LoginContext(JAAS_CLIENT_NAME, this);
715            context.login();
716          }
717          catch (Exception e)
718          {
719            debugException(e);
720    
721            throw new LDAPException(ResultCode.LOCAL_ERROR,
722                           ERR_GSSAPI_CANNOT_INITIALIZE_JAAS_CONTEXT.get(
723                                getExceptionMessage(e)), e);
724          }
725    
726          try
727          {
728            return (BindResult) Subject.doAs(context.getSubject(), this);
729          }
730          catch (Exception e)
731          {
732            debugException(e);
733            if (e instanceof LDAPException)
734            {
735              throw (LDAPException) e;
736            }
737            else
738            {
739              throw new LDAPException(ResultCode.LOCAL_ERROR,
740                             ERR_GSSAPI_AUTHENTICATION_FAILED.get(
741                                  getExceptionMessage(e)), e);
742            }
743          }
744        }
745        finally
746        {
747          conn.set(null);
748        }
749      }

753      /**
754       * Perform the privileged portion of the authentication processing.
755       *
756       * @return  {@code null}, since no return value is actually needed.
757       *
758       * @throws  LDAPException  If a problem occurs during processing.
759       */
760      @InternalUseOnly()
761      public Object run()
762             throws LDAPException
763      {
764        final LDAPConnection connection = conn.get();
765    
766        final String[] mechanisms = { GSSAPI_MECHANISM_NAME };
767    
768        final HashMap<String,Object> saslProperties = new HashMap<String,Object>();
769        saslProperties.put(Sasl.QOP, "auth");
770        saslProperties.put(Sasl.SERVER_AUTH, "true");
771    
772        final SaslClient saslClient;
773        try
774        {
775          saslClient = Sasl.createSaslClient(mechanisms, authorizationID, "ldap",
776                                             connection.getConnectedAddress(),
777                                             saslProperties, this);
778        }
779        catch (Exception e)
780        {
781          debugException(e);
782          throw new LDAPException(ResultCode.LOCAL_ERROR,
783               ERR_GSSAPI_CANNOT_CREATE_SASL_CLIENT.get(getExceptionMessage(e)), e);
784        }
785    
786        final SASLHelper helper = new SASLHelper(this, connection,
787             GSSAPI_MECHANISM_NAME, saslClient, getControls(),
788             getResponseTimeoutMillis(connection));
789    
790        try
791        {
792          return helper.processSASLBind();
793        }
794        finally
795        {
796          messageID = helper.getMessageID();
797        }
798      }

824      /**
825       * Handles any necessary callbacks required for SASL authentication.
826       *
827       * @param  callbacks  The set of callbacks to be handled.
828       */
829      @InternalUseOnly()
830      public void handle(final Callback[] callbacks)
831      {
832        for (final Callback callback : callbacks)
833        {
834          if (callback instanceof NameCallback)
835          {
836            ((NameCallback) callback).setName(authenticationID);
837          }
838          else if (callback instanceof PasswordCallback)
839          {
840            ((PasswordCallback) callback).setPassword(
841                 password.stringValue().toCharArray());
842          }
843          else if (callback instanceof RealmCallback)
844          {
845            if (realm != null)
846            {
847              ((RealmCallback) callback).setText(realm);
848            }
849          }
850          else
851          {
852            // This is an unexpected callback.
853            if (debugEnabled(DebugType.LDAP))
854            {
855              debug(Level.WARNING, DebugType.LDAP,
856                    "Unexpected GSSAPI SASL callback of type " +
857                    callback.getClass().getName());
858            }
859          }
860        }
861      }

944        buffer.append('\'');

946        final Control[] controls = getControls();
947        if (controls.length > 0)
948        {

953            {

955            }
