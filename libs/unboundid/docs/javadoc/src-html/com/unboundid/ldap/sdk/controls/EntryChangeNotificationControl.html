049    /**
050     * This class provides an implementation of the entry change notification
051     * control as defined in draft-ietf-ldapext-psearch.  It will be returned in
052     * search result entries that match the criteria associated with a persistent
053     * search (see the {@link PersistentSearchRequestControl} class) and have been
054     * changed in a way associated with the registered change types for that search.
055     * <BR><BR>
056     * The information that can be included in an entry change notification control
057     * includes:
058     * <UL>
059     *   <LI>A change type, which indicates the type of operation that was performed
060     *       to trigger this entry change notification control.  It will be one of
061     *       the values of the {@link PersistentSearchChangeType} enum.</LI>
062     *   <LI>An optional previous DN, which indicates the DN that the entry had
063     *       before the associated operation was processed.  It will only be present
064     *       if the associated operation was a modify DN operation.</LI>
065     *   <LI>An optional change number, which may be used to retrieve additional
066     *       information about the associated operation from the server.  This may
067     *       not be available in all directory server implementations.</LI>
068     * </UL>
069     * Note that the entry change notification control should only be included in
070     * search result entries that are associated with a search request that included
071     * the persistent search request control, and only if that persistent search
072     * request control had the {@code returnECs} flag set to {@code true} to
073     * indicate that entry change notification controls should be included in
074     * resulting entries.  Further, the entry change notification control will only
075     * be included in entries that are returned as the result of a change in the
076     * server and not any of the preliminary entries that may be returned if the
077     * corresponding persistent search request had the {@code changesOnly} flag set
078     * to {@code false}.
079     */
080    @NotMutable()
081    @ThreadSafety(level=ThreadSafetyLevel.COMPLETELY_THREADSAFE)
082    public final class EntryChangeNotificationControl
083           extends Control
084           implements DecodeableControl
085    {

087       * The OID (2.16.840.1.113730.3.4.7) for the entry change notification
088       * control.
089       */
090      public static final String ENTRY_CHANGE_NOTIFICATION_OID =
091           "2.16.840.1.113730.3.4.7";

128       * information.  It will not be critical.

131       *                       {@code null}.
132       * @param  previousDN    The previous DN of the entry, if applicable.

134       *                       -1 if there should not be a change number.

136      public EntryChangeNotificationControl(
137                  final PersistentSearchChangeType changeType,
138                  final String previousDN, final long changeNumber)
139      {

145      /**
146       * Creates a new entry change notification control with the provided
147       * information.
148       *
149       * @param  changeType    The change type for the change.  It must not be
150       *                       {@code null}.
151       * @param  previousDN    The previous DN of the entry, if applicable.
152       * @param  changeNumber  The change number to include in this control, or
153       *                       -1 if there should not be a change number.
154       * @param  isCritical    Indicates whether this control should be marked
155       *                       critical.
156       */
157      public EntryChangeNotificationControl(
158                  final PersistentSearchChangeType changeType,
159                  final String previousDN, final long changeNumber,
160                  final boolean isCritical)
161      {
162        super(ENTRY_CHANGE_NOTIFICATION_OID, isCritical,
163              encodeValue(changeType, previousDN, changeNumber));
164    
165        this.changeType   = changeType;
166        this.previousDN   = previousDN;
167        this.changeNumber = changeNumber;
168      }

172      /**
173       * Creates a new entry change notification control with the provided
174       * information.
175       *
176       * @param  oid         The OID for the control.
177       * @param  isCritical  Indicates whether the control should be marked
178       *                     critical.
179       * @param  value       The encoded value for the control.  This may be
180       *                     {@code null} if no value was provided.
181       *
182       * @throws  LDAPException  If the provided control cannot be decoded as an
183       *                         entry change notification control.
184       */
185      public EntryChangeNotificationControl(final String oid,
186                                            final boolean isCritical,
187                                            final ASN1OctetString value)
188             throws LDAPException
189      {
190        super(oid, isCritical, value);
191    
192        if (value == null)
193        {
194          throw new LDAPException(ResultCode.DECODING_ERROR,
195                                  ERR_ECN_NO_VALUE.get());
196        }
197    
198        final ASN1Sequence ecnSequence;
199        try
200        {
201          final ASN1Element element = ASN1Element.decode(value.getValue());
202          ecnSequence = ASN1Sequence.decodeAsSequence(element);
203        }
204        catch (final ASN1Exception ae)
205        {
206          debugException(ae);
207          throw new LDAPException(ResultCode.DECODING_ERROR,
208                                  ERR_ECN_VALUE_NOT_SEQUENCE.get(ae), ae);
209        }
210    
211        final ASN1Element[] ecnElements = ecnSequence.elements();
212        if ((ecnElements.length < 1) || (ecnElements.length > 3))
213        {
214          throw new LDAPException(ResultCode.DECODING_ERROR,
215                                  ERR_ECN_INVALID_ELEMENT_COUNT.get(
216                                       ecnElements.length));
217        }
218    
219        final ASN1Enumerated ecnEnumerated;
220        try
221        {
222          ecnEnumerated = ASN1Enumerated.decodeAsEnumerated(ecnElements[0]);
223        }
224        catch (final ASN1Exception ae)
225        {
226          debugException(ae);
227          throw new LDAPException(ResultCode.DECODING_ERROR,
228                                  ERR_ECN_FIRST_NOT_ENUMERATED.get(ae), ae);
229        }
230    
231        changeType = PersistentSearchChangeType.valueOf(ecnEnumerated.intValue());
232        if (changeType == null)
233        {
234          throw new LDAPException(ResultCode.DECODING_ERROR,
235                                  ERR_ECN_INVALID_CHANGE_TYPE.get(
236                                       ecnEnumerated.intValue()));
237        }
238    
239    
240        String prevDN = null;
241        long   chgNum = -1;
242        for (int i=1; i < ecnElements.length; i++)
243        {
244          switch (ecnElements[i].getType())
245          {
246            case ASN1Constants.UNIVERSAL_OCTET_STRING_TYPE:
247              prevDN = ASN1OctetString.decodeAsOctetString(
248                            ecnElements[i]).stringValue();
249              break;
250    
251            case ASN1Constants.UNIVERSAL_INTEGER_TYPE:
252              try
253              {
254                chgNum = ASN1Long.decodeAsLong(ecnElements[i]).longValue();
255              }
256              catch (final ASN1Exception ae)
257              {
258                debugException(ae);
259                throw new LDAPException(ResultCode.DECODING_ERROR,
260                                        ERR_ECN_CANNOT_DECODE_CHANGE_NUMBER.get(ae),
261                                        ae);
262              }
263              break;
264    
265            default:
266              throw new LDAPException(ResultCode.DECODING_ERROR,
267                                      ERR_ECN_INVALID_ELEMENT_TYPE.get(
268                                           toHex(ecnElements[i].getType())));
269          }
270        }
271    
272        previousDN   = prevDN;
273        changeNumber = chgNum;
274      }

278      /**
279       * {@inheritDoc}
280       */
281      public EntryChangeNotificationControl
282                  decodeControl(final String oid, final boolean isCritical,
283                                final ASN1OctetString value)
284             throws LDAPException
285      {
286        return new EntryChangeNotificationControl(oid, isCritical, value);
287      }

291      /**
292       * Encodes the provided information into an octet string that can be used as
293       * the value for this control.
294       *
295       * @param  changeType    The change type for the change.  It must not be
296       *                       {@code null}.
297       * @param  previousDN    The previous DN of the entry, if applicable.
298       * @param  changeNumber  The change number to include in this control, or
299       *                       -1 if there should not be a change number.
300       *
301       * @return  An ASN.1 octet string that can be used as the value for this
302       *          control.
303       */
304      private static ASN1OctetString encodeValue(
305                   final PersistentSearchChangeType changeType,
306                   final String previousDN, final long changeNumber)
307      {
308        ensureNotNull(changeType);
309    
310        final ArrayList<ASN1Element> elementList = new ArrayList<ASN1Element>(3);
311        elementList.add(new ASN1Enumerated(changeType.intValue()));
312    
313        if (previousDN != null)
314        {
315          elementList.add(new ASN1OctetString(previousDN));
316        }
317    
318        if (changeNumber > 0)
319        {
320          elementList.add(new ASN1Long(changeNumber));
321        }
322    
323        return new ASN1OctetString(new ASN1Sequence(elementList).encode());
324      }

379      @Override()
380      public void toString(final StringBuilder buffer)
381      {
382        buffer.append("EntryChangeNotificationControl(changeType=");
383        buffer.append(changeType.getName());
384    
385        if (previousDN != null)
386        {
387          buffer.append(", previousDN='");
388          buffer.append(previousDN);
389          buffer.append('\'');
390        }
391    
392        if (changeNumber > 0)
393        {
394          buffer.append(", changeNumber=");
395          buffer.append(changeNumber);
396        }
397    
398        buffer.append(", isCritical=");
399        buffer.append(isCritical());
400        buffer.append(')');
401      }
