061    /**
062     * This class provides a tool that can be used to search an LDAP directory
063     * server repeatedly using multiple threads, and then modify each entry
064     * returned by that server.  It can help provide an estimate of the combined
065     * search and modify performance that a directory server is able to achieve.
066     * Either or both of the base DN and the search filter may be a value pattern as
067     * described in the {@link ValuePattern} class.  This makes it possible to
068     * search over a range of entries rather than repeatedly performing searches
069     * with the same base DN and filter.
070     * <BR><BR>
071     * Some of the APIs demonstrated by this example include:
072     * <UL>
073     *   <LI>Argument Parsing (from the {@code com.unboundid.util.args}
074     *       package)</LI>
075     *   <LI>LDAP Command-Line Tool (from the {@code com.unboundid.util}
076     *       package)</LI>
077     *   <LI>LDAP Communication (from the {@code com.unboundid.ldap.sdk}
078     *       package)</LI>
079     *   <LI>Value Patterns (from the {@code com.unboundid.util} package)</LI>
080     * </UL>
081     * <BR><BR>
082     * All of the necessary information is provided using command line arguments.
083     * Supported arguments include those allowed by the {@link LDAPCommandLineTool}
084     * class, as well as the following additional arguments:
085     * <UL>
086     *   <LI>"-b {baseDN}" or "--baseDN {baseDN}" -- specifies the base DN to use
087     *       for the searches.  This must be provided.  It may be a simple DN, or it
088     *       may be a value pattern to express a range of base DNs.</LI>
089     *   <LI>"-s {scope}" or "--scope {scope}" -- specifies the scope to use for the
090     *       search.  The scope value should be one of "base", "one", "sub", or
091     *       "subord".  If this isn't specified, then a scope of "sub" will be
092     *       used.</LI>
093     *   <LI>"-f {filter}" or "--filter {filter}" -- specifies the filter to use for
094     *       the searches.  This must be provided.  It may be a simple filter, or it
095     *       may be a value pattern to express a range of filters.</LI>
096     *   <LI>"-A {name}" or "--attribute {name}" -- specifies the name of an
097     *       attribute that should be included in entries returned from the server.
098     *       If this is not provided, then all user attributes will be requested.
099     *       This may include special tokens that the server may interpret, like
100     *       "1.1" to indicate that no attributes should be returned, "*", for all
101     *       user attributes, or "+" for all operational attributes.  Multiple
102     *       attributes may be requested with multiple instances of this
103     *       argument.</LI>
104     *   <LI>"-m {name}" or "--modifyAttribute {name}" -- specifies the name of the
105     *       attribute to modify.  Multiple attributes may be modified by providing
106     *       multiple instances of this argument.  At least one attribute must be
107     *       provided.</LI>
108     *   <LI>"-l {num}" or "--valueLength {num}" -- specifies the length in bytes to
109     *       use for the values of the target attributes to modify.  If this is not
110     *       provided, then a default length of 10 bytes will be used.</LI>
111     *   <LI>"-C {chars}" or "--characterSet {chars}" -- specifies the set of
112     *       characters that will be used to generate the values to use for the
113     *       target attributes to modify.  It should only include ASCII characters.
114     *       Values will be generated from randomly-selected characters from this
115     *       set.  If this is not provided, then a default set of lowercase
116     *       alphabetic characters will be used.</LI>
117     *   <LI>"-t {num}" or "--numThreads {num}" -- specifies the number of
118     *       concurrent threads to use when performing the searches.  If this is not
119     *       provided, then a default of one thread will be used.</LI>
120     *   <LI>"-i {sec}" or "--intervalDuration {sec}" -- specifies the length of
121     *       time in seconds between lines out output.  If this is not provided,
122     *       then a default interval duration of five seconds will be used.</LI>
123     *   <LI>"-I {num}" or "--numIntervals {num}" -- specifies the maximum number of
124     *       intervals for which to run.  If this is not provided, then it will
125     *       run forever.</LI>
126     *   <LI>"-r {ops-per-second}" or "--ratePerSecond {ops-per-second}" --
127     *       specifies the target number of operations to perform per second.  Each
128     *       search and modify operation will be counted separately for this
129     *       purpose, so if a value of 1 is specified and a search returns two
130     *       entries, then a total of three seconds will be required (one for the
131     *       search and one for the modify for each entry).  It is still necessary
132     *       to specify a sufficient number of threads for achieving this rate.  If
133     *       this option is not provided, then the tool will run at the maximum rate
134     *       for the specified number of threads.</LI>
135     *   <LI>"--warmUpIntervals {num}" -- specifies the number of intervals to
136     *       complete before beginning overall statistics collection.</LI>
137     *   <LI>"--timestampFormat {format}" -- specifies the format to use for
138     *       timestamps included before each output line.  The format may be one of
139     *       "none" (for no timestamps), "with-date" (to include both the date and
140     *       the time), or "without-date" (to include only time time).</LI>
141     *   <LI>"-Y {authzID}" or "--proxyAs {authzID}" -- Use the proxied
142     *       authorization v2 control to request that the operations be processed
143     *       using an alternate authorization identity.  In this case, the bind DN
144     *       should be that of a user that has permission to use this control.  The
145     *       authorization identity may be a value pattern.</LI>
146     *   <LI>"--suppressErrorResultCodes" -- Indicates that information about the
147     *       result codes for failed operations should not be displayed.</LI>
148     *   <LI>"-c" or "--csv" -- Generate output in CSV format rather than a
149     *       display-friendly format.</LI>
150     * </UL>
151     */

495      /**
496       * Performs the actual processing for this tool.  In this case, it gets a
497       * connection to the directory server and uses it to perform the requested
498       * searches.
499       *
500       * @return  The result code for the processing that was performed.
501       */
502      @Override()
503      public ResultCode doToolProcessing()
504      {
505        // Convert the search scope from a string to an integer.
506        final SearchScope scope;
507        if (scopeStr.getValue().equalsIgnoreCase("base"))
508        {
509          scope = SearchScope.BASE;
510        }
511        else if (scopeStr.getValue().equalsIgnoreCase("one"))
512        {
513          scope = SearchScope.ONE;
514        }
515        else if (scopeStr.getValue().equalsIgnoreCase("subord"))
516        {
517          scope = SearchScope.SUBORDINATE_SUBTREE;
518        }
519        else
520        {
521          scope = SearchScope.SUB;
522        }
523    
524    
525        // Create value patterns for the base DN, filter, and proxied authorization
526        // DN.
527        final ValuePattern dnPattern;
528        try
529        {
530          dnPattern = new ValuePattern(baseDN.getValue());
531        }
532        catch (ParseException pe)
533        {
534          err("Unable to parse the base DN value pattern:  ", pe.getMessage());
535          return ResultCode.PARAM_ERROR;
536        }
537    
538        final ValuePattern filterPattern;
539        try
540        {
541          filterPattern = new ValuePattern(filter.getValue());
542        }
543        catch (ParseException pe)
544        {
545          err("Unable to parse the filter pattern:  ", pe.getMessage());
546          return ResultCode.PARAM_ERROR;
547        }
548    
549        final ValuePattern authzIDPattern;
550        if (proxyAs.isPresent())
551        {
552          try
553          {
554            authzIDPattern = new ValuePattern(proxyAs.getValue());
555          }
556          catch (ParseException pe)
557          {
558            err("Unable to parse the proxied authorization pattern:  ",
559                pe.getMessage());
560            return ResultCode.PARAM_ERROR;
561          }
562        }
563        else
564        {
565          authzIDPattern = null;
566        }
567    
568    
569        // Get the attributes to return.
570        final String[] returnAttrs;
571        if (returnAttributes.isPresent())
572        {
573          final List<String> attrList = returnAttributes.getValues();
574          returnAttrs = new String[attrList.size()];
575          attrList.toArray(returnAttrs);
576        }
577        else
578        {
579          returnAttrs = new String[0];
580        }
581    
582    
583        // Get the names of the attributes to modify.
584        final String[] modAttrs = new String[modifyAttributes.getValues().size()];
585        modifyAttributes.getValues().toArray(modAttrs);
586    
587    
588        // Get the character set as a byte array.
589        final byte[] charSet = getBytes(characterSet.getValue());
590    
591    
592        // If the --ratePerSecond option was specified, then limit the rate
593        // accordingly.
594        FixedRateBarrier fixedRateBarrier = null;
595        if (ratePerSecond.isPresent())
596        {
597          final int intervalSeconds = collectionInterval.getValue();
598          final int ratePerInterval = ratePerSecond.getValue() * intervalSeconds;
599    
600          fixedRateBarrier =
601               new FixedRateBarrier(1000L * intervalSeconds, ratePerInterval);
602        }
603    
604    
605        // Determine whether to include timestamps in the output and if so what
606        // format should be used for them.
607        final boolean includeTimestamp;
608        final String timeFormat;
609        if (timestampFormat.getValue().equalsIgnoreCase("with-date"))
610        {
611          includeTimestamp = true;
612          timeFormat       = "dd/MM/yyyy HH:mm:ss";
613        }
614        else if (timestampFormat.getValue().equalsIgnoreCase("without-date"))
615        {
616          includeTimestamp = true;
617          timeFormat       = "HH:mm:ss";
618        }
619        else
620        {
621          includeTimestamp = false;
622          timeFormat       = null;
623        }
624    
625    
626        // Determine whether any warm-up intervals should be run.
627        final long totalIntervals;
628        final boolean warmUp;
629        int remainingWarmUpIntervals = warmUpIntervals.getValue();
630        if (remainingWarmUpIntervals > 0)
631        {
632          warmUp = true;
633          totalIntervals = 0L + numIntervals.getValue() + remainingWarmUpIntervals;
634        }
635        else
636        {
637          warmUp = true;
638          totalIntervals = 0L + numIntervals.getValue();
639        }
640    
641    
642        // Create the table that will be used to format the output.
643        final OutputFormat outputFormat;
644        if (csvFormat.isPresent())
645        {
646          outputFormat = OutputFormat.CSV;
647        }
648        else
649        {
650          outputFormat = OutputFormat.COLUMNS;
651        }
652    
653        final ColumnFormatter formatter = new ColumnFormatter(includeTimestamp,
654             timeFormat, outputFormat, " ",
655             new FormattableColumn(12, HorizontalAlignment.RIGHT, "Recent",
656                      "Searches/Sec"),
657             new FormattableColumn(12, HorizontalAlignment.RIGHT, "Recent",
658                      "Srch Dur ms"),
659             new FormattableColumn(12, HorizontalAlignment.RIGHT, "Recent",
660                      "Mods/Sec"),
661             new FormattableColumn(12, HorizontalAlignment.RIGHT, "Recent",
662                      "Mod Dur ms"),
663             new FormattableColumn(12, HorizontalAlignment.RIGHT, "Recent",
664                      "Errors/Sec"),
665             new FormattableColumn(12, HorizontalAlignment.RIGHT, "Overall",
666                      "Searches/Sec"),
667             new FormattableColumn(12, HorizontalAlignment.RIGHT, "Overall",
668                      "Srch Dur ms"),
669             new FormattableColumn(12, HorizontalAlignment.RIGHT, "Overall",
670                      "Mods/Sec"),
671             new FormattableColumn(12, HorizontalAlignment.RIGHT, "Overall",
672                      "Mod Dur ms"));
673    
674    
675        // Create values to use for statistics collection.
676        final AtomicLong        searchCounter   = new AtomicLong(0L);
677        final AtomicLong        errorCounter    = new AtomicLong(0L);
678        final AtomicLong        modCounter      = new AtomicLong(0L);
679        final AtomicLong        modDurations    = new AtomicLong(0L);
680        final AtomicLong        searchDurations = new AtomicLong(0L);
681        final ResultCodeCounter rcCounter       = new ResultCodeCounter();
682    
683    
684        // Determine the length of each interval in milliseconds.
685        final long intervalMillis = 1000L * collectionInterval.getValue();
686    
687    
688        // Create the threads to use for the searches.
689        final Random random = new Random();
690        final CyclicBarrier barrier = new CyclicBarrier(numThreads.getValue() + 1);
691        final SearchAndModRateThread[] threads =
692             new SearchAndModRateThread[numThreads.getValue()];
693        for (int i=0; i < threads.length; i++)
694        {
695          final LDAPConnection connection;
696          try
697          {
698            connection = getConnection();
699          }
700          catch (LDAPException le)
701          {
702            err("Unable to connect to the directory server:  ",
703                getExceptionMessage(le));
704            return le.getResultCode();
705          }
706    
707          threads[i] = new SearchAndModRateThread(i, connection, dnPattern, scope,
708               filterPattern, returnAttrs, modAttrs, valueLength.getValue(),
709               charSet, authzIDPattern, random.nextLong(), barrier, searchCounter,
710               modCounter, searchDurations, modDurations, errorCounter, rcCounter,
711               fixedRateBarrier);
712          threads[i].start();
713        }
714    
715    
716        // Display the table header.
717        for (final String headerLine : formatter.getHeaderLines(true))
718        {
719          out(headerLine);
720        }
721    
722    
723        // Indicate that the threads can start running.
724        try
725        {
726          barrier.await();
727        } catch (Exception e) {}
728        long overallStartTime = System.nanoTime();
729        long nextIntervalStartTime = System.currentTimeMillis() + intervalMillis;
730    
731    
732        boolean setOverallStartTime = false;
733        long    lastSearchDuration  = 0L;
734        long    lastModDuration     = 0L;
735        long    lastNumEntries      = 0L;
736        long    lastNumErrors       = 0L;
737        long    lastNumSearches     = 0L;
738        long    lastNumMods          = 0L;
739        long    lastEndTime         = System.nanoTime();
740        for (long i=0; i < totalIntervals; i++)
741        {
742          final long startTimeMillis = System.currentTimeMillis();
743          final long sleepTimeMillis = nextIntervalStartTime - startTimeMillis;
744          nextIntervalStartTime += intervalMillis;
745          try
746          {
747            if (sleepTimeMillis > 0)
748            {
749              Thread.sleep(sleepTimeMillis);
750            }
751          } catch (Exception e) {}
752    
753          final long endTime          = System.nanoTime();
754          final long intervalDuration = endTime - lastEndTime;
755    
756          final long numSearches;
757          final long numMods;
758          final long numErrors;
759          final long totalSearchDuration;
760          final long totalModDuration;
761          if (warmUp && (remainingWarmUpIntervals > 0))
762          {
763            numSearches         = searchCounter.getAndSet(0L);
764            numMods             = modCounter.getAndSet(0L);
765            numErrors           = errorCounter.getAndSet(0L);
766            totalSearchDuration = searchDurations.getAndSet(0L);
767            totalModDuration    = modDurations.getAndSet(0L);
768          }
769          else
770          {
771            numSearches         = searchCounter.get();
772            numMods             = modCounter.get();
773            numErrors           = errorCounter.get();
774            totalSearchDuration = searchDurations.get();
775            totalModDuration    = modDurations.get();
776          }
777    
778          final long recentNumSearches = numSearches - lastNumSearches;
779          final long recentNumMods = numMods - lastNumMods;
780          final long recentNumErrors = numErrors - lastNumErrors;
781          final long recentSearchDuration =
782               totalSearchDuration - lastSearchDuration;
783          final long recentModDuration = totalModDuration - lastModDuration;
784    
785          final double numSeconds = intervalDuration / 1000000000.0D;
786          final double recentSearchRate = recentNumSearches / numSeconds;
787          final double recentModRate = recentNumMods / numSeconds;
788          final double recentErrorRate  = recentNumErrors / numSeconds;
789          final double recentAvgSearchDuration =
790                      1.0D * recentSearchDuration / recentNumSearches / 1000000;
791          final double recentAvgModDuration =
792                      1.0D * recentModDuration / recentNumMods / 1000000;
793    
794          if (warmUp && (remainingWarmUpIntervals > 0))
795          {
796            out(formatter.formatRow(recentSearchRate, recentAvgSearchDuration,
797                 recentModRate, recentAvgModDuration, recentErrorRate, "warming up",
798                 "warming up", "warming up", "warming up"));
799    
800            remainingWarmUpIntervals--;
801            if (remainingWarmUpIntervals == 0)
802            {
803              out("Warm-up completed.  Beginning overall statistics collection.");
804              setOverallStartTime = true;
805            }
806          }
807          else
808          {
809            if (setOverallStartTime)
810            {
811              overallStartTime    = lastEndTime;
812              setOverallStartTime = false;
813            }
814    
815            final double numOverallSeconds =
816                 (endTime - overallStartTime) / 1000000000.0D;
817            final double overallSearchRate = numSearches / numOverallSeconds;
818            final double overallModRate = numMods / numOverallSeconds;
819            final double overallAvgSearchDuration =
820                 1.0D * totalSearchDuration / numSearches / 1000000;
821            final double overallAvgModDuration =
822                 1.0D * totalModDuration / numMods / 1000000;
823    
824            out(formatter.formatRow(recentSearchRate, recentAvgSearchDuration,
825                 recentModRate, recentAvgModDuration, recentErrorRate,
826                 overallSearchRate, overallAvgSearchDuration, overallModRate,
827                 overallAvgModDuration));
828    
829            lastNumSearches    = numSearches;
830            lastNumMods        = numMods;
831            lastNumErrors      = numErrors;
832            lastSearchDuration = totalSearchDuration;
833            lastModDuration    = totalModDuration;
834          }
835    
836          final List<ObjectPair<ResultCode,Long>> rcCounts =
837               rcCounter.getCounts(true);
838          if ((! suppressErrors.isPresent()) && (! rcCounts.isEmpty()))
839          {
840            err("\tError Results:");
841            for (final ObjectPair<ResultCode,Long> p : rcCounts)
842            {
843              err("\t", p.getFirst().getName(), ":  ", p.getSecond());
844            }
845          }
846    
847          lastEndTime = endTime;
848        }
849    
850    
851        // Stop all of the threads.
852        ResultCode resultCode = ResultCode.SUCCESS;
853        for (final SearchAndModRateThread t : threads)
854        {
855          final ResultCode r = t.stopRunning();
856          if (resultCode == ResultCode.SUCCESS)
857          {
858            resultCode = r;
859          }
860        }
861    
862        return resultCode;
863      }
