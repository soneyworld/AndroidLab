055    /**
056     * This class implements the processing necessary to perform an LDAPv3 search
057     * operation, which can be used to retrieve entries that match a given set of
058     * criteria.  A search request may include the following elements:
059     * <UL>
060     *   <LI>Base DN -- Specifies the base DN for the search.  Only entries at or
061     *       below this location in the server (based on the scope) will be
062     *       considered potential matches.</LI>
063     *   <LI>Scope -- Specifies the range of entries relative to the base DN that
064     *       may be considered potential matches.</LI>
065     *   <LI>Dereference Policy -- Specifies the behavior that the server should
066     *       exhibit if any alias entries are encountered while processing the
067     *       search.  If no dereference policy is provided, then a default of
068     *       {@code DereferencePolicy.NEVER} will be used.</LI>
069     *   <LI>Size Limit -- Specifies the maximum number of entries that should be
070     *       returned from the search.  A value of zero indicates that there should
071     *       not be any limit enforced.  Note that the directory server may also
072     *       be configured with a server-side size limit which can also limit the
073     *       number of entries that may be returned to the client and in that case
074     *       the smaller of the client-side and server-side limits will be
075     *       used.  If no size limit is provided, then a default of zero (unlimited)
076     *       will be used.</LI>
077     *   <LI>Time Limit -- Specifies the maximum length of time in seconds that the
078     *       server should spend processing the search.  A value of zero indicates
079     *       that there should not be any limit enforced.  Note that the directory
080     *       server may also be configured with a server-side time limit which can
081     *       also limit the processing time, and in that case the smaller of the
082     *       client-side and server-side limits will be used.  If no time limit is
083     *       provided, then a default of zero (unlimited) will be used.</LI>
084     *   <LI>Types Only -- Indicates whether matching entries should include only
085     *       attribute names, or both attribute names and values.  If no value is
086     *       provided, then a default of {@code false} will be used.</LI>
087     *   <LI>Filter -- Specifies the criteria for determining which entries should
088     *       be returned.  See the {@link Filter} class for the types of filters
089     *       that may be used.
090     *       <BR><BR>
091     *       Note that filters can be specified using either their string
092     *       representations or as {@link Filter} objects.  As noted in the
093     *       documentation for the {@link Filter} class, using the string
094     *       representation may be somewhat dangerous if the data is not properly
095     *       sanitized because special characters contained in the filter may cause
096     *       it to be invalid or worse expose a vulnerability that could cause the
097     *       filter to request more information than was intended.  As a result, if
098     *       the filter may include special characters or user-provided strings,
099     *       then it is recommended that you use {@link Filter} objects created from
100     *       their individual components rather than their string representations.
101     * </LI>
102     *   <LI>Attributes -- Specifies the set of attributes that should be included
103     *       in matching entries.  If no attributes are provided, then the server
104     *       will default to returning all user attributes.  If a specified set of
105     *       attributes is given, then only those attributes will be included.
106     *       Values that may be included to indicate a special meaning include:
107     *       <UL>
108     *         <LI>{@code NO_ATTRIBUTES} -- Indicates that no attributes should be
109     *             returned.  That is, only the DNs of matching entries will be
110     *             returned.</LI>
111     *         <LI>{@code ALL_USER_ATTRIBUTES} -- Indicates that all user attributes
112     *             should be included in matching entries.  This is the default if
113     *             no attributes are provided, but this special value may be
114     *             included if a specific set of operational attributes should be
115     *             included along with all user attributes.</LI>
116     *         <LI>{@code ALL_OPERATIONAL_ATTRIBUTES} -- Indicates that all
117     *             operational attributes should be included in matching
118     *             entries.</LI>
119     *       </UL>
120     *       These special values may be used alone or in conjunction with each
121     *       other and/or any specific attribute names or OIDs.</LI>
122     *   <LI>An optional set of controls to include in the request to send to the
123     *       server.</LI>
124     *   <LI>An optional {@link SearchResultListener} which may be used to process
125     *       search result entries and search result references returned by the
126     *       server in the course of processing the request.  If this is
127     *       {@code null}, then the entries and references will be collected and
128     *       returned in the {@link SearchResult} object that is returned.</LI>
129     * </UL>
130     * When processing a search operation, there are three ways that the returned
131     * entries and references may be accessed:
132     * <UL>
133     *   <LI>If the {@link LDAPInterface#search(SearchRequest)} method is used and
134     *       the provided search request does not include a
135     *       {@link SearchResultListener} object, then the entries and references
136     *       will be collected internally and made available in the
137     *       {@link SearchResult} object that is returned.</LI>
138     *   <LI>If the {@link LDAPInterface#search(SearchRequest)} method is used and
139     *       the provided search request does include a {@link SearchResultListener}
140     *       object, then that listener will be used to provide access to the
141     *       entries and references, and they will not be present in the
142     *       {@link SearchResult} object (although the number of entries and
143     *       references returned will still be available).</LI>
144     *   <LI>The {@link LDAPEntrySource} object may be used to access the entries
145     *        and references returned from the search.  It uses an
146     *        {@code Iterator}-like API to provide access to the entries that are
147     *        returned, and any references returned will be included in the
148     *        {@link EntrySourceException} thrown on the appropriate call to
149     *        {@link LDAPEntrySource#nextEntry()}.</LI>
150     * </UL>
151     * <BR><BR>
152     * {@code SearchRequest} objects are mutable and therefore can be altered and
153     * re-used for multiple requests.  Note, however, that {@code SearchRequest}
154     * objects are not threadsafe and therefore a single {@code SearchRequest}
155     * object instance should not be used to process multiple requests at the same
156     * time.
157     * <BR><BR>
158     * <H2>Example</H2>
159     * The following example demonstrates a simple search operation in which the
160     * client performs a search to find all users in the "Sales" department and then
161     * prints out the name and e-mail address for each matching user:
162     * <PRE>
163     *   Filter filter = Filter.createEqualityFilter("ou", "Sales");
164     *
165     *   SearchRequest searchRequest =
166     *        new SearchRequest("dc=example,dc=com", SearchScope.SUB, filter,
167     *                          "cn", "mail");
168     *
169     *   try
170     *   {
171     *     SearchResult searchResult = connection.search(searchRequest);
172     *
173     *     for (SearchResultEntry entry : searchResult.getSearchEntries())
174     *     {
175     *       String name = entry.getAttributeValue("cn");
176     *       String mail = entry.getAttributeValue("mail");
177     *       System.out.println(name + "\t" + mail);
178     *     }
179     *   }
180     *   catch (LDAPSearchException lse)
181     *   {
182     *     System.err.println("The search failed.");
183     *   }
184     * </PRE>
185     */

1073      /**
1074       * Sends this search request to the directory server over the provided
1075       * connection and returns the associated response.  The search result entries
1076       * and references will either be collected and returned in the
1077       * {@code SearchResult} object that is returned, or will be interactively
1078       * returned via the {@code SearchResultListener} interface.
1079       *
1080       * @param  connection  The connection to use to communicate with the directory
1081       *                     server.
1082       * @param  depth       The current referral depth for this request.  It should
1083       *                     always be one for the initial request, and should only
1084       *                     be incremented when following referrals.
1085       *
1086       * @return  An object that provides information about the result of the
1087       *          search processing, potentially including the sets of matching
1088       *          entries and/or search references.
1089       *
1090       * @throws  LDAPException  If a problem occurs while sending the request or
1091       *                         reading the response.
1092       */
1093      @Override()
1094      protected SearchResult process(final LDAPConnection connection,
1095                                     final int depth)
1096                throws LDAPException
1097      {
1098        if (connection.synchronousMode())
1099        {
1100          return processSync(connection, depth);
1101        }
1102    
1103        final long requestTime = System.nanoTime();
1104        processAsync(connection, null);
1105    
1106        try
1107        {
1108          // Wait for and process the response.
1109          final ArrayList<SearchResultEntry> entryList;
1110          final ArrayList<SearchResultReference> referenceList;
1111          if (searchResultListener == null)
1112          {
1113            entryList     = new ArrayList<SearchResultEntry>(5);
1114            referenceList = new ArrayList<SearchResultReference>(5);
1115          }
1116          else
1117          {
1118            entryList     = null;
1119            referenceList = null;
1120          }
1121    
1122          int numEntries    = 0;
1123          int numReferences = 0;
1124          ResultCode intermediateResultCode = ResultCode.SUCCESS;
1125          final long responseTimeout = getResponseTimeoutMillis(connection);
1126          while (true)
1127          {
1128            final LDAPResponse response;
1129            try
1130            {
1131              if (responseTimeout > 0)
1132              {
1133                response =
1134                     responseQueue.poll(responseTimeout, TimeUnit.MILLISECONDS);
1135              }
1136              else
1137              {
1138                response = responseQueue.take();
1139              }
1140            }
1141            catch (InterruptedException ie)
1142            {
1143              debugException(ie);
1144              throw new LDAPException(ResultCode.LOCAL_ERROR,
1145                   ERR_SEARCH_INTERRUPTED.get(connection.getHostPort()), ie);
1146            }
1147    
1148            if (response == null)
1149            {
1150              final SearchResult searchResult =
1151                   new SearchResult(messageID, ResultCode.TIMEOUT,
1152                        ERR_SEARCH_CLIENT_TIMEOUT.get(responseTimeout,
1153                             connection.getHostPort()),
1154                        null, null, entryList, referenceList, numEntries,
1155                        numReferences, null);
1156              throw new LDAPSearchException(searchResult);
1157            }
1158    
1159            if (response instanceof ConnectionClosedResponse)
1160            {
1161              final ConnectionClosedResponse ccr =
1162                   (ConnectionClosedResponse) response;
1163              final String message = ccr.getMessage();
1164              if (message == null)
1165              {
1166                // The connection was closed while waiting for the response.
1167                final SearchResult searchResult =
1168                     new SearchResult(messageID, ccr.getResultCode(),
1169                          ERR_CONN_CLOSED_WAITING_FOR_SEARCH_RESPONSE.get(
1170                               connection.getHostPort(), toString()),
1171                          null, null, entryList, referenceList, numEntries,
1172                          numReferences, null);
1173                throw new LDAPSearchException(searchResult);
1174              }
1175              else
1176              {
1177                // The connection was closed while waiting for the response.
1178                final SearchResult searchResult =
1179                     new SearchResult(messageID, ccr.getResultCode(),
1180                          ERR_CONN_CLOSED_WAITING_FOR_SEARCH_RESPONSE_WITH_MESSAGE.
1181                               get(connection.getHostPort(), toString(), message),
1182                          null, null, entryList, referenceList, numEntries,
1183                          numReferences, null);
1184                throw new LDAPSearchException(searchResult);
1185              }
1186            }
1187            else if (response instanceof SearchResultEntry)
1188            {
1189              final SearchResultEntry searchEntry = (SearchResultEntry) response;
1190              numEntries++;
1191              if (searchResultListener == null)
1192              {
1193                entryList.add(searchEntry);
1194              }
1195              else
1196              {
1197                searchResultListener.searchEntryReturned(searchEntry);
1198              }
1199            }
1200            else if (response instanceof SearchResultReference)
1201            {
1202              final SearchResultReference searchReference =
1203                   (SearchResultReference) response;
1204              if (followReferrals(connection))
1205              {
1206                final LDAPResult result = followSearchReference(messageID,
1207                     searchReference, connection, depth);
1208                if (! result.getResultCode().equals(ResultCode.SUCCESS))
1209                {
1210                  // We couldn't follow the reference.  We don't want to fail the
1211                  // entire search because of this right now, so treat it as if
1212                  // referral following had not been enabled.  Also, set the
1213                  // intermediate result code to match that of the result.
1214                  numReferences++;
1215                  if (searchResultListener == null)
1216                  {
1217                    referenceList.add(searchReference);
1218                  }
1219                  else
1220                  {
1221                    searchResultListener.searchReferenceReturned(searchReference);
1222                  }
1223    
1224                  if (intermediateResultCode.equals(ResultCode.SUCCESS))
1225                  {
1226                    intermediateResultCode = result.getResultCode();
1227                  }
1228                }
1229                else if (result instanceof SearchResult)
1230                {
1231                  final SearchResult searchResult = (SearchResult) result;
1232                  numEntries += searchResult.getEntryCount();
1233                  if (searchResultListener == null)
1234                  {
1235                    entryList.addAll(searchResult.getSearchEntries());
1236                  }
1237                }
1238              }
1239              else
1240              {
1241                numReferences++;
1242                if (searchResultListener == null)
1243                {
1244                  referenceList.add(searchReference);
1245                }
1246                else
1247                {
1248                  searchResultListener.searchReferenceReturned(searchReference);
1249                }
1250              }
1251            }
1252            else
1253            {
1254              connection.getConnectionStatistics().incrementNumSearchResponses(
1255                   numEntries, numReferences,
1256                   (System.nanoTime() - requestTime));
1257              SearchResult result = (SearchResult) response;
1258              result.setCounts(numEntries, entryList, numReferences, referenceList);
1259    
1260              if ((result.getResultCode().equals(ResultCode.REFERRAL)) &&
1261                  followReferrals(connection))
1262              {
1263                if (depth >=
1264                    connection.getConnectionOptions().getReferralHopLimit())
1265                {
1266                  return new SearchResult(messageID,
1267                                          ResultCode.REFERRAL_LIMIT_EXCEEDED,
1268                                          ERR_TOO_MANY_REFERRALS.get(),
1269                                          result.getMatchedDN(),
1270                                          result.getReferralURLs(), entryList,
1271                                          referenceList, numEntries,
1272                                          numReferences,
1273                                          result.getResponseControls());
1274                }
1275    
1276                result = followReferral(result, connection, depth);
1277              }
1278    
1279              if ((result.getResultCode().equals(ResultCode.SUCCESS)) &&
1280                  (! intermediateResultCode.equals(ResultCode.SUCCESS)))
1281              {
1282                return new SearchResult(messageID, intermediateResultCode,
1283                                        result.getDiagnosticMessage(),
1284                                        result.getMatchedDN(),
1285                                        result.getReferralURLs(),
1286                                        entryList, referenceList, numEntries,
1287                                        numReferences,
1288                                        result.getResponseControls());
1289              }
1290    
1291              return result;
1292            }
1293          }
1294        }
1295        finally
1296        {
1297          connection.deregisterResponseAcceptor(messageID);
1298        }
1299      }

1308       *                         directory server.

1310       *                         when the response is received.  It may be
1311       *                         {@code null} only if the result is to be processed
1312       *                         by this class.

1321          throws LDAPException
1322      {
1323        // Create the LDAP message.
1324        messageID = connection.nextMessageID();
1325        final LDAPMessage message = new LDAPMessage(messageID, this, getControls());
1326

1328        // If the provided async result listener is {@code null}, then we'll use
1329        // this class as the message acceptor.  Otherwise, create an async helper
1330        // and use it as the message acceptor.
1331        if (resultListener == null)
1332        {
1333          connection.registerResponseAcceptor(messageID, this);
1334        }
1335        else
1336        {
1337          final AsyncSearchHelper helper = new AsyncSearchHelper(connection,
1338               resultListener, getIntermediateResponseListener());
1339          connection.registerResponseAcceptor(messageID, helper);
1340        }
1341    
1342    
1343        // Send the request to the server.
1344        try
1345        {
1346          debugLDAPRequest(this);
1347          connection.getConnectionStatistics().incrementNumSearchRequests();
1348          connection.sendMessage(message);
1349          return messageID;
1350        }
1351        catch (LDAPException le)
1352        {
1353          debugException(le);
1354    
1355          connection.deregisterResponseAcceptor(messageID);
1356          throw le;
1357        }

1380              throws LDAPException
1381      {
1382        // Create the LDAP message.
1383        messageID = connection.nextMessageID();
1384        final LDAPMessage message =
1385             new LDAPMessage(messageID,  this, getControls());
1386    
1387    
1388        // Set the appropriate timeout on the socket.
1389        final long responseTimeout = getResponseTimeoutMillis(connection);
1390        try
1391        {
1392          connection.getConnectionInternals().getSocket().setSoTimeout(
1393               (int) responseTimeout);
1394        }
1395        catch (Exception e)
1396        {
1397          debugException(e);
1398        }
1399    
1400    
1401        // Send the request to the server.
1402        final long requestTime = System.nanoTime();
1403        debugLDAPRequest(this);
1404        connection.getConnectionStatistics().incrementNumSearchRequests();
1405        connection.sendMessage(message);
1406    
1407        final ArrayList<SearchResultEntry> entryList;
1408        final ArrayList<SearchResultReference> referenceList;
1409        if (searchResultListener == null)
1410        {
1411          entryList     = new ArrayList<SearchResultEntry>(5);
1412          referenceList = new ArrayList<SearchResultReference>(5);
1413        }
1414        else
1415        {
1416          entryList     = null;
1417          referenceList = null;
1418        }
1419    
1420        int numEntries    = 0;
1421        int numReferences = 0;
1422        ResultCode intermediateResultCode = ResultCode.SUCCESS;
1423        while (true)
1424        {
1425          final LDAPResponse response = connection.readResponse(messageID);
1426    
1427          if (response == null)
1428          {
1429            throw new LDAPException(ResultCode.TIMEOUT,
1430                 ERR_SEARCH_CLIENT_TIMEOUT.get(responseTimeout,
1431                      connection.getHostPort()));
1432          }
1433          else if (response instanceof ConnectionClosedResponse)
1434          {
1435            final ConnectionClosedResponse ccr =
1436                 (ConnectionClosedResponse) response;
1437            final String msg = ccr.getMessage();
1438            if (msg == null)
1439            {
1440              // The connection was closed while waiting for the response.
1441              final SearchResult searchResult =
1442                   new SearchResult(messageID, ccr.getResultCode(),
1443                        ERR_CONN_CLOSED_WAITING_FOR_SEARCH_RESPONSE.get(
1444                             connection.getHostPort(), toString()),
1445                        null, null, entryList, referenceList, numEntries,
1446                        numReferences, null);
1447              throw new LDAPSearchException(searchResult);
1448            }
1449            else
1450            {
1451              // The connection was closed while waiting for the response.
1452              final SearchResult searchResult =
1453                   new SearchResult(messageID, ccr.getResultCode(),
1454                        ERR_CONN_CLOSED_WAITING_FOR_SEARCH_RESPONSE_WITH_MESSAGE.
1455                             get(connection.getHostPort(), toString(), msg),
1456                        null, null, entryList, referenceList, numEntries,
1457                        numReferences, null);
1458              throw new LDAPSearchException(searchResult);
1459            }
1460          }
1461          else if (response instanceof SearchResultEntry)
1462          {
1463            final SearchResultEntry searchEntry = (SearchResultEntry) response;
1464            numEntries++;
1465            if (searchResultListener == null)
1466            {
1467              entryList.add(searchEntry);
1468            }
1469            else
1470            {
1471              searchResultListener.searchEntryReturned(searchEntry);
1472            }
1473          }
1474          else if (response instanceof SearchResultReference)
1475          {
1476            final SearchResultReference searchReference =
1477                 (SearchResultReference) response;
1478            if (followReferrals(connection))
1479            {
1480              final LDAPResult result = followSearchReference(messageID,
1481                   searchReference, connection, depth);
1482              if (! result.getResultCode().equals(ResultCode.SUCCESS))
1483              {
1484                // We couldn't follow the reference.  We don't want to fail the
1485                // entire search because of this right now, so treat it as if
1486                // referral following had not been enabled.  Also, set the
1487                // intermediate result code to match that of the result.
1488                numReferences++;
1489                if (searchResultListener == null)
1490                {
1491                  referenceList.add(searchReference);
1492                }
1493                else
1494                {
1495                  searchResultListener.searchReferenceReturned(searchReference);
1496                }
1497    
1498                if (intermediateResultCode.equals(ResultCode.SUCCESS))
1499                {
1500                  intermediateResultCode = result.getResultCode();
1501                }
1502              }
1503              else if (result instanceof SearchResult)
1504              {
1505                final SearchResult searchResult = (SearchResult) result;
1506                numEntries += searchResult.getEntryCount();
1507                if (searchResultListener == null)
1508                {
1509                  entryList.addAll(searchResult.getSearchEntries());
1510                }
1511              }
1512            }
1513            else
1514            {
1515              numReferences++;
1516              if (searchResultListener == null)
1517              {
1518                referenceList.add(searchReference);
1519              }
1520              else
1521              {
1522                searchResultListener.searchReferenceReturned(searchReference);
1523              }
1524            }
1525          }
1526          else
1527          {
1528            return handleResponse(connection, response, requestTime, depth,
1529                                  numEntries, numReferences, entryList,
1530                                  referenceList, intermediateResultCode);
1531          }
1532        }
1533      }

1537      /**
1538       * Performs the necessary processing for handling a response.
1539       *
1540       * @param  connection              The connection used to read the response.
1541       * @param  response                The response to be processed.
1542       * @param  requestTime             The time the request was sent to the
1543       *                                 server.
1544       * @param  depth                   The current referral depth for this
1545       *                                 request.  It should always be one for the
1546       *                                 initial request, and should only be
1547       *                                 incremented when following referrals.
1548       * @param  numEntries              The number of entries received from the
1549       *                                 server.
1550       * @param  numReferences           The number of references received from
1551       *                                 the server.
1552       * @param  entryList               The list of search result entries received
1553       *                                 from the server, if applicable.
1554       * @param  referenceList           The list of search result references
1555       *                                 received from the server, if applicable.
1556       * @param  intermediateResultCode  The intermediate result code so far for the
1557       *                                 search operation.
1558       *
1559       * @return  The search result.
1560       *
1561       * @throws  LDAPException  If a problem occurs.
1562       */
1563      private SearchResult handleResponse(final LDAPConnection connection,
1564                   final LDAPResponse response, final long requestTime,
1565                   final int depth, final int numEntries, final int numReferences,
1566                   final List<SearchResultEntry> entryList,
1567                   final List<SearchResultReference> referenceList,
1568                   final ResultCode intermediateResultCode)
1569              throws LDAPException
1570      {
1571        connection.getConnectionStatistics().incrementNumSearchResponses(
1572             numEntries, numReferences,
1573             (System.nanoTime() - requestTime));
1574        SearchResult result = (SearchResult) response;
1575        result.setCounts(numEntries, entryList, numReferences, referenceList);
1576    
1577        if ((result.getResultCode().equals(ResultCode.REFERRAL)) &&
1578            followReferrals(connection))
1579        {
1580          if (depth >=
1581              connection.getConnectionOptions().getReferralHopLimit())
1582          {
1583            return new SearchResult(messageID,
1584                                    ResultCode.REFERRAL_LIMIT_EXCEEDED,
1585                                    ERR_TOO_MANY_REFERRALS.get(),
1586                                    result.getMatchedDN(),
1587                                    result.getReferralURLs(), entryList,
1588                                    referenceList, numEntries,
1589                                    numReferences,
1590                                    result.getResponseControls());
1591          }
1592    
1593          result = followReferral(result, connection, depth);
1594        }
1595    
1596        if ((result.getResultCode().equals(ResultCode.SUCCESS)) &&
1597            (! intermediateResultCode.equals(ResultCode.SUCCESS)))
1598        {
1599          return new SearchResult(messageID, intermediateResultCode,
1600                                  result.getDiagnosticMessage(),
1601                                  result.getMatchedDN(),
1602                                  result.getReferralURLs(),
1603                                  entryList, referenceList, numEntries,
1604                                  numReferences,
1605                                  result.getResponseControls());
1606        }
1607    
1608        return result;
1609      }

1613      /**
1614       * Attempts to follow a search result reference to continue a search in a
1615       * remote server.
1616       *
1617       * @param  messageID        The message ID for the LDAP message that is
1618       *                          associated with this result.
1619       * @param  searchReference  The search result reference to follow.
1620       * @param  connection       The connection on which the reference was
1621       *                          received.
1622       * @param  depth            The number of referrals followed in the course of
1623       *                          processing this request.
1624       *
1625       * @return  The result of attempting to follow the search result reference.
1626       *
1627       * @throws  LDAPException  If a problem occurs while attempting to establish
1628       *                         the referral connection, sending the request, or
1629       *                         reading the result.
1630       */
1631      private LDAPResult followSearchReference(final int messageID,
1632                              final SearchResultReference searchReference,
1633                              final LDAPConnection connection, final int depth)
1634              throws LDAPException
1635      {
1636        for (final String urlString : searchReference.getReferralURLs())
1637        {
1638          try
1639          {
1640            final LDAPURL referralURL = new LDAPURL(urlString);
1641            final String host = referralURL.getHost();
1642    
1643            if (host == null)
1644            {
1645              // We can't handle a referral in which there is no host.
1646              continue;
1647            }
1648    
1649            final String requestBaseDN;
1650            if (referralURL.baseDNProvided())
1651            {
1652              requestBaseDN = referralURL.getBaseDN().toString();
1653            }
1654            else
1655            {
1656              requestBaseDN = baseDN;
1657            }
1658    
1659            final SearchScope requestScope;
1660            if (referralURL.scopeProvided())
1661            {
1662              requestScope = referralURL.getScope();
1663            }
1664            else
1665            {
1666              requestScope = scope;
1667            }
1668    
1669            final Filter requestFilter;
1670            if (referralURL.filterProvided())
1671            {
1672              requestFilter = referralURL.getFilter();
1673            }
1674            else
1675            {
1676              requestFilter = filter;
1677            }
1678    
1679    
1680            final SearchRequest searchRequest =
1681                 new SearchRequest(searchResultListener, getControls(),
1682                                   requestBaseDN, requestScope, derefPolicy,
1683                                   sizeLimit, timeLimit, typesOnly, requestFilter,
1684                                   attributes);
1685    
1686            final LDAPConnection referralConn =
1687                 connection.getReferralConnection(referralURL, connection);
1688    
1689            try
1690            {
1691              return searchRequest.process(referralConn, depth+1);
1692            }
1693            finally
1694            {
1695              referralConn.setDisconnectInfo(DisconnectType.REFERRAL, null, null);
1696              referralConn.close();
1697            }
1698          }
1699          catch (LDAPException le)
1700          {
1701            debugException(le);
1702    
1703            if (le.getResultCode().equals(ResultCode.REFERRAL_LIMIT_EXCEEDED))
1704            {
1705              throw le;
1706            }
1707          }
1708        }
1709    
1710        // If we've gotten here, then we could not follow any of the referral URLs,
1711        // so we'll create a failure result.
1712        return new SearchResult(messageID, ResultCode.REFERRAL, null, null,
1713                                searchReference.getReferralURLs(), 0, 0, null);
1714      }
