040    /**
041     * This class provides an implementation of the LDAP post-read request control
042     * as defined in <A HREF="http://www.ietf.org/rfc/rfc4527.txt">RFC 4527</A>.  It
043     * may be used to request that the server retrieve a copy of the target entry as
044     * it appeared immediately after processing an add, modify, or modify DN
045     * operation.
046     * <BR><BR>
047     * If this control is included in an add, modify, or modify DN request, then the
048     * corresponding response may include a {@link PostReadResponseControl}
049     * containing a version of the entry as it appeared after applying that change.
050     * Note that this response control will only be included if the operation was
051     * successful, so it will not be provided if the operation failed for some
052     * reason (e.g., if the change would have violated the server schema, or if the
053     * requester did not have sufficient permission to perform that operation).
054     * <BR><BR>
055     * The value of this control should contain a set of requested attributes to
056     * include in the entry that is returned.  The server should treat this set of
057     * requested attributes exactly as it treats the requested attributes from a
058     * {@link com.unboundid.ldap.sdk.SearchRequest}.  As is the case with a search
059     * request, if no attributes are specified, then all user attributes will be
060     * included.
061     * <BR><BR>
062     * <H2>Example</H2>
063     * The following example demonstrates the use of the post-read controls.  It
064     * will modify an entry to increment the value of the {@code testCounter}
065     * attribute by one, and will use the post-read controls to determine what the
066     * new value is:
067     * <PRE>
068     *   Modification mod =
069     *        new Modification(ModificationType.INCREMENT, "testCounter", "1");
070     *   ModifyRequest modifyRequest =
071     *        new ModifyRequest("uid=john.doe,ou=People,dc=example,dc=com", mod);
072     *   modifyRequest.addControl(new PostReadRequestControl("testCounter"));
073     *   LDAPResult modifyResult = connection.modify(modifyRequest);
074     *
075     *   Integer newValue = null;
076     *   for (Control c : modifyResult.getResponseControls())
077     *   {
078     *     if (c instanceof PostReadResponseControl)
079     *     {
080     *       ReadOnlyEntry e = ((PostReadResponseControl) c).getEntry();
081     *       if (e.hasAttribute("testCounter"))
082     *       {
083     *         newValue = Integer.parseInt(e.getAttributeValue("testCounter"));
084     *       }
085     *     }
086     *   }
087     * </PRE>
088     */
089    @NotMutable()
090    @ThreadSafety(level=ThreadSafetyLevel.COMPLETELY_THREADSAFE)
091    public final class PostReadRequestControl
092           extends Control
093    {
094      /**
095       * The OID (1.3.6.1.1.13.2) for the post-read request control.
096       */
097      public static final String POST_READ_REQUEST_OID = "1.3.6.1.1.13.2";
098    
099    
100    
101      /**
102       * The set of requested attributes that will be used if none are provided.
103       */
104      private static final String[] NO_ATTRIBUTES = new String[0];
105    
106    
107    
108      /**
109       * The serial version UID for this serializable class.
110       */
111      private static final long serialVersionUID = -4210061989410209462L;
112    
113    
114    
115      // The set of requested attributes to retrieve from the target entry.
116      private final String[] attributes;
117    
118    
119    
120      /**
121       * Creates a new post-read request control that will retrieve the specified
122       * set of attributes from the target entry.  It will be marked critical.
123       *
124       * @param  attributes  The set of attributes to retrieve from the target
125       *                     entry.  It behaves in the same way as the set of
126       *                     requested attributes for a search operation.  If this
127       *                     is empty or {@code null}, then all user attributes
128       *                     will be returned.
129       */
130      public PostReadRequestControl(final String... attributes)
131      {
132        this(true, attributes);
133      }
134    
135    
136    
137      /**
138       * Creates a new post-read request control that will retrieve the specified
139       * set of attributes from the target entry.
140       *
141       * @param  isCritical  Indicates whether this control should be marked
142       *                     critical.
143       * @param  attributes  The set of attributes to retrieve from the target
144       *                     entry.  It behaves in the same way as the set of
145       *                     requested attributes for a search operation.  If this
146       *                     is empty or {@code null}, then all user attributes
147       *                     will be returned.
148       */
149      public PostReadRequestControl(final boolean isCritical,
150                                    final String... attributes)
151      {
152        super(POST_READ_REQUEST_OID, isCritical, encodeValue(attributes));
153    
154        if (attributes == null)
155        {
156          this.attributes = NO_ATTRIBUTES;
157        }
158        else
159        {
160          this.attributes = attributes;
161        }
162      }
163    
164    
165    
166      /**
167       * Creates a new post-read request control which is decoded from the provided
168       * generic control.
169       *
170       * @param  control  The generic control to be decoded as a post-read request
171       *                  control.
172       *
173       * @throws  LDAPException  If the provided control cannot be decoded as a
174       *                         post-read request control.
175       */
176      public PostReadRequestControl(final Control control)
177             throws LDAPException
178      {
179        super(control);
180    
181        final ASN1OctetString value = control.getValue();
182        if (value == null)
183        {
184          throw new LDAPException(ResultCode.DECODING_ERROR,
185                                  ERR_POST_READ_REQUEST_NO_VALUE.get());
186        }
187    
188        try
189        {
190          final ASN1Element valueElement = ASN1Element.decode(value.getValue());
191          final ASN1Element[] attrElements =
192               ASN1Sequence.decodeAsSequence(valueElement).elements();
193          attributes = new String[attrElements.length];
194          for (int i=0; i < attrElements.length; i++)
195          {
196            attributes[i] =
197                 ASN1OctetString.decodeAsOctetString(attrElements[i]).stringValue();
198          }
199        }
200        catch (Exception e)
201        {
202          debugException(e);
203          throw new LDAPException(ResultCode.DECODING_ERROR,
204                                  ERR_POST_READ_REQUEST_CANNOT_DECODE.get(e), e);
205        }
206      }
207    
208    
209    
210      /**
211       * Encodes the provided information into an octet string that can be used as
212       * the value for this control.
213       *
214       * @param  attributes  The set of attributes to retrieve from the target
215       *                     entry.  It behaves in the same way as the set of
216       *                     requested attributes for a search operation.  If this
217       *                     is empty or {@code null}, then all user attributes
218       *                     will be returned.
219       *
220       * @return  An ASN.1 octet string that can be used as the value for this
221       *          control.
222       */
223      private static ASN1OctetString encodeValue(final String[] attributes)
224      {
225        if ((attributes == null) || (attributes.length == 0))
226        {
227          return new ASN1OctetString(new ASN1Sequence().encode());
228        }
229    
230        final ASN1OctetString[] elements = new ASN1OctetString[attributes.length];
231        for (int i=0; i < attributes.length; i++)
232        {
233          elements[i] = new ASN1OctetString(attributes[i]);
234        }
235    
236        return new ASN1OctetString(new ASN1Sequence(elements).encode());
237      }
238    
239    
240    
241      /**
242       * Retrieves the set of attributes that will be requested for inclusion in the
243       * entry returned in the response control.
244       *
245       * @return  The set of attributes that will be requested for inclusion in the
246       *          entry returned in the response control, or an empty array if all
247       *          user attributes should be returned.
248       */
249      public String[] getAttributes()
250      {
251        return attributes;
252      }
253    
254    
255    
256      /**
257       * {@inheritDoc}
258       */
259      @Override()
260      public String getControlName()
261      {
262        return INFO_CONTROL_NAME_POST_READ_REQUEST.get();
263      }
264    
265    
266    
267      /**
268       * {@inheritDoc}
269       */
270      @Override()
271      public void toString(final StringBuilder buffer)
272      {
273        buffer.append("PostReadRequestControl(attributes={");
274        for (int i=0; i < attributes.length; i++)
275        {
276          if (i > 0)
277          {
278            buffer.append(", ");
279          }
280          buffer.append(attributes[i]);
281        }
282        buffer.append("}, isCritical=");
283        buffer.append(isCritical());
284        buffer.append(')');
285      }
286    }
