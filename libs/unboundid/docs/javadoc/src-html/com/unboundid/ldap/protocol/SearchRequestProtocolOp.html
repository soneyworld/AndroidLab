050    /**
051     * This class provides an implementation of an LDAP search request protocol op.
052     */
053    @InternalUseOnly()
054    @NotMutable()
055    @ThreadSafety(level=ThreadSafetyLevel.COMPLETELY_THREADSAFE)
056    public final class SearchRequestProtocolOp
057           implements ProtocolOp
058    {

061       */
062      private static final long serialVersionUID = -8521750809606744181L;

067      private final boolean typesOnly;

071    
072      // The filter for this search request.
073      private final Filter filter;
074    
075      // The size limit for this search request.
076      private final int sizeLimit;
077    
078      // The time limit for this search request.
079      private final int timeLimit;
080    
081      // The set of attributes for this search request.
082      private final List<String> attributes;
083    
084      // The scope for this search request.
085      private final SearchScope scope;
086    
087      // The base DN for this search request.
088      private final String baseDN;

092      /**
093       * Creates a new search request protocol op with the provided information.
094       *
095       * @param  baseDN       The base DN for this search request.
096       * @param  scope        The scope for this search request.
097       * @param  derefPolicy  The policy to use for aliases encountered during the
098       *                      search.
099       * @param  sizeLimit    The maximum number of entries to return for the
100       *                      search, or zero for no limit.
101       * @param  timeLimit    The maximum length of time to spend processing the
102       *                      search, or zero for no limit.
103       * @param  typesOnly    Indicates whether to return only attribute types or
104       *                      both types and values.
105       * @param  filter       The filter for this search request.
106       * @param  attributes   The names of attributes to include in matching
107       *                      entries.
108       */
109      public SearchRequestProtocolOp(final String baseDN, final SearchScope scope,
110                  final DereferencePolicy derefPolicy, final int sizeLimit,
111                  final int timeLimit, final boolean typesOnly, final Filter filter,
112                  final List<String> attributes)
113      {
114        this.scope       = scope;
115        this.derefPolicy = derefPolicy;
116        this.typesOnly   = typesOnly;
117        this.filter      = filter;
118    
119        if (baseDN == null)
120        {
121          this.baseDN = "";
122        }
123        else
124        {
125          this.baseDN = baseDN;
126        }
127    
128        if (sizeLimit > 0)
129        {
130          this.sizeLimit = sizeLimit;
131        }
132        else
133        {
134          this.sizeLimit = 0;
135        }
136    
137        if (timeLimit > 0)
138        {
139          this.timeLimit = timeLimit;
140        }
141        else
142        {
143          this.timeLimit = 0;
144        }
145    
146        if (attributes == null)
147        {
148          this.attributes = Collections.emptyList();
149        }
150        else
151        {
152          this.attributes = Collections.unmodifiableList(attributes);
153        }
154      }
155    
156    
157    
158      /**
159       * Creates a new search request protocol op read from the provided ASN.1
160       * stream reader.
161       *
162       * @param  reader  The ASN.1 stream reader from which to read the search
163       *                 request protocol op.
164       *
165       * @throws  LDAPException  If a problem occurs while reading or parsing the
166       *                         search request.
167       */
168      SearchRequestProtocolOp(final ASN1StreamReader reader)
169           throws LDAPException
170      {
171        try
172        {
173          reader.beginSequence();
174          baseDN      = reader.readString();
175          scope       = SearchScope.valueOf(reader.readEnumerated());
176          derefPolicy = DereferencePolicy.valueOf(reader.readEnumerated());
177          sizeLimit   = reader.readInteger();
178          timeLimit   = reader.readInteger();
179          typesOnly   = reader.readBoolean();
180          filter      = Filter.readFrom(reader);
181    
182          final ArrayList<String> attrs = new ArrayList<String>(5);
183          final ASN1StreamReaderSequence attrSequence = reader.beginSequence();
184          while (attrSequence.hasMoreElements())
185          {
186            attrs.add(reader.readString());
187          }
188    
189          attributes = Collections.unmodifiableList(attrs);
190        }
191        catch (LDAPException le)
192        {
193          debugException(le);
194          throw le;
195        }
196        catch (Exception e)
197        {
198          debugException(e);
199    
200          throw new LDAPException(ResultCode.DECODING_ERROR,
201               ERR_SEARCH_REQUEST_CANNOT_DECODE.get(getExceptionMessage(e)), e);
202        }
203      }

312      public byte getProtocolOpType()
313      {
314        return LDAPMessage.PROTOCOL_OP_TYPE_SEARCH_REQUEST;
315      }

317    
318    
319      /**
320       * {@inheritDoc}
321       */
322      public void writeTo(final ASN1Buffer buffer)
323      {
324        final ASN1BufferSequence opSequence =
325             buffer.beginSequence(LDAPMessage.PROTOCOL_OP_TYPE_SEARCH_REQUEST);
326        buffer.addOctetString(baseDN);
327        buffer.addEnumerated(scope.intValue());
328        buffer.addEnumerated(derefPolicy.intValue());
329        buffer.addInteger(sizeLimit);
330        buffer.addInteger(timeLimit);
331        buffer.addBoolean(typesOnly);
332        filter.writeTo(buffer);
333    
334        final ASN1BufferSequence attrSequence = buffer.beginSequence();
335        for (final String s : attributes)
336        {
337          buffer.addOctetString(s);
338        }
339        attrSequence.end();
340        opSequence.end();
341      }

353        final StringBuilder buffer = new StringBuilder();
354        toString(buffer);
355        return buffer.toString();
356      }

360      /**
361       * {@inheritDoc}
362       */
363      public void toString(final StringBuilder buffer)
364      {
365        buffer.append("SearchRequestProtocolOp(baseDN='");
366        buffer.append(baseDN);
367        buffer.append("', scope='");
368        buffer.append(scope.toString());
369        buffer.append("', derefPolicy='");
370        buffer.append(derefPolicy.toString());
371        buffer.append("', sizeLimit=");
372        buffer.append(sizeLimit);
373        buffer.append(", timeLimit=");
374        buffer.append(timeLimit);
375        buffer.append(", typesOnly=");
376        buffer.append(typesOnly);
377        buffer.append(", filter='");
378        filter.toString(buffer);
379        buffer.append("', attributes={");
380    
381        final Iterator<String> iterator = attributes.iterator();
382        while (iterator.hasNext())
383        {
384          buffer.append(iterator.next());
385          if (iterator.hasNext())
386          {
387            buffer.append(',');
388          }
389        }
390    
391        buffer.append("})");
392      }
