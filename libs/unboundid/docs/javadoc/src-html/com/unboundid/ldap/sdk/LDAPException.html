033    /**
034     * This class defines an exception that can be thrown if a problem occurs while
035     * performing LDAP-related processing.  An LDAP exception can include all of
036     * the elements of an {@link LDAPResult}, so that all of the response elements
037     * will be available.
038     */
039    @NotExtensible()
040    @NotMutable()
041    @ThreadSafety(level=ThreadSafetyLevel.COMPLETELY_THREADSAFE)
042    public class LDAPException
043           extends LDAPSDKException
044    {
045      /**
046       * The serial version UID for this serializable class.
047       */
048      private static final long serialVersionUID = -4257171063946350327L;
049    
050    
051    
052      /**
053       * An empty array that will be used when no controls were provided.
054       */
055      protected static final Control[] NO_CONTROLS = new Control[0];
056    
057    
058    
059      /**
060       * An empty array that will be used when no referrals were provided.
061       */
062      protected static final String[] NO_REFERRALS = new String[0];
063    
064    
065    
066      // The set of response controls for this LDAP exception.
067      private final Control[] responseControls;
068    
069      // The result code for this LDAP exception.
070      private final ResultCode resultCode;
071    
072      // The set of referral URLs for this LDAP exception.
073      private final String[] referralURLs;
074    
075      // The matched DN for this LDAP exception.
076      private final String matchedDN;
077    
078    
079    
080      /**
081       * Creates a new LDAP exception with the provided result code.  A default
082       * message (based on the result code) will be used.
083       *
084       * @param  resultCode  The result code for this LDAP exception.
085       */
086      public LDAPException(final ResultCode resultCode)
087      {
088        super(resultCode.getName());
089    
090        this.resultCode = resultCode;
091    
092        matchedDN        = null;
093        referralURLs     = NO_REFERRALS;
094        responseControls = NO_CONTROLS;
095      }
096

102       *
103       * @param  resultCode  The result code for this LDAP exception.
104       * @param  cause       The underlying exception that triggered this exception.
105       */
106      public LDAPException(final ResultCode resultCode, final Throwable cause)
107      {
108        super(resultCode.getName(), cause);
109    
110        this.resultCode = resultCode;
111

114        responseControls = NO_CONTROLS;
115      }

160      /**
161       * Creates a new LDAP exception with the provided information.
162       *
163       * @param  resultCode    The result code for this LDAP exception.
164       * @param  errorMessage  The error message for this LDAP exception.
165       * @param  matchedDN     The matched DN for this LDAP exception.
166       * @param  referralURLs  The set of referral URLs for this LDAP exception.
167       */
168      public LDAPException(final ResultCode resultCode, final String errorMessage,
169                           final String matchedDN, final String[] referralURLs)
170      {
171        super(errorMessage);
172    
173        this.resultCode = resultCode;
174        this.matchedDN  = matchedDN;
175    
176        if (referralURLs == null)
177        {
178          this.referralURLs = NO_REFERRALS;
179        }
180        else
181        {
182          this.referralURLs = referralURLs;
183        }
184    
185        responseControls = NO_CONTROLS;
186      }

303      /**
304       * Creates a new LDAP exception using the information contained in the
305       * provided LDAP result object.
306       *
307       * @param  ldapResult  The LDAP result object containing the information to
308       *                     use for this LDAP exception.
309       */
310      public LDAPException(final LDAPResult ldapResult)
311      {
312        super((ldapResult.getDiagnosticMessage() == null)
313              ? ldapResult.getResultCode().getName()
314              : ldapResult.getDiagnosticMessage());
315    
316        resultCode       = ldapResult.getResultCode();
317        matchedDN        = ldapResult.getMatchedDN();
318        referralURLs     = ldapResult.getReferralURLs();
319        responseControls = ldapResult.getResponseControls();
320      }

328       * @param  ldapResult  The LDAP result object containing the information to

330       * @param  cause       The underlying exception that triggered this exception.
331       */
332      public LDAPException(final LDAPResult ldapResult, final Throwable cause)
333      {

372      /**
373       * Retrieves the set of referral URLs for this LDAP exception.
374       *
375       * @return  The set of referral URLs for this LDAP exception, or an empty
376       *          list if there are none.
377       */
378      public final String[] getReferralURLs()
379      {
380        return referralURLs;
381      }

385      /**
386       * Indicates whether this result contains at least one control.
387       *
388       * @return  {@code true} if this result contains at least one control, or
389       *          {@code false} if not.
390       */
391      public final boolean hasResponseControl()
392      {
393        return (responseControls.length > 0);
394      }
395    
396    
397    
398      /**
399       * Indicates whether this result contains at least one control with the
400       * specified OID.
401       *
402       * @param  oid  The object identifier for which to make the determination.  It
403       *              must not be {@code null}.
404       *
405       * @return  {@code true} if this result contains at least one control with
406       *          the specified OID, or {@code false} if not.
407       */
408      public final boolean hasResponseControl(final String oid)
409      {
410        for (final Control c : responseControls)
411        {
412          if (c.getOID().equals(oid))
413          {
414            return true;
415          }
416        }
417    
418        return false;
419      }
420    
421    
422    
423      /**
424       * Retrieves the set of response controls for this LDAP exception.
425       *
426       * @return  The set of response controls for this LDAP exception, or an empty
427       *          list if there are none.
428       */
429      public final Control[] getResponseControls()
430      {
431        return responseControls;
432      }
433    
434    
435    
436      /**
437       * Retrieves the response control with the specified OID.
438       *
439       * @param  oid  The OID of the control to retrieve.
440       *
441       * @return  The response control with the specified OID, or {@code null} if
442       *          there is no such control.
443       */
444      public final Control getResponseControl(final String oid)
445      {
446        for (final Control c : responseControls)
447        {
448          if (c.getOID().equals(oid))
449          {
450            return c;
451          }
452        }
453    
454        return null;
455      }

459      /**
460       * Creates a new {@code LDAPResult} object from this exception.
461       *
462       * @return  The {@code LDAPResult} object created from this exception.
463       */
464      public final LDAPResult toLDAPResult()
465      {
466        return new LDAPResult(-1, resultCode, getMessage(), matchedDN, referralURLs,
467                              responseControls);
468      }

472      /**
473       * {@inheritDoc}
474       */
475      @Override()
476      public void toString(final StringBuilder buffer)
477      {
478        buffer.append("LDAPException(resultCode=");
479        buffer.append(resultCode);
480    
481        final String errorMessage = getMessage();
482        if (errorMessage != null)
483        {
484          buffer.append(", errorMessage='");
485          buffer.append(errorMessage);
486          buffer.append('\'');
487        }
488    
489        if (matchedDN != null)
490        {
491          buffer.append(", matchedDN='");
492          buffer.append(matchedDN);
493          buffer.append('\'');
494        }
495    
496        if (referralURLs.length > 0)
497        {
498          buffer.append(", referralURLs={");
499    
500          for (int i=0; i < referralURLs.length; i++)
501          {
502            if (i > 0)
503            {
504              buffer.append(", ");
505            }
506    
507            buffer.append('\'');
508            buffer.append(referralURLs[i]);
509            buffer.append('\'');
510          }
511    
512          buffer.append('}');
513        }
514    
515        if (responseControls.length > 0)
516        {
517          buffer.append(", responseControls={");
518    
519          for (int i=0; i < responseControls.length; i++)
520          {
521            if (i > 0)
522            {
523              buffer.append(", ");
524            }
525    
526            buffer.append(responseControls[i]);
527          }
528    
529          buffer.append('}');
530        }
531    
532        buffer.append(')');
533      }
