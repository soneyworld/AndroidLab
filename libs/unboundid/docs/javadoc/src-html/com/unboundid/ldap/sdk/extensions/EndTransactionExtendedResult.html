048    /**
049     * This class provides an implementation of the end transaction extended result
050     * as defined in RFC 5805.  It is able to decode a generic extended result to
051     * extract the appropriate response information.
052     * <BR><BR>
053     * See the documentation for the {@link StartTransactionExtendedRequest} class
054     * for an example of performing a transaction.
055     */
056    @NotMutable()
057    @ThreadSafety(level=ThreadSafetyLevel.COMPLETELY_THREADSAFE)
058    public final class EndTransactionExtendedResult
059           extends ExtendedResult
060    {
061      /**
062       * The serial version UID for this serializable class.
063       */
064      private static final long serialVersionUID = 1514265185948328221L;
065    
066    
067    
068      // The message ID for the operation that failed, if applicable.
069      private final int failedOpMessageID;
070    
071      // A mapping of the response controls for the operations performed as part of
072      // the transaction.
073      private final TreeMap<Integer,Control[]> opResponseControls;
074    
075    
076    
077      /**
078       * Creates a new end transaction extended result from the provided extended
079       * result.
080       *
081       * @param  extendedResult  The extended result to be decoded as an end
082       *                         transaction extended result.  It must not be
083       *                         {@code null}.
084       *
085       * @throws  LDAPException  If a problem occurs while attempting to decode the
086       *                         provided extended result as an end transaction
087       *                         extended result.
088       */
089      public EndTransactionExtendedResult(final ExtendedResult extendedResult)
090             throws LDAPException
091      {
092        super(extendedResult);
093    
094        opResponseControls = new TreeMap<Integer,Control[]>();
095    
096        final ASN1OctetString value = extendedResult.getValue();
097        if (value == null)
098        {
099          failedOpMessageID = -1;
100          return;
101        }
102    
103        final ASN1Sequence valueSequence;
104        try
105        {
106          final ASN1Element valueElement = ASN1Element.decode(value.getValue());
107          valueSequence = ASN1Sequence.decodeAsSequence(valueElement);
108        }
109        catch (final ASN1Exception ae)
110        {
111          debugException(ae);
112          throw new LDAPException(ResultCode.DECODING_ERROR,
113               ERR_END_TXN_RESPONSE_VALUE_NOT_SEQUENCE.get(ae.getMessage()), ae);
114        }
115    
116        final ASN1Element[] valueElements = valueSequence.elements();
117        if (valueElements.length == 0)
118        {
119          failedOpMessageID = -1;
120          return;
121        }
122        else if (valueElements.length > 2)
123        {
124          throw new LDAPException(ResultCode.DECODING_ERROR,
125               ERR_END_TXN_RESPONSE_INVALID_ELEMENT_COUNT.get(
126                    valueElements.length));
127        }
128    
129        int msgID = -1;
130        for (final ASN1Element e : valueElements)
131        {
132          if (e.getType() == ASN1Constants.UNIVERSAL_INTEGER_TYPE)
133          {
134            try
135            {
136              msgID = ASN1Integer.decodeAsInteger(e).intValue();
137            }
138            catch (final ASN1Exception ae)
139            {
140              debugException(ae);
141              throw new LDAPException(ResultCode.DECODING_ERROR,
142                   ERR_END_TXN_RESPONSE_CANNOT_DECODE_MSGID.get(ae), ae);
143            }
144          }
145          else if (e.getType() == ASN1Constants.UNIVERSAL_SEQUENCE_TYPE)
146          {
147            decodeOpControls(e, opResponseControls);
148          }
149          else
150          {
151            throw new LDAPException(ResultCode.DECODING_ERROR,
152                 ERR_END_TXN_RESPONSE_INVALID_TYPE.get(toHex(e.getType())));
153          }
154        }
155    
156        failedOpMessageID = msgID;
157      }

161      /**
162       * Creates a new end transaction extended result with the provided
163       * information.
164       *
165       * @param  messageID           The message ID for the LDAP message that is
166       *                             associated with this LDAP result.
167       * @param  resultCode          The result code from the response.
168       * @param  diagnosticMessage   The diagnostic message from the response, if
169       *                             available.
170       * @param  matchedDN           The matched DN from the response, if available.
171       * @param  referralURLs        The set of referral URLs from the response, if
172       *                             available.
173       * @param  failedOpMessageID   The message ID for the operation that failed,
174       *                             or {@code null} if there was no failure.
175       * @param  opResponseControls  A map containing the response controls for each
176       *                             operation, indexed by message ID.  It may be
177       *                             {@code null} if there were no response
178       *                             controls.
179       * @param  responseControls    The set of controls from the response, if
180       *                             available.
181       */
182      public EndTransactionExtendedResult(final int messageID,
183                  final ResultCode resultCode, final String diagnosticMessage,
184                  final String matchedDN, final String[] referralURLs,
185                  final Integer failedOpMessageID,
186                  final Map<Integer,Control[]> opResponseControls,
187                  final Control[] responseControls)
188      {
189        super(messageID, resultCode, diagnosticMessage, matchedDN, referralURLs,
190              null, encodeValue(failedOpMessageID, opResponseControls),
191                                responseControls);
192    
193        if ((failedOpMessageID == null) || (failedOpMessageID <= 0))
194        {
195          this.failedOpMessageID = -1;
196        }
197        else
198        {
199          this.failedOpMessageID = failedOpMessageID;
200        }
201    
202        if (opResponseControls == null)
203        {
204          this.opResponseControls = new TreeMap<Integer,Control[]>();
205        }
206        else
207        {
208          this.opResponseControls =
209               new TreeMap<Integer,Control[]>(opResponseControls);
210        }
211      }

215      /**
216       * Decodes the provided ASN.1 element as an update controls sequence.  Each
217       * element of the sequence should itself be a sequence containing the message
218       * ID associated with the operation in which the control was returned and a
219       * sequence of the controls included in the response for that operation.
220       *
221       * @param  element     The ASN.1 element to be decoded.
222       * @param  controlMap  The map into which to place the decoded controls.
223       *
224       * @throws  LDAPException  If a problem occurs while attempting to decode the
225       *                         contents of the provided ASN.1 element.
226       */
227      private static void decodeOpControls(final ASN1Element element,
228                                           final Map<Integer,Control[]> controlMap)
229              throws LDAPException
230      {
231        final ASN1Sequence ctlsSequence;
232        try
233        {
234          ctlsSequence = ASN1Sequence.decodeAsSequence(element);
235        }
236        catch (final ASN1Exception ae)
237        {
238          debugException(ae);
239          throw new LDAPException(ResultCode.DECODING_ERROR,
240                         ERR_END_TXN_RESPONSE_CONTROLS_NOT_SEQUENCE.get(ae), ae);
241        }
242    
243        for (final ASN1Element e : ctlsSequence.elements())
244        {
245          final ASN1Sequence ctlSequence;
246          try
247          {
248            ctlSequence = ASN1Sequence.decodeAsSequence(e);
249          }
250          catch (final ASN1Exception ae)
251          {
252            debugException(ae);
253            throw new LDAPException(ResultCode.DECODING_ERROR,
254                           ERR_END_TXN_RESPONSE_CONTROL_NOT_SEQUENCE.get(ae), ae);
255          }
256    
257          final ASN1Element[] ctlSequenceElements = ctlSequence.elements();
258          if (ctlSequenceElements.length != 2)
259          {
260            throw new LDAPException(ResultCode.DECODING_ERROR,
261                           ERR_END_TXN_RESPONSE_CONTROL_INVALID_ELEMENT_COUNT.get(
262                                ctlSequenceElements.length));
263          }
264    
265          final int msgID;
266          try
267          {
268            msgID = ASN1Integer.decodeAsInteger(ctlSequenceElements[0]).intValue();
269          }
270          catch (final ASN1Exception ae)
271          {
272            debugException(ae);
273            throw new LDAPException(ResultCode.DECODING_ERROR,
274                           ERR_END_TXN_RESPONSE_CONTROL_MSGID_NOT_INT.get(ae), ae);
275          }
276    
277          final ASN1Sequence controlsSequence;
278          try
279          {
280            controlsSequence =
281                 ASN1Sequence.decodeAsSequence(ctlSequenceElements[1]);
282          }
283          catch (final ASN1Exception ae)
284          {
285            debugException(ae);
286            throw new LDAPException(ResultCode.DECODING_ERROR,
287                 ERR_END_TXN_RESPONSE_CONTROLS_ELEMENT_NOT_SEQUENCE.get(ae), ae);
288          }
289    
290          final Control[] controls = Control.decodeControls(controlsSequence);
291          if (controls.length == 0)
292          {
293            continue;
294          }
295    
296          controlMap.put(msgID, controls);
297        }
298      }

302      /**
303       * Encodes the provided information into an appropriate value for this
304       * control.
305       *
306       * @param  failedOpMessageID   The message ID for the operation that failed,
307       *                             or {@code null} if there was no failure.
308       * @param  opResponseControls  A map containing the response controls for each
309       *                             operation, indexed by message ID.  It may be
310       *                             {@code null} if there were no response
311       *                             controls.
312       *
313       * @return  An ASN.1 octet string containing the encoded value for this
314       *          control, or {@code null} if there should not be a value.
315       */
316      private static ASN1OctetString encodeValue(final Integer failedOpMessageID,
317                          final Map<Integer,Control[]> opResponseControls)
318      {
319        if ((failedOpMessageID == null) && (opResponseControls == null))
320        {
321          return null;
322        }
323    
324        final ArrayList<ASN1Element> elements = new ArrayList<ASN1Element>(2);
325        if (failedOpMessageID != null)
326        {
327          elements.add(new ASN1Integer(failedOpMessageID));
328        }
329    
330        if ((opResponseControls != null) && (! opResponseControls.isEmpty()))
331        {
332          final ArrayList<ASN1Element> controlElements =
333               new ArrayList<ASN1Element>();
334          for (final Map.Entry<Integer,Control[]> e : opResponseControls.entrySet())
335          {
336            final ASN1Element[] ctlElements =
337            {
338              new ASN1Integer(e.getKey()),
339              Control.encodeControls(e.getValue())
340            };
341            controlElements.add(new ASN1Sequence(ctlElements));
342          }
343    
344          elements.add(new ASN1Sequence(controlElements));
345        }
346    
347        return new ASN1OctetString(new ASN1Sequence(elements).encode());
348      }

353      /**
354       * Retrieves the message ID of the operation that caused the transaction
355       * processing to fail, if applicable.
356       *
357       * @return  The message ID of the operation that caused the transaction
358       *          processing to fail, or -1 if no message ID was included in the
359       *          end transaction response.
360       */
361      public int getFailedOpMessageID()
362      {
363        return failedOpMessageID;
364      }

368      /**
369       * Retrieves the set of response controls returned by the operations
370       * processed as part of the transaction.  The value returned will contain a
371       * mapping between the message ID of the associated request message and a list
372       * of the response controls for that operation.
373       *
374       * @return  The set of response controls returned by the operations processed
375       *          as part of the transaction.  It may be an empty map if none of the
376       *          operations had any response controls.
377       */
378      public Map<Integer,Control[]> getOperationResponseControls()
379      {
380        return opResponseControls;
381      }
382    
383    
384    
385      /**
386       * Retrieves the set of response controls returned by the specified operation
387       * processed as part of the transaction.
388       *
389       * @param  messageID  The message ID of the operation for which to retrieve
390       *                    the response controls.
391       *
392       * @return  The response controls for the specified operation, or
393       *          {@code null} if there were no controls returned for the specified
394       *          operation.
395       */
396      public Control[] getOperationResponseControls(final int messageID)
397      {
398        return opResponseControls.get(messageID);
399      }

414      /**
415       * Appends a string representation of this extended result to the provided
416       * buffer.
417       *
418       * @param  buffer  The buffer to which a string representation of this
419       *                 extended result will be appended.
420       */
421      @Override()
422      public void toString(final StringBuilder buffer)
423      {
424        buffer.append("EndTransactionExtendedResult(resultCode=");
425        buffer.append(getResultCode());
426    
427        if (failedOpMessageID > 0)
428        {
429          buffer.append(", failedOpMessageID=");
430          buffer.append(failedOpMessageID);
431        }
432    
433        if (! opResponseControls.isEmpty())
434        {
435          buffer.append(", opResponseControls={");
436    
437          for (final int msgID : opResponseControls.keySet())
438          {
439            buffer.append("opMsgID=");
440            buffer.append(msgID);
441            buffer.append(", opControls={");
442    
443            boolean first = true;
444            for (final Control c : opResponseControls.get(msgID))
445            {
446              if (first)
447              {
448                first = false;
449              }
450              else
451              {
452                buffer.append(", ");
453              }
454    
455              buffer.append(c);
456            }
457            buffer.append('}');
458          }
459    
460          buffer.append('}');
461        }
462    
463        final String diagnosticMessage = getDiagnosticMessage();
464        if (diagnosticMessage != null)
465        {
466          buffer.append(", diagnosticMessage='");
467          buffer.append(diagnosticMessage);
468          buffer.append('\'');
469        }
470    
471        final String matchedDN = getMatchedDN();
472        if (matchedDN != null)
473        {
474          buffer.append(", matchedDN='");
475          buffer.append(matchedDN);
476          buffer.append('\'');
477        }
478    
479        final String[] referralURLs = getReferralURLs();
480        if (referralURLs.length > 0)
481        {
482          buffer.append(", referralURLs={");
483          for (int i=0; i < referralURLs.length; i++)
484          {
485            if (i > 0)
486            {
487              buffer.append(", ");
488            }
489    
490            buffer.append('\'');
491            buffer.append(referralURLs[i]);
492            buffer.append('\'');
493          }
494          buffer.append('}');
495        }
496    
497        final Control[] responseControls = getResponseControls();
498        if (responseControls.length > 0)
499        {
500          buffer.append(", responseControls={");
501          for (int i=0; i < responseControls.length; i++)
502          {
503            if (i > 0)
504            {
505              buffer.append(", ");
506            }
507    
508            buffer.append(responseControls[i]);
509          }
510          buffer.append('}');
511        }
512    
513        buffer.append(')');
514      }
