049    /**
050     * This class provides a facility for interacting with an LDAPv3 directory
051     * server.  It provides a means of establishing a connection to the server,
052     * sending requests, and reading responses.  See
053     * <A HREF="http://www.ietf.org/rfc/rfc4511.txt">RFC 4511</A> for the LDAPv3
054     * protocol specification and more information about the types of operations
055     * defined in LDAP.
056     * <BR><BR>
057     * <H2>Creating, Establishing, and Authenticating Connections</H2>
058     * An LDAP connection can be established either at the time that the object is
059     * created or as a separate step.  Similarly, authentication can be performed on
060     * the connection at the time it is created, at the time it is established, or
061     * as a separate process.  For example:
062     * <BR><BR>
063     * <PRE>
064     *   // Create a new, unestablished connection.  Then connect and perform a
065     *   // simple bind as separate operations.
066     *   LDAPConnection c = new LDAPConnection();
067     *   c.connect(address, port);
068     *   BindResult bindResult = c.bind(bindDN, password);
069     *
070     *   // Create a new connection that is established at creation time, and then
071     *   // authenticate separately using simple authentication.
072     *   LDAPConnection c = new LDAPConnection(address, port);
073     *   BindResult bindResult = c.bind(bindDN, password);
074     *
075     *   // Create a new connection that is established and bound using simple
076     *   // authentication all in one step.
077     *   LDAPConnection c = new LDAPConnection(address, port, bindDN, password);
078     * </PRE>
079     * <BR><BR>
080     * When authentication is performed at the time that the connection is
081     * established, it is only possible to perform a simple bind and it is not
082     * possible to include controls in the bind request, nor is it possible to
083     * receive response controls if the bind was successful.  Therefore, it is
084     * recommended that authentication be performed as a separate step if the server
085     * may return response controls even in the event of a successful authentication
086     * (e.g., a control that may indicate that the user's password will soon
087     * expire).  See the {@link BindRequest} class for more information about
088     * authentication in the UnboundID LDAP SDK for Java.
089     * <BR><BR>
090     * By default, connections will use standard unencrypted network sockets.
091     * However, it may be desirable to create connections that use SSL/TLS to
092     * encrypt communication.  This can be done by specifying a
093     * {@link javax.net.SocketFactory} that should be used to create the socket to
094     * use to communicate with the directory server.  The
095     * {@link javax.net.ssl.SSLSocketFactory#getDefault} method or the
096     * {@link javax.net.ssl.SSLContext#getSocketFactory} method may be used to
097     * obtain a socket factory for performing SSL communication.  See the
098     * <A HREF=
099     * "http://java.sun.com/j2se/1.5.0/docs/guide/security/jsse/JSSERefGuide.html">
100     * JSSE Reference Guide</A> for more information on using these classes.
101     * Alternately, you may use the {@link com.unboundid.util.ssl.SSLUtil} class to
102     * simplify the process.
103     * <BR><BR>
104     * Whenever the connection is no longer needed, it may be terminated using the
105     * {@link LDAPConnection#close} method.
106     * <BR><BR>
107     * <H2>Processing LDAP Operations</H2>
108     * This class provides a number of methods for processing the different types of
109     * operations.  The types of operations that can be processed include:
110     * <UL>
111     *   <LI>Abandon -- This may be used to request that the server stop processing
112     *      on an operation that has been invoked asynchronously.</LI>
113     *   <LI>Add -- This may be used to add a new entry to the directory
114     *       server.  See the {@link AddRequest} class for more information about
115     *       processing add operations.</LI>
116     *   <LI>Bind -- This may be used to authenticate to the directory server.  See
117     *       the {@link BindRequest} class for more information about processing
118     *       bind operations.</LI>
119     *   <LI>Compare -- This may be used to determine whether a specified entry has
120     *       a given attribute value.  See the {@link CompareRequest} class for more
121     *       information about processing compare operations.</LI>
122     *   <LI>Delete -- This may be used to remove an entry from the directory
123     *       server.  See the {@link DeleteRequest} class for more information about
124     *       processing delete operations.</LI>
125     *   <LI>Extended -- This may be used to process an operation which is not
126     *       part of the core LDAP protocol but is a custom extension supported by
127     *       the directory server.  See the {@link ExtendedRequest} class for more
128     *       information about processing extended operations.</LI>
129     *   <LI>Modify -- This may be used to alter an entry in the directory
130     *       server.  See the {@link ModifyRequest} class for more information about
131     *       processing modify operations.</LI>
132     *   <LI>Modify DN -- This may be used to rename an entry or subtree and/or move
133     *       that entry or subtree below a new parent in the directory server.  See
134     *       the {@link ModifyDNRequest} class for more information about processing
135     *       modify DN operations.</LI>
136     *   <LI>Search -- This may be used to retrieve a set of entries in the server
137     *       that match a given set of criteria.  See the {@link SearchRequest}
138     *       class for more information about processing search operations.</LI>
139     * </UL>
140     * <BR><BR>
141     * Most of the methods in this class used to process operations operate in a
142     * synchronous manner.  In these cases, the SDK will send a request to the
143     * server and wait for a response to arrive before returning to the caller.  In
144     * these cases, the value returned will include the contents of that response,
145     * including the result code, diagnostic message, matched DN, referral URLs, and
146     * any controls that may have been included.  However, it also possible to
147     * process operations asynchronously, in which case the SDK will return control
148     * back to the caller after the request has been sent to the server but before
149     * the response has been received.  In this case, the SDK will return an
150     * {@link AsyncRequestID} object which may be used to later abandon or cancel
151     * that operation if necessary, and will notify the client when the response
152     * arrives via a listener interface.
153     * <BR><BR>
154     * This class is mostly threadsafe.  It is possible to process multiple
155     * concurrent operations over the same connection as long as the methods being
156     * invoked will not change the state of the connection in a way that might
157     * impact other operations in progress in unexpected ways.  In particular, the
158     * following should not be attempted while any other operations may be in
159     * progress on this connection:
160     * <UL>
161     *   <LI>
162     *     Using one of the {@code connect} methods to re-establish the connection.
163     *   </LI>
164     *   <LI>
165     *     Using one of the {@code close} methods to terminate the connection.
166     *   </LI>
167     *   <LI>
168     *     Using one of the {@code bind} methods to attempt to authenticate the
169     *     connection (unless you are certain that the bind will not impact the
170     *     identity of the associated connection, for example by including the
171     *     retain identity request control in the bind request).
172     *   </LI>
173     *   <LI>
174     *     Attempting to make a change to the way that the underlying communication
175     *     is processed (e.g., by using the StartTLS extended operation to convert
176     *     an insecure connection into a secure one).
177     *   </LI>
178     * </UL>
179     */

647      @ThreadSafety(level=ThreadSafetyLevel.METHOD_NOT_THREADSAFE)
648      public void connect(final String host, final int port, final int timeout)
649             throws LDAPException
650      {
651        ensureNotNull(host, port);
652    
653        hostPort = host + ':' + port;
654    
655        if (isConnected())
656        {
657          setDisconnectInfo(DisconnectType.RECONNECT, null, null);
658          close();
659        }
660    
661        lastUsedSocketFactory = socketFactory;
662        disconnectType        = null;
663        disconnectMessage     = null;
664        disconnectCause       = null;
665        reconnectAddress      = host;
666        reconnectPort         = port;
667        cachedSchema          = null;
668        unbindRequestSent     = false;
669    
670        try
671        {
672          connectionStatistics.incrementNumConnects();
673          connectionInternals = new LDAPConnectionInternals(this, connectionOptions,
674               lastUsedSocketFactory, host, port, timeout);
675          connectionInternals.startConnectionReader();
676        }
677        catch (Exception e)
678        {
679          debugException(e);
680          setDisconnectInfo(DisconnectType.LOCAL_ERROR, null, e);
681          connectionInternals = null;
682          throw new LDAPException(ResultCode.CONNECT_ERROR,
683               ERR_CONN_CONNECT_ERROR.get(getHostPort(), getExceptionMessage(e)),
684               e);
685        }
686    
687        if (connectionOptions.useSchema())
688        {
689          try
690          {
691            cachedSchema = getSchema();
692          }
693          catch (Exception e)
694          {
695            debugException(e);
696          }
697        }
698      }

702      /**
703       * Attempts to re-establish a connection to the server and re-authenticate if
704       * appropriate.
705       *
706       * @throws  LDAPException  If a problem occurs while attempting to re-connect
707       *                         or re-authenticate.
708       */
709      public void reconnect()
710             throws LDAPException
711      {
712        if ((System.currentTimeMillis() - lastReconnectTime) < 1000L)
713        {
714          // If the last reconnect attempt was less than 1 second ago, then abort.
715          throw new LDAPException(ResultCode.SERVER_DOWN,
716                                  ERR_CONN_MULTIPLE_FAILURES.get());
717        }
718    
719        BindRequest bindRequest = null;
720        if (lastBindRequest != null)
721        {
722          bindRequest = lastBindRequest.getRebindRequest(reconnectAddress,
723                                                         reconnectPort);
724          if (bindRequest == null)
725          {
726            throw new LDAPException(ResultCode.SERVER_DOWN,
727                 ERR_CONN_CANNOT_REAUTHENTICATE.get(getHostPort()));
728          }
729        }
730    
731        setDisconnectInfo(DisconnectType.RECONNECT, null, null);
732        terminate(null);
733        connect(reconnectAddress, reconnectPort);
734    
735        if (bindRequest != null)
736        {
737          try
738          {
739            bind(bindRequest);
740          }
741          catch (LDAPException le)
742          {
743            debugException(le);
744            setDisconnectInfo(DisconnectType.BIND_FAILED, null, le);
745            terminate(null);
746    
747            throw le;
748          }
749        }
750    
751        lastReconnectTime = System.currentTimeMillis();
752      }
