035    /**
036     * This class defines an {@link Entry} subclass in which the contents of the
037     * entry cannot be modified.  Any attempt to call a method which could be used
038     * to alter the contents of the entry will result in an
039     * {@link UnsupportedOperationException}.
040     */
041    @NotExtensible()
042    @NotMutable()
043    @ThreadSafety(level=ThreadSafetyLevel.COMPLETELY_THREADSAFE)
044    public class ReadOnlyEntry
045           extends Entry
046    {

048       * The serial version UID for this serializable class.
049       */
050      private static final long serialVersionUID = -6482574870325012756L;

059       *                     {@code null}.
060       */
061      public ReadOnlyEntry(final String dn, final Attribute... attributes)
062      {
063        super(dn, attributes);
064      }
