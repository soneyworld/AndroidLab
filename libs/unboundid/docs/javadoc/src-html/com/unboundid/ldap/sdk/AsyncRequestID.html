021    package com.unboundid.ldap.sdk;
022    
023    
024    
025    import java.io.Serializable;
026    
027    import com.unboundid.util.NotMutable;
028    import com.unboundid.util.ThreadSafety;
029    import com.unboundid.util.ThreadSafetyLevel;
030    
031    
032    
033    /**
034     * This class defines an object that provides information about a request that
035     * was initiated asynchronously.  It may be used to abandon or cancel the
036     * associated request.
037     * <BR><BR>
038     * <H2>Example</H2>
039     * The following example initiates an asynchronous modify operation and then
040     * attempts to abandon it:
041     * <PRE>
042     *   Modification mod = new Modification(ModificationType.REPLACE,
043     *        "description", "This is the new description.");
044     *   ModifyRequest modifyRequest =
045     *        new ModifyRequest("dc=example,dc=com", mod);
046     *
047     *   AsyncRequestID asyncRequestID =
048     *        connection.asyncModify(modifyRequest, myAsyncResultListener);
049     *
050     *   // Assume that we've waited a reasonable amount of time but the modify
051     *   // hasn't completed yet so we'll try to abandon it.
052     *
053     *   connection.abandon(asyncRequestID);
054     * </PRE>
055     */
056    @NotMutable()
057    @ThreadSafety(level=ThreadSafetyLevel.COMPLETELY_THREADSAFE)
058    public final class AsyncRequestID
059           implements Serializable
060    {
061      /**
062       * The serial version UID for this serializable class.
063       */
064      private static final long serialVersionUID = 8244005138437962030L;
065    
066    
067    
068      // The message ID for the request message.
069      private final int messageID;
070    
071    
072    
073      /**
074       * Creates a new async request ID with the provided message ID.
075       *
076       * @param  messageID  The message ID for the associated request.
077       */
078      AsyncRequestID(final int messageID)
079      {
080        this.messageID = messageID;
081      }
082    
083    
084    
085      /**
086       * Retrieves the message ID for the associated request.
087       *
088       * @return  The message ID for the associated request.
089       */
090      public int getMessageID()
091      {
092        return messageID;
093      }
094    
095    
096    
097      /**
098       * Retrieves a hash code for this async request ID.
099       *
100       * @return  A hash code for this async request ID.
101       */
102      @Override()
103      public int hashCode()
104      {
105        return messageID;
106      }
107    
108    
109    
110      /**
111       * Indicates whether the provided object is equal to this async request ID.
112       *
113       * @param  o  The object for which to make the determination.
114       *
115       * @return  {@code true} if the provided object is equal to this async request
116       *          ID, or {@code false} if not.
117       */
118      @Override()
119      public boolean equals(final Object o)
120      {
121        if (o == null)
122        {
123          return false;
124        }
125    
126        if (o == this)
127        {
128          return true;
129        }
130    
131        if (o instanceof AsyncRequestID)
132        {
133          return (((AsyncRequestID) o).messageID == messageID);
134        }
135        else
136        {
137          return false;
138        }
139      }
140    
141    
142    
143      /**
144       * Retrieves a string representation of this async request ID.
145       *
146       * @return  A string representation of this async request ID.
147       */
148      @Override()
149      public String toString()
150      {
151        return "AsyncRequestID(messageID=" + messageID + ')';
152      }
153    }
