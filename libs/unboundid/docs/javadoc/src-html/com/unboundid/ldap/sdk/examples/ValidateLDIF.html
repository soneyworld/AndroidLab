059    /**
060     * This class provides a simple tool that can be used to validate that the
061     * contents of an LDIF file are valid.  This includes ensuring that the contents
062     * can be parsed as valid LDIF, and it can also ensure that the LDIF content
063     * conforms to the server schema.  It will obtain the schema by connecting to
064     * the server and retrieving the default schema (i.e., the schema which governs
065     * the root DSE).  By default, a thorough set of validation will be performed,
066     * but it is possible to disable certain types of validation.
067     * <BR><BR>
068     * Some of the APIs demonstrated by this example include:
069     * <UL>
070     *   <LI>Argument Parsing (from the {@code com.unboundid.util.args}
071     *       package)</LI>
072     *   <LI>LDAP Command-Line Tool (from the {@code com.unboundid.util}
073     *       package)</LI>
074     *   <LI>LDIF Processing (from the {@code com.unboundid.ldif} package)</LI>
075     *   <LI>Schema Parsing (from the {@code com.unboundid.ldap.sdk.schema}
076     *       package)</LI>
077     * </UL>
078     * <BR><BR>
079     * Supported arguments include those allowed by the {@link LDAPCommandLineTool}
080     * class (to obtain the information to use to connect to the server to read the
081     * schema), as well as the following additional arguments:
082     * <UL>
083     *   <LI>"--schemaDirectory {path}" -- specifies the path to a directory
084     *       containing files with schema definitions.  If this argument is
085     *       provided, then no attempt will be made to communicate with a directory
086     *       server.</LI>
087     *   <LI>"-f {path}" or "--ldifFile {path}" -- specifies the path to the LDIF
088     *       file to be validated.</LI>
089     *   <LI>"-R {path}" or "--rejectFile {path}" -- specifies the path to the file
090     *       to be written with information about all entries that failed
091     *       validation.</LI>
092     *   <LI>"-t {num}" or "--numThreads {num}" -- specifies the number of
093     *       concurrent threads to use when processing the LDIF.  If this is not
094     *       provided, then a default of one thread will be used.</LI>
095     *   <LI>"--ignoreUndefinedObjectClasses" -- indicates that the validation
096     *       process should ignore validation failures due to entries that contain
097     *       object classes not defined in the server schema.</LI>
098     *   <LI>"--ignoreUndefinedAttributes" -- indicates that the validation process
099     *       should ignore validation failures due to entries that contain
100     *       attributes not defined in the server schema.</LI>
101     *   <LI>"--ignoreMalformedDNs" -- indicates that the validation process should
102     *       ignore validation failures due to entries with malformed DNs.</LI>
103     *   <LI>"--ignoreStructuralObjectClasses" -- indicates that the validation
104     *       process should ignore validation failures due to entries that either do
105     *       not have a structural object class or that have multiple structural
106     *       object classes.</LI>
107     *   <LI>"--ignoreProhibitedObjectClasses" -- indicates that the validation
108     *       process should ignore validation failures due to entries containing
109     *       auxiliary classes that are not allowed by a DIT content rule, or
110     *       abstract classes that are not subclassed by an auxiliary or structural
111     *       class contained in the entry.</LI>
112     *   <LI>"--ignoreProhibitedAttributes" -- indicates that the validation process
113     *       should ignore validation failures due to entries including attributes
114     *       that are not allowed or are explicitly prohibited by a DIT content
115     *       rule.</LI>
116     *   <LI>"--ignoreMissingAttributes" -- indicates that the validation process
117     *       should ignore validation failures due to entries missing required
118     *       attributes.</LI>
119     *   <LI>"--ignoreSingleValuedAttributes" -- indicates that the validation
120     *       process should ignore validation failures due to single-valued
121     *       attributes containing multiple values.</LI>
122     *   <LI>"--ignoreAttributeSyntax" -- indicates that the validation process
123     *       should ignore validation failures due to attribute values which violate
124     *       the associated attribute syntax.</LI>
125     *   <LI>"--ignoreNameForms" -- indicates that the validation process should
126     *       ignore validation failures due to name form violations (in which the
127     *       entry's RDN does not comply with the associated name form).</LI>
128     * </UL>
129     */

365      /**
366       * Performs the actual processing for this tool.  In this case, it gets a
367       * connection to the directory server and uses it to retrieve the server
368       * schema.  It then reads the LDIF file and validates each entry accordingly.
369       *
370       * @return  The result code for the processing that was performed.
371       */
372      @Override()
373      public ResultCode doToolProcessing()
374      {
375        // Get the connection to the directory server and use it to read the schema.
376        final Schema schema;
377        if (schemaDirectory.isPresent())
378        {
379          final File schemaDir = schemaDirectory.getValue();
380    
381          try
382          {
383            final TreeMap<String,File> fileMap = new TreeMap<String,File>();
384            for (final File f : schemaDir.listFiles())
385            {
386              final String name = f.getName();
387              if (f.isFile() && name.endsWith(".ldif"))
388              {
389                fileMap.put(name, f);
390              }
391            }
392    
393            if (fileMap.isEmpty())
394            {
395              err("No LDIF files found in directory " +
396                  schemaDir.getAbsolutePath());
397              return ResultCode.PARAM_ERROR;
398            }
399    
400            final ArrayList<File> fileList = new ArrayList<File>(fileMap.values());
401            schema = Schema.getSchema(fileList);
402          }
403          catch (Exception e)
404          {
405            err("Unable to read schema from files in directory " +
406                schemaDir.getAbsolutePath() + ":  " + getExceptionMessage(e));
407            return ResultCode.LOCAL_ERROR;
408          }
409        }
410        else
411        {
412          try
413          {
414            final LDAPConnection connection = getConnection();
415            schema = connection.getSchema();
416            connection.close();
417          }
418          catch (LDAPException le)
419          {
420            err("Unable to connect to the directory server and read the schema:  ",
421                le.getMessage());
422            return le.getResultCode();
423          }
424        }
425    
426    
427        // Create the entry validator and initialize its configuration.
428        entryValidator = new EntryValidator(schema);
429        entryValidator.setCheckAttributeSyntax(!ignoreAttributeSyntax.isPresent());
430        entryValidator.setCheckMalformedDNs(!ignoreMalformedDNs.isPresent());
431        entryValidator.setCheckMissingAttributes(
432             !ignoreMissingAttributes.isPresent());
433        entryValidator.setCheckNameForms(!ignoreNameForms.isPresent());
434        entryValidator.setCheckProhibitedAttributes(
435             !ignoreProhibitedAttributes.isPresent());
436        entryValidator.setCheckProhibitedObjectClasses(
437             !ignoreProhibitedObjectClasses.isPresent());
438        entryValidator.setCheckSingleValuedAttributes(
439             !ignoreSingleValuedAttributes.isPresent());
440        entryValidator.setCheckStructuralObjectClasses(
441             !ignoreStructuralObjectClasses.isPresent());
442        entryValidator.setCheckUndefinedAttributes(
443             !ignoreUndefinedAttributes.isPresent());
444        entryValidator.setCheckUndefinedObjectClasses(
445             !ignoreUndefinedObjectClasses.isPresent());
446    
447    
448        // Create an LDIF reader that can be used to read through the LDIF file.
449        final LDIFReader ldifReader;
450        rejectWriter = null;
451        try
452        {
453          ldifReader = new LDIFReader(new FileInputStream(ldifFile.getValue()),
454                                      numThreads.getValue(), this);
455        }
456        catch (Exception e)
457        {
458          err("Unable to open the LDIF reader:  ", getExceptionMessage(e));
459          return ResultCode.LOCAL_ERROR;
460        }
461    
462        ldifReader.setSchema(schema);
463        ldifReader.setIgnoreDuplicateValues(ignoreDuplicateValues.isPresent());
464    
465        try
466        {
467          // Create an LDIF writer that can be used to write information about
468          // rejected entries.
469          try
470          {
471            if (rejectFile.isPresent())
472            {
473              rejectWriter = new LDIFWriter(rejectFile.getValue());
474            }
475          }
476          catch (Exception e)
477          {
478            err("Unable to create the reject writer:  ", getExceptionMessage(e));
479            return ResultCode.LOCAL_ERROR;
480          }
481    
482          ResultCode resultCode = ResultCode.SUCCESS;
483          while (true)
484          {
485            try
486            {
487              final Entry e = ldifReader.readEntry();
488              if (e == null)
489              {
490                // Because we're performing parallel processing and returning null
491                // from the translate method, LDIFReader.readEntry() should never
492                // return a non-null value.  However, it can throw an LDIFException
493                // if it encounters an invalid entry, or an IOException if there's
494                // a problem reading from the file, so we should still iterate
495                // through all of the entries to catch and report on those problems.
496                break;
497              }
498            }
499            catch (LDIFException le)
500            {
501              malformedEntries.incrementAndGet();
502    
503              if (resultCode == ResultCode.SUCCESS)
504              {
505                resultCode = ResultCode.DECODING_ERROR;
506              }
507    
508              if (rejectWriter != null)
509              {
510                try
511                {
512                  rejectWriter.writeComment(
513                       "Unable to parse an entry read from LDIF:", false, false);
514                  if (le.mayContinueReading())
515                  {
516                    rejectWriter.writeComment(getExceptionMessage(le), false, true);
517                  }
518                  else
519                  {
520                    rejectWriter.writeComment(getExceptionMessage(le), false,
521                                              false);
522                    rejectWriter.writeComment("Unable to continue LDIF processing.",
523                                              false, true);
524                    err("Aborting LDIF processing:  ", getExceptionMessage(le));
525                    return ResultCode.LOCAL_ERROR;
526                  }
527                }
528                catch (IOException ioe)
529                {
530                  err("Unable to write to the reject file:",
531                      getExceptionMessage(ioe));
532                  err("LDIF parse failure that triggered the rejection:  ",
533                      getExceptionMessage(le));
534                  return ResultCode.LOCAL_ERROR;
535                }
536              }
537            }
538            catch (IOException ioe)
539            {
540    
541              if (rejectWriter != null)
542              {
543                try
544                {
545                  rejectWriter.writeComment("I/O error reading from LDIF:", false,
546                                            false);
547                  rejectWriter.writeComment(getExceptionMessage(ioe), false,
548                                            true);
549                  return ResultCode.LOCAL_ERROR;
550                }
551                catch (Exception ex)
552                {
553                  err("I/O error reading from LDIF:", getExceptionMessage(ioe));
554                  return ResultCode.LOCAL_ERROR;
555                }
556              }
557            }
558          }
559    
560          if (malformedEntries.get() > 0)
561          {
562            out(malformedEntries.get() + " entries were malformed and could not " +
563                "be read from the LDIF file.");
564          }
565    
566          if (entryValidator.getInvalidEntries() > 0)
567          {
568            if (resultCode == ResultCode.SUCCESS)
569            {
570              resultCode = ResultCode.OBJECT_CLASS_VIOLATION;
571            }
572    
573            for (final String s : entryValidator.getInvalidEntrySummary(true))
574            {
575              out(s);
576            }
577          }
578          else
579          {
580            if (malformedEntries.get() == 0)
581            {
582              out("No errors were encountered.");
583            }
584          }
585    
586          return resultCode;
587        }
588        finally
589        {
590          try
591          {
592            ldifReader.close();
593          }
594          catch (Exception e) {}
595    
596          try
597          {
598            if (rejectWriter != null)
599            {
600              rejectWriter.close();
601            }
602          }
603          catch (Exception e) {}
604        }
605      }

609      /**
610       * Examines the provided entry to determine whether it conforms to the
611       * server schema.
612       *
613       * @param  entry           The entry to be examined.
614       * @param  firstLineNumber The line number of the LDIF source on which the
615       *                         provided entry begins.
616       *
617       * @return  The updated entry.  This method will always return {@code null}
618       *          because all of the real processing needed for the entry is
619       *          performed in this method and the entry isn't needed any more
620       *          after this method is done.
621       */
622      public Entry translate(final Entry entry, final long firstLineNumber)
623      {
624        final ArrayList<String> invalidReasons = new ArrayList<String>(5);
625        if (! entryValidator.entryIsValid(entry, invalidReasons))
626        {
627          if (rejectWriter != null)
628          {
629            synchronized (this)
630            {
631              try
632              {
633                rejectWriter.writeEntry(entry, listToString(invalidReasons));
634              }
635              catch (IOException ioe) {}
636            }
637          }
638        }
639    
640        final long numEntries = entriesProcessed.incrementAndGet();
641        if ((numEntries % 1000L) == 0L)
642        {
643          out("Processed ", numEntries, " entries.");
644        }
645    
646        return null;
647      }

651      /**
652       * Converts the provided list of strings into a single string.  It will
653       * contain line breaks after all but the last element.
654       *
655       * @param  l  The list of strings to convert to a single string.
656       *
657       * @return  The string from the provided list, or {@code null} if the provided
658       *          list is empty or {@code null}.
659       */
660      private static String listToString(final List<String> l)
661      {
662        if ((l == null) || (l.isEmpty()))
663        {
664          return null;
665        }
666    
667        final StringBuilder buffer = new StringBuilder();
668        final Iterator<String> iterator = l.iterator();
669        while (iterator.hasNext())
670        {
671          buffer.append(iterator.next());
672          if (iterator.hasNext())
673          {
674            buffer.append(EOL);
675          }
676        }
677    
678        return buffer.toString();
679      }
