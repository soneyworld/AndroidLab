064    /**
065     * This class provides an LDIF reader, which can be used to read and decode
066     * entries and change records from a data source using the LDAP Data Interchange
067     * Format as per <A HREF="http://www.ietf.org/rfc/rfc2849.txt">RFC 2849</A>.
068     * <BR>
069     * This class is not synchronized.  If multiple threads read from the
070     * LDIFReader, they must be synchronized externally.
071     * <BR><BR>
072     * <H2>Example</H2>
073     * The following example iterates through all entries contained in an LDIF file
074     * and attempts to add them to a directory server:
075     * <PRE>
076     *   LDIFReader ldifReader = new LDIFReader(pathToLDIFFile);
077     *
078     *   while (true)
079     *   {
080     *     Entry entry;
081     *     try
082     *     {
083     *       entry = ldifReader.readEntry();
084     *       if (entry == null)
085     *       {
086     *         System.err.println("All entries have been processed.");
087     *         break;
088     *       }
089     *     }
090     *     catch (LDIFException le)
091     *     {
092     *       if (le.mayContinueReading())
093     *       {
094     *         System.err.println("A recoverable occurred while attempting to " +
095     *              "read an entry at or near line number " + le.getLineNumber() +
096     *              ":  " + le.getMessage());
097     *         System.err.println("The entry will be skipped.");
098     *         continue;
099     *       }
100     *       else
101     *       {
102     *         System.err.println("An unrecoverable occurred while attempting to " +
103     *              "read an entry at or near line number " + le.getLineNumber() +
104     *              ":  " + le.getMessage());
105     *         System.err.println("LDIF processing will be aborted.");
106     *         break;
107     *       }
108     *     }
109     *     catch (IOException ioe)
110     *     {
111     *       System.err.println("An I/O error occurred while attempting to read " +
112     *            "from the LDIF file:  " + ioe.getMessage());
113     *       System.err.println("LDIF processing will be aborted.");
114     *       break;
115     *     }
116     *
117     *     try
118     *     {
119     *       connection.add(entry);
120     *       System.out.println("Successfully added entry " + entry.getDN());
121     *     }
122     *     catch (LDAPException le)
123     *     {
124     *       System.err.println("Unable to add entry " + entry.getDN() + " -- " +
125     *            le.getMessage());
126     *     }
127     *   }
128     *
129     *   ldifReader.close();
130     * </PRE>
131     */

377      /**
378       * Creates a new LDIF reader that will read data from the specified buffered
379       * reader and parses the LDIF records asynchronously using the specified
380       * number of threads.  The encoding of the underlying Reader must be set to
381       * "UTF-8" as required by RFC 2849.
382       *
383       * @param reader The buffered reader that will be used to read the LDIF data.
384       *               It must not be {@code null}.
385       * @param  numParseThreads  If this value is greater than zero, then the
386       *                          specified number of threads will be used to
387       *                          asynchronously read and parse the LDIF file.
388       *                          This should only be set to greater than zero when
389       *                          performance analysis has demonstrated that reading
390       *                          and parsing the LDIF is a bottleneck.  The default
391       *                          synchronous processing is normally fast enough.
392       *                          There is little benefit in passing in a value
393       *                          greater than four (unless there is an
394       *                          LDIFReaderEntryTranslator that does time-consuming
395       *                          processing).  A value of zero implies the
396       *                          default behavior of reading and parsing LDIF
397       *                          records synchronously when one of the read
398       *                          methods is called.
399       * @param entryTranslator  The LDIFReaderEntryTranslator to apply to read
400       *                         EntryS before they are returned.  This is normally
401       *                         {@code null}, which causes EntryS to be returned
402       *                         unaltered. This is particularly useful when parsing
403       *                         the input file in parallel because the Entry
404       *                         translation is also done in parallel.
405       */
406      public LDIFReader(final BufferedReader reader,
407                        final int numParseThreads,
408                        final LDIFReaderEntryTranslator entryTranslator)
409      {
410        ensureNotNull(reader);
411        ensureTrue(numParseThreads >= 0,
412                   "LDIFReader.numParseThreads must not be negative.");
413    
414        this.reader = reader;
415        this.entryTranslator = entryTranslator;
416    
417        if (numParseThreads == 0)
418        {
419          isAsync = false;
420          asyncParser = null;
421          asyncParsingComplete = null;
422          asyncParsedRecords = null;
423        }
424        else
425        {
426          isAsync = true;
427          asyncParsingComplete = new AtomicBoolean(false);
428    
429          // Decodes entries in parallel.
430          final ParallelProcessor<UnparsedLDIFRecord, LDIFRecord> parallelParser =
431              new ParallelProcessor<UnparsedLDIFRecord, LDIFRecord>(
432                 new RecordParser(), numParseThreads, ASYNC_MIN_PER_PARSING_THREAD);
433    
434          final BlockingQueue<UnparsedLDIFRecord> pendingQueue = new
435               ArrayBlockingQueue<UnparsedLDIFRecord>(ASYNC_QUEUE_SIZE);
436    
437          // The output queue must be a little more than twice as big as the input
438          // queue to more easily handle being shutdown in the middle of processing
439          // when the queues are full and threads are blocked.
440          asyncParsedRecords = new ArrayBlockingQueue
441               <Result<UnparsedLDIFRecord, LDIFRecord>>(2 * ASYNC_QUEUE_SIZE + 100);
442    
443          asyncParser = new AsynchronousParallelProcessor
444               <UnparsedLDIFRecord, LDIFRecord>(pendingQueue, parallelParser,
445                                                asyncParsedRecords);
446    
447          final LineReaderThread lineReaderThread = new LineReaderThread();
448          lineReaderThread.start();
449        }
450      }

454      /**
455       * Closes this LDIF reader and the underlying LDIF source.
456       *
457       * @throws  IOException  If a problem occurs while closing the underlying LDIF
458       *                       source.
459       */
460      public void close()
461             throws IOException
462      {
463        reader.close();
464    
465        if (isAsync())
466        {
467          // Closing the reader will trigger the LineReaderThread to complete, but
468          // not if it's blocked submitting the next UnparsedLDIFRecord.  To avoid
469          // this, we clear out the completed output queue, which is larger than
470          // the input queue, so the LineReaderThread will stop reading and
471          // shutdown the asyncParser.
472          asyncParsedRecords.clear();
473        }
474      }

676      /**
677       * Reads an entry asynchronously from the LDIF source.
678       *
679       * @return The entry read from the LDIF source, or {@code null} if there are
680       *         no more entries to be read.
681       *
682       * @throws IOException   If a problem occurs while attempting to read from the
683       *                       LDIF source.
684       * @throws LDIFException If the data read could not be parsed as an entry.
685       */
686      private Entry readEntryAsync()
687              throws IOException, LDIFException
688      {
689        Result<UnparsedLDIFRecord, LDIFRecord> result = null;
690        LDIFRecord record = null;
691        while (record == null)
692        {
693          result = readLDIFRecordResultAsync();
694          if (result == null)
695          {
696            return null;
697          }
698    
699          record = result.getOutput();
700    
701          // This is a special value that means we should skip this Entry.  We have
702          // to use something different than null because null means EOF.
703          if (record == SKIP_ENTRY)
704          {
705            record = null;
706          }
707        }
708    
709        if (!(record instanceof Entry))
710        {
711          try
712          {
713            // Some LDIFChangeRecord can be converted to an Entry.  This is really
714            // an edge case though.
715            return ((LDIFChangeRecord)record).toEntry();
716          }
717          catch (LDIFException e)
718          {
719            debugException(e);
720            final long firstLineNumber = result.getInput().getFirstLineNumber();
721            throw new LDIFException(e.getExceptionMessage(),
722                                    firstLineNumber, true, e);
723          }

786      /**
787       * Reads the next LDIF record, which was read and parsed asynchronously by
788       * separate threads.
789       *
790       * @return  The next LDIF record or {@code null} if there are no more records.
791       *
792       * @throws  IOException  If a problem occurs while attempting to read from the
793       *                       LDIF source.
794       *
795       * @throws  LDIFException  If the data read could not be parsed as an entry.
796       */
797      private Result<UnparsedLDIFRecord, LDIFRecord> readLDIFRecordResultAsync()
798              throws IOException, LDIFException
799      {
800        Result<UnparsedLDIFRecord, LDIFRecord> result = null;
801    
802        // If the asynchronous reading and parsing is complete, then we don't have
803        // to block waiting for the next record to show up on the queue.  If there
804        // isn't a record there, then return null (EOF) right away.
805        if (asyncParsingComplete.get())
806        {
807          result = asyncParsedRecords.poll();
808        }
809        else
810        {
811          try
812          {
813            // We probably could just do a asyncParsedRecords.take() here, but
814            // there are some edge case error scenarios where
815            // asyncParsingComplete might be set without a special EOF sentinel
816            // Result enqueued.  So to guard against this, we have a very cautious
817            // polling interval of 1 second.  During normal processing, we never
818            // have to wait for this to expire, when there is something to do
819            // (like shutdown).
820            while ((result == null) && (!asyncParsingComplete.get()))
821            {
822              result = asyncParsedRecords.poll(1, TimeUnit.SECONDS);
823            }
824    
825            // There's a very small chance that we missed the value, so double-check
826            if (result == null)
827            {
828              result = asyncParsedRecords.poll();
829            }
830          }
831          catch (InterruptedException e)
832          {
833            debugException(e);
834            throw new IOException(getExceptionMessage(e));
835          }
836        }
837        if (result == null)
838        {
839          return null;
840        }
841    
842        rethrow(result.getFailureCause());
843    
844        // Check if we reached the end of the input
845        final UnparsedLDIFRecord unparsedRecord = result.getInput();
846        if (unparsedRecord.isEOF())
847        {
848          // This might have been set already by the LineReaderThread, but
849          // just in case it hasn't gotten to it yet, do so here.
850          asyncParsingComplete.set(true);
851    
852          // Enqueue this EOF result again for any other thread that might be
853          // blocked in asyncParsedRecords.take() even though having multiple
854          // threads call this method concurrently breaks the contract of this
855          // class.
856          try
857          {
858            asyncParsedRecords.put(result);
859          }
860          catch (InterruptedException e)
861          {
862            // We shouldn't ever get interrupted because the put won't ever block.
863            // Once we are done reading, this is the only item left in the queue,
864            // so we should always be able to re-enqueue it.
865            debugException(e);
866          }
867          return null;
868        }
869    
870        return result;
871      }

1023      /**
1024       * Reads a record (either an entry or a change record) from the LDIF source
1025       * and places it in the line list.
1026       *
1027       * @return  The line number for the first line of the entry that was read.
1028       *
1029       * @throws  IOException  If a problem occurs while attempting to read from the
1030       *                       LDIF source.
1031       *
1032       * @throws  LDIFException  If the data read could not be parsed as a valid
1033       *                         LDIF record.
1034       */
1035      private UnparsedLDIFRecord readUnparsedRecord()
1036             throws IOException, LDIFException
1037      {
1038        final ArrayList<StringBuilder> lineList = new ArrayList<StringBuilder>(20);
1039        boolean lastWasComment = false;
1040        long firstLineNumber = lineNumberCounter + 1;
1041        while (true)
1042        {
1043          final String line = reader.readLine();
1044          lineNumberCounter++;
1045    
1046          if (line == null)
1047          {
1048            // We've hit the end of the LDIF source.  If we haven't read any entry
1049            // data, then return null.  Otherwise, the last entry wasn't followed by
1050            // a blank line, which is OK, and we should decode that entry.
1051            if (lineList.isEmpty())
1052            {
1053              return new UnparsedLDIFRecord(new ArrayList<StringBuilder>(0),
1054                                            ignoreDuplicateValues, schema, -1);
1055            }
1056            else
1057            {
1058              break;
1059            }
1060          }
1061    
1062          if (line.length() == 0)
1063          {
1064            // It's a blank line.  If we have read entry data, then this signals the
1065            // end of the entry.  Otherwise, it's an extra space between entries,
1066            // which is OK.
1067            if (lineList.isEmpty())
1068            {
1069              firstLineNumber++;
1070              continue;
1071            }
1072            else
1073            {
1074              break;
1075            }
1076          }
1077    
1078          if (line.charAt(0) == ' ')
1079          {
1080            // The line starts with a space, which means that it must be a
1081            // continuation of the previous line.
1082            if (lineList.isEmpty())
1083            {
1084              throw new LDIFException(
1085                             ERR_READ_UNEXPECTED_FIRST_SPACE.get(lineNumberCounter),
1086                             lineNumberCounter, false);
1087            }
1088            else if(! lastWasComment)
1089            {
1090              lineList.get(lineList.size() - 1).append(line.substring(1));
1091            }
1092          }
1093          else if (line.charAt(0) == '#')
1094          {
1095            lastWasComment = true;
1096          }
1097          else
1098          {
1099            // We want to make sure that we skip over the "version:" line if it
1100            // exists, but that should only occur at the beginning of an entry where
1101            // it can't be confused with a possible "version" attribute.
1102            if (lineList.isEmpty() && line.startsWith("version:"))
1103            {
1104              lastWasComment = true;
1105            }
1106            else
1107            {
1108              lineList.add(new StringBuilder(line));
1109              lastWasComment = false;
1110            }
1111          }
1112        }
1113    
1114        return new UnparsedLDIFRecord(lineList, ignoreDuplicateValues, schema,
1115                                      firstLineNumber);
1116      }

1275      /**
1276       * Parses the provided set of lines into a list of {@code StringBuilder}
1277       * objects suitable for decoding into an entry or LDIF change record.
1278       * Comments will be ignored and wrapped lines will be unwrapped.
1279       *
1280       * @param  ignoreDuplicateValues  Indicates whether to ignore duplicate
1281       *                                attribute values encountered while parsing.
1282       * @param  schema                 The schema to use when parsing the record,
1283       *                                if applicable.
1284       * @param  ldifLines              The set of lines that comprise the record to
1285       *                                decode.  It must not be {@code null} or
1286       *                                empty.
1287       *
1288       * @return  The prepared list of {@code StringBuilder} objects ready to be
1289       *          decoded.
1290       *
1291       * @throws  LDIFException  If the provided lines do not contain valid LDIF
1292       *                         content.
1293       */
1294      private static UnparsedLDIFRecord
1295                          prepareRecord(final boolean ignoreDuplicateValues,
1296                                        final Schema schema,
1297                                        final String... ldifLines)
1298              throws LDIFException
1299      {
1300        ensureNotNull(ldifLines);
1301        ensureFalse(ldifLines.length == 0,
1302                    "LDIFReader.prepareRecord.ldifLines must not be empty.");
1303    
1304        boolean lastWasComment = false;
1305        final ArrayList<StringBuilder> lineList =
1306             new ArrayList<StringBuilder>(ldifLines.length);
1307        for (int i=0; i < ldifLines.length; i++)
1308        {
1309          final String line = ldifLines[i];
1310          if (line.length() == 0)
1311          {
1312            // This is only acceptable if there are no more non-empty lines in the
1313            // array.
1314            for (int j=i+1; j < ldifLines.length; j++)
1315            {
1316              if (ldifLines[j].length() > 0)
1317              {
1318                throw new LDIFException(ERR_READ_UNEXPECTED_BLANK.get(i), i, true,
1319                                        ldifLines, null);
1320              }
1321    
1322              // If we've gotten here, then we know that we're at the end of the
1323              // entry.  If we have read data, then we can decode it as an entry.
1324              // Otherwise, there was no real data in the provided LDIF lines.
1325              if (lineList.isEmpty())
1326              {
1327                throw new LDIFException(ERR_READ_ONLY_BLANKS.get(), 0, true,
1328                                        ldifLines, null);
1329              }
1330              else
1331              {
1332                return new UnparsedLDIFRecord(lineList, ignoreDuplicateValues,
1333                                              schema, 0);
1334              }
1335            }
1336          }
1337    
1338          if (line.charAt(0) == ' ')
1339          {
1340            if (i > 0)
1341            {
1342              if (! lastWasComment)
1343              {
1344                lineList.get(lineList.size() - 1).append(line.substring(1));
1345              }
1346            }
1347            else
1348            {
1349              throw new LDIFException(
1350                             ERR_READ_UNEXPECTED_FIRST_SPACE_NO_NUMBER.get(), 0,
1351                             true, ldifLines, null);
1352            }
1353          }
1354          else if (line.charAt(0) == '#')
1355          {
1356            lastWasComment = true;
1357          }
1358          else
1359          {
1360            lineList.add(new StringBuilder(line));
1361            lastWasComment = false;
1362          }
1363        }
1364    
1365        if (lineList.isEmpty())
1366        {
1367          throw new LDIFException(ERR_READ_NO_DATA.get(), 0, true, ldifLines, null);
1368        }
1369        else
1370        {
1371          return new UnparsedLDIFRecord(lineList, ignoreDuplicateValues, schema, 0);
1372        }
1373      }

1377      /**
1378       * Decodes the unparsed record that was read from the LDIF source.  It may be
1379       * either an entry or an LDIF change record.
1380       *
1381       * @param  unparsedRecord  The unparsed LDIF record that was read from the
1382       *                         input.  It must not be {@code null} or empty.
1383       *
1384       * @return  The parsed record, or {@code null} if there are no more entries to
1385       *          be read.
1386       *
1387       * @throws  LDIFException  If the data read could not be parsed as an entry or
1388       *                         an LDIF change record.
1389       */
1390      private static LDIFRecord decodeRecord(
1391                                     final UnparsedLDIFRecord unparsedRecord)
1392           throws LDIFException
1393      {
1394        // If there was an error reading from the input, then we rethrow it here.
1395        final Exception readError = unparsedRecord.getFailureCause();
1396        if (readError != null)
1397        {
1398          if (readError instanceof LDIFException)
1399          {
1400            // If the error was an LDIFException, which will normally be the case,
1401            // then rethrow it with all of the same state.  We could just
1402            //   throw (LDIFException) readError;
1403            // but that's considered bad form.
1404            final LDIFException ldifEx = (LDIFException) readError;
1405            throw new LDIFException(ldifEx.getMessage(),
1406                                    ldifEx.getLineNumber(),
1407                                    ldifEx.mayContinueReading(),
1408                                    ldifEx.getDataLines(),
1409                                    ldifEx.getCause());
1410          }
1411          else
1412          {
1413            throw new LDIFException(getExceptionMessage(readError),
1414                                    -1, true, readError);
1415          }
1416        }
1417    
1418        if (unparsedRecord.isEOF())
1419        {
1420          return null;
1421        }
1422    
1423        final ArrayList<StringBuilder> lineList = unparsedRecord.getLineList();
1424        if (unparsedRecord.getLineList() == null)
1425        {
1426          return null;  // We can get here if there was an error reading the lines.
1427        }
1428    
1429        final LDIFRecord r;
1430        if ((lineList.size() > 1) &&
1431            toLowerCase(lineList.get(1).toString()).startsWith("changetype:"))
1432        {
1433          r = decodeChangeRecord(unparsedRecord, false);
1434        }
1435        else
1436        {
1437          r = decodeEntry(unparsedRecord);
1438        }
1439    
1440        debugLDIFRead(r);
1441        return r;
1442      }

1446      /**
1447       * Decodes the provided set of LDIF lines as an entry.  The provided list must
1448       * not contain any blank lines or comments, and lines are not allowed to be
1449       * wrapped.
1450       *
1451       * @param  unparsedRecord   The unparsed LDIF record that was read from the
1452       *                          input.  It must not be {@code null} or empty.
1453       *
1454       * @return  The entry read from LDIF.
1455       *
1456       * @throws  LDIFException  If the provided LDIF data cannot be read as an
1457       *                         entry.
1458       */
1459      private static Entry decodeEntry(final UnparsedLDIFRecord unparsedRecord)
1460              throws LDIFException
1461      {
1462        final ArrayList<StringBuilder> ldifLines = unparsedRecord.getLineList();
1463        final long firstLineNumber = unparsedRecord.getFirstLineNumber();
1464    
1465        final Iterator<StringBuilder> iterator = ldifLines.iterator();
1466    
1467        // The first line must be the entry DN, and it must start with "dn:".
1468        final StringBuilder line = iterator.next();
1469        final int colonPos = line.indexOf(":");
1470        if ((colonPos < 0) ||
1471            (! line.substring(0, colonPos).equalsIgnoreCase("dn")))
1472        {
1473          throw new LDIFException(
1474                         ERR_READ_DN_LINE_DOESNT_START_WITH_DN.get(firstLineNumber),
1475                         firstLineNumber, true, ldifLines, null);
1476        }
1477    
1478        final String dn;
1479        final int length = line.length();
1480        if (length == (colonPos+1))
1481        {
1482          // The colon was the last character on the line.  This is acceptable and
1483          // indicates that the entry has the null DN.
1484          dn = "";
1485        }
1486        else if (line.charAt(colonPos+1) == ':')
1487        {
1488          // Skip over any spaces leading up to the value, and then the rest of the
1489          // string is the base64-encoded DN.
1490          int pos = colonPos+2;
1491          while ((pos < length) && (line.charAt(pos) == ' '))
1492          {
1493            pos++;
1494          }
1495    
1496          try
1497          {
1498            final byte[] dnBytes = Base64.decode(line.substring(pos));
1499            dn = new String(dnBytes, "UTF-8");
1500          }
1501          catch (final ParseException pe)
1502          {
1503            debugException(pe);
1504            throw new LDIFException(
1505                           ERR_READ_CANNOT_BASE64_DECODE_DN.get(firstLineNumber,
1506                                                                pe.getMessage()),
1507                           firstLineNumber, true, ldifLines, pe);
1508          }
1509          catch (final Exception e)
1510          {
1511            debugException(e);
1512            throw new LDIFException(
1513                           ERR_READ_CANNOT_BASE64_DECODE_DN.get(firstLineNumber, e),
1514                           firstLineNumber, true, ldifLines, e);
1515          }
1516        }
1517        else
1518        {
1519          // Skip over any spaces leading up to the value, and then the rest of the
1520          // string is the DN.
1521          int pos = colonPos+1;
1522          while ((pos < length) && (line.charAt(pos) == ' '))
1523          {
1524            pos++;
1525          }
1526    
1527          dn = line.substring(pos);
1528        }
1529    
1530    
1531        // The remaining lines must be the attributes for the entry.  However, we
1532        // will allow the case in which an entry does not have any attributes, to be
1533        // able to support reading search result entries in which no attributes were
1534        // returned.
1535        if (! iterator.hasNext())
1536        {
1537          return new Entry(dn);
1538        }
1539    
1540        return new Entry(dn, parseAttributes(dn,
1541                                             unparsedRecord.ignoreDuplicateValues(),
1542                                             unparsedRecord.getSchema(),
1543                                             ldifLines, iterator, firstLineNumber));
1544      }

1548      /**
1549       * Decodes the provided set of LDIF lines as a change record.  The provided
1550       * list must not contain any blank lines or comments, and lines are not
1551       * allowed to be wrapped.
1552       *
1553       * @param  unparsedRecord   The unparsed LDIF record that was read from the
1554       *                          input.  It must not be {@code null} or empty.
1555       * @param  defaultAdd       Indicates whether an LDIF record not containing a
1556       *                          changetype should be retrieved as an add change
1557       *                          record.  If this is {@code false} and the record
1558       *                          read does not include a changetype, then an
1559       *                          {@link LDIFException} will be thrown.
1560       *
1561       * @return  The change record read from LDIF.
1562       *
1563       * @throws  LDIFException  If the provided LDIF data cannot be decoded as a
1564       *                         change record.
1565       */
1566      private static LDIFChangeRecord decodeChangeRecord(
1567                                           final UnparsedLDIFRecord unparsedRecord,
1568                                           final boolean defaultAdd)
1569              throws LDIFException
1570      {
1571        final ArrayList<StringBuilder> ldifLines = unparsedRecord.getLineList();
1572        final long firstLineNumber = unparsedRecord.getFirstLineNumber();
1573    
1574        final Iterator<StringBuilder> iterator = ldifLines.iterator();
1575    
1576        // The first line must be the entry DN, and it must start with "dn:".
1577        StringBuilder line = iterator.next();
1578        int colonPos = line.indexOf(":");
1579        if ((colonPos < 0) ||
1580            (! line.substring(0, colonPos).equalsIgnoreCase("dn")))
1581        {
1582          throw new LDIFException(
1583               ERR_READ_CR_DN_LINE_DOESNT_START_WITH_DN.get(firstLineNumber),
1584               firstLineNumber, true, ldifLines, null);
1585        }
1586    
1587        final String dn;
1588        int length = line.length();
1589        if (length == (colonPos+1))
1590        {
1591          // The colon was the last character on the line.  This is acceptable and
1592          // indicates that the entry has the null DN.
1593          dn = "";
1594        }
1595        else if (line.charAt(colonPos+1) == ':')
1596        {
1597          // Skip over any spaces leading up to the value, and then the rest of the
1598          // string is the base64-encoded DN.
1599          int pos = colonPos+2;
1600          while ((pos < length) && (line.charAt(pos) == ' '))
1601          {
1602            pos++;
1603          }
1604    
1605          try
1606          {
1607            final byte[] dnBytes = Base64.decode(line.substring(pos));
1608            dn = new String(dnBytes, "UTF-8");
1609          }
1610          catch (final ParseException pe)
1611          {
1612            debugException(pe);
1613            throw new LDIFException(
1614                           ERR_READ_CR_CANNOT_BASE64_DECODE_DN.get(firstLineNumber,
1615                                                                   pe.getMessage()),
1616                           firstLineNumber, true, ldifLines, pe);
1617          }
1618          catch (final Exception e)
1619          {
1620            debugException(e);
1621            throw new LDIFException(
1622                           ERR_READ_CR_CANNOT_BASE64_DECODE_DN.get(firstLineNumber,
1623                                                                   e),
1624                           firstLineNumber, true, ldifLines, e);
1625          }
1626        }
1627        else
1628        {
1629          // Skip over any spaces leading up to the value, and then the rest of the
1630          // string is the DN.
1631          int pos = colonPos+1;
1632          while ((pos < length) && (line.charAt(pos) == ' '))
1633          {
1634            pos++;
1635          }
1636    
1637          dn = line.substring(pos);
1638        }
1639    
1640    
1641        // The second line must be the change type, and it must start with
1642        // "changetype:".
1643        if (! iterator.hasNext())
1644        {
1645          throw new LDIFException(ERR_READ_CR_TOO_SHORT.get(firstLineNumber),
1646                                  firstLineNumber, true, ldifLines, null);
1647        }
1648    
1649    
1650        // If defaultAdd is true, then the change record may or may not have a
1651        // changetype.  If it is false, then the record must have a changetype.
1652        final String changeType;
1653        if (defaultAdd &&
1654            (! toLowerCase(ldifLines.get(1).toString()).startsWith("changetype:")))
1655        {
1656          changeType = "add";
1657        }
1658        else
1659        {
1660          line = iterator.next();
1661          colonPos = line.indexOf(":");
1662          if ((colonPos < 0) ||
1663              (! line.substring(0, colonPos).equalsIgnoreCase("changetype")))
1664          {
1665            throw new LDIFException(
1666                 ERR_READ_CR_CT_LINE_DOESNT_START_WITH_CT.get(firstLineNumber),
1667                 firstLineNumber, true, ldifLines, null);
1668          }
1669    
1670          length = line.length();
1671          if (length == (colonPos+1))
1672          {
1673            // The colon was the last character on the line.  This is not
1674            // acceptable.
1675            throw new LDIFException(
1676                 ERR_READ_CT_LINE_NO_CT_VALUE.get(firstLineNumber), firstLineNumber,
1677                 true, ldifLines, null);
1678          }
1679          else if (line.charAt(colonPos+1) == ':')
1680          {
1681            // Skip over any spaces leading up to the value, and then the rest of
1682            // the string is the base64-encoded changetype.  This is unusual and
1683            // unnecessary, but is nevertheless acceptable.
1684            int pos = colonPos+2;
1685            while ((pos < length) && (line.charAt(pos) == ' '))
1686            {
1687              pos++;
1688            }
1689    
1690            try
1691            {
1692              final byte[] changeTypeBytes = Base64.decode(line.substring(pos));
1693              changeType = new String(changeTypeBytes, "UTF-8");
1694            }
1695            catch (final ParseException pe)
1696            {
1697              debugException(pe);
1698              throw new LDIFException(
1699                             ERR_READ_CANNOT_BASE64_DECODE_CT.get(firstLineNumber,
1700                                                                  pe.getMessage()),
1701                             firstLineNumber, true, ldifLines, pe);
1702            }
1703            catch (final Exception e)
1704            {
1705              debugException(e);
1706              throw new LDIFException(
1707                   ERR_READ_CANNOT_BASE64_DECODE_CT.get(firstLineNumber, e),
1708                   firstLineNumber, true, ldifLines, e);
1709            }
1710          }
1711          else
1712          {
1713            // Skip over any spaces leading up to the value, and then the rest of
1714            // the string is the changetype.
1715            int pos = colonPos+1;
1716            while ((pos < length) && (line.charAt(pos) == ' '))
1717            {
1718              pos++;
1719            }
1720    
1721            changeType = line.substring(pos);
1722          }
1723        }
1724    
1725    
1726        // Make sure that the change type is acceptable and then decode the rest of
1727        // the change record accordingly.
1728        final String lowerChangeType = toLowerCase(changeType);
1729        if (lowerChangeType.equals("add"))
1730        {
1731          // There must be at least one more line.  If not, then that's an error.
1732          // Otherwise, parse the rest of the data as attribute-value pairs.
1733          if (iterator.hasNext())
1734          {
1735            final Collection<Attribute> attrs =
1736                 parseAttributes(dn, unparsedRecord.ignoreDuplicateValues(),
1737                                 unparsedRecord.getSchema(), ldifLines,
1738                                 iterator, firstLineNumber);
1739            final Attribute[] attributes = new Attribute[attrs.size()];
1740            final Iterator<Attribute> attrIterator = attrs.iterator();
1741            for (int i=0; i < attributes.length; i++)
1742            {
1743              attributes[i] = attrIterator.next();
1744            }
1745    
1746            return new LDIFAddChangeRecord(dn, attributes);
1747          }
1748          else
1749          {
1750            throw new LDIFException(ERR_READ_CR_NO_ATTRIBUTES.get(firstLineNumber),
1751                                    firstLineNumber, true, ldifLines, null);
1752          }
1753        }
1754        else if (lowerChangeType.equals("delete"))
1755        {
1756          // There shouldn't be any more data.  If there is, then that's an error.
1757          // Otherwise, we can just return the delete change record with what we
1758          // already know.
1759          if (iterator.hasNext())
1760          {
1761            throw new LDIFException(
1762                           ERR_READ_CR_EXTRA_DELETE_DATA.get(firstLineNumber),
1763                           firstLineNumber, true, ldifLines, null);
1764          }
1765          else
1766          {
1767            return new LDIFDeleteChangeRecord(dn);
1768          }
1769        }
1770        else if (lowerChangeType.equals("modify"))
1771        {
1772          // There must be at least one more line.  If not, then that's an error.
1773          // Otherwise, parse the rest of the data as a set of modifications.
1774          if (iterator.hasNext())
1775          {
1776            final Modification[] mods =
1777                 parseModifications(ldifLines, iterator, firstLineNumber);
1778            return new LDIFModifyChangeRecord(dn, mods);
1779          }
1780          else
1781          {
1782            throw new LDIFException(ERR_READ_CR_NO_MODS.get(firstLineNumber),
1783                                    firstLineNumber, true, ldifLines, null);
1784          }
1785        }
1786        else if (lowerChangeType.equals("moddn") ||
1787                 lowerChangeType.equals("modrdn"))
1788        {
1789          // There must be at least one more line.  If not, then that's an error.
1790          // Otherwise, parse the rest of the data as a set of modifications.
1791          if (iterator.hasNext())
1792          {
1793            return parseModifyDNChangeRecord(ldifLines, iterator, dn,
1794                                             firstLineNumber);
1795          }
1796          else
1797          {
1798            throw new LDIFException(ERR_READ_CR_NO_NEWRDN.get(firstLineNumber),
1799                                    firstLineNumber, true, ldifLines, null);
1800          }
1801        }
1802        else
1803        {
1804          throw new LDIFException(ERR_READ_CR_INVALID_CT.get(changeType,
1805                                                             firstLineNumber),
1806                                  firstLineNumber, true, ldifLines, null);
1807        }
1808      }

1812      /**
1813       * Parses the data available through the provided iterator as a collection of
1814       * attributes suitable for use in an entry or an add change record.
1815       *
1816       * @param  dn                     The DN of the record being read.
1817       * @param  ignoreDuplicateValues  Indicates whether to ignore duplicate
1818       *                                attribute values encountered while parsing.
1819       * @param  schema                 The schema to use when parsing the
1820       *                                attributes, or {@code null} if none is
1821       *                                needed.
1822       * @param  ldifLines              The lines that comprise the LDIF
1823       *                                representation of the full record being
1824       *                                parsed.
1825       * @param  iterator               The iterator to use to access the attribute
1826       *                                lines.
1827       * @param  firstLineNumber        The line number for the start of the record.
1828       *
1829       * @return  The collection of attributes that were read.
1830       *
1831       * @throws  LDIFException  If the provided LDIF data cannot be decoded as a
1832       *                         set of attributes.
1833       */
1834      private static ArrayList<Attribute> parseAttributes(final String dn,
1835           final boolean ignoreDuplicateValues, final Schema schema,
1836           final ArrayList<StringBuilder> ldifLines,
1837           final Iterator<StringBuilder> iterator, final long firstLineNumber)
1838              throws LDIFException
1839      {
1840        final LinkedHashMap<String,Object> attributes =
1841             new LinkedHashMap<String,Object>(ldifLines.size());
1842        while (iterator.hasNext())
1843        {
1844          final StringBuilder line = iterator.next();
1845          final int colonPos = line.indexOf(":");
1846          if (colonPos <= 0)
1847          {
1848            throw new LDIFException(ERR_READ_NO_ATTR_COLON.get(firstLineNumber),
1849                                    firstLineNumber, true, ldifLines, null);
1850          }
1851    
1852          final String attributeName = line.substring(0, colonPos);
1853          final String lowerName     = toLowerCase(attributeName);
1854    
1855          final MatchingRule matchingRule;
1856          if (schema == null)
1857          {
1858            matchingRule = CaseIgnoreStringMatchingRule.getInstance();
1859          }
1860          else
1861          {
1862            matchingRule =
1863                 MatchingRule.selectEqualityMatchingRule(attributeName, schema);
1864          }
1865    
1866          Attribute attr;
1867          final LDIFAttribute ldifAttr;
1868          final Object attrObject = attributes.get(lowerName);
1869          if (attrObject == null)
1870          {
1871            attr     = null;
1872            ldifAttr = null;
1873          }
1874          else
1875          {
1876            if (attrObject instanceof Attribute)
1877            {
1878              attr     = (Attribute) attrObject;
1879              ldifAttr = new LDIFAttribute(attr.getName(), matchingRule,
1880                                           attr.getRawValues()[0]);
1881              attributes.put(lowerName, ldifAttr);
1882            }
1883            else
1884            {
1885              attr     = null;
1886              ldifAttr = (LDIFAttribute) attrObject;
1887            }
1888          }
1889    
1890          final int length = line.length();
1891          if (length == (colonPos+1))
1892          {
1893            // This means that the attribute has a zero-length value, which is
1894            // acceptable.
1895            if (attrObject == null)
1896            {
1897              attr = new Attribute(attributeName, "");
1898              attributes.put(lowerName, attr);
1899            }
1900            else
1901            {
1902              try
1903              {
1904                if (! ldifAttr.addValue(new ASN1OctetString()))
1905                {
1906                  if (! ignoreDuplicateValues)
1907                  {
1908                    throw new LDIFException(ERR_READ_DUPLICATE_VALUE.get(dn,
1909                         firstLineNumber, attributeName), firstLineNumber, true,
1910                         ldifLines, null);
1911                  }
1912                }
1913              }
1914              catch (LDAPException le)
1915              {
1916                throw new LDIFException(ERR_READ_VALUE_SYNTAX_VIOLATION.get(dn,
1917                     firstLineNumber, attributeName, getExceptionMessage(le)),
1918                     firstLineNumber, true, ldifLines, le);
1919              }
1920            }
1921          }
1922          else if (line.charAt(colonPos+1) == ':')
1923          {
1924            // Skip over any spaces leading up to the value, and then the rest of
1925            // the string is the base64-encoded attribute value.
1926            int pos = colonPos+2;
1927            while ((pos < length) && (line.charAt(pos) == ' '))
1928            {
1929              pos++;
1930            }
1931    
1932            try
1933            {
1934              final byte[] valueBytes = Base64.decode(line.substring(pos));
1935              if (attrObject == null)
1936              {
1937                attr = new Attribute(attributeName, valueBytes);
1938                attributes.put(lowerName, attr);
1939              }
1940              else
1941              {
1942                try
1943                {
1944                  if (! ldifAttr.addValue(new ASN1OctetString(valueBytes)))
1945                  {
1946                    if (! ignoreDuplicateValues)
1947                    {
1948                      throw new LDIFException(ERR_READ_DUPLICATE_VALUE.get(dn,
1949                           firstLineNumber, attributeName), firstLineNumber, true,
1950                           ldifLines, null);
1951                    }
1952                  }
1953                }
1954                catch (LDAPException le)
1955                {
1956                  throw new LDIFException(ERR_READ_VALUE_SYNTAX_VIOLATION.get(dn,
1957                       firstLineNumber, attributeName, getExceptionMessage(le)),
1958                       firstLineNumber, true, ldifLines, le);
1959                }
1960              }
1961            }
1962            catch (final ParseException pe)
1963            {
1964              debugException(pe);
1965              throw new LDIFException(ERR_READ_CANNOT_BASE64_DECODE_ATTR.get(
1966                                           attributeName,  firstLineNumber,
1967                                           pe.getMessage()),
1968                                      firstLineNumber, true, ldifLines, pe);
1969            }
1970          }
1971          else if (line.charAt(colonPos+1) == '<')
1972          {
1973            // Skip over any spaces leading up to the value, and then the rest of
1974            // the string is a URL that indicates where to get the real content.
1975            // At the present time, we'll only support the file URLs.
1976            int pos = colonPos+2;
1977            while ((pos < length) && (line.charAt(pos) == ' '))
1978            {
1979              pos++;
1980            }
1981    
1982            final String urlString = line.substring(pos);
1983            if (! toLowerCase(urlString).startsWith("file:/"))
1984            {
1985              throw new LDIFException(ERR_READ_URL_INVALID_SCHEME.get(attributeName,
1986                                           urlString, firstLineNumber),
1987                                      firstLineNumber, true, ldifLines, null);
1988            }
1989    
1990            pos = 6;
1991            while ((pos < urlString.length()) && (urlString.charAt(pos) == '/'))
1992            {
1993              pos++;
1994            }
1995    
1996            try
1997            {
1998              final File f = new File(urlString.substring(pos-1));
1999              if (! f.exists())
2000              {
2001                throw new LDIFException(ERR_READ_URL_NO_SUCH_FILE.get(attributeName,
2002                                             urlString, firstLineNumber,
2003                                             f.getAbsolutePath()),
2004                                        firstLineNumber, true, ldifLines, null);
2005              }
2006    
2007              // In order to conserve memory, we'll only allow values to be read
2008              // from files no larger than 10 megabytes.
2009              final long fileSize = f.length();
2010              if (fileSize > (10 * 1024 * 1024))
2011              {
2012                throw new LDIFException(ERR_READ_URL_FILE_TOO_LARGE.get(
2013                                             attributeName, urlString,
2014                                             firstLineNumber, f.getAbsolutePath(),
2015                                             (10*1024*1024)),
2016                                        firstLineNumber, true, ldifLines, null);
2017              }
2018    
2019              int fileBytesRead              = 0;
2020              int fileBytesRemaining         = (int) fileSize;
2021              final byte[]          fileData = new byte[(int) fileSize];
2022              final FileInputStream fis      = new FileInputStream(f);
2023              try
2024              {
2025                while (fileBytesRead < fileSize)
2026                {
2027                  final int bytesRead =
2028                       fis.read(fileData, fileBytesRead, fileBytesRemaining);
2029                  if (bytesRead < 0)
2030                  {
2031                    // We hit the end of the file before we expected to.  This
2032                    // shouldn't happen unless the file size changed since we first
2033                    // looked at it, which we won't allow.
2034                    throw new LDIFException(ERR_READ_URL_FILE_SIZE_CHANGED.get(
2035                                                 attributeName, urlString,
2036                                                 firstLineNumber,
2037                                                 f.getAbsolutePath()),
2038                                            firstLineNumber, true, ldifLines, null);
2039                  }
2040    
2041                  fileBytesRead      += bytesRead;
2042                  fileBytesRemaining -= bytesRead;
2043                }
2044    
2045                if (fis.read() != -1)
2046                {
2047                  // There is still more data to read.  This shouldn't happen unless
2048                  // the file size changed since we first looked at it, which we
2049                  // won't allow.
2050                  throw new LDIFException(ERR_READ_URL_FILE_SIZE_CHANGED.get(
2051                                               attributeName, urlString,
2052                                               firstLineNumber,
2053                                               f.getAbsolutePath()),
2054                                          firstLineNumber, true, ldifLines, null);
2055                }
2056              }
2057              finally
2058              {
2059                fis.close();
2060              }
2061    
2062              if (attrObject == null)
2063              {
2064                attr = new Attribute(attributeName, fileData);
2065                attributes.put(lowerName, attr);
2066              }
2067              else
2068              {
2069                if (! ldifAttr.addValue(new ASN1OctetString(fileData)))
2070                {
2071                  if (! ignoreDuplicateValues)
2072                  {
2073                    throw new LDIFException(ERR_READ_DUPLICATE_VALUE.get(dn,
2074                         firstLineNumber, attributeName), firstLineNumber, true,
2075                         ldifLines, null);
2076                  }
2077                }
2078              }
2079            }
2080            catch (LDIFException le)
2081            {
2082              debugException(le);
2083              throw le;
2084            }
2085            catch (Exception e)
2086            {
2087              debugException(e);
2088              throw new LDIFException(ERR_READ_URL_EXCEPTION.get(attributeName,
2089                                           urlString, firstLineNumber, e),
2090                                      firstLineNumber, true, ldifLines, e);
2091            }
2092          }
2093          else
2094          {
2095            // Skip over any spaces leading up to the value, and then the rest of
2096            // the string is the value.
2097            int pos = colonPos+1;
2098            while ((pos < length) && (line.charAt(pos) == ' '))
2099            {
2100              pos++;
2101            }
2102

2810      /**
2811       * When processing in asynchronous mode, this thread is responsible for
2812       * reading the raw unparsed records from the input and submitting them for
2813       * processing.
2814       */
2815      private final class LineReaderThread
2816           extends Thread
2817      {
2818        /**
2819         * Constructor.
2820         */
2821        private LineReaderThread()
2822        {
2823          super("Asynchronous LDIF line reader");
2824          setDaemon(true);
2825        }
2826    
2827    
2828    
2829        /**
2830         * Reads raw, unparsed records from the input and submits them for
2831         * processing until the input is finished or closed.
2832         */
2833        @Override()
2834        public void run()
2835        {
2836          try
2837          {
2838            boolean stopProcessing = false;
2839            while (!stopProcessing)
2840            {
2841              UnparsedLDIFRecord unparsedRecord = null;
2842              try
2843              {
2844                unparsedRecord = readUnparsedRecord();
2845              }
2846              catch (IOException e)
2847              {
2848                debugException(e);
2849                unparsedRecord = new UnparsedLDIFRecord(e);
2850                stopProcessing = true;
2851              }
2852              catch (Exception e)
2853              {
2854                debugException(e);
2855                unparsedRecord = new UnparsedLDIFRecord(e);
2856              }
2857    
2858              try
2859              {
2860                asyncParser.submit(unparsedRecord);
2861              }
2862              catch (InterruptedException e)
2863              {
2864                debugException(e);
2865                // If this thread is interrupted, then someone wants us to stop
2866                // processing, so that's what we'll do.
2867                stopProcessing = true;
2868              }
2869    
2870              if ((unparsedRecord == null) || (unparsedRecord.isEOF()))
2871              {
2872                stopProcessing = true;
2873              }
2874            }
2875          }
2876          finally
2877          {
2878            try
2879            {
2880              asyncParser.shutdown();
2881            }
2882            catch (InterruptedException e)
2883            {
2884              debugException(e);
2885            }
2886            finally
2887            {
2888              asyncParsingComplete.set(true);
2889            }
2890          }
2891        }
2892      }

2900                                                            LDIFRecord>

2903           * {@inheritDoc}
2904           */
2905          public LDIFRecord process(final UnparsedLDIFRecord input)
2906               throws LDIFException

2909
