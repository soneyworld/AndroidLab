 \begin{block}{Funktionale Anforderungen}

\section{Stand der Dinge}
\frame{
	\frametitle{Probleme}
	
 	  \begin{block}{Probleme}
   	      \begin{itemize}
                  \item Kontakte Anwendung von Google
                    \begin{itemize}
                      \item Nur Eingabe von Name moeglich
                    \end{itemize}
                  \item Importieren funktioniert
		  \item neue Kontakte im LDAP werden automatisch im Kontaktbuch erstellt
		  \item Kontakte löschen ist implementiert
		  \item Exportieren/Synchronisieren in Arbeit
                  \item Grafischen Oberfläche erstellt
                  \item LDAP Server läuft
                \end{itemize}
           \end{block}
}

\frame{
	\frametitle{Loesungen}
	
 	  \begin{block}{Loesungen}
   	      \begin{itemize}

                    \begin{itemize}
                      \item Aufgerufen durch Intent aus der Kontakte Anwendung
                    \end{itemize}
                  \item Importieren funktioniert
		  \item neue Kontakte im LDAP werden automatisch im Kontaktbuch erstellt
		  \item Kontakte löschen ist implementiert

\section{Änderungen}
\frame{
 \frametitle{Wichtige Komponenten} 
\begin{block}{RawContacts}
    \begin{itemize}
        \item Kontaktdaten werden pro Account gespeichert
	\item Sichtbarkeit im Kontaktbuch je nach Wunsch 
    \end{itemize}
 \end{block}
 \begin{block}{SyncAdapter}
    \begin{itemize}
        \item Schnittstelle für Android zur Synchronisation
    \end{itemize}
 \end{block}
